183 RexxObject *RexxMessage::notify(RexxMessage *_message)
184 /******************************************************************************/
185 /* Function: Add a message object to the notification list */
186 /******************************************************************************/
187 {
188  /* is argument a real message object?*/
189  if (message != OREF_NULL && isOfClass(Message, _message))
190  {
191  /* Yes, then add it to the */
192  /* toBeNotified list. */
193 
194  if (this->allNotified()) /* Have all notifications been sent? */
195  {
196  /* Yes, then send notification right */
197  _message->send(OREF_NULL); /* away */
198  }
199  else
200  {
201  /* nope, add it to list and wait for */
202  /* for result. */
203  this->interestedParties->addLast(_message);
204  }
205  }
206  else
207  { /* nope, its and error, report it. */
208  if ( message == OREF_NULL)
209  {
210  reportException(Error_Incorrect_method_noarg, OREF_positional, IntegerOne);
211  }
212  else
213  {
214  reportException(Error_Incorrect_method_nomessage, OREF_positional, _message);
215  }
216  }
217  return OREF_NULL; /* all done, we return nothing */
218 }

220 RexxObject *RexxMessage::result(void)
221 /******************************************************************************/
222 /* Function: Return the message result...will wait if the message isn't done */
223 /******************************************************************************/
224 {
225 
226  /* Did send/satrt cause an error */
227  /*condition Yes, we need to raise it */
228  /*here. */
229  if (this->raiseError())
230  {
231  ActivityManager::currentActivity->reraiseException(this->condition);
232  }
233  else
234  {
235  /* Quick test to see if result */
236  /*already present */
237  if (!this->resultReturned())
238  {
239  /* got an activity available? */
240  if (this->startActivity != OREF_NULL)
241  {
242  /* go perform dead lock checks */
243  this->startActivity->checkDeadLock(ActivityManager::currentActivity);
244  }
245 
246  /* No result yet, now we need to wait*/
247  /* until we get a result. */
248  /* Is anyone else waiting ???? */
249  if (this->waitingActivities == OREF_NULL)
250  {
251  /* No, Create a waiting list */
252  OrefSet(this, this->waitingActivities, new_list());
253  }
254  /* add this activity to the list */
255  this->waitingActivities->addLast((RexxObject *)ActivityManager::currentActivity);
256  /* now go wait to be woken up */
257  ActivityManager::currentActivity->waitReserve((RexxObject *)this);
258  if (this->raiseError()) /* do we need to raise an error. */
259  {
260  /* yes, */
261  this->setErrorReported(); /* indicate error was reported, and */
262  /* report and error. */
263  ActivityManager::currentActivity->reraiseException(this->condition);
264  }
265  }
266  }
267  return this->resultObject; /* ok, return the result. */
268 }

270 RexxObject *RexxMessage::send(RexxObject *_receiver)
271 /******************************************************************************/
272 /* Function: Send the message contained by this message object */
273 /******************************************************************************/
274 {
275  if (this->msgSent())
276  {
277  /* Yes, this is an error */
278  reportException(Error_Execution_message_reuse);
279  }
280 
281  /* get the activity I'm running under*/
282  RexxActivity *myActivity = (RexxActivity *)ActivityManager::currentActivity;
283  /* If we have a pending start message*/
284  /* sure this send is a result of */
285  /*that message dispatch. */
286  if (this->startPending() && myActivity != this->startActivity )
287  {
288  /* Yes, this is an error */
289  reportException(Error_Execution_message_reuse);
290  }
291  this->setMsgSent(); /* indicate we were sent a message */
292 
293  if (_receiver != OREF_NULL) /* new receiver specified? */
294  {
295  /* Yes, indicate this is the receiver*/
296  OrefSet(this, this->receiver, _receiver);
297  }
298  /* validate startscope */
299  if (startscope != OREF_NULL)
300  {
301  if (!this->receiver->behaviour->checkScope(this->startscope))
302  {
303  reportException(Error_Incorrect_method_array_noclass, IntegerTwo);
304  }
305  }
306  /* this is a primitive object? */
307  /* tell the activation/nativeact, we */
308  /*are running under to let us know */
309  /*if an error occured. */
310  myActivity->getTopStackFrame()->setObjNotify(this);
311  /* set this for resource deadlock */
312  /* checking purposes */
313  OrefSet(this, this->startActivity, myActivity);
314  ProtectedObject p(myActivity);
315  /* call message_send to do the send */
316  /* and assign our result. */
317  if (this->startscope != OREF_NULL)/* have a starting scope? */
318  {
319  /* send it with an override */
320  // this->receiver->messageSend(this->message, (RexxObject **)this->args->data(), this->args->size(), this->startscope, p);
321  this->receiver->messageSend(this->message, this->arglist, this->argcount, this->named_argcount, this->startscope, p);
322  }
323  else /* no over ride */
324  {
325  // this->receiver->messageSend(this->message, (RexxObject **)this->args->data(), this->args->size(), p);
326  this->receiver->messageSend(this->message, this->arglist, this->argcount, this->named_argcount, p);
327  }
328  this->resultObject = (RexxObject *)p;
329  this->setResultReturned(); /* Indicate we have a result. */
330  this->sendNotification();
331  return this->resultObject; /* return the result of the send. */
332 }

334 RexxObject *RexxMessage::start(RexxObject *_receiver)
335 /******************************************************************************/
336 /* Function: Since a start is to happen Async, we create a new activity */
337 /* using ourselves as a dispatch target. */
338 /******************************************************************************/
339 {
340  /* has message already been sent or */
341  /* is another start message pending? */
342  if (this->msgSent() || this->startPending())
343  {
344  /* Yes, this is an error */
345  reportException(Error_Execution_message_reuse);
346  }
347  /* indicate object has received a */
348  /*start we need this additional bit */
349  /*so that the send message will */
350  /*accept this msg */
351  this->setStartPending();
352 
353 
354  if (_receiver != OREF_NULL) /* new receiver specified? */
355  {
356  /* Yes, indicate this is the receiver*/
357  OrefSet(this, this->receiver, _receiver);
358  }
359 
360  /* get the current activity */
361  RexxActivity *oldActivity = ActivityManager::currentActivity;
362  /* Create the new activity */
363  RexxActivity *newActivity = oldActivity->spawnReply();
364  /* indicate the activity the send */
365  /*message should come in on. */
366  OrefSet(this, this->startActivity, newActivity);
367  // tell the activity to run this
368  newActivity->run(this);
369  return OREF_NULL; /* all done here, return to caller. */
370 }

372 void RexxMessage::sendNotification(void)
373 /******************************************************************************/
374 /* Function : we now have a result from message send/start, so notify */
375 /* all interested parties, and post the waitResult semopohore if it exists */
376 /******************************************************************************/
377 {
378  /* no longer care about any error */
379  /*condition */
380  ActivityManager::currentActivity->getTopStackFrame()->setObjNotify(OREF_NULL);
381  /* others waiting for a result? */
382  if (this->waitingActivities != OREF_NULL)
383  {
384  size_t i = this->waitingActivities->getSize();/* get the waiting count */
385  while (i--) /* while we have items */
386  {
387  /* get the first item */
388  RexxActivity *waitingActivity = (RexxActivity *)this->waitingActivities->removeFirst();
389  waitingActivity->postDispatch(); /* go wake it up */
390  }
391  }
392  /* now traverse the list of Iterested*/
393  /* parties, and let them know we */
394  /*have a result */
395  for (size_t listIndex = this->interestedParties->firstIndex() ;
396  listIndex != LIST_END;
397  listIndex = this->interestedParties->nextIndex(listIndex) )
398  {
399  /* Get the next message object to */
400  /*process */
401  RexxMessage *thisMessage = (RexxMessage *)this->interestedParties->getValue(listIndex);
402  /* now just have this message send */
403  /*its message */
404  thisMessage->send(OREF_NULL);
405  }
406 
407  /* indicate we notified all */
408  /*Interested parties. Not used */
409  /*yet.... */
410  this->setAllNotified();
411 }

533 RexxObject *RexxMessage::newRexx(
534  RexxObject **msgArgs, /* message argument array */
535  size_t argCount, /* the number of arguments */
536  size_t named_argCount)
537 /******************************************************************************/
538 /* Function: Rexx level new routine */
539 /******************************************************************************/
540 {
541  // this method is defined on the object class, but this is actually attached
542  // to a class object instance. Therefore, any use of the this pointer
543  // will be touching the wrong data. Use the classThis pointer for calling
544  // any methods on this object from this method.
545  RexxClass *classThis = (RexxClass *)this;
546  classThis->checkAbstract(); // ooRexx5
547 
548  RexxArray *argPtr = NULL; // the arguments used with the message.
549  ProtectedObject p_argPtr;
550  size_t argCountMsg = 0; // the count of positional arguments used with the message
551  size_t named_argCountMsg = 0; // the count of named arguments used with the message
552 
553  size_t num_args = argCount; /* get number of args passed */
554 
555  if (num_args < 2 ) /* passed less than 2 args? */
556  {
557  /* Yes, this is an error. */
558  reportException(Error_Incorrect_method_minarg, OREF_positional, IntegerTwo);
559  }
560  RexxObject *_target = msgArgs[0]; /* Get the receiver object */
561  requiredArgument(_target, OREF_positional, ARG_ONE);
562  RexxObject *_message = msgArgs[1]; /* get the message . */
563  requiredArgument(_message, OREF_positional, ARG_TWO);
564  RexxString *msgName;
565  RexxObject *_startScope;
566  // decode the message argument into name and scope
567  RexxObject::decodeMessageName(_target, _message, msgName, _startScope);
568  ProtectedObject m(msgName);
569 
570  /*
571  >>-new(target,messagename-+---------------------------------------------+-)--><
572  +-,Individual--| Arguments |------------------+
573  +--+-----------------+--+---------------------+
574  +-,Array,argument-+ +-,Directory,argument-+
575  */
576 
577  // TODO named arguments : add support for named argumenst
578 
579  /* are there arguments to be sent */
580  /*with the message? */
581  if (num_args > 2 )
582  {
583  /* get 3rd arg only concerned w/ 1st */
584  RexxString *optionString = (RexxString *)msgArgs[2];
585  /* Did we really get an option */
586  /*passed? */
587  if (optionString == OREF_NULL)
588  {
589  // /* nope, use null array as argument */
590  argPtr = (RexxArray *)TheNullArray->copy();
591  p_argPtr = argPtr;
592  argCountMsg = 0; // 0 positional argument
593  named_argCountMsg = 0;
594  }
595  else
596  {
597  /* Convert it into a string. */
598  optionString = stringArgument(optionString, OREF_positional, ARG_THREE);
599  /* char and make it lower case */
600  char option = tolower((int)optionString->getChar(0));
601  if (option == 'a') /* args passed as an array? */
602  {
603  /* are there less than 4 required */
604  /*args? */
605  if (num_args < 4) /* this is an error */
606  {
607  reportException(Error_Incorrect_method_minarg, OREF_positional, IntegerFour);
608  }
609 
610  /* are there more than 4 required */
611  /*args? */
612  if (num_args > 4) /* this is an error */
613  {
614  reportException(Error_Incorrect_method_maxarg, OREF_positional, IntegerFour);
615  }
616 
617  /* get the array of arguments */
618  argPtr = (RexxArray *)msgArgs[3];
619  if (argPtr == OREF_NULL) /* no array given? */
620  {
621  /* this is an error */
622  reportException(Error_Incorrect_method_noarg, OREF_positional, IntegerFour);
623  }
624  /* force to array form */
625  argPtr = (RexxArray *)REQUEST_ARRAY(argPtr);
626  /* not an array? */
627  if (argPtr == TheNilObject || argPtr->getDimension() != 1)
628  {
629  /* raise an error */
630  reportException(Error_Incorrect_method_noarray, OREF_positional, msgArgs[3]);
631  }
632 
633  p_argPtr = argPtr; // Protected against GC
634  argCountMsg = argPtr->size(); // count of positional arguments
635 
636  }
637  else if (option == 'i' ) /* specified as individual? */
638  {
639  /* yes, build array of all arguments */
640 
641  // makes a copy of msgArgs, including the named arguments
642  argPtr = new (argCount - 3 + 2 * named_argCount, msgArgs + 3) RexxArray;
643  p_argPtr = argPtr;
644  argCountMsg = argCount - 3;
645  named_argCountMsg = named_argCount;
646  }
647  else
648  {
649  reportException(Error_Incorrect_method_option, "AI", msgArgs[2]);
650  }
651  }
652  }
653  else
654  {
655  // /* no args, use a null array. */
656  argPtr = (RexxArray *)TheNullArray->copy();
657  p_argPtr = argPtr;
658  argCountMsg = 0; // 0 positional argument
659  named_argCountMsg = 0;
660  }
661  /* all args are parcelled out, go */
662  /*create the new message object... */
663  RexxMessage *newMessage = new RexxMessage(_target, msgName, _startScope, argPtr->data(), argCountMsg, named_argCountMsg);
664  /* actually a subclassed item? */
665  if (classThis->isPrimitive())
666  {
667  ProtectedObject p(newMessage);
668  /* Give new object its behaviour */
669  newMessage->setBehaviour(classThis->getInstanceBehaviour());
670  newMessage->sendMessage(OREF_INIT);/* call any rexx inits */
671  }
672  return newMessage; /* return the new message */
673 }
