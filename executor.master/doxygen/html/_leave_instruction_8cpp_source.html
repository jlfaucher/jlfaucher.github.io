94 void RexxInstructionLeave::execute(
95  RexxActivation *context, /* current activation context */
96  RexxExpressionStack *stack ) /* evaluation stack */
97 /****************************************************************************/
98 /* Function: Execute a REXX LEAVE instruction */
99 /****************************************************************************/
100 {
101  context->traceInstruction(this); /* trace if necessary */
102  /* is it a LEAVE? */
103  if (this->instructionType == KEYWORD_LEAVE)
104  {
105  /* tell the activation to exit the */
106  context->leaveLoop(this->name); /* appopriate loop */
107  }
108  else
109  {
110  /* tell the activation to iterate */
111  context->iterate(this->name); /* the appopriate loop */
112  }
113  context->pauseInstruction(); /* do debug pause if necessary */
114 }
