49 class ActivationFrame
50 {
51 friend class RexxActivity;
52 public:
53  inline ActivationFrame(RexxActivity *a) : activity(a)
54  {
55  // it would be better to have the activity class do this, but because
56  // we're doing this with inline methods, we run into a bit of a
57  // circular reference problem
58  next = activity->activationFrames;
59  activity->activationFrames = this;
60  }
61 
62  inline ~ActivationFrame()
63  {
64  // remove ourselves from the list and give this object a
65  // little hold protection.
66  activity->activationFrames = next;
67  }
68 
69  virtual RexxString *messageName() = 0;
70  virtual RexxMethod *method() = 0;
71  virtual StackFrameClass *createStackFrame() = 0;
72  virtual RexxSource *getSource() = 0;
73 
74 protected:
75  ActivationFrame *next; // the next activation frame in the chain
76  RexxActivity *activity; // the activity we're running on
77 };
