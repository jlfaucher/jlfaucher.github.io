59 void RexxInstructionOptions::execute(
60  RexxActivation *context, /* current activation context */
61  RexxExpressionStack *stack) /* evaluation stack */
62 /******************************************************************************/
63 /* Function: Execute a REXX OPTIONS instruction */
64 /******************************************************************************/
65 {
66  RexxObject *value; /* output value */
67  RexxString *stringVal; /* string version of the value */
68  size_t i; /* loop counter */
69  RexxString *word; /* current word */
70 
71  context->traceInstruction(this); /* trace if necessary */
72  /* get the expression value */
73  value = this->expression->evaluate(context, stack);
74  stringVal = REQUEST_STRING(value); /* get the string version */
75  ProtectedObject p(stringVal);
76  context->traceResult(value); /* trace the output value */
77  for (i = 1; ;i++) /* now process each word */
78  {
79  /* get the next word */
80  word = (RexxString *)(stringVal->word(new_integer(i)));
81  if (word->getBLength() == 0) /* get the length of the word */
82  {
83  break; /* if length of word = 0 then stop */
84  }
85 
86  if (word->strCaselessCompare(CHAR_COMMANDS))
87  {
88  context->enableCommands(true);
89  }
90 
91  else if (word->strCaselessCompare(CHAR_NOCOMMANDS))
92  {
93  context->enableCommands(false);
94  }
95 
96  else if (word->strCaselessCompare(CHAR_MACROSPACE))
97  {
98  context->enableMacrospace(true);
99  }
100 
101  else if (word->strCaselessCompare(CHAR_NOMACROSPACE))
102  {
103  context->enableMacrospace(false);
104  }
105 
106 #ifdef _DEBUG
107  else if (word->strCaselessCompare("DUMPMEMORY"))
108  {
109  memoryObject.dumpEnable = true;
110  memoryObject.dump();
111  }
112 #endif
113  }
114  context->pauseInstruction(); /* do debug pause if necessary */
115 }
