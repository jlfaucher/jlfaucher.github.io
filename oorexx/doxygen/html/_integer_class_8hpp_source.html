210 inline RexxInteger *new_integer(wholenumber_t v) { return TheIntegerClass->newCache(v); }
211 inline RexxInteger *new_integer(bool sign, const char *digits, wholenumber_t intDigits, wholenumber_t pad = 0)
212 {
213  wholenumber_t v, i;
214  v = digits[0];
215  for (i = 1; i < intDigits; i++)
216  {
217  // add digits
218  v = v * 10 + digits[i];
219  }
220  for (i = 1; i <= pad; i++)
221  {
222  // add zeros
223  v = v * 10;
224  }
225  // return value with correct sign
226  return TheIntegerClass->newCache(sign ? -v : v);
227 }

229 inline wholenumber_t length_in_bits(wholenumber_t v)
230 {
231  wholenumber_t r = 0;
232 
233  if (v < 0) v = -v;
234 #ifdef __REXX64__
235  if (v & 0xFFFFFFFF00000000) { v >>= 32; r |= 32; }
236 #endif
237  if (v & 0xFFFF0000) { v >>= 16; r |= 16; }
238  if (v & 0xFF00) { v >>= 8; r |= 8; }
239  if (v & 0xF0) { v >>= 4; r |= 4; }
240  if (v & 0xC) { v >>= 2; r |= 2; }
241  if (v & 0x2) { v >>= 1; r |= 1; }
242  return r + 1;
243 }
