103 /**
104  * Special stack for managing the stack frames used by the
105  * activations running on an Activity.
106  */
107 class ActivationStack
108 {
109  public:
110 
111  enum { DefaultFrameBufferSize = 2048 };
112  ActivationStack() { ; }
113 
114  void live(size_t);
115  void liveGeneral(MarkReason reason);
116 
117  void init();
118  void expandCapacity(size_t entries);
119 
120  inline void ensureCapacity(size_t entries) { if (!current->hasCapacity(entries)) { expandCapacity(entries); } }
121  inline RexxInternalObject **allocateFrame(size_t entries)
122  {
123  // make sure we have space first
124  ensureCapacity(entries);
125  // now allocate from the current stack buffer
126  return current->allocateFrame(entries);
127  }
128  void releaseFrame(RexxInternalObject **frame)
129  {
130  // we may be popping back one or more buffers. We deactivate the newer ones
131  while (!current->contains(frame))
132  {
133  // we need to pop at least one buffer off of the stack
134  ActivationFrameBuffer *released = current;
135  current = released->getPrevious();
136  // we'll keep at least one buffer around for reuse. If
137  // we've already got one in the cache, just let this one
138  // get GCed.
139  if (unused == OREF_NULL)
140  {
141  unused = released;
142  unused->reset(); // reset to clean state
143  }
144  }
145 
146  // now back this up to the release point
147  current->releaseFrame(frame);
148  }
149 
150 protected:
151 
152  ActivationFrameBuffer *current; // our current frame buffer
153  ActivationFrameBuffer *unused; // our cached unused buffer(s)
154 };
