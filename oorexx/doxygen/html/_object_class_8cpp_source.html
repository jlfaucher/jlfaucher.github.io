406 /**
407  * Hash an exported object. If we're a non-primitive one, this
408  * will require us to send the HASHCODE message to request a
409  * hash value.
410  *
411  * @return A "hashed hash" that can be used by the map collections.
412  */
413 HashCode RexxObject::hash()
414 {
415  // if this is a primitive object, we can just return the primitive hash code.
416  if (isBaseClass())
417  {
418  return getHashValue();
419  }
420  else
421  {
422  ProtectedObject result;
423  // we have some other type of object, so we need to request a hash code
424  // by sending the HASHCODE() message.
425  sendMessage(GlobalNames::HASHCODE, result);
426 
427  // we need to have a return value for this.
428  if (result.isNull())
429  {
430  reportException(Error_No_result_object_message, GlobalNames::HASHCODE);
431  }
432 
433  // the default version sends us a string containing binary data.
434  // if the string is long enough for that, we reverse the process. Otherwise,
435  // we'll just take the hash code from the string object.
436  return result->stringValue()->getObjectHashCode();
437  }
438 }

546 /**
547  * Copy an object that is visible to the Rexx programmer.
548  * This copy ensures that all object variables are copied,
549  * as well as ensuring that the new instance has the
550  * same behaviour as the initial object.
551  *
552  * @return A new instance of this object.
553  */
554 RexxInternalObject *RexxObject::copy()
555 {
556  // Instead of calling new_object and memcpy, ask the memory object to make
557  // a copy of ourself. This way, any header information can be correctly
558  // initialized by memory.
559  RexxObject *newObj = (RexxObject *)clone();
560  ProtectedObject p(newObj);
561 
562  // do we have object variables? We need to give that object
563  // a copy of the variables
564  copyObjectVariables(newObj);
565 
566  // have instance methods?
567  if (behaviour->hasInstanceMethods())
568  {
569  // need to copy the behaviour
570  newObj->setBehaviour((RexxBehaviour *)newObj->behaviour->copy());
571  }
572 
573  // if the source object has an uninit method, then so does the copy. Make
574  // sure it is added to the table.
575  if (hasUninit())
576  {
577  newObj->requiresUninit();
578  }
579  return newObj;
580 }

601 /**
602  * Check a private method for accessibility.
603  *
604  * @param method The method object to check
605  * @param error An error to be raised if this is not permitted.
606  *
607  * @return An executable method, or OREF_NULL if this cannot be called.
608  */
609 MethodClass *RexxObject::checkPrivate(MethodClass *method, RexxErrorCodes &error)
610 {
611  // get the calling activation context
612  ActivationBase *activation = ActivityManager::currentActivity->getTopStackFrame();
613  if (activation != OREF_NULL)
614  {
615  // if the sending and receiving object are the same, this is allowed.
616  RexxObject *sender = activation->getReceiver();
617  if (sender == this)
618  {
619  return method;
620  }
621  // no sender means this is a routine or program context. Definitely not allowed.
622  if (sender == OREF_NULL)
623  {
624  error = Error_No_method_private;
625  return OREF_NULL;
626  }
627  // ok, now we check the various scope possibilities
628  RexxClass *scope = method->getScope();
629  // 1) Another instance of the same class that defined the method?
630  if (sender->isInstanceOf(scope) )
631  {
632  return method; // ok, we'll allow this
633  }
634  // if the sender is a class object, check the class for compatibility with the
635  // method scope
636  if (isOfClassType(Class, sender))
637  {
638  // if this class is part of the compatible hierarchy, this is also permitted
639  if (((RexxClass *)sender)->isCompatibleWith(scope))
640  {
641  return method;
642  }
643  }
644  }
645  // can't touch this...
646  error = Error_No_method_private;
647  return OREF_NULL;
648 }

691 /**
692  * Check that methods like RUN, SETMETHOD, and UNSETMETHOD are
693  * issued from a method on the same object instance.
694  *
695  * @return An executable method, or OREF_NULL if this cannot be called.
696  */
697 void RexxObject::checkRestrictedMethod(const char *methodName)
698 {
699  // get the calling activation context
700  ActivationBase *activation = ActivityManager::currentActivity->getTopStackFrame();
701  if (activation != OREF_NULL)
702  {
703  // Unlike private methods, this is only allowed from the instance, or from
704  // the class. Because RUN, SETMETHOD, and UNSETMETHOD are defined by Object,
705  // this essentially means the only restrictions on these private methods under
706  // the new rules is that it must be called from another method. Seriously
707  // NOT the intent of making private methods interact a little more. Therefore,
708  // we add an additional check for any private methods defined by .object
709  RexxObject *sender = activation->getReceiver();
710  if (sender == this)
711  {
712  return;
713  }
714 
715  // no sender means this is a routine or program context. Definitely not allowed.
716  if (sender == OREF_NULL)
717  {
718  reportException(Error_Execution_private_access, methodName);
719  }
720 
721  // if the sender is a class object, check the class for compatibility with the
722  // method scope
723  if (isOfClassType(Class, sender))
724  {
725  // if this class is part of the object's hierarchy, this is also permitted
726  if (isInstanceOf((RexxClass *)sender))
727  {
728  return;
729  }
730  }
731  // we have this issued from an invalid context.
732  reportException(Error_Execution_private_access, methodName);
733  }
734 }

1231  * through the whole search order to do the conversion.
1232  *
1233  * @return The converted string value or TheNilObject.
1234  */
1235 RexxString *RexxInternalObject::requestString()
1236 {
1237  // if we have a primitive object, we can get the string value directly.
1238  if (isBaseClass())
1239  {
1240  RexxString *string_value = primitiveMakeString();
1241  // if this didn't convert, we need to try to raise the NOSTRING condition.
1242  if (string_value == TheNilObject)
1243  {
1244  // get the ultimate, really needs to return a string string value for raising the condition.
1245  string_value = stringValue();
1246  // and raise nostring. If not trapped, this returns here and we return the final string value as a result.
1247  ActivityManager::currentActivity->raiseCondition(GlobalNames::NOSTRING, OREF_NULL, string_value, (RexxObject *)this, OREF_NULL);
1248  }
1249  return string_value;
1250  }
1251  else
1252  {
1253  // we need to do this via a real request message.
1254  ProtectedObject string_value;
1255 
1256  ((RexxObject *)this)->sendMessage(GlobalNames::REQUEST, GlobalNames::STRING, string_value);
1257  // The returned value might be an Integer or NumberString value. We need to
1258  // force this to be a real string value.

1266  // here and returning a value. It is possible there is a

1268  // isn't the most ideal message since the error is raised at the

1270  // situation. As a fallback, use the default object STRING method,
1271  // then raise an error if we still don't get anything. This at least
1272  // keeps the interpreter from crashing, there's a good chance the
1273  // program will run. Frankly, there's something seriously wrong
1274  // if this error ever gets issued.
