128 /**
129  * Create a new thread.
130  *
131  * @param threadNumber
132  * The returned thread ID
133  * @param stackSize The required stack size
134  * @param startRoutine
135  * The thread startup routine.
136  * @param startArgument
137  * The typeless argument passed to the startup routine.
138  *
139  * @return The success/failure return code.
140  */
141 int SysThread::createThread(pthread_t &id, bool &idValid, size_t stackSize, void *(*startRoutine)(void *), void *startArgument)
142 {
143  pthread_attr_t newThreadAttr;
144 
145  // Create an attr block for Thread.
146  pthread_attr_init(&newThreadAttr);
147 
148  // Set the stack size.
149  pthread_attr_setstacksize(&newThreadAttr, stackSize);
150 
151  // Now create the thread
152  int rc = pthread_create(&id, &newThreadAttr, startRoutine, (void *)startArgument);
153  idValid = rc == 0; // only on success we have a valid id
154  pthread_attr_destroy(&newThreadAttr);
155  return rc;
156 }

183 /**
184  * Platform wrapper around a sleep function that can sleep for long periods of time. .
185  *
186  * @param microseconds
187  * The number of microseconds to delay.
188  *
189  * @return The success/failure return code.
190  */
191 int SysThread::longSleep(uint64_t microseconds)
192 {
193  // split into two part: secs and nanoseconds
194  long secs = (long)microseconds / 1000000;
195  long nanoseconds = (long)(microseconds % 1000000) * 1000;
196 
197 #if defined( HAVE_NANOSLEEP )
198  struct timespec Rqtp, Rmtp;
199  Rqtp.tv_sec = secs;
200  Rqtp.tv_nsec = nanoseconds;
201  return nanosleep(&Rqtp, &Rmtp) == -1 ? errno : 0;
202 #elif defined( HAVE_NSLEEP )
203  struct timestruc_t Rqtp, Rmtp;
204  Rqtp.tv_sec = secs;
205  Rqtp.tv_nsec = nanoseconds;
206  return nsleep(&Rqtp, &Rmtp) == -1 ? errno : 0;
207 #else
208  return usleep(microseconds) == -1 ? errno : 0;
209 #endif
210 }
