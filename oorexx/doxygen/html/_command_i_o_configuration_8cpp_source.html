127 /**
128  * Create an active IO redirection context for issuing
129  * a command.
130  *
131  * @param context The current activation context the command is running under
132  * @param stack The expression stack used for evaluations
133  * @param commandConfig
134  * A configuration explicitly specified with this command
135  * which can override the global configuration.
136  *
137  * @return An instance of an I/O context for this configuration.
138  */
139 CommandIOContext *CommandIOConfiguration::createIOContext(RexxActivation *context, ExpressionStack *stack, CommandIOConfiguration *commandConfig)
140 {
141  // create the runtime I/O context object and fill in the needed sections
142  Protected<CommandIOContext> ioContext = new CommandIOContext();
143 
144  // we use a chaining strategy here. If we have a configuration from the command, then we ask
145  // it to merge each item. Otherwise we create the item directly from us
146  if (commandConfig != OREF_NULL)
147  {
148  ioContext->input = commandConfig->createInputSource(context, stack, this);
149  ioContext->output = commandConfig->createOutputTarget(context, stack, this);
150  ioContext->error = commandConfig->createErrorTarget(context, stack, this);
151  }
152  // we just use what we already have
153  else
154  {
155  ioContext->input = createInputSource(context, stack);
156  ioContext->output = createOutputTarget(context, stack);
157  ioContext->error = createErrorTarget(context, stack);
158  }
159 
160  // if the same objects are getting used for both input and output, then
161  // we need to insert a buffering layer in front of the output object.
162  ioContext->resolveConflicts();
163  return ioContext;
164 }

263 /**
264  * Create an input source redirector from an evaluated
265  * source object.
266  *
267  * @param context The current execution context.
268  * @param stack The current context stack.
269  *
270  * @return An appropriate input source for the object.
271  */
272 InputRedirector *CommandIOConfiguration::createInputSource(RexxActivation *context, ExpressionStack *stack)
273 {
274  // if there's nothing configured, just return null
275  if (inputSource == OREF_NULL)
276  {
277  return OREF_NULL;
278  }
279 
280  RexxObject *inputObject = inputSource->evaluate(context, stack);
281  // need to trace this if on
282  context->traceKeywordResult(GlobalNames::INPUT, inputObject);
283 
284  // process based on the type of object we've been presented with.
285  switch (inputType)
286  {
287  // if STEM was specified, then the syntax checks requires a STEM variable
288  // symbol follow the option. Therefore, we know this will have evaluated
289  // to stem object. This is the easiest of the checks;
290  case RedirectionType::STEM_VARIABLE:
291  {
292  return new StemInputSource((StemClass *)inputObject);
293  }
294  // a stream specified by name. This must be convertable to string form and
295  // will be used to create a stream object.
296  case RedirectionType::STREAM_NAME:
297  {
298  // this must be a string
299  Protected<RexxString> streamName = inputObject->requestString();
300 
301  // because we need to detect conflicts between input and outputs,
302  // we need to use the fully qualified file name here.
303  streamName = Interpreter::qualifyFileSystemName(streamName);
304 
305  return new StreamInputSource(streamName);
306  }
307 
308  // this was ADDRESS .. WITH INPUT USING expr. We dynamically determine from
309  // the type what to do.
310  default:
311  {
312  // Checks proceed in the following order, with the result
313  // 1) string object. This will be a single line written to the input pipe
314  // (uses the array input source to produce).
315  // 2) stem object. This is handled like the explicit STEM option.
316  // 3) stream object. This is a stream input source.
317  // 4) a makearray is performed on the object. If convertible, we use
318  // an array source.
319 
320  if (::isString(inputObject))
321  {
322  Protected<ArrayClass> source = new_array(inputObject);
323  return new ArrayInputSource(source);
324  }
325  // an evaluated stem object
326  if (::isStem(inputObject))
327  {
328  return new StemInputSource((StemClass *)inputObject);
329  }
330 
331  // Now we need to check for input streams
332  RexxClass *streamClass = TheRexxPackage->findClass(GlobalNames::INPUTSTREAM);
333  if (inputObject->isInstanceOf(streamClass))
334  {
335  return new StreamObjectInputSource(inputObject);
336  }
337 
338  // Now we treat monitors as if they are input streams.
339  RexxClass *monitorClass = TheRexxPackage->findClass(GlobalNames::MONITOR);
340  if (inputObject->isInstanceOf(monitorClass))
341  {
342  // for the purposes of this, a monitor and a stream object are treated the same.
343  return new StreamObjectInputSource(inputObject);
344  }
345 
346  // OK, now try for a FILE object
347  RexxClass *fileClass = TheRexxPackage->findClass(GlobalNames::FILE);
348  if (inputObject->isInstanceOf(fileClass))
349  {
350  ProtectedObject result;
351  // this just uses lineout
352  RexxString *streamName = (RexxString *)inputObject->sendMessage(GlobalNames::ABSOLUTEPATH, result);
353 
354  return new StreamObjectInputSource(streamName);
355  }
356 
357  // this must be convertable to some sort of array past this point
358  Protected<ArrayClass> array;
359  if (isArray(inputObject))
360  {
361  array = ((ArrayClass *)inputObject)->makeArray();
362  }
363  else
364  {
365  // some other type of collection, use the less direct means
366  // of requesting an array
367  array = inputObject->requestArray();
368  // raise an error if this did not convert ok, or we got
369  // back something other than a real Rexx array.
370  if (!isArray(array))
371  {
372  reportException(Error_Execution_address_input_source, inputObject);
373  }
374  }
375  return new ArrayInputSource(array);
376  }
377  }
378  // nothing to create
379  return OREF_NULL;
380 }

424 /**
425  * Create an appropriate output target for an ADDRESS WITH
426  * instruction.
427  *
428  * @param context The current execution context.
429  * @param stack The current context stack.
430  * @param outputObject
431  * The resolved output target
432  * @param type The specified type of target
433  *
434  * @return A resolved and constructed output target.
435  */
436 OutputRedirector *CommandIOConfiguration::createOutputTarget(RexxString *keyword, RexxActivation *context, ExpressionStack *stack, RexxInternalObject *outputTarget, RedirectionType::Enum type, OutputOption::Enum option)
437 {
438  RexxObject *outputObject = outputTarget->evaluate(context, stack);
439  // need to trace this if on
440  context->traceKeywordResult(keyword, outputObject);
441  switch (type)
442  {
443  case RedirectionType::STEM_VARIABLE:
444  {
445  return new StemOutputTarget((StemClass *)outputObject, option);
446  }
447  case RedirectionType::STREAM_NAME:
448  {
449  // this must be a string
450  Protected<RexxString> streamName = outputObject->requestString();
451  // because we need to detect conflicts between input and outputs,
452  // we need to use the fully qualified file name here.
453  streamName = Interpreter::qualifyFileSystemName(streamName);
454 
455  return new StreamOutputTarget(streamName, option);
456  }
457  default:
458  {
459  // an evaluated stem object
460  if (::isStem(outputObject))
461  {
462  return new StemOutputTarget((StemClass *)outputObject, option);
463  }
464 
465  // Now we need to check for output streams
466  RexxClass *streamClass = TheRexxPackage->findClass(GlobalNames::OUTPUTSTREAM);
467  // Now we treat monitors as if they are input streams.
468  RexxClass *monitorClass = TheRexxPackage->findClass(GlobalNames::MONITOR);
469 
470  // monitors are treated as if they are stream objects.
471  if (outputObject->isInstanceOf(streamClass) || outputObject->isInstanceOf(monitorClass))
472  {
473  // REPLACE or APPEND does not make sense for an arbitrary inputstream
474  // object that might not even be a file stream. raise an error for anything
475  // other than default
476  if (option != OutputOption::DEFAULT)
477  {
478  reportException(Error_Execution_using_stream_option);
479  }
480 
481  return new StreamObjectOutputTarget(outputObject, option);
482  }
483 
484  // OK, now try for a RexxQueue object
485  RexxClass *rexxQueueClass = TheRexxPackage->findClass(GlobalNames::REXXQUEUE);
486  if (outputObject->isInstanceOf(rexxQueueClass))
487  {
488  // REPLACE or APPEND does not make sense for a queue
489  // object. raise an error for anything
490  // other than default
491  if (option != OutputOption::DEFAULT)
492  {
493  reportException(Error_Execution_using_rexxqueue_option);
494  }
495  return new RexxQueueOutputTarget(outputObject);
496  }
497 
498  // OK, now try for a FILE object
499  RexxClass *fileClass = TheRexxPackage->findClass(GlobalNames::FILE);
500  if (outputObject->isInstanceOf(fileClass))
501  {
502  ProtectedObject result;
503  // this just uses lineout
504  RexxString *streamName = (RexxString *)outputObject->sendMessage(GlobalNames::ABSOLUTEPATH, result);
505 
506  return new StreamOutputTarget(streamName, option);
507  }
508 
509  // Now some sort of ordered collection
510  RexxClass *orderedCollection = TheRexxPackage->findClass(GlobalNames::ORDEREDCOLLECTION);
511  if (outputObject->isInstanceOf(orderedCollection))
512  {
513  return new CollectionOutputTarget(outputObject, option);
514  }
515  // an unknown type of target
516  reportException(Error_Execution_address_output_target, outputObject);
517  }
518  }
519  return OREF_NULL;
520 }
