92 /**
93  * This was an undocumented API prior to 4.0, but is known to have been used by
94  * some IBM applications. Therefore this was maintained solely for binary
95  * compatibility.
96  *
97  * However, it is now also used to turn off processing Windows messages in a
98  * special case situation. See the SysSemaphore::waitHandle() comments for more
99  * info. This is done on a per-thread basis, RexxSetProcessMessages() must be
100  * invoked while executing on the proper thread.
101  *
102  * @return TRUE always.
103  */
104 BOOL APIENTRY RexxSetProcessMessages(BOOL turnOn)
105 {
106  if ( ! turnOn )
107  {
108  SysSemaphore::setNoMessageLoop();
109  }
110  return TRUE;
111 }
