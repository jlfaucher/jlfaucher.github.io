52  * Take in an argument passed to a method, convert it to a length
53  * object, verifying that the number is a non-negative value.
54  * If the argument is omitted, an error is raised.
55  *
56  * @param argument The argument reference to test.
57  * @param position The position of the argument (used for error reporting.)
58  *
59  * @return The argument converted to a non-negative integer value.
60  */
61 size_t lengthArgument(RexxObject *argument, size_t position)
62 {
63  if (argument == OREF_NULL)
64  {
65  missingArgument(position);
66  }
67  size_t value;
68  // converted using the ARGUMENT_DIGITS value
69  if (!argument->unsignedNumberValue(value, Numerics::ARGUMENT_DIGITS))
70  {
71  reportException(Error_Incorrect_method_length, argument);
72  }
73  return value;
74 }

410 /**
411  * Validate that an array is non-space up to the last item
412  * and contains nothing but string values.
413  *
414  * @param argArray The argument array.
415  * @param position The argument name for error reporting.
416  */
417 void stringArrayArgument(ArrayClass *argArray, const char *position)
418 {
419  for (size_t counter = 1; counter <= argArray->lastIndex(); counter++)
420  {
421  RexxString *sourceString = (RexxString *)argArray->get(counter);
422  // we must have something here
423  if (sourceString == OREF_NULL)
424  {
425  reportException(Error_Incorrect_method_nostring_inarray, position);
426  }
427  // and it must be convertable to a string value
428  sourceString = sourceString->makeString();
429  // if this did not convert, this is an error
430  if (sourceString == (RexxString *)TheNilObject)
431  {
432  reportException(Error_Incorrect_method_nostring_inarray, position);
433  }
434  else
435  {
436  // replace the original item in the array
437  argArray->put(sourceString, counter);
438  }
439  }
