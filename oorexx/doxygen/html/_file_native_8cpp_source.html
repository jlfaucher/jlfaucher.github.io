258 /**
259  * Get a list of the file children for a directory.
260  */
261 RexxMethod1(RexxObjectPtr, file_list, CSTRING, name)
262 {
263  if (!SysFileSystem::isDirectory(name))
264  {
265  return context->Nil();
266  }
267 
268  // create an empty array to start
269  RexxArrayObject result = context->NewArray(0);
270 
271  MethodFileNameBuffer buffer(context);
272  MethodFileNameBuffer nextFile(context);
273 
274  SysFileIterator iterator(name, NULL, buffer);
275  SysFileIterator::FileAttributes attributes;
276 
277  while (iterator.hasNext())
278  {
279  iterator.next(nextFile, attributes);
280  // don't include the "." and ".." directories in this list
281  if (nextFile != "." && nextFile != "..")
282  {
283  context->ArrayAppendString(result, nextFile, nextFile.length());
284  }
285  }
286 
287  return result;
288 }
