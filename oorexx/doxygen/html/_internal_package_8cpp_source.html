86 /********************************************************************************************/
87 /* sysFilespec */
88 /********************************************************************************************/
89 RexxRoutine2(RexxStringObject, sysFilespec, CSTRING, option, CSTRING, name)
90 {
91  const char *endPtr = name + strlen(name); // point to last character
92 
93  const char *pathStart = SysFileSystem::getPathStart(name);
94  const char *pathEnd = SysFileSystem::getPathEnd(name);
95 
96  // get the end of the path portion (if any)
97  // note that pathend is one character past the end of the path.
98  // this means the length is easily calculated as pathEnd - pathStart,
99  // even in the cases where there is no patch portion
100  pathEnd = pathEnd == NULL ? pathStart : pathEnd + 1;
101  // this one needs a little adjustment for the case where this is all name
102  const char *nameStart = pathEnd == name ? name : pathEnd;
103 
104  switch (Utilities::toUpper(*option)) // process each option
105  {
106  case FILESPEC_PATH: // extract the path (without drive)
107  {
108  return context->String(pathStart, pathEnd - pathStart);
109  }
110 
111  case FILESPEC_NAME: // extract the file name
112  {
113  return context->String(nameStart, endPtr - nameStart);
114  }
115 
116  case FILESPEC_LOCATION: // extract the path (including drive)
117  {
118  return context->String(name, pathEnd - name);
119  }
120 
121  case FILESPEC_DRIVE: // extract the drive
122  {
123  // this will return a null string if nothing is before the pathStart
124  return context->String(name, pathStart - name);
125  }
126 
127  case FILESPEC_EXTENSION: // extract the file extension
128  {
129  // find the position of the last dot
130  const char *lastDot = strrchr(name, '.');
131 
132  if (lastDot >= nameStart)
133  {
134  // we don't extract the period
135  lastDot++;
136  return context->String(lastDot, endPtr - lastDot);
137  }
138  else
139  {
140  return context->NullString(); // nothing found, return the empty string
141  }
142 
143  }
144  default: /* unknown option */
145  {
146  char optionChar[2];
147  optionChar[0] = *option;
148  optionChar[1] = '\0';
149 
150  RexxArrayObject subs = context->Array(context->String("FILESPEC"), context->WholeNumberToObject(1),
151  context->String("DELNP"), context->String(optionChar));
152  /* raise an error */
153  context->RaiseException(Rexx_Error_Incorrect_call_list, subs);
154  return NULLOBJECT;
155  }
156  }
157 }

165 /*********************************************************************/
166 /* */
167 /* Subroutine Name: sysBeep */
168 /* */
169 /* Descriptive Name: BEEP function */
170 /* */
171 /* Function: sounds the speaker at frequency Hertz for */
172 /* specified duration (in milliseconds) */
173 /*********************************************************************/
174 RexxRoutine2(CSTRING, sysBeep, wholenumber_t, Frequency, wholenumber_t, Duration)
175 {
176  /* out of range? */
177  if (Frequency > MAX_FREQUENCY || Frequency < MIN_FREQUENCY)
178  {
179  RexxArrayObject subs = context->NewArray(4);
180  context->ArrayAppend(subs, context->NewStringFromAsciiz("frequency"));
181  context->ArrayAppend(subs, context->WholeNumberToObject(MIN_FREQUENCY));
182  context->ArrayAppend(subs, context->WholeNumberToObject(MAX_FREQUENCY));
183  context->ArrayAppend(subs, context->WholeNumberToObject(Frequency));
184  context->RaiseException(Rexx_Error_Invalid_argument_range, subs);
185  return NULL;
186  }
187  /* out of range? */
188  if (Duration > MAX_DURATION || Duration < MIN_DURATION)
189  {
190  RexxArrayObject subs = context->NewArray(4);
191  context->ArrayAppend(subs, context->NewStringFromAsciiz("duration"));
192  context->ArrayAppend(subs, context->WholeNumberToObject(MIN_DURATION));
193  context->ArrayAppend(subs, context->WholeNumberToObject(MAX_DURATION));
194  context->ArrayAppend(subs, context->WholeNumberToObject(Duration));
195  context->RaiseException(Rexx_Error_Invalid_argument_range, subs);
196  return NULL;
197  }
198 
199  SysProcess::beep((int)Frequency, (int)Duration);
200  return ""; /* always returns a null */
201 }
