109 /**
110  * Block guard lock on an object instance. This allows us to
111  * grab a guard lock, while ensuring that the lock is released
112  * during exception unwind.
113  */
114 class GuardLock
115 {
116 public:
117  inline GuardLock(Activity *a, RexxObject *o, RexxClass *s) : activity(a), target(o), scope(s)
118  {
119  // just acquire the scope
120  target->guardOn(activity, scope);
121  }
122 
123  inline ~GuardLock()
124  {
125  target->guardOff(activity, scope);
126  }
127 
128 private:
129 
130  Activity *activity; // the activity we're running on
131  RexxObject *target; // the target object for the lock
132  RexxClass *scope; // the scope of the required guard lock
133 };
