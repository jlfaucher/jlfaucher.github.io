165 /**
166  * Simple class for managing returning results as a stem or an array object.
167  */
168 class StemHandler
169 {
170  public:
171  StemHandler(RexxCallContext *c, RexxStemObject s = NULLOBJECT) : context(c), stem(s), arrayCount(0), array(NULLOBJECT), isArray(false) { }
172 
173  StemHandler(RexxCallContext *c, RexxObjectPtr s, int position) : context(c), arrayCount(0)
174  {
175  setStem(s, position);
176  }
177 
178  ~StemHandler()
179  {
180  complete();
181  }
182 
183  void complete()
184  {
185  if (!isArray && stem != NULLOBJECT)
186  {
187  context->SetStemArrayElement(stem, 0, context->StringSizeToObject(arrayCount));
188  stem = NULLOBJECT;
189  }
190  }
191 
192  void setStem(RexxObjectPtr stemArgument, int position)
193  {
194  // as an alternative to a stem, is this an array?
195  isArray = context->IsArray(stemArgument);
196  if (isArray)
197  {
198  array = (RexxArrayObject)stemArgument;
199  context->SendMessage0(array, "EMPTY");
200  }
201  else
202  {
203  // try to resolve a stem. This is either a stem object already, or
204  // a valid stem name.
205  stem = context->ResolveStemVariable(stemArgument);
206 
207  // this could be an invalid argument, or it might be a name like "a.1"
208  if (stem == NULLOBJECT)
209  {
210  // Argument must be an array object, a stem object, or a stem name value
211  context->ThrowException2(Rexx_Error_Incorrect_call_noarray_nostem, context->WholeNumberToObject(position), stem);
212  }
213  }
214  }
215 
216  void addList(const char *values)
217  {
218  // values is a list of null terminated strings, terminated with a double null
219  while (*values != '\0')
220  {
221  size_t valueLen = strlen(values);
222  addValue(values, valueLen);
223  values += valueLen + 1;
224  }
225  }
226 
227  void addValue(const char *value)
228  {
229  addValue(value, strlen(value));
230  }
231 
232  void addValue(const char *value, size_t valueLen)
233  {
234  if (isArray)
235  {
236  context->ArrayAppendString(array, value, valueLen);
237  }
238  else
239  {
240  RexxStringObject s = context->NewString(value, valueLen);
241  context->SetStemArrayElement(stem, ++arrayCount, s);
242  context->ReleaseLocalReference(s);
243  }
244  }
245 
246  protected:
247  RexxCallContext *context; // the call context
248  RexxStemObject stem; // the stem we're managing
249  size_t arrayCount; // the current stem count
250  RexxArrayObject array; // the array we're managing
251  bool isArray; // true if we manage an array, false if a stem
252 };

286  class AttributeMask
287  {
288  public:
289 
290  typedef enum
291  {
292  Archive,
293  Directory,
294  Hidden,
295  ReadOnly,
296  System,
297  Control,
298  } AttributeType;
299 
300  AttributeMask()
301  {
302  memset(mask, 0, sizeof(mask));
303  }
304 
305  bool parseMask(const char *maskArg)
306  {
307  // everything is off until a + or - is found in the mask
308  mask[Control] = IgnoreAll;
309  int y = 0;
310 
311  while (*maskArg != '\0')
312  {
313  switch(*maskArg)
314  {
315  case '+':
316  {
317  mask[y] = AttributeOn;
318  mask[Control] = HaveMask;
319  break;
320  }
321  case '-':
322  {
323  mask[y] = AttributeOff;
324  mask[Control] = HaveMask;
325  break;
326  }
327  case '*':
328  {
329  mask[y] = AttributeIgnore;
330  break;
331  }
332  default:
333  return false;
334  }
335 
336  y++;
337  maskArg++;
338  }
339  return true;
340  }
341 
342  bool noNewSettings() { return mask[Control] == IgnoreAll; }
343  bool isOn(AttributeType maskSetting) { return mask[maskSetting] == AttributeOn; }
344  bool isOff(AttributeType maskSetting) { return mask[maskSetting] == AttributeOff; }
345  bool isIgnored(AttributeType maskSetting) { return mask[maskSetting] == AttributeIgnore; }
346  void set(AttributeType maskSetting, bool on) { mask[maskSetting] = on ? AttributeOn : AttributeOff; }
347  char maskMarker(AttributeType maskSetting) { return mask[maskSetting] == AttributeOn ? maskChars[maskSetting] : '-'; }
348  bool isSelected(AttributeType maskSetting, bool value) { return value ? !isOff(maskSetting) : !isOn(maskSetting); }
349  bool acceptAll() { return mask[Control] == IgnoreAll; }
350 
351  protected:
352 
353  const char AttributeOff = -1;
354  const char AttributeOn = 1;
355  const char AttributeIgnore = 0;
356  const char IgnoreAll = 0;
357  const char HaveMask = 1;
358  static const char maskChars[6];
359 
360  int8_t mask[8]; // our map for matching
361  };
