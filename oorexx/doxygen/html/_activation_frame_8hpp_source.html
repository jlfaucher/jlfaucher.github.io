50 /**
51  * The base class for all stack frames.
52  */
53 class ActivationFrame
54 {
55 friend class Activity;
56  public:
57  inline ActivationFrame(Activity *a) : activity(a)
58  {
59  // it would be better to have the activity class do this, but because
60  // we're doing this with inline methods, we run into a bit of a
61  // circular reference problem
62  next = activity->activationFrames;
63  activity->activationFrames = this;
64  }
65 
66  // explicitly disable a frame for situations where we no longer
67  // wish to have the frame show up in the tracebacks.
68  inline void disableFrame()
69  {
70  // only do this once
71  if (activity != OREF_NULL)
72  {
73  // remove ourselves from the list
74  activity->activationFrames = next;
75  // this will keep us from attempting to do this twice
76  activity = OREF_NULL;
77  }
78 
79  }
80 
81  inline ~ActivationFrame()
82  {
83  // make this no longer active
84  disableFrame();
85  }
86 
87  virtual RexxString *messageName() = 0;
88  virtual BaseExecutable *executable() = 0;
89  virtual StackFrameClass *createStackFrame() = 0;
90  virtual PackageClass *getPackage() = 0;
91 
92  protected:
93 
94  ActivationFrame *next; // the next activation frame in the chain
95  Activity *activity; // the activity we're running on
96 };
