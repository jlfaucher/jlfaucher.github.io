117  inline void putVariable(RexxVariable *variable, size_t index)
118  {
119  // this may be a dynamic addition, so we might not know the index
120  if (index != 0)
121  {
122  locals[index] = variable;
123  if (dictionary != OREF_NULL)
124  {
125  dictionary->addVariable(variable->getName(), variable);
126  }
127  }
128  else
129  {
130  if (dictionary == OREF_NULL)
131  {
132  createDictionary();
133  }
134  dictionary->addVariable(variable->getName(), variable);
135  }
136  }
