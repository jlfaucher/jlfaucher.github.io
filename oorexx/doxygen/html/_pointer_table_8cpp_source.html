109 /**
110  * Place an item into a hash collection using a key.
111  *
112  * @param value The inserted value.
113  * @param index The insertion key.
114  *
115  * @return The retrieved object. Returns OREF_NULL if the object
116  * was not found.
117  */
118 void PointerTable::put(RexxInternalObject *value, void *index)
119 {
120  // try to insert in the existing hash tables...if this
121  // fails, we're full and need to reallocate.
122  if (!contents->put(value, index))
123  {
124  // reallocate and try again
125  reallocateContents();
126  contents->put(value, index);
127  }
128 }
129

131 /**
132  * Reallocate the hash bucket to a larger version after
133  * a failed put() operation.
134  */
135 void PointerTable::reallocateContents()
136 {
137  // create a new bucket and merge the old bucket into it, then replace the contents
138  // with the new ones.
139  PointerBucket *newContents = new (contents->totalSize * 2) PointerBucket(contents->totalSize * 2);
140  contents->merge(newContents);
141  contents = newContents;
142 }
