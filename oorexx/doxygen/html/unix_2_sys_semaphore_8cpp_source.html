65 /**
66  * Check to see if we have gone beyond the appointed time.
67  *
68  * @param ts The target time
69  * @param nextInterval
70  * The return next wait interval if this still has not timed out.
71  *
72  * @return true if we have gone beyond the target time, false otherwise.
73  */
74 bool checkTimeOut(const struct timespec *ts, long &nextInterval)
75 {
76 
77  struct timespec currentTime;
78  clock_gettime(CLOCK_REALTIME, &currentTime); // get the current time
79 
80  // if target time seconds are greater, we have not timed out
81  if (ts->tv_sec > currentTime.tv_sec)
82  {
83  return false;
84  }
85  // if the seconds are the same, we calculate the delta
86  // and check
87  if (ts->tv_sec == currentTime.tv_sec)
88  {
89  // calculate unconditionally, we use the sign to then determine if this was a time out
90  nextInterval = ts->tv_nsec - currentTime.tv_nsec;
91  if (nextInterval < 0)
92  {
93  return true; // a negative result is a timeout
94  }
95  // we never wait longer than 10 milliseconds
96  nextInterval = std::min(nextInterval, TIMESLICE);
97  return false;
98  }
99 
100  // the target seconds are less, so we are done
101  return true;
102 }

105 /**
106  * Fix for systems that don't have pthread_mutex_timedlock
107  *
108  * @param mutex The mutex we're waiting on
109  * @param abs_timeout
110  * The time timeout value
111  *
112  * @return zero or an error code.
113  */
114 int pthread_mutex_timedlock(pthread_mutex_t *mutex, const struct timespec *abs_timeout)
115 {
116  int pthread_rc;
117 
118  long waitInterval;
119 
120  // we might already be timed out
121  if (checkTimeOut(abs_timeout, waitInterval))
122  {
123  return ETIMEDOUT;
124  }
125 
126  while ((pthread_rc = pthread_mutex_trylock(mutex)) == EBUSY)
127  {
128  struct timespec ts;
129  ts.tv_sec = 0;
130  ts.tv_nsec = waitInterval;
131 
132  struct timespec slept;
133  nanosleep(&ts, &slept);
134 
135  // check for a timeout immediately, this also gives us our
136  // next wait interval
137  if (checkTimeOut(abs_timeout, waitInterval))
138  {
139  return ETIMEDOUT;
140  }
141  }
142 
143  return pthread_rc;
144 }

169 /**
170  * Create the semaphore
171  */
172 void SysSemaphore::create()
173 {
174  // don't create this multiple times
175  if (created)
176  {
177  return;
178  }
179  int rc = 0;
180 
181  pthread_mutexattr_t mutexattr;
182 
183  rc = pthread_mutexattr_init(&mutexattr);
184  if (rc == 0)
185  {
186  // our mutexes must allow recursive (multiple) locks
187  rc = pthread_mutexattr_settype(&mutexattr, PTHREAD_MUTEX_RECURSIVE);
188  }
189  if (rc == 0)
190  {
191  rc = pthread_mutex_init(&semMutex, &mutexattr);
192  }
193  if (rc == 0)
194  {
195  // destroying our mutexattr doesn't affect mutexes created with it
196  rc = pthread_mutexattr_destroy(&mutexattr);
197  }
198  if (rc == 0)
199  {
200  rc = pthread_cond_init(&semCond, NULL);
201  }
202  if (rc != 0)
203  {
204  fprintf(stderr, "*** Internal error in SysSemaphore::create: pthread_mutex rc=%d\n", rc);
205  if (rc == EINVAL)
206  {
207  fprintf(stderr, "*** Internal error in SysSemaphore::create: not built thread-safe\n");
208  }
209  }
210  postedCount = 0;
211  created = true;
212 }

238  rc = pthread_mutex_unlock(&semMutex); // unlock access to semaphore mutex

242 /**
243  * Wait for an event semaphore to be posted.
244  */
245 void SysSemaphore::wait()
246 {
247  int rc;
248 
249  rc = pthread_mutex_lock(&semMutex); // lock access to semaphore
250  if (this->postedCount == 0) // has it been posted?
251  {
252  rc = pthread_cond_wait(&semCond, &semMutex); // nope, then wait on it
253  }
254  rc = pthread_mutex_unlock(&semMutex); // release mutex lock
255 }

258 /**
259  * Fill in a timespec given a timeout value.
260  *
261  * @param t The time to wait, in milliseconds.
262  * @param ts
263  */
264 void SysSemaphore::createTimeOut(uint32_t t, timespec &ts)
265 {
266  int result = 0;
267  clock_gettime(CLOCK_REALTIME, &ts); // get the current time
268  time_t deltaSeconds = (time_t)(t / 1000); // get the seconds part of this
269  // we need to convert to nanoseconds
270  long deltaNanoSeconds = ((long)(t % 1000)) * 1000000l;
271 
272  ts.tv_nsec += deltaNanoSeconds; // add fractions of seconds
273  if (ts.tv_nsec > 1000000000l) // did nanoseconds overflow?
274  {
275  ts.tv_nsec -= 1000000000l; // correct microsecond overflow ..
276  ts.tv_sec += 1; // .. by adding a second
277  }
278 
279  ts.tv_sec += deltaSeconds; // add requested seconds
280 }

283 /**
284  * Wait for the event, with a timeout.
285  *
286  * @param t The timeout value, in milliseconds.
287  *
288  * @return true if the event occurrec, false if there was
289  * a timeout.
290  */
291 bool SysSemaphore::wait(uint32_t t) // takes a timeout in msecs
292 {
293  struct timespec ts; // fill in the timeout spec
294  int result = 0;
295  createTimeOut(t, ts);
296 
297  pthread_mutex_lock(&semMutex); // lock access to semaphore
298  while (result == 0 && !this->postedCount) // Has it been posted? Spurious wakeups may occur
299  { // wait with timeout
300  result = pthread_cond_timedwait(&semCond, &semMutex, &ts);
301  }
302  pthread_mutex_unlock(&semMutex); // release mutex lock
303  // a false return means this timed out
304  return result != ETIMEDOUT;
305 }
