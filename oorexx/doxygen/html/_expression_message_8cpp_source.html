57  * @param argCount The count of arguments for the message.
58  *
59  * @return Storage for building a message object.
60  */
61 void *RexxExpressionMessage::operator new(size_t size, size_t argCount)
62 {
63  if (argCount == 0)
64  {
65  // allocate with singleton item chopped off
66  return new_object(size - sizeof(RexxObject *), T_MessageSendTerm);
67  }
68  else
69  {
70  // allocate with the space needed for the arguments
71  return new_object(size + (argCount - 1) * sizeof(RexxObject *), T_MessageSendTerm);
72  }
73 }

147 /**
148  * Evaluate a message expression term.
149  *
150  * @param context The current execution context.
151  * @param stack The evaluation stack.
152  *
153  * @return The message result.
154  */
155 RexxObject *RexxExpressionMessage::evaluate(RexxActivation *context, ExpressionStack *stack)
156 {
157  // evaluate the target object
158  RexxObject *_target = target->evaluate(context, stack);
159  RexxClass *_super = OREF_NULL;
160 
161  // do we have a super class override?
162  if (super != OREF_NULL)
163  {
164  _super = (RexxClass *)super->evaluate(context, stack);
165  // _super an instance of TheClassClass
166  if (!_super->isInstanceOf(TheClassClass))
167  {
168  reportException(Error_Invalid_argument_noclass, "SCOPE", "Class");
169  }
170  // validate the starting scope
171  _target->validateScopeOverride(_super);
172  // we send the message using the stack, which
173  // expects to find the target and the arguments
174  // on the stack, but not the super. We need to
175  // pop this item off after evaluation. Since this
176  // comes either from a variable or the environment, this
177  // is already protected from GC.
178  stack->toss();
179  }
180 
181  // evaluate the arguments first
182  RexxInstruction::evaluateArguments(context, stack, arguments, argumentCount);
183 
184  ProtectedObject result;
185 
186  // issue based on whether we have the override
187  if (_super == OREF_NULL)
188  {
189  stack->send(messageName, argumentCount, result);
190  }
191  else
192  {
193  stack->send(messageName, _super, argumentCount, result);
194  }
195 
196  // remove any arguments from the stack
197  stack->popn(argumentCount);
198 
199  // double twidde form? replace the result with the target .
200  // NOTE: currently, the target is the top item on the stack, so
201  // we don't need to fix the stack
202  if (doubleTilde)
203  {
204  result = _target;
205  }
206  // use the actual return value, replace it on the stack
207  else
208  {
209  stack->prefixResult(result);
210  }
211 
212  // we're in an expression here, so a result is required.
213  if (result.isNull())
214  {
215  reportException(Error_No_result_object_message, messageName);
216  }
217 
218  // trace if necessary
219  context->traceMessage(messageName, result);
220 
221  return result;
222 }

225 /**
226  * Perform an assignment operation for a message term
227  * used in USE ARG or PARSE situation.
228  *
229  * @param context The current activation context.
230  * @param value The new value.
231  */
232 void RexxExpressionMessage::assign(RexxActivation *context, RexxObject *value)
233 {
234  // the stack is not passed to assignment operations but fortunately, we can
235  // get that from the context.
236  ExpressionStack *stack = context->getStack();
237 
238  // evaluate the target (protected on the stack)
239  RexxObject *_target = target->evaluate(context, stack);
240  RexxClass *_super = OREF_NULL;
241 
242  // message override?
243  if (super != OREF_NULL)
244  {
245  // evaluate the superclass override
246  _super = (RexxClass *)super->evaluate(context, stack);
247  // we need to remove this from the stack for the send operation to work.
248  stack->toss();
249  }
250 
251  // push the assignment value on to the stack as the firt argument
252  stack->push(value);
253  // now push the rest of the arguments. This might be something like a[1,2,3,4] as
254  // an assignment term. The assignment value is the first argument, followed by
255  // any other arguments that are part of the encoded message term.
256  size_t argcount = argumentCount;
257 
258  for (size_t i = 0; i < argcount; i++)
259  {
260  // non-omitted argument?
261  if (arguments[i] != OREF_NULL)
262  {
263  // evaluate and potentially trace
264  RexxObject *resultArg = arguments[i]->evaluate(context, stack);
265  context->traceResult(resultArg);
266  }
267  else
268  {
269  // non existant arg....we may still need to trace that
270  stack->push(OREF_NULL);
271  context->traceResult(GlobalNames::NULLSTRING);
272  }
273  }
274 
275  ProtectedObject result;
276 
277  // now send the message the appropriate way. Note we
278  // have an extra arg from the assignment target.
279  if (_super == OREF_NULL)
280  {
281  // normal message send
282  stack->send(messageName, argcount + 1, result);
283  }
284  else
285  {
286  // send with an override
287  stack->send(messageName, _super, argcount + 1, result);
288  }
289 
290  context->traceAssignment(messageName, result);
291  // remove all arguments (arguments + target + assignment value)
292  stack->popn(argcount + 2);
293 }
