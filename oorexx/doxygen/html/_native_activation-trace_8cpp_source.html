208 /**
209  * Process the arguments for a typed function/method call.
210  *
211  * @param argcount The count of arguments.
212  * @param arglist The original Rexx arguments.
213  * @param argumentTypes
214  * The type descriptor from the target.
215  * @param descriptors
216  * The maximum argument count for the target.
217  * @param maximumArgumentCount
218  */
219 void NativeActivation::processArguments(size_t _argcount, RexxObject **_arglist, uint16_t *argumentTypes,
220  ValueDescriptor *descriptors, size_t maximumArgumentCount)
221 {
222  size_t inputIndex = 0; // no arguments used yet */
223  size_t outputIndex = 1; // we start filling in from the second (first is return value)
224  bool usedArglist = false; // if we request the argument list, then there's no requirement
225  // to use up all of the passed arguments individually.
226 
227  // set up the return value descriptor
228  descriptors[0].type = *argumentTypes;
229  descriptors[0].value.value_int64_t = 0;
230  // the first type in the signature is the return value, which we skip for now
231  uint16_t *currentType = argumentTypes + 1;
232  // now loop through the type information. Some of the types are "created"
233  // arguments that don't appear directly in the method/function invocation.
234  // We can't directly do a 1-for-1 mapping
235  for (; *currentType != REXX_ARGUMENT_TERMINATOR; currentType++)
236  {
237  // make sure we don't get a buffer overload
238  if (outputIndex >= maximumArgumentCount)
239  {
240  reportSignatureError();
241  }
242 
243  uint16_t type = ARGUMENT_TYPE(*currentType);
244  bool isOptional = IS_OPTIONAL_ARGUMENT(*currentType);
245 
246  descriptors[outputIndex].type = type; // fill in the type
247  switch (type)
248  {
249  // reference to the receiver object...if this is a function call,
250  // then this will be OREF NULL.
251  case REXX_VALUE_OSELF:
252  {
253  // this doesn't make any sense for a function call
254  if (!isMethod())
255  {
256  reportSignatureError();
257  }
258  // fill in the receiver object and mark it...
259  descriptors[outputIndex].value.value_RexxObjectPtr = (RexxObjectPtr)receiver;
260  descriptors[outputIndex].flags = ARGUMENT_EXISTS | SPECIAL_ARGUMENT;
261  break;
262  }
263 
264  // reference to the method scope...if this is a function call,
265  // then this will be OREF NULL.
266  case REXX_VALUE_SCOPE:
267  {
268  // this doesn't make any sense for a function call
269  if (!isMethod())
270  {
271  reportSignatureError();
272  }
273  // fill in the receiver object and mark it...
274  descriptors[outputIndex].value.value_RexxObjectPtr = (RexxObjectPtr)getScope();
275  descriptors[outputIndex].flags = ARGUMENT_EXISTS | SPECIAL_ARGUMENT;
276  break;
277  }
278 
279  // reference to the superclass scope...if this is a function call,
280  // then this will be OREF NULL.
281  case REXX_VALUE_SUPER:
282  {
283  // this doesn't make any sense for a function call
284  if (!isMethod())
285  {
286  reportSignatureError();
287  }
288  // fill in the receiver object and mark it...
289  descriptors[outputIndex].value.value_RexxObjectPtr = (RexxClassObject)getSuper();
290  descriptors[outputIndex].flags = ARGUMENT_EXISTS | SPECIAL_ARGUMENT;
291  break;
292  }
293 
294  case REXX_VALUE_CSELF: /* reference to CSELF */
295  {
296  // this doesn't make any sense for a function call
297  if (!isMethod())
298  {
299  reportSignatureError();
300  }
301  descriptors[outputIndex].value.value_POINTER = cself();
302  descriptors[outputIndex].flags = ARGUMENT_EXISTS | SPECIAL_ARGUMENT;
303  break;
304  }
305 
306  case REXX_VALUE_ARGLIST: /* need the argument list */
307  {
308  descriptors[outputIndex].flags = ARGUMENT_EXISTS | SPECIAL_ARGUMENT;
309  descriptors[outputIndex].value.value_RexxArrayObject = (RexxArrayObject)getArguments();
310  // we've used this
311  usedArglist = true;
312  break;
313  }
314 
315  // This is either the message name used to invoke a method or
316  // the name used to call a function
317  case REXX_VALUE_NAME:
318  {
319  descriptors[outputIndex].flags = ARGUMENT_EXISTS | SPECIAL_ARGUMENT;
320  descriptors[outputIndex].value.value_CSTRING = (CSTRING)messageName->getStringData();
321  break;
322  }
323 
324  // this is a real argument taken from the argument list
325  default:
326  {
327  if (inputIndex < _argcount && _arglist[inputIndex] != OREF_NULL)
328  {
329  // all of these arguments exist
330  descriptors[outputIndex].flags = ARGUMENT_EXISTS;
331  RexxObject *argument = _arglist[inputIndex];
332  switch (type)
333  {
334  // arbitrary object reference
335  case REXX_VALUE_RexxObjectPtr:
336  {
337  descriptors[outputIndex].value.value_RexxObjectPtr = (RexxObjectPtr)argument;
338  break;
339  }
340 
341  case REXX_VALUE_int:
342  {
343  // convert and copy */
344  descriptors[outputIndex].value.value_int = (int)signedIntegerValue(argument, inputIndex, INT_MAX, INT_MIN);
345  break;
346  }
347 
348  case REXX_VALUE_int8_t:
349  {
350  descriptors[outputIndex].value.value_int8_t = (int8_t)signedIntegerValue(argument, inputIndex, INT8_MAX, INT8_MIN);
351  break;
352  }
353 
354  case REXX_VALUE_int16_t:
355  {
356  descriptors[outputIndex].value.value_int16_t = (int16_t)signedIntegerValue(argument, inputIndex, INT16_MAX, INT16_MIN);
357  break;
358  }
359 
360  case REXX_VALUE_int32_t:
361  {
362  descriptors[outputIndex].value.value_int32_t = (int32_t)signedIntegerValue(argument, inputIndex, INT32_MAX, INT32_MIN);
363  break;
364  }
365 
366  case REXX_VALUE_int64_t:
367  {
368  descriptors[outputIndex].value.value_int64_t = (int64_t)int64Value(argument, inputIndex);
369  break;
370  }
371 
372  case REXX_VALUE_ssize_t:
373  {
374  descriptors[outputIndex].value.value_ssize_t = (ssize_t)signedIntegerValue(argument, inputIndex, SSIZE_MAX, -SSIZE_MAX - 1);
375  break;
376  }
377 
378  case REXX_VALUE_intptr_t:
379  {
380  descriptors[outputIndex].value.value_intptr_t = (intptr_t)signedIntegerValue(argument, inputIndex, INTPTR_MAX, INTPTR_MIN);
381  break;
382  }
383 
384  case REXX_VALUE_uint8_t:
385  {
386  descriptors[outputIndex].value.value_uint8_t = (uint8_t)unsignedIntegerValue(argument, inputIndex, UINT8_MAX);
387  break;
388  }
389 
390  case REXX_VALUE_uint16_t:
391  {
392  descriptors[outputIndex].value.value_uint16_t = (uint16_t)unsignedIntegerValue(argument, inputIndex, UINT16_MAX);
393  break;
394  }
395 
396  case REXX_VALUE_uint32_t:
397  {
398  descriptors[outputIndex].value.value_uint32_t = (uint32_t)unsignedIntegerValue(argument, inputIndex, UINT32_MAX);
399  break;
400  }
401 
402  case REXX_VALUE_uint64_t:
403  {
404  descriptors[outputIndex].value.value_uint64_t = (uint64_t)unsignedInt64Value(argument, inputIndex);
405  break;
406  }
407 
408  case REXX_VALUE_size_t:
409  {
410  descriptors[outputIndex].value.value_size_t = (size_t)unsignedIntegerValue(argument, inputIndex, SIZE_MAX);
411  break;
412  }
413 
414  case REXX_VALUE_uintptr_t:
415  {
416  descriptors[outputIndex].value.value_uintptr_t = (uintptr_t)unsignedIntegerValue(argument, inputIndex, UINTPTR_MAX);
417  break;
418  }
419 
420  case REXX_VALUE_logical_t:
421  {
422  descriptors[outputIndex].value.value_logical_t = argument->truthValue(Error_Logical_value_method);
423  break;
424  }
425 
426  // The Rexx whole number one is checked against the human digits limit
427  case REXX_VALUE_wholenumber_t:
428  {
429  descriptors[outputIndex].value.value_wholenumber_t = (wholenumber_t)signedIntegerValue(argument, inputIndex, Numerics::MAX_WHOLENUMBER, Numerics::MIN_WHOLENUMBER);
430  break;
431  }
432 
433  // The Rexx whole number one is checked against the human digits limit
434  case REXX_VALUE_positive_wholenumber_t:
435  {
436  descriptors[outputIndex].value.value_wholenumber_t = positiveWholeNumberValue(argument, inputIndex);
437  break;
438  }
439 
440  // The Rexx whole number one is checked against the human digits limit
441  case REXX_VALUE_nonnegative_wholenumber_t:
442  {
443  descriptors[outputIndex].value.value_wholenumber_t = nonnegativeWholeNumberValue(argument, inputIndex);
444  break;
445  }
446 
447  // an unsigned string number value
448  case REXX_VALUE_stringsize_t:
449  {
450  descriptors[outputIndex].value.value_stringsize_t = (stringsize_t)unsignedIntegerValue(argument, inputIndex, Numerics::MAX_STRINGSIZE);
451  break;
452  }
453 
454  case REXX_VALUE_double:
455  {
456  descriptors[outputIndex].value.value_double = getDoubleValue(argument, inputIndex);
457  break;
458  }
459 
460 
461  case REXX_VALUE_float:
462  {
463  descriptors[outputIndex].value.value_float = (float)getDoubleValue(argument, inputIndex);
464  break;
465  }
466 
467  case REXX_VALUE_CSTRING:
468  {
469  descriptors[outputIndex].value.value_CSTRING = cstring(argument, inputIndex + 1);
470  break;
471  }
472 
473  case REXX_VALUE_RexxStringObject:
474  {
475  // force to a string value
476  RexxString *temp = stringArgument(argument, inputIndex + 1);
477  // if this forced a string object to be created,
478  // we need to protect it here.
479  if (temp != argument)
480  {
481  createLocalReference(temp);
482  }
483  descriptors[outputIndex].value.value_RexxStringObject = (RexxStringObject)temp;
484  break;
485 
486  }
487 
488  case REXX_VALUE_RexxArrayObject:
489  {
490  ArrayClass *temp = arrayArgument(argument, inputIndex + 1);
491  // if this forced a string object to be created,
492  // we need to protect it here.
493  if (temp != argument)
494  {
495  createLocalReference(temp);
496  }
497  descriptors[outputIndex].value.value_RexxArrayObject = (RexxArrayObject)temp;
498  break;
499 
500  }
501 
502  case REXX_VALUE_RexxStemObject:
503  {
504  // Stem arguments get special handling. If the argument
505  // object is already a stem object, we're done. Otherwise,
506  // we get the string value of the argument and use that
507  // to resolve a stem name in the current context. If the
508  // trailing period is not given on the name, one will be added.
509  // Note that the second form is only available if this is a
510  // call context. This is an error for a method context.
511 
512  // is this a stem already?
513  if (isStem(argument))
514  {
515  descriptors[outputIndex].value.value_RexxStemObject = (RexxStemObject)argument;
516  break;
517  }
518 
519  // this doesn't make any sense for a method call...requires
520  // variable pool access
521  if (isMethod())
522  {
523  reportStemError(inputIndex, argument);
524  }
525 
526  RexxString *temp = argument->requestString();
527  if ((RexxObject *)temp == TheNilObject)
528  {
529  reportStemError(inputIndex, argument);
530  }
531  // if this forced a string object to be created,
532  // we need to protect it here.
533  if (temp != argument)
534  {
535 
536  createLocalReference(temp);
537  }
538 
539  // see if we can retrieve this stem
540  RexxObject *stem = getContextStem(temp);
541  if (stem == OREF_NULL)
542  {
543  reportStemError(inputIndex, argument);
544  }
545  descriptors[outputIndex].value.value_RexxStemObject = (RexxStemObject)stem;
546  break;
547  }
548 
549  case REXX_VALUE_RexxClassObject: // required class object
550  {
551  // this must be a class object
552  if (!argument->isInstanceOf(TheClassClass))
553  {
554  reportException(Error_Invalid_argument_noclass, inputIndex + 1, TheClassClass->getId());
555  }
556  descriptors[outputIndex].value.value_RexxClassObject = (RexxClassObject)argument;
557  break;
558  }
559 
560  case REXX_VALUE_POINTER:
561  {
562  // this must be a pointer object
563  if (!argument->isInstanceOf(ThePointerClass))
564  {
565  reportException(Error_Invalid_argument_noclass, inputIndex + 1, ThePointerClass->getId());
566  }
567  descriptors[outputIndex].value.value_POINTER = pointer(argument);
568  break;
569  }
570 
571  case REXX_VALUE_POINTERSTRING:
572  {
573  descriptors[outputIndex].value.value_POINTERSTRING = pointerString(argument, inputIndex);
574  break;
575  }
576 
577  case REXX_VALUE_RexxMutableBufferObject:
578  {
579  // this must be a pointer object
580  if (!argument->isInstanceOf(TheMutableBufferClass))
581  {
582  reportException(Error_Invalid_argument_noclass, inputIndex + 1, TheMutableBufferClass->getId());
583  }
584  descriptors[outputIndex].value.value_RexxMutableBufferObject = (RexxMutableBufferObject)argument;
585  break;
586  }
587 
588  case REXX_VALUE_RexxVariableReferenceObject:
589  {
590  // this must be a pointer object
591  if (!argument->isInstanceOf(TheVariableReferenceClass))
592  {
593  reportException(Error_Invalid_argument_noclass, inputIndex + 1, TheVariableReferenceClass->getId());
594  }
595  descriptors[outputIndex].value.value_RexxVariableReferenceObject = (RexxVariableReferenceObject)argument;
596  break;
597  }
598 
599  default:
600  {
601  reportSignatureError();
602  break;
603  }
604  }
605  }
606  else
607  {
608  // if this was not an option argument
609  if (!isOptional)
610  {
611  reportException(Error_Invalid_argument_noarg, inputIndex + 1);
612  }
613 
614  // this is a non-specified argument
615  descriptors[outputIndex].flags = 0;
616  switch (type)
617  {
618 
619  case REXX_VALUE_RexxObjectPtr: // no object here
620  case REXX_VALUE_int: // non-integer value
621  case REXX_VALUE_wholenumber_t: // non-existent long
622  case REXX_VALUE_positive_wholenumber_t: // non-existent long
623  case REXX_VALUE_nonnegative_wholenumber_t: // non-existent long
624  case REXX_VALUE_CSTRING: // missing character string
625  case REXX_VALUE_POINTER:
626  case REXX_VALUE_RexxStringObject: // no object here
627  case REXX_VALUE_stringsize_t: // non-existent long
628  case REXX_VALUE_int8_t: // non-integer value
629  case REXX_VALUE_int16_t: // non-integer value
630  case REXX_VALUE_int32_t: // non-integer value
631  case REXX_VALUE_int64_t: // non-integer value
632  case REXX_VALUE_uint8_t: // non-integer value
633  case REXX_VALUE_uint16_t: // non-integer value
634  case REXX_VALUE_uint32_t: // non-integer value
635  case REXX_VALUE_uint64_t: // non-integer value
636  case REXX_VALUE_size_t:
637  case REXX_VALUE_ssize_t:
638  case REXX_VALUE_intptr_t:
639  case REXX_VALUE_uintptr_t:
640  case REXX_VALUE_logical_t: // this must be a boolean value
641  case REXX_VALUE_RexxArrayObject: // no object here
642  case REXX_VALUE_RexxStemObject:
643  case REXX_VALUE_RexxClassObject:
644  case REXX_VALUE_RexxMutableBufferObject:
645  case REXX_VALUE_POINTERSTRING:
646  {
647  // set this as a 64-bit value to clear everything out
648  descriptors[outputIndex].value.value_int64_t = 0;
649  break;
650  }
651  // non-existent double
652  case REXX_VALUE_double:
653  {
654  descriptors[outputIndex].value.value_double = 0.0;
655  break;
656  }
657  // non-existent float
658  case REXX_VALUE_float:
659  {
660  descriptors[outputIndex].value.value_float = 0.0;
661  break;
662  }
663 
664  // a bad signature
665  default:
666  {
667  reportSignatureError();
668  break;
669  }
670  }
671  }
672  inputIndex++; // we've used up one more input argument
673  break;
674  }
675  }
676  outputIndex++; // step to the next argument
677  argumentTypes++; // and the next output position pointer
678  }
679  // do we have additional unwanted arguments? that's an error
680  if (inputIndex < _argcount && !usedArglist) /* extra, unwanted arguments? */
681  {
682  reportException(Error_Invalid_argument_maxarg, inputIndex);
683  }
684 }

687 /**
688  * Convert an collection of value descriptors into a Rexx array
689  * object. This is useful for creating arrays of arguments for
690  * the various call APIs.
691  *
692  * @param value The self-describing value descriptors.
693  * @param count The number of descriptors in the list.
694  *
695  * @return The described value converted to an appropriate Rexx object.
696  */
697 ArrayClass *NativeActivation::valuesToObject(ValueDescriptor *value, size_t count)
698 {
699  ArrayClass *r = new_array(count);
700  ProtectedObject p(r);
701 
702  for (size_t i = 0; i < count; i++)
703  {
704  // convert each of the values in turn
705  r->put(valueToObject(value++), i);

1255 /**
1256  * Run a native method or function under the context of this activation.
1257  *
1258  * @param _receiver The receiver object (NULL if not a method invocation).
1259  * @param _msgname The message name.
1260  * @param _argcount The argument count
1261  * @param _arglist The list of arguments.
1262  * @param resultObj The returned result object.
1263  */
1264 void NativeActivation::run(MethodClass *_method, NativeMethod *_code, RexxObject *_receiver,
1265  RexxString *_msgname, RexxObject **_arglist, size_t _argcount, ProtectedObject &resultObj)
1266 {
1267  // add the frame to the execution stack
1268  NativeActivationFrame frame(activity, this);
1269  // anchor the relevant context information
1270  executable = _method;
1271  receiver = _receiver;
1272  messageName = _msgname;
1273  argList = _arglist;
1274  argCount = _argcount;
1275  activationType = METHOD_ACTIVATION; // this is for running a method
1276 
1277  ValueDescriptor arguments[MaxNativeArguments];
1278 
1279  MethodContext context; // the passed out method context
1280 
1281  // sort out our active security manager
1282  securityManager = _code->getSecurityManager();
1283  if (securityManager == OREF_NULL)
1284  {
1285  securityManager = activity->getInstanceSecurityManager();
1286  }
1287 
1288  // build a context pointer to pass out
1289  activity->createMethodContext(context, this);
1290 
1291  context.threadContext.arguments = arguments;
1292 
1293  // get the entry point address of the target method
1294  PNATIVEMETHOD methp = _code->getEntry();
1295 
1296  // retrieve the argument signatures and process them
1297  uint16_t *types = (*methp)((RexxMethodContext *)&context, NULL);
1298  processArguments(argCount, argList, types, arguments, MaxNativeArguments);
1299 
1300  size_t activityLevel = activity->getActivationLevel();
1301  trapErrors = true; // we trap errors from here
1302  try
1303  {
1304  activity->releaseAccess(); /* force this to "safe" mode */
1305  /* process the method call */
1306  (*methp)((RexxMethodContext *)&context, arguments);
1307  activity->requestAccess(); /* now in unsafe mode again */
1308 
1309  // process the returned result
1310  result = valueToObject(arguments);
1311  }
1312  catch (NativeActivation *)
1313  {
1314  }
1315  // there is a subtle interaction between native and rexx activations when
1316  // native calls make calls to APIs that in turn invoke Rexx code. When conditions
1317  // occur and the stack is being unwound, the ApiContext destructors will release
1318  // the kernel access, which can leave us with no active Activity. Bad things happen
1319  // when that occurs. We'll grab the exception when it goes past and make sure
1320  // things remain consistent.
1321  catch (RexxActivation *r)
1322  {
1323  // if we're not the current kernel holder when things return, make sure we
1324  // get the lock before we continue
1325  if (ActivityManager::currentActivity != activity)
1326  {
1327  activity->requestAccess();
1328  }
1329  throw r;
1330  }
1331 
1332  // if we're not the current kernel holder when things return, make sure we
1333  // get the lock before we continue
1334  if (ActivityManager::currentActivity != activity)
1335  {
1336  activity->requestAccess();
1337  }
1338  guardOff(); // release any variable locks
1339  argCount = 0; // make sure we don't try to mark any arguments after this
1340 
1341  // the lock holder gets here by a throw from a kernel reentry. We need to
1342  // make sure the activation count gets reset, else we'll accumulate bogus
1343  // nesting levels that will make it look like this activity is still in use
1344  // when in fact we're done with it.
1345  activity->restoreActivationLevel(activityLevel);
1346 
1347  // give up reference to receiver so that it can be garbage collected
1348  receiver = OREF_NULL;
1349 
1350  checkConditions(); // see if we have conditions to raise now
1351 
1352  // set the return value and get outta here
1353  resultObj = result;
1354 
1355  // disable the stack frame used to generate the tracebacks since we are
1356  // noh longer active. An error in an uninit method might pick up bogus information
1357  // if we're still in the list
1358  frame.disableFrame();
1359 
1360  // good place to check for uninits
1361  memoryObject.checkUninitQueue();
1362 
1363  activity->popStackFrame(this); // pop this from the activity
1364  setHasNoReferences(); // mark this as not having references in case we get marked
1365 }
