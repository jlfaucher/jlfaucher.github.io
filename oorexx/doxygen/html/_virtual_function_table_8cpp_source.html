188 /******************************************************************************/
189 /* Function: This small function is necessary to void optimizer problems on */
190 /* some versions of GCC. The optimizer appears to keep storing */
191 /* the same value in the VFT rather than picking up the new VFT */
192 /* for each class. Making this a separate routine avoids this. */
193 /******************************************************************************/
194 void *getVftPointer(void *loc)
195 {
196  return *((void **)loc);
197 }
