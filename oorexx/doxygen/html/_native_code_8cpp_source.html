302 /**
303  * Run a method call (vs a straight program call).
304  *
305  * @param activity The current activity.
306  * @param method The method we're attached to.
307  * @param receiver The method receiver object (the "self" object).
308  * @param messageName
309  * The name of the message used to invoke the method.
310  * @param count The count of arguments.
311  * @param argPtr The pointer to the arguments.
312  * @param result The protected object used to return the result.
313  */
314 void NativeMethod::run(Activity *activity, MethodClass *method, RexxObject *receiver, RexxString *messageName,
315  RexxObject **argPtr, size_t count, ProtectedObject &result)
316 {
317  // if this is NULL currently, we need to lazy resolve this entry
318  if (entry == NULL)
319  {
320  // have the package manager resolve this for us before we make a call
321  entry = PackageManager::resolveMethodEntry(packageName, name);
322  }
323 
324  // create a new native activation
325  NativeActivation *newNActa = ActivityManager::newNativeActivation(activity);
326  activity->pushStackFrame(newNActa); /* push it on the activity stack */
327  /* and go run it */
328  newNActa->run(method, this, receiver, messageName, argPtr, count, result);
329 }
