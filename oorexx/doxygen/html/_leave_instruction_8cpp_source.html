104 /**
105  * Execute a LEAVE or ITERATE instruction.
106  *
107  * @param context The current excution context.
108  * @param stack The current evaluation stack.
109  */
110 void RexxInstructionLeave::execute(RexxActivation *context, ExpressionStack *stack )
111 {
112  context->traceInstruction(this);
113 
114  // if this is a LEAVE, we tell the context to leave the appropriate loop.
115  if (instructionType == KEYWORD_LEAVE)
116  {
117  context->leaveLoop(name);
118  }
119  // we are iterating the loop
120  else

122  context->iterate(name);
123  }
124 
125  context->pauseInstruction();
126 }
