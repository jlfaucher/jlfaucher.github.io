59  /**
60  * Initialize an API context from a thread context.
61  *
62  * @param c The source context.
63  */
64  inline ApiContext(RexxThreadContext *c)
65  {
66  // we need to cleanup on exit
67  releaseLock = true;
68  clearConditions = false;
69  activity = contextToActivity(c);
70  context = activity->getApiContext();
71  context->enableConditionTraps();
72  // go acquire the kernel lock and take care of nesting
73  activity->enterCurrentThread();
74  // we need to validate the thread call context to ensure this
75  // is the correct thread
76  activity->validateThread();
77  }

250  /**
251  * Destructor for an API context. Releases the interpreter
252  * access lock on exit.
253  */
254  inline ~ApiContext()
255  {
256  // clear the condition if we've been requested not to propagate
257  if (clearConditions)
258  {
259  context->clearCondition();
260  }
261 
262  // we only do this sort of cleanup if we really entered on the
263  // activity
264  if (releaseLock)
265  {
266  context->disableConditionTraps();
267  activity->exitCurrentThread();
268  }
269  }

272  {

274  return (RexxObjectPtr)o;
275  }
