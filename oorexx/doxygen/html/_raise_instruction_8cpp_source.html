55 /**
56  * Construct a RAISE instruction object.
57  *
58  * @param _condition The condition name.
59  * @param _expression
60  * The RC expression.
61  * @param _description
62  * An optional description string.
63  * @param _additional
64  * An expression to resolve additional information.
65  * @param _result A result value to return when the context exits.
66  * @param _arrayCount
67  * A count of items specified with the ARRAY() option.
68  * @param array The subTerm queue holding the ARRAY() option expressions.
69  * @param flags Additional control flags.
70  */
71 RexxInstructionRaise::RexxInstructionRaise(RexxString *_condition, RexxInternalObject *_expression,
72  RexxInternalObject *_description, RexxInternalObject *_additional, RexxInternalObject *_result,
73  FlagSet<RaiseInstructionFlags, 32> flags)
74 {
75  // just copy the argument information
76  conditionName = _condition;
77  rcValue = _expression;
78  description = _description;
79  resultValue = _result;
80  instructionFlags = flags;
81  // is this the array form? We need to copy the expressions
82  if (flags[raise_array])
83  {
84  ArrayClass *arrayItems = (ArrayClass *)_additional;
85 
86  arrayCount = arrayItems->size();
87  // copy each of the argument expressions
88  for (size_t i = 0; i < arrayCount; i++)
89  {
90  additional[i] = (RexxObject *)arrayItems->get(i + 1);
91  }
92  }
93  // store the expression for retrieving the additional item from the
94  // first additional slot.
95  else
96  {
97  // we have just one item
98  additional[0] = _additional;
99  arrayCount = 1;
100  }
101 }

158 /**
159  * Execute a RAISE instruction.
160  *
161  * @param context The current execution context.
162  * @param stack The current context evaluation stack.
163  */
164 void RexxInstructionRaise::execute(RexxActivation *context, ExpressionStack *stack)
165 {
166  // trace if needed
167  context->traceInstruction(this);
168 
169  // set defaults for anything we need to evaluate.
170  RexxObject *_additional = OREF_NULL;
171  RexxString *_description = OREF_NULL;
172  RexxObject *rc = OREF_NULL;
173  RexxObject *_result = OREF_NULL;
174 
175  // and start evaluating
176  // extra RC information (SYNTAX, ERROR, and FAILURE only)
177  if (rcValue != OREF_NULL)
178  {
179  /* get the expression value */
180  rc = rcValue->evaluate(context, stack);
181  // this is traced using the condition name as the keyword
182  context->traceKeywordResult(conditionName, rc);
183  }
184  // syntax conditions have some special requirements, so process those
185  // up front.
186  if (instructionFlags[raise_syntax])
187  {
188  // give this a default additional information of an empty array
189  _additional = new_array();
190  // The description is a null string
191  _description = GlobalNames::NULLSTRING;
192  // the RC must have a string value...this is an error if it doesn.
193  RexxString *errorcode = rc->requestString();
194  if (errorcode == TheNilObject)
195  {
196  reportException(Error_Conversion_raise, rc);
197  }
198 
199  // convert this to decimal, then create an integer object
200  // that we replace the input rc value with
201  wholenumber_t msgNum = Interpreter::messageNumber(errorcode);
202  rc = new_integer(msgNum);
203  }
204 
205  // Reasonable defaults are set up, now see if we have explicit things given
206  if (description != OREF_NULL)
207  {
208  _description = (RexxString *)description->evaluate(context, stack);
209  context->traceKeywordResult(GlobalNames::DESCRIPTION, _description);
210  }
211 
212  // is this the ARRAY form of passing information?
213  if (instructionFlags[raise_array])
214  {
215  // we need to build an array of the additional information
216  size_t count = arrayCount;
217  _additional = new_array(count);
218  // push this on the eval stack for safekeeping
219  stack->push(_additional);
220  for (size_t i = 0; i < count; i++)
221  {
222  // we can have ommitted ones here, so only try to evaluate the
223  // ones that have been specified
224  if (additional[i] != OREF_NULL)
225  {
226  // trace each of these as arguments
227  RexxObject *arg = (additional[i])->evaluate(context, stack);
228  // trace each of these as arguments
229  context->traceArgument(arg);
230  ((ArrayClass *)_additional)->put(arg, i + 1);
231  // trace each of these as arguments
232  context->traceArgument(arg);
233  }
234  else
235  {
236  // just trace a null argument
237  context->traceArgument(GlobalNames::NULLSTRING);
238  }
239  }
240  context->traceKeywordResult(GlobalNames::ARRAY, _additional);
241  }
242  // we might have had this via the ADDITIONAL() option.
243  else if (this->additional[0] != OREF_NULL)
244  {
245  // get this expression value
246  _additional = additional[0]->evaluate(context, stack);
247  // trace what we got
248  context->traceKeywordResult(GlobalNames::ADDITIONAL, _additional);
249  }
250  // given a return result value to pass back to the caller?
251  if (resultValue != OREF_NULL)
252  {
253  _result = resultValue->evaluate(context, stack);
254  context->traceKeywordResult(GlobalNames::RESULT, _result);
255  }
256 
257  // set a default condition object
258  DirectoryClass *conditionobj = OREF_NULL;
259  // propagating an existing condition?
260  if (instructionFlags[raise_propagate])
261  {
262  conditionobj = context->getConditionObj();
263  if (conditionobj == OREF_NULL) /* no current active condition? */
264  {
265  reportException(Error_Execution_propagate);
266  }
267  }
268 
269  // if we have additional information, fill in a few more things
270  if (_additional != OREF_NULL)
271  {
272  // we may need some additional checks based on the syntax
273  // condition. Since this might come from a PROPAGATE, we
274  // have to rely on the string name of the condition.
275  RexxString *errorCode = conditionName;
276  // if this is a propagate, get CONDITION name from the condition object.
277  if (instructionFlags[raise_propagate])
278  {
279  errorCode = (RexxString *)conditionobj->get(GlobalNames::CONDITION);
280  }
281  // If this is a SYNTAX condition, than the Additional information MUST
282  // be an array of items used for substitutions.
283  if (errorCode->strCompare(GlobalNames::SYNTAX))
284  {
285  // get the array version, and it must be single dimension.
286  _additional = _additional->requestArray();
287  if (_additional == TheNilObject || ((ArrayClass *)_additional)->isMultiDimensional())
288  {
289  reportException(Error_Execution_syntax_additional);
290  }
291  }
292  }
293 
294  // we have two forms, both of which are handled by context
295  // raise return is processed as a return instruction
296  if (instructionFlags[raise_return])
297  {
298  context->raise(conditionName, rc, _description, _additional, _result, conditionobj);
299  }
300  // the default is handled like an EXIT
301  else
302  {
303  context->raiseExit(conditionName, rc, _description, _additional, _result, conditionobj);
304  }
305 }
