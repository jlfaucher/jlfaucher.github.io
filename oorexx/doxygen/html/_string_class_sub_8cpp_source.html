51 /**
52  * The center/centre method for the string class.
53  *
54  * @param _length The length to center in.
55  * @param pad The optional pad character
56  *
57  * @return The centered string.
58  */
59 RexxString *RexxString::center(RexxInteger *_length, RexxString *pad)
60 {
61  // this will be our final length
62  size_t width = lengthArgument(_length, ARG_ONE);
63 
64  // the pad character is optional, with a blank default
65  char padChar = optionalPadArgument(pad, ' ', ARG_TWO);
66  size_t len = getLength();
67  // if the width and the length are the same, we just return the target
68  // string unchanged
69  if (width == len)
70  {
71  return this;
72  }
73 
74  // centered in zero characters? This is a null string
75  if (width == 0)
76  {
77  return GlobalNames::NULLSTRING;
78  }
79 
80  // if the width is bigger than the length, we need to add pad characters
81  if (width > len)
82  {
83  // half the pad is on the left side
84  size_t leftPad = (width - len) / 2;
85  // the remainder on the right, which also gets the extra if an odd number
86  // is required
87  size_t rightPad = (width - len) - leftPad;
88  RexxString *retval = raw_string(width);
89 
90  StringBuilder builder(retval);
91  // set left pad characters
92  builder.pad(padChar, leftPad);
93  // add the string data
94  builder.append(getStringData(), len);
95  // now the trailing pad chars
96  builder.pad(padChar, rightPad);
97  return retval;
98  }
99  // the request width is smaller than the input, so we have to truncate
100  else
101  {
102  // we really only need to calculate the left side truncation
103  size_t leftPad = (len - width) / 2;
104  return new_string(getStringData() + leftPad, width);
105  }
106 }

109 /**
110  * The delstr() method of the string class.
111  *
112  * @param position The starting postion
113  * @param _length The length to delete.
114  *
115  * @return The string after the deletion.
116  */
117 RexxString *RexxString::delstr(RexxInteger *position, RexxInteger *_length)
118 {
119  size_t stringLen = getLength();
120  size_t deletePos = optionalPositionArgument(position, 1, ARG_ONE);
121  size_t deleteLen = optionalLengthArgument(_length, stringLen - deletePos + 1, ARG_TWO);
122 
123  // if the delete position is beyond the
124  // length of the string, just return unchanged
125  if (deletePos > stringLen)
126  {
127  return this;
128  }
129 
130  // delete all characters? This is a null string
131  if (deletePos == 1 && deleteLen >= stringLen)
132  {
133  return GlobalNames::NULLSTRING;
134  }
135 
136  // easier to do if origin zero
137  deletePos--;
138 
139  size_t backLen = 0;
140  // if we're not deleting up to or past the end, calculate the
141  // size of the trailing section
142  if (deleteLen < (stringLen - deletePos))
143  {
144  backLen = stringLen - (deletePos + deleteLen);
145  }
146 
147  RexxString *retval = raw_string(deletePos + backLen);
148  StringBuilder builder(retval);
149 
150  // copy any leading part, unless we're deleting from the
151  // start of the string
152  builder.append(getStringData(), deletePos);
153  // add the trailing section
154  builder.append(getStringData() + deletePos + deleteLen, backLen);
155 
156  return retval;
157 }

160 /**
161  * The String insert method.
162  *
163  * @param newStrObj The string to insert.
164  * @param position The insert postion.
165  * @param _length The optional insert length.
166  * @param pad The optional padding character.
167  *
168  * @return The string with the new string inserted.
169  */
170 RexxString *RexxString::insert(RexxString *newStrObj, RexxInteger *position, RexxInteger *_length, RexxString *pad)
171 {
172  size_t targetLength = getLength();
173 
174  newStrObj = stringArgument(newStrObj, ARG_ONE);
175  size_t newStringLength = newStrObj->getLength();
176 
177  // we're parsing this as a length argument because a postion of zero
178  // is valid for insert
179  size_t insertPosition = optionalLengthArgument(position, 0, ARG_TWO);
180  size_t insertLength = optionalLengthArgument(_length, newStringLength, ARG_THREE);
181 
182  // default pad character is a blank
183  char padChar = optionalPadArgument(pad, ' ', ARG_FOUR);
184  size_t leadPad = 0;
185  size_t frontLength;
186  size_t backLength;
187 
188  // inserting at the beginning? No leading pad required, no leading part
189  // to copy, the back length is the entire string
190  if (insertPosition == 0)
191  {
192  leadPad = 0;
193  frontLength = 0;
194  backLength = targetLength;
195  }
196  // insert position beyond the length? We might need to insert some leading
197  // pad characters, the front part is everything and there is no back part to copy
198  else if (insertPosition >= targetLength)
199  {
200  leadPad = insertPosition - targetLength;
201  frontLength = targetLength;
202  backLength = 0;
203  }
204 
205  // we're inserting in the middle. No leading pad, the insert position
206  // determines the length of the front and back sections
207  else
208  {
209  leadPad = 0;
210  frontLength = insertPosition;
211  backLength = targetLength - insertPosition;
212  }
213 
214  // we might need to truncate the inserted string if the specified length is
215  // shorter than the inserted string
216  newStringLength = std::min(newStringLength, insertLength);
217  size_t padLength = insertLength - newStringLength;
218 
219  size_t resultLength = targetLength + insertLength + leadPad;
220  RexxString *retval = raw_string(resultLength);
221  StringBuilder builder(retval);
222 
223  // if we have a front length, copy it now
224  builder.append(getStringData(), frontLength);
225 
226  // if there are leading pad characters required, add them now
227  builder.pad(padChar, leadPad);
228 
229  // if we have new string data to copy, this is next
230  builder.append(newStrObj->getStringData(), newStringLength);
231 
232  // and trailing pad required?
233  builder.pad(padChar, padLength);
234 
235  // and the back length value
236  builder.append(getStringData() + frontLength, backLength);
237  return retval;
238 }

244  *

246  * @param pad The pad character for padding out to the given length.

253 
254  // default padd is a blank
255  char padChar = optionalPadArgument(pad, ' ', ARG_TWO);
256 
257  // a request size if zero is a simple null string
258  if (size == 0)
259  {
260  return GlobalNames::NULLSTRING;
261  }
262 
263  size_t length = getLength();
264  RexxString *retval = raw_string(size);
265  StringBuilder builder(retval);
266

268  size_t copyLength = std::min(length, size);
269 
270  // if we have data to copy, add to the result

272

274  // pad characters.
275 
276  if (size > length)
277  {
278  builder.pad(padChar, size - length);
279  }
280  return retval;
281 }

284 /**
285  * Perform an overlay operation on a string.
286  *
287  * @param newStrObj The string being overlayed on the target string.
288  * @param position The overlay position.
289  * @param _length The optional length of the overlay.
290  * @param pad The optional pad character
291  *
292  * @return A new string with the overlay performed.
293  */
294 RexxString *RexxString::overlay(RexxString *newStrObj, RexxInteger *position, RexxInteger *_length, RexxString *pad)
295 {
296  size_t targetLen = getLength();
297  // make sure we have a real string value for the overlay
298  newStrObj = stringArgument(newStrObj, ARG_ONE);
299  size_t newLen = newStrObj->getLength();
300 
301  // the overlay postion defaults to the first character
302  size_t overlayPos = optionalPositionArgument(position, 1, ARG_TWO);
303  // the length default is the overlay string length
304  size_t overlayLen = optionalLengthArgument(_length, newLen, ARG_THREE);
305  // default pad is a blank
306  char padChar = optionalPadArgument(pad, ' ', ARG_FOUR);
307 
308  size_t backPad = 0;
309 
310  // if the requested length is larger than the string length, we
311  // need to pad
312  if (overlayLen > newLen)
313  {
314  backPad = overlayLen - newLen;
315  }
316  // might be shorter than the string, so reduce the copy length to
317  // the requested size
318  else
319  {
320  newLen = overlayLen;
321  }
322 
323  size_t frontPad = 0;
324  // calculate the default split positions
325  size_t frontLen = overlayPos - 1;
326  size_t backLen = targetLen - (overlayPos + overlayLen - 1);
327 
328  // if the overlay position is beyond the length of the target
329  // string, we need to add padding in front of the new string and
330  // copy just the target string length
331  if (overlayPos > targetLen)
332  {
333  frontPad = overlayPos - targetLen - 1;
334  frontLen = targetLen;
335  }
336 
337  // if the end of the overlay position extends past the
338  // end of the target string, there's no trailing part to copy
339  if (overlayPos + overlayLen > targetLen)
340  {
341  backLen = 0;
342  }
343 
344 
345  RexxString *retval = raw_string(frontLen + backLen + frontPad + overlayLen);
346  StringBuilder builder(retval);
347 
348  // copy the front part
349  builder.append(getStringData(), frontLen);
350  // add any leading padding
351  builder.pad(padChar, frontPad);
352  // copy the overlay string (or portion of the overlay string)
353  builder.append(newStrObj->getStringData(), newLen);
354  // add any possible back padding
355  builder.pad(padChar, backPad);
356  // and finally the trailing section
357  builder.append(getStringData() + overlayPos + overlayLen - 1, backLen);
358 
359  return retval;
360 }

363 /**
364  * Replace a substring starting at a given position and
365  * length with another string. This operation is essentially
366  * a delstr() followed by an insert() operation.
367  *
368  * @param newStrObj The replacement string
369  * @param position The replacement position (required)
370  * @param _length The length of string to replace (optional). If omitted,
371  * the length of the replacement string is used and this
372  * is essentially an overlay operation.
373  * @param pad The padding character if padding is required. The default
374  * is a ' '
375  *
376  * @return A new instance of the string with the value replace.
377  */
378 RexxString *RexxString::replaceAt(RexxString *newStrObj, RexxInteger *position, RexxInteger *_length, RexxString *pad)
379 {
380  size_t targetLen = getLength();
381  // the replacement value is required and must be a string
382  RexxString *newStr = stringArgument(newStrObj, ARG_ONE);
383 
384  // the length of the replacement string is the default replacement length
385  size_t newLen = newStr->getLength();
386  // the overlay position is required
387  size_t replacePos = positionArgument(position, ARG_TWO);
388  // the replacement length is optional, and defaults to the length of the replacement string
389  size_t replaceLen = optionalLengthArgument(_length, newLen, ARG_THREE);
390 
391  // we only pad if the start position is past the end of the string
392  char padChar = optionalPadArgument(pad, ' ', ARG_FOUR);
393 
394  size_t padding = 0;
395  size_t frontLen = 0;
396  size_t backLen = 0;
397 
398  // the only time we need to pad is if the replacement position is past the
399  // end of the string
400  if (replacePos > targetLen)
401  {
402  padding = replacePos - targetLen - 1;
403  frontLen = targetLen;
404  }
405  else
406  {
407  // this is within bounds, so we copy up to that position
408  frontLen = replacePos - 1;
409  }
410 
411  // is this within the bounds of the string?
412  if (replacePos + replaceLen - 1 < targetLen)
413  {
414  // calculate the back part we need to copy
415  backLen = targetLen - (replacePos + replaceLen - 1);
416  }
417 
418  // allocate a result string
419  RexxString *retval = raw_string(frontLen + backLen + padding + newLen);
420  StringBuilder builder(retval);
421 
422  // add the front section
423  builder.append(getStringData(), frontLen);
424 
425  // padding only happens if we've copied the entire front portion
426  builder.pad(padChar, padding);
427 
428  // copy the replacement string
429  builder.append(newStr->getStringData(), newLen);
430 
431  // the remainder, if there is any, gets copied after the
432  // replacement string with no padding
433  builder.append(getStringData() + replacePos + replaceLen - 1, backLen);
434  return retval;
435 }

438 /**
439  * Reverse the byte positions in a string object.
440  *
441  * @return The target string, reversed.
442  */
443 RexxString *RexxString::reverse()
444 {
445  size_t length = getLength();
446  // if this is a null string or just a single character,
447  // there's nothing to reverse
448  if (length <= 1)
449  {
450  return this;
451  }
452 
453  RexxString *retval = raw_string(length);
454  char *string = retval->getWritableData();
455 
456  // copy from the end in the original
457  const char *end = getStringData() + length - 1;
458 
459  // now perform the whole length copy
460  while (length--)
461  {
462 // avoid warning: writing 1 byte into a region of size 0 [-Wstringop-overflow=]
463 // (or similar) in gcc 12 and above, due to our RexxString char stringData[4]
464 #ifdef HAVE_PRAGMA_GCC_STRINGOPOVERFLOW
465 #pragma GCC diagnostic push
466 #pragma GCC diagnostic ignored "-Wstringop-overflow"
467 #endif
468  *string++ = *end--;
469 #ifdef HAVE_PRAGMA_GCC_STRINGOPOVERFLOW
470 #pragma GCC diagnostic pop
471 #endif
472  }
473 
474  return retval;
475 }

479  * Extract a substring of the target relative to the
480  * right end.
481  *
482  * @param _length The extract length.
483  * @param pad The optional padding character.
484  *
485  * @return The extracted substring.
486  */
487 RexxString *RexxString::right(RexxInteger *_length, RexxString *pad)
488 {
489  size_t size = lengthArgument(_length, ARG_ONE);
490 
491  char padChar = optionalPadArgument(pad, ' ', ARG_TWO);
492  size_t sourceLength = getLength();
493 
494  // if the extraction length is zero, return a null string
495  if (size == 0)
496  {
497  return GlobalNames::NULLSTRING;
498  }
499 
500  // asking for the same size (not uncommon, really), no extraction necessary
501  if (sourceLength == size)
502  {
503  return this;
504  }
505 
506  RexxString *retval = raw_string(size);
507  StringBuilder builder(retval);
508 
509  // the requested length might be longer than the target string, so
510  // cap at that size
511  size_t copyLength = std::min(sourceLength, size);
512  size_t padLength = size - copyLength;
513 
514  // padding, if required, occurs before the extracted string piece
515  builder.pad(padChar, padLength);
516  // copy on the string portion, copying from the end of the string
517  builder.append(getStringData() + sourceLength - copyLength, copyLength);
518  return retval;
519 }

522 /**
523  * Strip a set of leading and/or trailing characters from a string,
524  * returning a new string value, or if unchanged, the original string.
525  *
526  * @param optionString The option indicating where to strip characters.
527  * @param stripchar The set of characters to strip.
528  *
529  * @return A string instance with the target characters removed.
530  */
531 RexxString *RexxString::strip(RexxString *optionString, RexxString *stripchar)
532 {
533  // get the option character
534  char option = optionalOptionArgument(optionString, "BLT", STRIP_BOTH, ARG_ONE);
535 
536  // get the strip character set
537  stripchar = optionalStringArgument(stripchar, OREF_NULL, ARG_TWO);
538 
539  // the default is to remove spaces and horizontal tabs
540  const char *chars = stripchar == OREF_NULL ? " \t" : stripchar->getStringData();
541  size_t charsLen = stripchar == OREF_NULL ? strlen(" \t") : stripchar->getLength();
542 
543  const char *front = getStringData();
544  size_t length = getLength();
545 
546  if (option == STRIP_LEADING || option == STRIP_BOTH)
547  {
548  // loop while more string or we don't find one of the stripped characters
549  while (length > 0)
550  {
551  if (!StringUtil::matchCharacter(*front, chars, charsLen))
552  {
553  break;
554  }
555  front++;
556  length--;
557  }
558  }
559 
560  // need to strip trailing?
561  if (option == STRIP_TRAILING || option == STRIP_BOTH)
562  {
563  // point to the end and scan backwards now
564  const char *back = front + length - 1;
565  while (length > 0)
566  {
567  if (!StringUtil::matchCharacter(*back, chars, charsLen))
568  {
569  break;
570  }
571  back--;
572  length--;
573  }
574  }
575 
576  // if there is anything left, extract the remaining part
577  if (length > 0)
578  {
579  // it's quite common that nothing was stripped, in which case we can
580  // just return the original string instead of creating a new one
581  return length == getLength() ? this : new_string(front, length);
582  }
583  else
584  {
585  // null string, everything stripped away
586  return GlobalNames::NULLSTRING;
587  }
588 }
