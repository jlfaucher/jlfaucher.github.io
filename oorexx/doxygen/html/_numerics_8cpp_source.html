218 /**
219  * Convert an object into a whole number value.
220  *
221  * @param source The source object.
222  * @param result The returned converted value.
223  * @param maxValue The maximum range value for the target.
224  * @param minValue The minimum range value for this number.
225  *
226  * @return true if the number converted properly, false for any
227  * conversion errors.
228  */
229 bool Numerics::objectToWholeNumber(RexxObject *source, wholenumber_t &result, wholenumber_t maxValue, wholenumber_t minValue)
230 {
231  // is this an integer value (very common)
232  if (isInteger(source))
233  {
234  result = ((RexxInteger *)source)->wholeNumber();
235  return result <= maxValue && result >= minValue ? true : false;
236  }
237  else
238  {
239  // get this as a numberstring (which it might already be)
240  NumberString *nString = source->numberString();
241  // not convertible to number string? get out now
242  if (nString == OREF_NULL)
243  {
244  return false;
245  }
246  int64_t temp;
247 
248  // if not valid or outside of the minimum range, reject this too
249  if (nString->int64Value(&temp, ARGUMENT_DIGITS))
250  {
251  if (temp <= maxValue && temp >= minValue)
252  {
253  result = (wholenumber_t)temp;
254  return true;
255  }
256  }
257  return false;
258  }
259 }

556 /**
557  * Do portable formatting of a wholenumber value into an ascii
558  * string.
559  *
560  * @param integer The value to convert.
561  * @param dest The location to store the formatted string.
562  *
563  * @return The length of the converted number.
564  */
565 size_t Numerics::formatWholeNumber(wholenumber_t integer, char *dest)
566 {
567  // zero? this is pretty easy
568  if (integer == 0)
569  {
570  strcpy(dest, "0");
571  return 1;
572  }
573 
574  size_t sign = 0;
575  // we convert this directly because portable numeric-to-ascii routines
576  // don't really exist for the various 32/64 bit values.
577  char buffer[24];
578  size_t index = sizeof(buffer);
579 
580  // negative number? copy a negative sign, and take the abs value
581  if (integer < 0)
582  {
583  *dest++ = '-';
584  // work from an unsigned version that can hold all of the digits
585  // we need to use a version we can negate first, then add the
586  // digit back in
587  size_t working = (size_t)(-(integer + 1));
588  working++; // undoes the +1 above
589  sign = 1; // added in to the length
590 
591  while (working > 0)
592  {
593  // get the digit and reduce the size of the integer
594  int digit = (int)(working % 10);
595  working = working / 10;
596  // store the digit
597  buffer[--index] = digit + '0';
598  }
599  }
600  else
601  {
602  while (integer > 0)
603  {
604  // get the digit and reduce the size of the integer
605  int digit = (int)(integer % 10) + '0';
606  integer = integer / 10;
607  // store the digit
608  buffer[--index] = digit;
609  }
610  }
611 
612  // copy into the buffer and set the length
613  size_t length = sizeof(buffer) - index;
614  memcpy(dest, &buffer[index], length);
615  // make sure we have a terminating null
616  dest[length] = '\0';
617  return length + sign;
618 }
