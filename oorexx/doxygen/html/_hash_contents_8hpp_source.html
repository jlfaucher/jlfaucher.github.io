53  * collection. This version bases all index and item
54  * searches on object equality. Subclasses can override these

196  // default index hashing method. bypass the hash() method and directly use the hash value
197  virtual ItemLink hashIndex(RexxInternalObject *index)
198  {
199  // Note: even though we are using object reference identity to find a match,
200  // we use the hash value returned from getHashValue() rather than the identityHash
201  // because identityTables stored in the saved image or compiled programs will have
202  // a different reference value (and thus a different identifyHash) on restore, meaning
203  // that lookups against these tables may fail.
204  return (ItemLink)(index->getHashValue() % bucketSize);
