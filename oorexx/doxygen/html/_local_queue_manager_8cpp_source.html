61 /**
62  * Validate a queue name
63  *
64  * @param username The name to validate.
65  * @param isSession A bool to return to indicate the queue name is the session queue.
66  *
67  * @return True if the name is valid, false if invalid or "SESSION"
68  */
69 bool LocalQueueManager::validateQueueName(const char *username, bool &isSession)
70 {
71  isSession = false;
72 
73  if (username == NULL) /* NULL is OK. */
74  {
75  return true;
76  }
77  // "SESSION" is a reserved name, reject this in this context
78  if (Utilities::strCaselessCompare(username, "SESSION") == 0)
79  {
80  // indicate to the caller this is the session queue
81  isSession = true;
82  return false;
83  }
84 
85  size_t namelen = strlen(username);
86  if (namelen > 0 && namelen <= MAX_QUEUE_NAME_LENGTH)
87  {
88  const char *valptr = username; /* point to name */
89  char ch;
90  while ((ch = *(valptr++)))
91  {
92  if (!Utilities::isAlpha(ch) && !Utilities::isDigit(ch) && ch != ch_PERIOD &&
93  ch != ch_QUESTION_MARK && ch != ch_EXCLAMATION && ch != ch_UNDERSCORE)
94  {
95  return false;
96  }
97  }
98  return true;
99  }
100  else
101  {
102  return false;
103  }
104 }

146 /**
147  * Create the session queue for this process.
148  *
149  * @param session
150  *
151  * @return
152  */
153 QueueHandle LocalQueueManager::initializeSessionQueue(SessionID session)
154 {
155  // first check to see if we have an env variable set...if we do we
156  // inherit from our parent session
157  QueueHandle mysessionQueue;
158  // we could be inheriting the session queue from a caller process...check first.
159  if (!createdSessionQueue && SysLocalAPIManager::getActiveSessionQueue(mysessionQueue))
160  {
161  // make sure we update the nest count
162  // this might end up creating a new queue instance
163  mysessionQueue = nestSessionQueue(session, mysessionQueue);
164  }
165  else
166  {
167  // create a new session queue
168  mysessionQueue = createSessionQueue(session);
169  // remember that we created this initially. We'll need to create
170  // the queue each time one is needed for this session.
171  createdSessionQueue = true;
172  }
173  SysLocalAPIManager::setActiveSessionQueue(mysessionQueue);
174  return mysessionQueue;
175 }

193 /**
194  * Create a named queue
195  *
196  * @param name The requested queue name.
197  * @param size The size of the buffer for the returned name.
198  * @param createdName
199  * The buffer for the returned name.
200  * @param dup The duplicate flag.
201  *
202  * @return true if the requested name already exists and a new name
203  * was provided.
204  */
205 RexxReturnCode LocalQueueManager::createNamedQueue(const char *name, size_t size, char *createdName, size_t *dup)
206 {
207  // validation and copying of the request name only is necessary if given.
208  if (name != NULL)
209  {
210  bool isSession = false;
211  // make sure this is a valid name (session is not valid here)
212  if (!validateQueueName(name, isSession))
213  {
214  return RXQUEUE_BADQNAME;
215  }
216 
217  ClientMessage message(QueueManager, CREATE_NAMED_QUEUE, name);
218  // the session id is used for creating unique queue names
219  message.parameter1 = localManager->getSession();
220 
221  message.send();
222  Utilities::strncpy(createdName, message.nameArg, size);
223  // return the dup name indicator
224  *dup = message.result == DUPLICATE_QUEUE_NAME;
225  // check if we had enough space to copy the queue name
226  return strlen(message.nameArg) < size ? RXQUEUE_OK : RXQUEUE_STORAGE;
227  }
228  else
229  {
230  // always returning a generated name
231  ClientMessage message(QueueManager, CREATE_NAMED_QUEUE, "");
232  // the session id is used for creating unique queue names
233  message.parameter1 = localManager->getSession();
234 
235  message.send();
236  Utilities::strncpy(createdName, message.nameArg, size);
237  // by definition, this is not a duplicate
238  *dup = false;
239  // check if we had enough space to copy the queue name
240  return strlen(message.nameArg) < size ? RXQUEUE_OK : RXQUEUE_STORAGE;
241  }
242 }

246  * Create a named queue
247  *
248  * @param name The requested queue name.
249  * @param dup The duplicate flag.
250  *
251  * @return true if the requested name already exists
252  */
253 RexxReturnCode LocalQueueManager::openNamedQueue(const char *name, size_t *dup)
254 {
255  bool isSession = false;
256 
257  if (!validateQueueName(name, isSession)) // make sure this is a valid name
258  {
259  // this might have been rejected because the name was "SESSION". The session
260  // queue always exists
261  if (isSession)
262  {
263  // this is considered a duplicate
264  *dup = true;
265  return RXQUEUE_OK;
266  }
267  return RXQUEUE_BADQNAME;
268  }
269 
270  ClientMessage message(QueueManager, OPEN_NAMED_QUEUE, name);
271 
272  message.send();
273  // return the dup name indicator
274  *dup = message.result == QUEUE_EXISTS;
275  // everything worked here.
276  return RXQUEUE_OK;
277 }

460 RexxReturnCode LocalQueueManager::pullFromQueue(const char *name, RXSTRING &data, size_t waitFlag, RexxQueueTime *timeStamp)
461 {
462  bool isSession = false;
463  if (!validateQueueName(name, isSession)) // make sure this is a valid name
464  {
465  return RXQUEUE_BADQNAME;
466  }
467 
468  ClientMessage message(QueueManager, PULL_FROM_NAMED_QUEUE);
469  // set up for either name or session queue read
470  if (name != NULL)
471  {
472  Utilities::strncpy(message.nameArg, name, ServiceMessage::NAMESIZE);
473  }
474  else
475  {
476  message.operation = PULL_FROM_SESSION_QUEUE;
477  message.parameter3 = sessionQueue;
478  }
479  message.parameter1 = waitFlag != 0 ? QUEUE_WAIT_FOR_DATA : QUEUE_NO_WAIT;
480  message.send();
481  if (message.result == QUEUE_ITEM_PULLED)
482  {
483  message.transferMessageData(data);
484  // if this was a null string, then an empty buffer is sent back. Allocate a minimal
485  // buffer to distinguish between nothing and a null string value
486  if (data.strptr == NULL)
487  {
488  data.strptr = (char *)RexxAllocateMemory(1);
489  if (data.strptr == NULL)
490  {
491  throw new ServiceException(MEMORY_ERROR, "LocalQueueManager::pullFromQueue() Failure allocating memory");
492  }
493  // add a courtesy null terminator
494  data.strptr[data.strlength] = '\0';
495  }
496  // if the timestamp was requested, return it.
497  if (timeStamp != NULL)
498  {
499  memcpy(timeStamp, message.nameArg, sizeof(RexxQueueTime));
500  }
501  }
502  // map the server result to an API return code.
503  return mapReturnResult(message);
504 }
