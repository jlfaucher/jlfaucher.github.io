205 public:
206  inline ServiceRegistrationData(const char *module, const char *proc, bool drop, const char *userPointer)
207  {
208  Utilities::strncpy(moduleName, module, MAX_NAME_LENGTH);
209  Utilities::strncpy(procedureName, proc, MAX_NAME_LENGTH);
210  dropAuthority = !drop ? DROP_ANY : OWNER_ONLY;

214 
215  inline ServiceRegistrationData(const char *module)
216  {
217  if (module != NULL)
218  {

220  }
221  else
222  {
223  strcpy(moduleName, "");
224  }
225  strcpy(procedureName, "");
226  dropAuthority = DROP_ANY;
227  setUserData(NULL);
228  entryPoint = 0;
229  }

231  inline ServiceRegistrationData(REXXPFN entry, const char *userPointer)
232  {
233  strcpy(moduleName, "");
234  strcpy(procedureName, "");
235  entryPoint = (uintptr_t)entry;
236  dropAuthority = OWNER_ONLY;
237  setUserData(userPointer);
238  }

241  inline void setUserData(const char *userPointer)
242  {
243  // we have two bits of user data to copy
244  if (userPointer != NULL)
245  {
246  userData[0] = ((uintptr_t*)userPointer)[0];
247  userData[1] = ((uintptr_t*)userPointer)[1];
248  }
249  else
250  {
251  userData[0] = 0;
252  userData[1] = 0;
253  }
254  }
255

257  inline void retrieveUserData(char *userPointer)
258  {
259  // we have two bits of user data to copy
260  if (userPointer != NULL)
261  {
262  ((uintptr_t*)userPointer)[0] = userData[0];
263  ((uintptr_t*)userPointer)[1] = userData[1];
264  }
265  }

320  {
321  if (errorCode != NO_ERROR_CODE)

324  throw new ServiceException(errorCode, nameArg);

334  inline void clearMessageData()
335  {
336  messageData = NULL;
337  messageDataLength = 0;
338  retainMessageData = false;
339  }
340 
341  inline void *getMessageData() { return messageData; }
342  inline size_t getMessageDataLength() { return messageDataLength; }
343 
344  inline void transferMessageData(RXSTRING &data)
345  {
346  // if provided a buffer, then use it if large enough
347  if (data.strptr != NULL && messageDataLength < data.strlength)
348  {
349  memcpy(data.strptr, messageData, messageDataLength);
350  data.strlength = messageDataLength;
351  // we need to free our copy
352  freeMessageData();
353  }
354  else
355  {
356  MAKERXSTRING(data, (char *)messageData, messageDataLength);
357  // we've given up ownership of this data, so clear out the
358  // pointers so we don't try to free
359  clearMessageData();
360  }
361  }
362 
363 
364  inline void setMessageData(void *data, size_t length)
365  {
366  freeMessageData(); // make sure any existing data is cleared.
367  messageData = data;
368  messageDataLength = length;
369  // make sure we only set a pointer if this is non-zero, else
370  // we end up transmitting a stale pointer to the other side of the connection.
371  if (length == 0)
372  {
373  messageData = NULL;
374  }
375  // this is not freeable data
376  retainMessageData = true;
377  }
378 
379 
380  inline void *allocateMessageData(size_t length)
381  {
382  messageData = allocateResultMemory(length);
383  messageDataLength = length;
384  retainMessageData = false;
385  return messageData;
386  }
387 
388  inline void freeMessageData()
389  {
390  if (messageData != NULL && !retainMessageData)
391  {
392  // this is real message information
393  releaseResultMemory(messageData);
394  clearMessageData();
395  }
396  }
