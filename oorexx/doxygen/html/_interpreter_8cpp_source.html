139 /**
140  * Perform interpreter startup processing.
141  *
142  * @param mode The startup mode. This indicates whether we're saving the
143  * image or in shutdown mode.
144  */
145 void Interpreter::startInterpreter(InterpreterStartupMode mode, const char *imageTarget)
146 {
147  ResourceSection lock;
148 
149  // has everything been shutdown?
150  if (!isActive())
151  {
152  SystemInterpreter::startInterpreter(); // perform system specific initialization
153  // initialize the memory manager , and restore the
154  // memory image
155  memoryObject.initialize(mode == RUN_MODE, imageTarget);
156  RexxCreateSessionQueue();
157  // create our instances list
158  interpreterInstances = new_queue();
159  // if we have a local server created already, don't recurse.
160  if (localServer == OREF_NULL)
161  {
162  // Get an instance. This also gives the root activity of the instance
163  // the kernel lock.
164  InstanceBlock instance;
165  RexxObject *t = OREF_NULL; // required for the findClass call
166 
167  // get the server class from the REXX package where it is a non-public class
168  RexxClass *server_class = TheRexxPackage->findClass(new_string("LOCALSERVER"), t);
169 
170  // NOTE: This is a second block so that the
171  // protected object's destructor gets run before
172  // the activity is removed as the current activity.
173  {
174  ProtectedObject result;
175  // create a new server object
176  server_class->messageSend(GlobalNames::NEW, OREF_NULL, 0, result);
177  localServer = (RexxObject *)result;
178  }
179  }
180  }
181  // we're live now
182  active = true;
183 }

201 /**
202  * Terminate the global interpreter environment, shutting down
203  * all of the interpreter instances that we can and releasing
204  * the object heap memory.
205  *
206  * @return true if everything was shutdown, false if there are reasons
207  * why this can't be shutdown.
208  */
209 bool Interpreter::terminateInterpreter()
210 {
211  {
212  ResourceSection lock; // lock in this section
213  // if never even started up, then we've got a quick return
214  if (!isActive())
215  {
216  return true;
217  }
218 
219  // we can only shutdown interpreter instances from the
220  // threads that created them. If we have active instances,
221  // this is a "no can do" situation
222  if (interpreterInstances->items() != 0)
223  {
224  return false;
225  }
226 
227  }
228 
229  // the resource lock needs to be released here because unloading packages
230  // will require the kernel lock, which can never be requested while holding
231  // the resource lock
232  try
233  {
234  // this may seem funny, but we need to create an instance
235  // to shut down so that the package manager can unload
236  // the libraries (it needs to pass a RexxThreadContext
237  // pointer out to package unloaders, if they are defined)
238  InstanceBlock instance;
239  // run whatever uninits we can before we start releasing the libraries
240  memoryObject.lastChanceUninit();
241 
242  PackageManager::unload();
243  }
244  catch (ActivityException)
245  {
246  // we're shutting down, so ignore any failures while processing this
247  }
248 
249 
250  {
251  ResourceSection lock; // Now that we've released the kernel lock, we need to reacquire the resource lock
252 
253  // perform system-specific cleanup
254  SystemInterpreter::terminateInterpreter();
255 
256  // most interpreter resources will be cleanup automatically, but
257  // we need to poke the rxapi daemon and tell it to clean up our session
258  // resources.
259  RexxDeleteSessionQueue();
260  }
261  return true;
262 }

277  * Create a new instance and return the instance context pointers
278  * and thread context pointer for the instance.
279  *
280  * @param instance The returned instance pointer.
281  * @param threadContext
282  * The returned thread context pointer.
283  * @param options Options to apply to this interpreter instance.
284  *
285  * @return 0 if the instance was created ok.
286  */
287 int Interpreter::createInstance(RexxInstance *&instance, RexxThreadContext *&threadContext, RexxOption *options)
288 {
289  try
290  {
291  // create the instance
292  InterpreterInstance *newInstance = createInterpreterInstance(options);
293  instance = newInstance->getInstanceContext();
294  threadContext = newInstance->getRootThreadContext();
295  // we need to ensure we release the kernel lock before returning
296  Activity *activity = newInstance->getRootActivity();
297  activity->releaseAccess();
298  // the activity needs to be in a deactivated state when we return.
299  activity->deactivate();
300  return 0;
301  } catch (ActivityException)
302  {
303  // not everything works until an instance is actually created, so
304  // it's possible we'll see a true failure here, so give back an
305  // error condition.
306  return RXAPI_MEMFAIL;
307  }
308 }

311 /**
312  * Create a new interpreter instance. An interpreter instance
313  * is an accessible set of threads that constitutes an interpreter
314  * environment for the purposes API access.
315  *
316  * @param exits The set of exits to use for this invocation.
317  * @param defaultEnvironment
318  * The default addressible environment.
319  *
320  * @return The new interpreter instance.
321  */
322 InterpreterInstance *Interpreter::createInterpreterInstance(RexxOption *options)
323 {
324  // We need to ensure that the interpreter is initialized before we create an
325  // interpreter instance. There are some nasty recursion problems that can result,
326  // so this needs to be done carefully and the initialization needs to be protected by
327  // the resource lock during the entire process.
328  {
329  ResourceSection lock;
330  // if our instances list has not been created yet, then the memory subsystem has not
331  // been created yet. Keep the lock during the entire process.
332  if (interpreterInstances == OREF_NULL)
333  {
334  startInterpreter(RUN_MODE, NULL);
335  }
336  }
337 
338 
339  // get a new root activity for this instance. This might result in pushing a prior level down the
340  // stack
341  Activity *rootActivity = ActivityManager::getRootActivity();
342  // ok, we have an active activity here, so now we can allocate a new instance and bootstrap everything.
343  Protected<InterpreterInstance> instance = new InterpreterInstance();
344 
345  {
346  ResourceSection lock;
347 
348  // add this to the active list
349  interpreterInstances->append(instance);
350  }
351 
352  // now that this is protected from garbage collection, go and initialize everything
353  instance->initialize(rootActivity, options);
354  return instance;
355 }

490  * Decode a condition directory into a easier to use
491  * structure form for a native code user. This breaks

506  // just return the major part

508  condData->conditionName = (RexxStringObject)conditionObj->get(GlobalNames::CONDITION);
509 
510  RexxObject *temp = (RexxObject *)conditionObj->get(GlobalNames::MESSAGE);

515 
516  temp = (RexxObject *)conditionObj->get(GlobalNames::ERRORTEXT);
517  if (temp != OREF_NULL)
518  {
519  condData->errortext = (RexxStringObject)temp;
520  }
521 
522  temp = (RexxObject *)conditionObj->get(GlobalNames::DESCRIPTION);
523  if (temp != OREF_NULL)
524  {
525  condData->description = (RexxStringObject)temp;
526  }
527 
528  // this could be raised by a termination exit, so there might not be
529  // position information available
530  temp = (RexxObject *)conditionObj->get(GlobalNames::POSITION);
531  if (temp != OREF_NULL)
532  {
533  condData->position = ((RexxInteger *)temp)->wholeNumber();
534  }
535  else
536  {
537  condData->position = 0;
538  }
539 
540  temp = (RexxObject *)conditionObj->get(GlobalNames::PROGRAM);
541  if (temp != OREF_NULL)
542  {
543  condData->program = (RexxStringObject)temp;
544  }
545 
546  temp = (RexxObject *)conditionObj->get(GlobalNames::ADDITIONAL);
547  if (temp != OREF_NULL)

549  condData->additional = (RexxArrayObject)temp;

633 /**
634  * Parse out the error code string into the messagecode value
635  *
636  * @param errorcode The string version of the error code.
637  *
638  * @return The composite error code number.
639  */
640 wholenumber_t Interpreter::messageNumber(RexxString *errorcode)
641 {
642  wholenumber_t primary = 0; /* Primary part of error code, major */
643  wholenumber_t secondary = 0; /* Secondary protion (minor code) */
644  wholenumber_t count;
645 
646  // make sure this is a string
647  errorcode = (RexxString *)errorcode->stringValue();
648  // scan to decimal Point or end of error code.
649  const char *decimalPoint;
650  for (decimalPoint = errorcode->getStringData(), count = 0; *decimalPoint && *decimalPoint != '.'; decimalPoint++, count++);
651 
652  // must be a whole number in the correct range
653  if (!new_string(errorcode->getStringData(), count)->numberValue(primary) || primary < 1 || primary >= 100)
654  {
655  reportException(Error_Expression_result_raise);
656 
657  }
658 
659  // now shift over the decimal position.
660  primary *= 1000;
661 
662  // was there a decimal point in this?
663  if (*decimalPoint)
664  {
665  if (!new_string(decimalPoint + 1, errorcode->getLength() - count -1)->numberValue(secondary) || secondary < 0 || secondary >= 1000)
666  {
667  reportException(Error_Expression_result_raise);
668  }
669  }
670  return primary + secondary; // add two portions together, return
671 }
