49 /**
50  * A simple and fast stack object for doing memory management
51  * live marking. This has minimal overrun protections, so
52  * the using code needs to perform appropriate "stack is full"
53  * checks.
54  */
55 class LiveStack
56 {
57  public:
58  void *operator new(size_t, size_t);
59  void operator delete(void *);
60 
61  LiveStack(size_t size);
62 
63  // the position is origin zero, relative to the top, which is an empty slot. So, position 0
64  // is the top element, 1 is the penultimate elements, etc.
65  inline RexxInternalObject *get(size_t pos)
66  {
67  // we only return something if within the bounds
68  if (pos < top)
69  {
70  return stack[top - (pos + 1)];
71  }
72  else
73  {
74  return OREF_NULL;
75  }
76  }
77 
78 
79  /**
80  * Push an object on to the stack
81  *
82  * @param obj The object to push
83  *
84  * @return
85  */
86  inline void push(RexxInternalObject *obj)
87  {
88  // we have no overrun protection here. The using piece either
89  // needs to accurately predict how large the stack needs to be or
90  // it needs to check if things are full before pushing.
91  stack[top++] = obj;
92  }
93 
94 
95  /**
96  * Pop an item off of the stack. Has some underrun protection,
97  * but that's it.
98  *
99  * @return The popped object (or OREF_NULL if we're empty)
100  */
101  RexxInternalObject *pop()
102  {
103  // protect from an underrun
104  if (top == 0)
105  {
106  return OREF_NULL;
107  }
108  // decrement the top and return the pointer
109  return stack[--top];
110  }
111 
112  LiveStack *reallocate(size_t increment);
113  LiveStack *ensureSpace(size_t needed);
114 
115  inline bool checkRoom() { return top < size; }
116  inline bool checkRoom(size_t needed) { return size - top > needed; }
117  inline size_t stackSize() { return size; };
118  inline RexxInternalObject *stackTop() { return top == 0 ? OREF_NULL : stack[top - 1]; };
119  inline void copyEntries(LiveStack *other) { memcpy((char *)stack, (char *)other->stack, other->size * sizeof(RexxInternalObject *)); top = other->top; }
120  inline void clear() { memset(stack, 0, sizeof(RexxInternalObject*) * size); }
121 
122  protected:
123 
124  size_t size; // the stack size
125  size_t top; // the next position we push on the stack
126  RexxInternalObject *stack[1]; // the stack entries
127 };

130 /**
131  * A wrap-around marking stack. This stack can hold n
132  * elements, and if more than n elements are pushed on the
133  * stack, the oldest element is removed.
134  */
135 class PushThroughStack : public RexxInternalObject
136 {
137  public:
138  void *operator new(size_t, size_t);
139  void *operator new(size_t, size_t, bool temporary);
140  inline void operator delete(void *) { };
141 
142  inline PushThroughStack(RESTORETYPE restoreType) { ; };
143  PushThroughStack(size_t size);
144 
145  void init(size_t);
146 
147  void live(size_t) override;
148  void liveGeneral(MarkReason reason) override;
149 
150  // the position is origin zero, relative to the current. Current
151  // is the position of the last item pushed on to the stack.
152  inline RexxInternalObject *get(size_t pos)
153  {
154  // if they a really searching back, reduce it modulo size.
155  pos = pos % size;
156  // Since we wrap, the position within the bounds
157  if (pos > current)
158  {
159  // this is relative to the top of the stack
160  // if stack
161  return stack[size + current - pos];
162  }
163  else
164  {
165  // just pos items back from current
166  return stack[current - pos];
167  }
168  }
169 
170 
171  /**
172  * Push an object on to the stack
173  *
174  * @param obj The object to push
175  *
176  * @return
177  */
178  inline void push(RexxInternalObject *obj)
179  {
180  // this will wrap, as necessary, wiping out the reference at the bottom of the stack
181  incrementCurrent();
182  stack[current] = obj;
183  }
184 
185 
186  /**
187  * Pop an item off of the stack. Has some underrun protection,
188  * but that's it.
189  *
190  * @return The popped object (or OREF_NULL if we're empty)
191  */
192  RexxInternalObject *pop()
193  {
194  // get the current referenced item
195  RexxInternalObject *obj = stack[current];
196  // because we make everything in the stack and this is used for GC
197  // protection, null out the removed entry
198  stack[current] = OREF_NULL;
199  // now move back an item, with wrap protection.
200  decrementCurrent();
201  return obj;
202  }
203 
204  LiveStack *reallocate(size_t increment);
205 
206  inline size_t stackSize() { return size; };
207  inline RexxInternalObject *stackTop() { return stack[current]; };
208 
209  // increment and decrement will wrap
210  inline void decrementCurrent() { current = (current == 0) ? size - 1 : current - 1; }
211  inline void incrementCurrent() { if (++current >= size) current = 0; }
212 
213  inline void copyEntries(PushThroughStack *other) { memcpy((char *)stack, other->stack, other->size * sizeof(RexxInternalObject *)); current = other->current; }
214  inline void clear() { memset(stack, 0, sizeof(RexxInternalObject*) * size); }
215  void extend(size_t);
216  void remove(RexxInternalObject *, bool search = false);
217 
218  protected:
219 
220  size_t size; // the stack size
221  size_t current; // the last object on the stack (starts as if OREF_NULL had been written first)
222  RexxInternalObject *stack[1]; // the stack entries
223 };
