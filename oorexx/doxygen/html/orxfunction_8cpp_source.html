538 // test REDIRECTING command handler for TestAddCommandEnvironment
539 // just sets a return code identifying various I/O context functions
540 RexxObjectPtr RexxEntry rHandler(RexxExitContext *context,
541  RexxStringObject address,
542  RexxStringObject command,
543  RexxIORedirectorContext *ioContext)
544 {
545  size_t rc = 0;
546  // we don't do actual redirection, but we want to return a five-digit
547  // number with each digit in sequence representing the status of:
548  // - IsRedirectionRequested()
549  // - IsInputRedirected()
550  // - IsOutputRedirected()
551  // - IsErrorRedirected()
552  // - AreOutputAndErrorSameTarget()
553  rc = rc * 10 + ioContext->IsRedirectionRequested();
554  rc = rc * 10 + ioContext->IsInputRedirected();
555  rc = rc * 10 + ioContext->IsOutputRedirected();
556  rc = rc * 10 + ioContext->IsErrorRedirected();
557  rc = rc * 10 + ioContext->AreOutputAndErrorSameTarget();
558 
559  return context->WholeNumberToObject(rc);
560 }

563 // test command handler for the I/O redirection APIs
564 RexxObjectPtr RexxEntry ioHandler(RexxExitContext *context, RexxStringObject address, RexxStringObject command, RexxIORedirectorContext *ioContext)
565 {
566  CSTRING commandString = context->CString(command);
567 
568  if (strcmp(commandString, "INPUTOUTPUT") == 0)
569  {
570  CSTRING data;
571  size_t length;
572  size_t count = 0;
573 
574  ioContext->ReadInput(&data, &length);
575  while (data != NULL)
576  {
577  count++;
578  ioContext->WriteOutput(data, length);
579  ioContext->ReadInput(&data, &length);
580  }
581 
582  return context->StringSizeToObject(count);
583  }
584 
585  if (strcmp(commandString, "INPUTERROR") == 0)
586  {
587  CSTRING data;
588  size_t length;
589  size_t count = 0;
590 
591  ioContext->ReadInput(&data, &length);
592  while (data != NULL)
593  {
594  count++;
595  ioContext->WriteError(data, length);
596  ioContext->ReadInput(&data, &length);
597  }
598 
599  return context->StringSizeToObject(count);
600  }
601 
602  if (strcmp(commandString, "INPUTBOTH") == 0)
603  {
604  CSTRING data;
605  size_t length;
606  bool useError = false;
607  size_t count = 0;
608 
609  ioContext->ReadInput(&data, &length);
610  while (data != NULL)
611  {
612  count++;
613  if (useError)
614  {
615  ioContext->WriteError(data, length);
616  }
617  else
618  {
619  ioContext->WriteOutput(data, length);
620  }
621  useError = !useError;
622  ioContext->ReadInput(&data, &length);
623  }
624 
625  return context->StringSizeToObject(count);
626  }
627 
628 
629  if (strcmp(commandString, "NOBLANKOUTPUT") == 0)
630  {
631  CSTRING data;
632  size_t length;
633  size_t count = 0;
634 
635  ioContext->ReadInput(&data, &length);
636  while (data != NULL)
637  {
638  count++;
639  // only write non blank lines
640  if (length > 0)
641  {
642  ioContext->WriteOutput(data, length);
643  }
644  ioContext->ReadInput(&data, &length);
645  }
646 
647  return context->StringSizeToObject(count);
648  }
649 
650 
651  if (strcmp(commandString, "NOBLANKERROR") == 0)
652  {
653  CSTRING data;
654  size_t length;
655  size_t count = 0;
656 
657  ioContext->ReadInput(&data, &length);
658  while (data != NULL)
659  {
660  count++;
661  // only write non blank lines
662  if (length > 0)
663  {
664  ioContext->WriteError(data, length);
665  }
666  ioContext->ReadInput(&data, &length);
667  }
668 
669  return context->StringSizeToObject(count);
670  }
671 
672  if (strcmp(commandString, "BUFFEROUTPUT") == 0)
673  {
674  CSTRING data;
675  size_t length;
676  size_t count = 0;
677 
678  ioContext->ReadInput(&data, &length);
679  while (data != NULL)
680  {
681  count++;
682  ioContext->WriteOutputBuffer(data, length);
683  ioContext->ReadInput(&data, &length);
684  }
685 
686  return context->StringSizeToObject(count);
687  }
688 
689  if (strcmp(commandString, "BUFFERERROR") == 0)

691  CSTRING data;
692  size_t length;
693  size_t count = 0;
694 
695  ioContext->ReadInput(&data, &length);

698  count++;
699  ioContext->WriteErrorBuffer(data, length);
700  ioContext->ReadInput(&data, &length);
701  }
702 
703  return context->StringSizeToObject(count);
704  }
705 
706  if (strcmp(commandString, "BUFFERINPUT") == 0)
707  {
708  CSTRING data;
709  size_t length;
710 
711  ioContext->ReadInputBuffer(&data, &length);
712  if (data != NULL)
713  {
714  ioContext->WriteOutputBuffer(data, length);
715  }
716 
717  return context->StringSizeToObject(length);
718  }
719 
720  if (strcmp(commandString, "INPUTREDIRECTED") == 0)
721  {
722  return ioContext->IsInputRedirected() ? context->True() : context->False();
723  }
724 
725  if (strcmp(commandString, "OUTPUTREDIRECTED") == 0)
726  {
727  return ioContext->IsOutputRedirected() ? context->True() : context->False();
728  }
729

743  }
744
