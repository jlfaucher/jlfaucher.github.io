128 /**
129  * Perform the initial loading of a package. The loading
130  * process involves resolving the external library and
131  * attempting to resolve a Rexx package exporter routine
132  * in the library. If the library loads, but does not have
133  * the package exporter function, this is a classic library.
134  *
135  * If we do find a package exporter, then we can load all of
136  * the routines immediately. Method loads are deferred until
137  * the first request.
138  *
139  * @param manager The package manager we're attached to.
140  *
141  * @return True if we were able to load this as proper ooRexx package
142  * file, false if either step failed. We do not throw
143  * exceptions here, since these are usually loaded in the
144  * context of operations that return an error result instead
145  * of an exception.
146  */
147 bool LibraryPackage::load()
148 {
149  // try to load the package table.
150  package = getPackageTable();
151  // if this is NULL, return false to the manager
152  if (package == NULL)
153  {
154  return false;
155  }
156  // call the loader to get the package tables and set them up.
157  loadPackage();
158  return true;
159 }

185 /**
186  * Load a library and see if it is possible to retrieve
187  * a package entry from the library.
188  *
189  * @return A package table entry, if possible. A load failure or
190  * no package loading routines returns NULL.
191  */
192 RexxPackageEntry *LibraryPackage::getPackageTable()
193 {
194  // first try to load the libary
195  PACKAGE_LOADER loader;
196  // reset the library handle that was saved in the image.
197  lib.reset();
198 
199  if (!lib.load(libraryName->getStringData()))
200  {
201  // we don't report an exception here. This may have
202  // just been a probe attempt to see if we're real. We'll
203  // leave the exception decisions up to the package manager.
204  return NULL;
205  }
206 
207  // we're loaded now, vs. just a package fronting a name.
208  loaded = true;
209  // the try to resolve a package getting structure
210  // resolve the function address
211  void *entry = lib.getProcedure("RexxGetPackage");
212  if (entry == NULL)
213  {
214  // again, this is not an exception...this could just be
215  // a classic style function registration.
216  return NULL;
217  }
218  loader = (PACKAGE_LOADER)entry;
219  // call the loader to get the package tables and set them up.
220  return (*loader)();
221 }

250 /**
251  * Load all of the routines in a package, registering them with
252  * the package manager.
253  *
254  * @param table The package table describing this package.
255  */
256 void LibraryPackage::loadRoutines(RexxRoutineEntry *table)
257 {
258  // no routines exported by this package? Just return without
259  // doing anything.
260  if (table == NULL)
261  {
262  return;
263  }
264 
265  // create a directory of loaded routines
266  setField(routines, new_string_table());
267  // and a version using an uppercase name map
268  setField(publicRoutines, new_string_table());
269 
270  while (table->style != 0)
271  {
272  // table names tend to be specified in friendly form, we need to
273  // convert them to uppercase because "normal" Rexx function names
274  // tend to be uppercase.
275  Protected<RexxString> target = new_upper_string(table->name);
276  Protected<RexxString> routineName = new_string(table->name);
277 
278  Protected<BaseNativeRoutine> func;
279  if (table->style == ROUTINE_CLASSIC_STYLE)
280  {
281  func = new RegisteredRoutine(libraryName, routineName, (RexxRoutineHandler *)table->entryPoint);
282  }
283  else
284  {
285  func = new NativeRoutine(libraryName, routineName, (PNATIVEROUTINE)table->entryPoint);
286  }
287 
288  Protected<RoutineClass> routine = new RoutineClass(routineName, func);
289  // add this to our local table. Our local table needs to keep the original case,
290  // since those will be referenced by ::ROUTINE statements.
291  routines->put(routine, routineName);
292  // and also keep an upper case version for merging with packages
293  publicRoutines->put(routine, target);
294 
295  // add this to the global function pool
296  PackageManager::addPackageRoutine(target, routine);
297  // step to the next table entry
298  table++;
299  }
300 }

304 /**
305  * Locate a named method entry from the package registration
306  * table.
307  *
308  * @param name The target name.
309  *
310  * @return The entry associated with the target entry, if it exists.
311  * NULL indicates a not found condition.
312  */
313 RexxMethodEntry *LibraryPackage::locateMethodEntry(RexxString *name)
314 {
315  RexxMethodEntry *entry = package->methods;
316 
317  if (entry != NULL)
318  {
319  // scan the exported method table for the required method
320  while (entry->style != 0)
321  {
322  // is this one a name match? Make a method, add it to
323  // the table, and return.
324  if (name->strCaselessCompare(entry->name))
325  {
326  return entry;
327  }
328  entry++;
329  }
330  }
331  return NULL;
332 }

366 /**
367  * Get a NativeCode object for a method associated with a
368  * package.
369  *
370  * @param name Name of the target method.
371  *
372  * @return A NativeCode object for this method, if located.
373  */
374 NativeMethod *LibraryPackage::resolveMethod(RexxString *name)
375 {
376  // create our methods table if not yet created.
377  if (methods == OREF_NULL)
378  {
379  setField(methods, new_string_table());
380  }
381 
382  // see if this is in the table yet.
383  Protected<NativeMethod> code = (NativeMethod *)methods->get(name);
384  if (code == OREF_NULL)
385  {
386  // find the package definition
387  RexxMethodEntry *entry = locateMethodEntry(name);
388  // if we found one with this name, create a native method out of it.
389  if (entry != NULL)
390  {
391  code = new NativeMethod(libraryName, name, (PNATIVEMETHOD)entry->entryPoint);
392  methods->put(code, name);
393  return code;
394  }
395  // This, we know from nothing....
396  return OREF_NULL;
397  }
398  // had this cached already.
399  return code;
400 }

403 /**
404  * Get a Routine object from a native library package.
405  *
406  * @param name Name of the target routine, searching is done caseless.
407  *
408  * @return A Routine object, if located.
409  */
410 RoutineClass *LibraryPackage::resolveRoutine(RexxString *name)
411 {
412  // the package might not define any routines, in which case routines is NULL. We
413  // need to protect against that.
414  if (routines == OREF_NULL)
415  {
416  return OREF_NULL;
417  }
418 
419  // we resolve all of these at load time, so this is either in the table, or it's not.
420  RoutineClass *code = (RoutineClass *)routines->get(name);
421  if (code == OREF_NULL)
422  {
423  // try to locate name caseless
424  RexxRoutineEntry *entry = locateRoutineEntry(name);
425  // if we found one with a caseless equal name, return the routine
426  if (entry != NULL)
427  {
428  return (RoutineClass *)routines->get(new_string(entry->name));
429  }
430  // This, we know from nothing....
431  return OREF_NULL;
432  }
433  // had this resolved already.
434  return code;
435 }
