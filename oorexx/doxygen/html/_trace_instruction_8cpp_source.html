132 /**
133  * Execute a TRACE instruction
134  *
135  * @param context The current execution context.
136  * @param stack The current evaluation stack.
137  */
138 void RexxInstructionTrace::execute(RexxActivation *context, ExpressionStack *stack)
139 {
140  // trace if needed.
141  context->traceInstruction(this);
142 
143  // if this is the debug skip form, turn on the trace suppression
144  if (skip)
145  {
146  // turn on the skip mode in the context. A negative value also suppresses
147  // the tracing.
148  context->debugSkip(debugSkip);
149  }
150  // non-dynamic form?
151  else if (expression == OREF_NULL)
152  {
153  // if not in debug mode, we just do the setting. The TRACE instruction
154  // is ignored in debug mode, although we do trace everything
155  if (!context->inDebug())
156  {
157  context->setTrace(settings);
158  }
159  else
160  {
161  // we're in debug, so do the pause
162  context->pauseInstruction();
163  }
164  }
165  // dynamic form, requiring an expression evaluation.
166  else
167  {
168  // evaluate, and get as a string value
169  RexxObject *result = expression->evaluate(context, stack);
170  // Even trace gets traced :-)
171  context->traceKeywordResult(GlobalNames::VALUE, result);
172  Protected<RexxString> value = result->requestString();
173 
174  // if this is not numeric, it is a TRACE option.
175  wholenumber_t number;
176  if (!value->requestNumber(number, number_digits()))
177  {
178  // again, we don't change anything if we're already in debug mode.
179  if (!context->inDebug())
180  {
181  context->setTrace(value);
182  }
183  else
184  {
185  // in debug mode means we do need to pause.
186  context->pauseInstruction();
187  }
188  }
189  // a valid number
190  else
191  {
192  // turn on the skip mode in the context. A negative value also suppresses
193  // the tracing.
194  context->debugSkip(number);
195  }
196  }
197 }
