122  public:
123  inline NumberBuilder(RexxString *s) : current(s->getWritableData()) {}
124 
125  inline void addSign(bool isNegative) { if (isNegative) { *current = RexxString::ch_MINUS; current++; } }
126  inline void addExponentSign(bool isNegative) { *current = isNegative ? RexxString::ch_MINUS : RexxString::ch_PLUS; current++; }
127  inline void addDecimal() { append(RexxString::ch_PERIOD); }
128  inline void addExponent(const char *exp)
129  {
130  size_t len = strlen(exp);
131  memcpy(current, exp, len);
132  current += len;
133  }
134  inline void append(const char *d, size_t l) { memcpy(current, d, l); current += l; }
135  inline void append(char c)
136  {
137 // avoid warning: writing 1 byte into a region of size 0 [-Wstringop-overflow=]
138 // (or similar) in gcc 12 and above, due to our RexxString char stringData[4]
139 #ifdef HAVE_PRAGMA_GCC_STRINGOPOVERFLOW
140 #pragma GCC diagnostic push
141 #pragma GCC diagnostic ignored "-Wstringop-overflow"
142 #endif
143  *current++ = c;
144 #ifdef HAVE_PRAGMA_GCC_STRINGOPOVERFLOW
145 #pragma GCC diagnostic pop
146 #endif
147  }
148  inline void addDigits(const char *d, wholenumber_t len)
149  {
150  for (wholenumber_t i = 0; i < len; i++)
151  {
152  append(d[i] + RexxString::ch_ZERO);
153  }
154  }
155  inline void addZeroDecimal() { append('0'); append('.'); }
156  inline void addZeros(wholenumber_t count)
157  {
158  memset(current, '0', count);
159  current += count;
160  }
161 
162  inline void addSpaces(wholenumber_t count)
163  {
164  memset(current, ' ', count);
165  current += count;
166  }
167 
168  inline void addIntegerPart(bool sign, const char *digits, wholenumber_t intDigits, wholenumber_t pad = 0)
169  {
170  addSign(sign);
171  addDigits(digits, intDigits);
172  addZeros(pad);
173  }
174 
175  inline void addDecimalPart(const char *digits, wholenumber_t decimalDigits, wholenumber_t leadPad = 0, wholenumber_t trailingPad = 0)
176  {
177  addDecimal();
178  addZeros(leadPad);
179  addDigits(digits, decimalDigits);
180  addZeros(trailingPad);
181  }

184  protected:
185  char *current; // current output pointer
186  };
