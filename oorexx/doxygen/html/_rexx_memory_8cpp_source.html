138 /**
139  * Initialize the memory object, including getting the
140  * initial memory pool allocations.
141  *
142  * @param restoringImage
143  * True if we are initializing during an image restore, false
144  * if we need to build the initial image environemnt (i.e., called
145  * via rexximage during a build).
146  * @param imageTarget
147  * The location to store the image if this is a save operation.
148  */
149 void MemoryObject::initialize(bool restoringImage, const char *imageTarget)
150 {
151  // The constructor makes sure some crucial aspects of the Memory object are set up.
152  new (this) MemoryObject;
153 
154  // create our various segment pools
155  new (&newSpaceNormalSegments) NormalSegmentSet(this);
156  new (&newSpaceLargeSegments) LargeSegmentSet(this);
157  new (&newSpaceSingleSegments) SingleObjectSegmentSet(this);
158 
159  // and the new/old Space segments
160  new (&oldSpaceSegments) OldSpaceSegmentSet(this);
161 
162  collections = 0;
163  allocations = 0;
164  globalStrings = OREF_NULL;
165 
166  // get our table of virtual functions setup first thing. We need this
167  // before we can allocate our first real object.
168  buildVirtualFunctionTable();
169 
170  // This is the live stack used for
171  // sweep marking (not allocated from the object heap)
172  liveStack = new(Memory::LiveStackSize) LiveStack(Memory::LiveStackSize);
173 
174  // if we're restoring, load everything from the image file. All of the
175  // classes will exist then, as well as restoring all of the
176  // behaviours
177  if (restoringImage)
178  {
179  restoreImage();
180  }
181 
182  // set the object behaviour
183  memoryObject.setBehaviour(TheMemoryBehaviour);
184 
185  // make sure we have an inital segment set to allocate from.
186  newSpaceNormalSegments.getInitialSet();
187 
188  // we also get an initial segment set for large object, since we
189  // will need this for the globalReferences table.
190  newSpaceLargeSegments.getInitialSet();
191 
192  // get the initial uninit table
193  uninitTable = new_identity_table();
194 
195  // and our global references table
196  globalReferences = new MapTable(Memory::DefaultGlobalReferenceSize);
197 
198  // this is our size of uninit objects awaiting processing.
199  pendingUninits = 0;
200 
201  // is this image creation? This will build and save the image, then
202  // terminate
203  if (!restoringImage)
204  {
205  createImage(imageTarget);
206  }
207 
208  restore(); // go restore the state of the memory object
209 }

226 /**
227  * Main memory_mark driving loop
228  *
229  * @param rootObject The root object used for the marking (usually the
230  * MemoryObject).
231  */
232 void MemoryObject::markObjectsMain(RexxInternalObject *rootObject)
233 {
234  // for some of the root objects, we get called to mark them before they get allocated.
235  // make sure we don't process any null references.
236  if (rootObject == OREF_NULL)
237  {
238  return;
239  }
240 
241  // flag that we're in the middle of a mark operation
242  markingObjects = true;
243 
244  // set up the live marking word passed to the live() routines
245  // we include the OldSpaceBit here to allow both conditions to be tested
246  // in one shot.
247  size_t liveMark = markWord | ObjectHeader::OldSpaceBit;
248 
249  allocations = 0;
250  // add a fence to the stack to act as a terminator.
251  pushLiveStack(OREF_NULL);
252  // mark the root object and start processing the stacked item.
253  // we terminate once we hit the null fence item.
254  mark(rootObject);
255  for (RexxInternalObject *markObject = popLiveStack(); markObject != OREF_NULL; markObject = popLiveStack())
256  {
257  // mark the behaviour as live
258  memory_mark(markObject->behaviour);
259  // Mark other referenced obj. We can do this without checking
260  // the references flag because we only push the object on to
261  // the stack if it has references.
262  allocations++;
263  markObject->live(liveMark);
264  }
265 
266  // we are done marking, no longer in a critical section
267  markingObjects = false;
268 }

271 /**
272  * Check for objects that require an uninit method run.
273  */
274 void MemoryObject::checkUninit()
275 {
276  // we might not actually have a table yet, so make sure we check
277  // before using it.
278  if (uninitTable == NULL)
279  {
280  return;
281  }
282 
283  // scan the uninit table looking for objects that are elegable for collection.
284  for (HashContents::TableIterator iterator = uninitTable->iterator(); iterator.isAvailable(); iterator.next())
285  {
286  RexxInternalObject *uninitObject = iterator.value();
287 
288  // was this object not marked by the last sweep operation?
289  if (uninitObject != OREF_NULL && uninitObject->isObjectDead(markWord))
290  {
291  // mark this as ready for uninit
292  uninitObject->setReadyForUninit();
293  // up the pending uninit count
294  pendingUninits++;
295  }
296  }
297 }

334 /**
335  * Run any pending uninit methods for this activity.
336  */
337 void MemoryObject::runUninits()
338 {
339  // if we're already processing this, don't try to do this
340  // recursively.
341  if (processingUninits)
342  {
343  return;
344  }
345 
346  // ok, turn on the interlock
347  processingUninits = true;
348  verboseMessage("Starting to process pending uninit methods\n");
349 
350  // get the current activity for running the uninits
351  Activity *activity = ActivityManager::currentActivity;
352 
353  // scan the uninit table looking for objects that are elegable for collection.
354  for (HashContents::TableIterator iterator = uninitTable->iterator(); iterator.isAvailable();)
355  {
356  RexxInternalObject *uninitObject = iterator.value();
357 
358  // was this object already marked for running the uninit? run it now
359  if (uninitObject != OREF_NULL && uninitObject->isReadyForUninit())
360  {
361  // we remove this item here and advance the iterator.
362  iterator.removeAndAdvance();
363  // remove the pending item count
364  pendingUninits--;
365 
366  // because we've removed this from the uninit table, it is no longer
367  // proctected. We need to ensure it does not get GC'd until after the
368  // uninit method runs
369  ProtectedObject p(uninitObject);
370 
371  // run this method with appropriate error trapping
372  UninitDispatcher dispatcher(uninitObject);
373  activity->run(dispatcher);
374  }
375  // not processing that item, so just step the iterator
376  else
377  {
378  iterator.next();
379  }
380  }
381 
382  // turn off the interlock
383  processingUninits = false; ;
384  verboseMessage("Done processing pending uninit methods\n");
385 }

413 /**
414  * Main mark routine for garbage collection.
415  */
416 void MemoryObject::markObjects()
417 {
418  verboseMessage("Beginning mark operation\n");
419 
420  // do the marking using the memory object as the root object.
421  markObjectsMain(this);
422  // now process the weak reference queue...We check this before the
423  // uninit list is processed so that the uninit list doesn't mark any of the
424  // weakly referenced items. We don't want an object placed on the uninit queue
425  // to end up strongly referenced later.
426  checkWeakReferences();
427 
428  // check for any objects we're holding for uninits that have
429  // gone out of scope
430  checkUninit();
431  // make the uninit table and the pending uninits queue to keep those
432  // objects from getting reclaimed.
433  markObjectsMain(uninitTable);
434 
435  // if we had to expand the live stack previously, we allocated a temporary
436  // one from malloc() storage rather than the object heap. We will hold on
437  // to the expanded one because once we've hit the threshold where we expand,
438  // it's likely we'll need to in the future.
439  verboseMessage("Mark operation completed\n");
440 }

443 /**
444  * Scan the weak reference queue looking for either dead weak
445  * objects or weak references that refer to objects that have gone out of
446  * scope.
447  */
448 void MemoryObject::checkWeakReferences()
449 {
450  WeakReference *current = weakReferenceList;
451  // list of "live" weak references...built while scanning
452  WeakReference *newList = OREF_NULL;
453 
454  // loop through the list
455  while (current != OREF_NULL)
456  {
457  // we have to save the next one in the list
458  WeakReference *next = current->nextReferenceList;
459  // this reference still in scope?
460  if (current->isObjectLive(markWord))
461  {
462  // keep this one in the list
463  current->nextReferenceList = newList;
464  newList = current;
465  // have a reference?
466  if (current->referentObject != OREF_NULL)
467  {
468  // if the object is not alive, null out the reference
469  if (!current->referentObject->isObjectLive(markWord))
470  {
471  current->referentObject = OREF_NULL;
472  }
473  }
474  }
475  // step to the new next item
476  current = next;
477  }
478 
479  // update the list
480  weakReferenceList = newList;
481 }

547 /**
548  * Allocate a segment of the requested size. The requested size
549  * is the desired size, while the minimum is the absolute minimum we can
550  * handle. This takes care of the overhead accounting and additional
551  * rounding. The requested size is assumed to have been rounded up to the
552  * next "appropriate" boundary already, and the segment overhead will be
553  * allocated from that part, if possible. Otherwise, an
554  * additional page is added.
555  *
556  * @param requestedBytes
557  * The suggested number of bytes to allocate.
558  * @param minBytes The minimum number of bytes we can accept. Failure to
559  * allocate the minimum is an out-of-memory situation.
560  *
561  * @return The new memory segment.
562  */
563 MemorySegment *MemoryObject::newSegment(size_t requestedBytes, size_t minBytes)
564 {
565  // first make sure we've got enough space for the control
566  // information. We don't do any rounding. The segment set has already
567  // decided what the apropriate size should be. In some cases (i.e., the single object
568  // segment set), rounding would just waste memory.
569  requestedBytes = requestedBytes + MemorySegment::MemorySegmentOverhead;
570  verboseMessage("Allocating a new segment of %d bytes\n", requestedBytes);
571  // try to allocate a new segment
572  MemorySegment *segment = newSegment(requestedBytes);
573  if (segment == NULL)
574  {
575  verboseMessage("Allocating a boundary new segment of %d bytes\n", requestedBytes);
576  // Segmentsize is the minimum size request we handle. If
577  // minbytes is small, then we're just adding a segment to the
578  // small pool. Reduce the request to SegmentSize and try again.
579  // For all other requests, try once more with the minimum.
580  minBytes = minBytes + MemorySegment::MemorySegmentOverhead;
581  verboseMessage("Allocating a fallback new segment of %d bytes\n", minBytes);
582  // try to allocate once more...if this fails, the caller will
583  // have to handle it.
584  segment = newSegment(minBytes);
585  }
586  return segment; // return the allocated segment
587 }

624 /**
625  * Restore a saved image to usefulness.
626  */
627 void MemoryObject::restoreImage()
628 {
629  // Nothing to restore if we have a buffer already
630  if (restoredImage != NULL)
631  {
632  return;
633  }
634 
635  size_t imageSize;
636 
637  // load the image file
638  loadImage(restoredImage, imageSize);
639  // we write a size to the start of the image when the image is created.
640  // the restoredImage buffer does not include that image size, so we
641  // need to pretend the buffer is slightly before the start.
642  // image data is just past that information.
643  char *relocation = restoredImage - sizeof(size_t);
644 
645  // create a handler for fixing up reference addresses.
646  ImageRestoreMarkHandler markHandler(relocation);
647  setMarkHandler(&markHandler);
648 
649  // address the start and end of the image.
650  RexxInternalObject *objectPointer = (RexxInternalObject *)restoredImage;
651  RexxInternalObject *endPointer = (RexxInternalObject *)(restoredImage + imageSize);
652 
653  // the save array is the 1st object in the buffer
654  ArrayClass *saveArray = (ArrayClass *)objectPointer;
655 
656  // now loop through the image buffer fixing up the objects.
657  while (objectPointer < endPointer)
658  {
659  size_t primitiveTypeNum;
660 
661  // Fixup the Behaviour pointer for the object.
662 
663  // If this is a primitive object, we can just pick up the hard coded
664  // behaviour.
665  if (objectPointer->isNonPrimitive())
666  {
667  // Working with a copy, so don't use the static table version.
668  // the behaviour will have been packed into the image, so we need to
669  // pick up the reference.
670  RexxBehaviour *imageBehav = (RexxBehaviour *)(relocation + (uintptr_t)objectPointer->behaviour);
671  // and rewrite the offset version with the resolved pointer
672  objectPointer->behaviour = imageBehav;
673  // get the type number from the behaviour. This tells us what
674  // virtual function pointer to use.
675  primitiveTypeNum = imageBehav->getClassType();
676  }
677  else
678  {
679  // the original behaviour pointer has been encoded as a type number and class
680  // category to allow us to convert back to the appropriate type.
681  objectPointer->behaviour = RexxBehaviour::restoreSavedPrimitiveBehaviour(objectPointer->behaviour);
682  primitiveTypeNum = objectPointer->behaviour->getClassType();
683  }
684  // This will be an OldSpace object. We delay setting this
685  // until now, because the oldspace bit is overloaded with the
686  // NonPrimitive bit. Since the we are done with the
687  // non-primitive bit now, we can use this for the oldspace
688  // flag too.
689  objectPointer->setOldSpace();
690  // now fix up the virtual function table for the object using the type number
691  objectPointer->setVirtualFunctions(virtualFunctionTable[primitiveTypeNum]);
692 
693  // if the object has references, call liveGeneral() to cause these references to
694  // be converted back into real pointers
695  if (objectPointer->hasReferences())
696  {
697  objectPointer->liveGeneral(RESTORINGIMAGE);
698  }
699  // advance to the next object in the image
700  objectPointer = objectPointer->nextObject();
701  }
702 
703  // now start restoring the critical objects from the save array
704  TheEnvironment = (DirectoryClass *)saveArray->get(saveArray_ENV);
705  // restore all of the primitive behaviour data...right now, these are
706  // all default values.
707  ArrayClass *primitiveBehaviours = (ArrayClass *)saveArray->get(saveArray_PBEHAV);
708  for (size_t i = 0; i <= T_Last_Exported_Class; i++)
709  {
710  RexxBehaviour::primitiveBehaviours[i].restore((RexxBehaviour *)primitiveBehaviours->get(i + 1));
711  }
712 
713  TheSystem = (StringTable *)saveArray->get(saveArray_SYSTEM);
714  TheTrueObject = (RexxInteger *)saveArray->get(saveArray_TRUE);

972 /**
973  * allocate a new object of the requested size.
974  *
975  * @param requestLength
976  * The required object length.
977  * @param type The object type this should be initialized to.
978  *
979  * @return The storage for creating a new object.
980  */
981 RexxInternalObject *MemoryObject::newObject(size_t requestLength, size_t type)
982 {
983  allocations++;
984 
985  // Compute size of new object and determine where we should
986  // allocate from
987  requestLength = Memory::roundObjectBoundary(requestLength);
988 
989  RexxInternalObject *newObj;
990 
991  // is this a typical small object? These come from a segment set that optimizes
992  // for small size objects.
993  if (requestLength <= MemorySegment::LargeBlockThreshold)
994  {
995  // make sure we don't go below our minimum size.
996  if (requestLength < Memory::MinimumObjectSize)
997  {
998  requestLength = Memory::MinimumObjectSize;
999  }
1000  newObj = newSpaceNormalSegments.allocateObject(requestLength);
1001  // if we could not allocate, process an allocation failure. This will
1002  // drive a garbage collection and potentially expand the heap size. This also
1003  // raises an error if all of the recovery steps result in an allocation failure.
1004  if (newObj == NULL)
1005  {
1006  newObj = newSpaceNormalSegments.handleAllocationFailure(requestLength);
1007  }
1008  }
1009  // between the small size threshold and "really big", we allocate from a segment set
1010  // designed to handle this range of allocation.
1011  else if (requestLength <= MemorySegment::SingleBlockThreshold)
1012  {
1013  // these are larger objects, but we still try to pack them tightly in the
1014  // segment sets.
1015  newObj = newSpaceLargeSegments.allocateObject(requestLength);
1016

1018  if (newObj == NULL)

1021  }
1022  }
1023  // once we get into really large objects (segment size or larger), we manage those allocations separately

1025  else
1026  {
1027  // since we're only to have one object in the segment, there's no need to round
1028  // the size to any boundary.
1029  newObj = newSpaceSingleSegments.allocateObject(requestLength);
1030  if (newObj == NULL)
1031  {
1032  // Recovery here will always force a GC and try again. Because previously
1033  // allocated large objects may have been returned to the system, this might succeed
1034  // this time.
1035  newObj = newSpaceSingleSegments.handleAllocationFailure(requestLength);

1052  //and safely anchored by caller.
1053  pushSaveStack(newObj);
