78  memset(getData() + offset, character, l);
79  }
80 protected:
81 
82  // the following field is padding required to get the start of the of the
83  // buffer data aligned on an object grain boundary. Since we unflatten saved programs
84  // by reducing the size of the surrounding buffer to reveal the exposed data, we need
85  // to ensure appropriate data alignment. Fortunately, because the sizes of all of the
86  // fields doubles when going to 64-bit, this single padding item is sufficient to
87  // get everything lined up on all platforms.
88  size_t reserved;
89  size_t bufferSize; // size of the buffer
90  size_t dataLength; // length of the buffer data (freqently the same)
