52 /**
53  * Locate the first occurrence of any character from a given set.
54  *
55  * @param string The string we're searching in.
56  * @param set The set of characters to search for as an ASCII-Z string.
57  * The terminating null is not part of the search set.
58  * @param length The length of the string we search in.
59  *
60  * @return The hit position or null if none of the characters are found.
61  */
62 const char *Utilities::locateCharacter(const char *string, const char *set, size_t length )
63 {
64  while (length-- > 0)
65  {
66  // check for a match in the string. NOTE: this will return a hit
67  // for a null character, so we need to check for this after the fact
68  if (strchr(set, *string))
69  {
70  if (*string != '\0')
71  {
72  return string;
73  }
74  }
75  string++;
76  }
77  // no matched position
78  return NULL;
79 }

82 /**
83  * Portable implementation of an ascii-z caseless string compare.
84  *
85  * @param opt1 First string argument
86  * @param opt2 Second string argument.
87  *
88  * @return The compare result. Returns 0, negative, or positive depending
89  * one the ordering compare result.
90  */
91 int Utilities::strCaselessCompare(const char *op1, const char *op2)
92 {
93  while (toLower(*op1) == toLower(*op2))
94  {
95  if (*op1 == 0)
96  {
97  return 0;
98  }
99  op1++;
100  op2++;
101  }
102 
103  return (toLower(*op1) - toLower(*op2));
104 }

106 /**
107  * Portable implementation of a caseless memory compare.
108  *
109  * @param opt1 First memory location to compare.
110  * @param opt2 Second memory location.
111  * @param len Length to compare.
112  *
113  * @return The compare result. Returns 0, negative, or positive depending
114  * one the ordering compare result.
115  */
116 int Utilities::memicmp(const void *mem1, const void *mem2, size_t len)
117 {
118  const char *op1 = (const char *)mem1;
119  const char *op2 = (const char *)mem2;
120  while(len != 0)
121  {
122  if (toLower(*op1) != toLower(*op2))
123  {
124  return toLower(*op1) - toLower(*op2);
125 
126  }
127  op1++;
128  op2++;
129  len--;
130  }
131  return 0;
132 }

172 /**
173  * Bounded strchr() function.
174  *
175  * @param data The data pointer.
176  * @param n The maximum length to scan.
177  * @param ch The character of interest.
178  *
179  * @return The pointer to the located character, or NULL if it isn't found.
180  */
181 const char *Utilities::strnchr(const char *data, size_t n, char ch)
182 {
183  const char *endPtr = data + n;
184  while (data < endPtr && *data != '\0')
185  {
186  if (*data == ch)
187  {
188  return data;
189  }
190  data++;
191  }
192  return NULL;
193 }

196 /**
197  * A strncpy() implementation that will always null-terminate.
198  *
199  * @param dst The destination pointer.
200  * @param src The source pointer.
201  * @param len The destination buffer length.
202  *
203  * @return false when no truncation has occured, true otherwise.
204  */
205 bool Utilities::strncpy(char *dest, const char *src, size_t len)
206 {
207  // ::strncpy() is dangerous as it doesn't NUL-terminate dest when truncating,
208  // and inefficient as it always writes len bytes to dest, filling up with
209  // NUL characters when not truncating. Alternatives like strlcpy() in libbsd
210  // or strncpy_s() aren't readily available, so we implement our own a safe
211  // version here.
212  char *end = (char *)memchr(src, '\0', len);
213  if (end != NULL)
214  {
215  // we found a terminating NUL character
216  memcpy(dest, src, end - src + 1);
217  return false; // success
218  }
219  else
220  {
221  // no NUL char found
222  memcpy(dest, src, len - 1);
223  dest[len - 1] = '\0';
224  return true; // truncated
225  }
226 }
