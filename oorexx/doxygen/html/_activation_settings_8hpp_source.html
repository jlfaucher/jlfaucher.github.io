61 // activationContext values
62 // these are done as bit settings to
63 // allow multiple potential values
64 // to be checked with a single test
65 typedef enum
66 {
67  DEBUGPAUSE = 0x00000001,
68  METHODCALL = 0x00000002,
69  INTERNALCALL = 0x00000004,
70  INTERPRET = 0x00000008,
71  PROGRAMCALL = 0x00000010,
72  EXTERNALCALL = 0x00000020,
73  // check for top level execution
74  TOP_LEVEL_CALL = (PROGRAMCALL | METHODCALL | EXTERNALCALL),
75  // non-method top level execution
76  PROGRAM_LEVEL_CALL = (PROGRAMCALL | EXTERNALCALL),
77  // non-method top level execution
78  PROGRAM_OR_METHOD = (PROGRAMCALL | METHODCALL),
79  // call is within an activation
80  INTERNAL_LEVEL_CALL = (INTERNALCALL | INTERPRET),
81 } ActivationContext;

95 /**
96  * Main activation settings section created for easy
97  * copying between related activations.
98  */
99 class ActivationSettings
100 {
101  typedef enum
102  {
103  singleStep, // we are single stepping execution
104  singleStepNested, // this is a nested stepping
105  debugPromptIssued, // debug prompt already issued
106  debugBypass, // skip next debug pause
107  procedureValid, // procedure instruction is valid
108  haltCondition, // a HALT condition occurred
109  traceOn, // external trace ON condition occurred
110  traceOff, // external trace OFF condition occurred
111  sourceTraced, // source string has been traced
112  clauseExits, // need to call clause boundary exits
113  forwarded, // forward instruction active
114  replyIssued, // reply has already been issued
115  setTraceOn, // trace turned on externally
116  setTraceOff, // trace turned off externally
117  trapsCopied, // copy of trap info has been made
118  returnStatusSet, // had our first host command
119  transferFailed, // transfer of variable lock failure
120  traceSuppress, // tracing is currently suppressed
121  elapsedReset, // The elapsed time stamp was reset via time('r')
122  guardedMethod, // this is a guarded method
123  ioConfigCopied, // We have made a copy of the config table
124  } ActivationFlag;
