128 /**
129  * Execute an IF/WHEN instruction.
130  *
131  * @param context The current execution context.
132  * @param stack The current evaluation stack.
133  */
134 void RexxInstructionIf::execute(RexxActivation *context, ExpressionStack *stack)
135 {
136  context->traceInstruction(this);
137 
138  // evaluate and trace the condition expression.
139  RexxObject *result = condition->evaluate(context, stack);
140  context->traceResult(result);
141 
142  // the comparison methods return either .true or .false, so we
143  // can to a quick test against those.
144  if (result == TheFalseObject)
145  {
146  // we execute the ELSE branch
147  context->setNext(else_location->nextInstruction);
148  }
149 
150  // if it is not the .true object, we need to perform a fuller
151  // evaluation of the result.
152  else if (result != TheTrueObject)
153  {
154  // evaluate and decide if we take the ELSE branch
155  if (!result->truthValue(instructionType == KEYWORD_WHEN ? Error_Logical_value_when : Error_Logical_value_if))
156  {
157  context->setNext(else_location->nextInstruction);
158  }
159  }
160 
161  // We do nothing for true...we just continue on to the next instruction.
162 
163  context->pauseInstruction();
164 }
