107 /**
108  * Runtime execution of an END statement.
109  *
110  * @param context The current execution context.
111  * @param stack The current evaluation stack.
112  */
113 void RexxInstructionEnd::execute(RexxActivation *context, ExpressionStack *stack )
114 {
115  // SIGNAL will disable all active block instructions, so it is possible that
116  // this END has been encountered without its corresponding block instruction actually
117  // being active.
118  if (!context->hasActiveBlockInstructions())
119  {
120  // trace anyway, then give an error
121  context->traceInstruction(this);
122  reportException(Error_Unexpected_end_nodo);
123  }
124 
125  // we have different actions depending on the type of instruction
126  // we're hooked to. Probably not worth trying to create separate END
127  // instruction types.
128  switch (getStyle())
129  {
130  // All types of loops. There will be a block managed by the
131  // context that holds the state of an active loop.
132  case LOOP_BLOCK:
133  {
134  // get the top block from the context.
135  DoBlock *doBlock = context->topBlockInstruction();
136  // For tracing the end, reset the indentation to the block beginning indent.
137  // then trace.
138  context->setIndent(doBlock->getIndent());
139  context->traceInstruction(this);
140  // tell the DO/LOOP instruction we're back around.
141  ((RexxInstructionBaseLoop *)(doBlock->getParent()))->reExecute(context, stack, doBlock);
142  break;
143  }
144 
145  // END of a select block. If we have fallen through to this, then we have an
146  // error with a SELECT instruction that does not have an OTHERWISE section.
147  // This is an error.
148  case SELECT_BLOCK:
149  {
150  // back off the indentation and trace
151  context->unindent();
152  context->traceInstruction(this);
153 
154  // we've traced, now raise an error.
155  reportException(Error_When_expected_nootherwise);
156  break;
157  }
158 
159  // for labeled BLOCK types, we need to remove the active marker.
160  case OTHERWISE_BLOCK:
161  case LABELED_OTHERWISE_BLOCK:
162  case LABELED_DO_BLOCK:
163  {
164  // terminateBlock will also reset the indentation for tracing
165  context->terminateBlockInstruction();
166  context->traceInstruction(this);
167  break;
168  }
169 
170  // probably a simple unlabeled DO...back off and fall through.
171  default:
172  {
173  context->removeBlockInstruction();
174  context->traceInstruction(this);
175  break;
176  }
177  }
178 }
