137 /**
138  * Initialize an interpreter instance.
139  *
140  * @param activity The root activity for the interpreter instance.
141  * @param handlers The exit handlers used by all threads running under this instance.
142  * @param defaultEnvironment
143  * The default address environment for this interpreter instance. Each
144  * active interpreter instance can define its own default environment.
145  */
146 void InterpreterInstance::initialize(Activity *activity, RexxOption *options)
147 {
148  rootActivity = activity;
149  allActivities = new_queue();
150  searchExtensions = new_array(); // this will be filled in during options processing
151  requiresFiles = new_string_table(); // our list of loaded requires packages
152  // this gets added to the entire active list.
153  allActivities->append(activity);
154  // create a default wrapper for this security manager
155  securityManager = new SecurityManager(OREF_NULL);
156  // set the default system address environment (can be overridden by options)
157  defaultEnvironment = SystemInterpreter::getDefaultAddressName();
158  // our list of command handlers (must be done before options are processed)
159  commandHandlers = new_string_table();
160 
161  // associate the thread with this instance
162  activity->setupAttachedActivity(this);
163  // create a local environment
164  localEnvironment = new_directory();
165  processOptions(options);
166  // when handled originally, we didn't have the exits setup
167  // do this now.
168  activity->setupExits();
169  // do system specific initialization
170  sysInstance.initialize(this, options);
171  // register the system command handlers for this platform.
172  sysInstance.registerCommandHandlers(this);
173  // now do the local initialization;
174  Interpreter::initLocal();
175 }

209 /**
210  * Attach a thread to an interpreter instance.
211  *
212  * @return The attached activity.
213  */
214 Activity *InterpreterInstance::attachThread()
215 {
216  // first check for an existing activity
217  Activity *oldActivity = findActivity();
218  // do we have an activity for this already? There are possible
219  // situations here. Normally, this will be a case where the thread
220  // exits the interpreter and calls external code that is then
221  // reattaches to the interpreter using the same interpreter instance.
222  // We treat this as a nested activity and keep the same activity
223  // object for use.
224  //
225  // On Windows, however there's another situation that can occur
226  // that's more difficult to handle. When we wait for a semaphore on
227  // Windows, we also process the Windows message queue. If the activity
228  // is waiting for the kernel semaphore and a
229  // dispatched messages make ooRexx API calls on the same thread, then
230  // the original activity can get its semaphore posted, but it cannot wake
231  // up until the message returns, which might also be sitting waiting to be
232  // dispatched. The result is a hang.
233  if (oldActivity != OREF_NULL && !oldActivity->isNestable())
234  {
235  // make sure we mark this as attached...we might be nested and don't want to
236  // clean this up until we complete
237  oldActivity->nestAttach();
238  return oldActivity;
239  }
240 
241  // we need to get a new activity set up for this particular thread
242  // NB: The activity manager handles the special case of an activity
243  // on the dispatch queue.
244  Activity *activity = ActivityManager::attachThread();
245 
246  // The creation of the new activity also made that activity the current
247  // thread.
248 
249  // resource lock must come AFTER we attach the thread and acquire the
250  // kernel lock. We must never try to acquire the kernel lock while holding
251  // the resource lock.
252  ResourceSection lock;
253  // add this to the activity lists
254  allActivities->append(activity);
255  // associate the thread with this instance
256  activity->setupAttachedActivity(this);
257  return activity;
258 }

261 /**
262  * Detach a thread from this interpreter instance.
263  *
264  * @param activity The activity to detach
265  *
266  * @return true if this worked ok.
267  */
268 bool InterpreterInstance::detachThread(Activity *activity)
269 {
270  // if the thread in question is not found or this is not an attached thread,
271  // this fails
272  if (activity == OREF_NULL || !activity->isAttached())
273  {
274  return false;
275  }
276 
277  if (activity->isNestedAttach())
278  {
279  // if we reused the activity because of a nested callback attach, then
280  // we just decrement the nesting count and return without cleaning up
281  // any resources.
282  activity->returnAttach();
283  return true;
284  }
285 
286 
287  // this activity owned the kernel semaphore before entering here...release it
288  // now.
289  activity->releaseAccess();
290  ResourceSection lock;
291 
292  allActivities->removeItem(activity);
293  // have the activity manager remove this from the global tables
294  // and perform resource cleanup
295  ActivityManager::returnActivity(activity);
296 
297  // Was this the last detach of an thread? Signal the shutdown event
298  if (allActivities->items() <= 1 && terminating)
299  {
300  terminationSem.post();
301  }
302  return true;
303 }

340 /**
341  * Return a spawned activity back to the activity pool. This
342  * will disassociate the activity from the interpreter instance
343  * and place the thread back into the global thread pool.
344  *
345  * @param activity The activity to return.
346  *
347  * @return true if the activity was added to the pool, false if the
348  * activity should continue with termination.
349  */
350 bool InterpreterInstance::poolActivity(Activity *activity)
351 {
352  ResourceSection lock;
353  // detach from this instance
354  activity->detachInstance();
355  // remove from the activities lists for the instance
356  allActivities->removeItem(activity);
357  if (terminating)
358  {
359  // is this the last one to finish up? Generally, the main thread
360  // will be waiting for this to terminate. That is thread 1, we're thread
361  // 2. In reality, this is the test for the last "spawned" thread.
362  if (allActivities->items() <= 1)
363  {
364  terminationSem.post();
365  }
366  // don't allow this to be pooled
367  return false;
368  }
369  // and move this to the global activity pool
370  return ActivityManager::poolActivity(activity);
371 }

374 /**
375  * Locate an activity for a specific thread ID that's attached
376  * to this instance.
377  *
378  * @param threadId The target thread id.
379  *
380  * @return The associated activity, or OREF_NULL if the current thread
381  * is not attached.
382  */
383 Activity* InterpreterInstance::findActivity(thread_id_t threadId)
384 {
385  // this is a critical section
386  ResourceSection lock;
387  // NB: New activities are pushed on to the end, so it's prudent to search
388  // from the list end toward the front of the list. Also, this ensures we
389  // will find the toplevel activity nested on a given thread first.
390  for (size_t listIndex = allActivities->items(); listIndex > 0; listIndex--)
391  {
392  Activity *activity = (Activity *)allActivities->get(listIndex);
393  // this should never happen, but we never return suspended threads
394  if (activity->isThread(threadId) && !activity->isSuspended())
395  {
396  return activity;
397  }
398  }
399  return OREF_NULL;
400 }

415 /**
416  * Enter on the current thread context, making sure the interpreter
417  * lock is obtained.
418  *
419  * @return The activity object associated with this thread/instance
420  * combination.
421  */
422 Activity* InterpreterInstance::enterOnCurrentThread()
423 {
424  // attach this thread to the current activity
425  Activity *activity = attachThread();
426  // indicate this is a nested entry
427  activity->activate();
428  // from this point forward, we want to be the active activity, so
429  // acquire the kernel lock
430  activity->requestApiAccess();
431  // return the activity for use
432  return activity;

436 void InterpreterInstance::removeInactiveActivities()
437 {
438  size_t count = allActivities->items();
439 
440  // This is a bit complicated. Each activity will be removed from the
441  // head of the list, and any activity not ready for termination is
442  // put back on the end. Since we're using the initial count of the
443  // items for handling this, we'll look at each activity at most once.
444  // If there are any items left, those are activities we can't release yet.
445  for (size_t i = 0; i < count; i++)
446  {
447  Activity *activity = (Activity *)allActivities->pull();
448  // we never terminate the root activity or any activity current in use
449  if (activity == rootActivity || activity->isActive())
450  {
451  allActivities->append(activity);
452  }
453  else
454  {
455  // have the inactive thread wake up and terminate
456  activity->terminatePoolActivity();
457  }
458  }
459 }

462 /**
463  * Attempt to shutdown the interpreter instance. This can only be done
464  * from the root activity.
465  *
466  * @return true if shutdown has been initiated, false otherwise.
467  */
468 bool InterpreterInstance::terminate()
469 {
470  // check this is in fact a valid instance
471  if (!Interpreter::isInstanceActive(this))
472  {
473  return false;
474  }
475 
476 
477  // we can't be doing active work on the root thread
478  if (rootActivity->isActive())
479  {
480  return false;
481  }
482 
483  {
484 
485  ResourceSection lock;
486 
487  // it's possible to get a call on a second thread or even recursively, let's make sure
488  // we're not already in the process of shutting down
489 
490  if (terminating)
491  {
492  return false;
493  }
494 
495  terminated = false;
496  // turn on the global termination in process flag
497  terminating = true;
498 
499  // go remove all of the activities that are not doing work for this instance
500  removeInactiveActivities();
501  // if we just have the single root activity left, then we can shutdown
502  terminated = allActivities->items() == 1;
503  }
504 
505  // if there are active threads still running, we need to wait until
506  // they all finish
507  if (!terminated)
508  {
509  terminationSem.wait();
510  }
511 
512  Activity *current;
513 
514  try
515  {
516  // if everything has terminated, then make sure we run the uninits before shutting down.
517  // This activity is currently the current activity. We're going to run the
518  // uninits on this one, so reactivate it until we're done running. If we were not actually
519  // called on an attached thread, an attach will be performed.
520  current = enterOnCurrentThread();
521 
522  // this might be holding some local references. Make sure we clear these
523  // before running the garbage collector
524  rootActivity->clearLocalReferences();
525 
526  // before we update of the data structures, make sure we process any
527  // pending uninit activity.
528  memoryObject.collectAndUninit(Interpreter::lastInstance());
529 
530  // do system specific termination of an instance
531  sysInstance.terminate();
532 
533  // ok, deactivate this again...this will return the activity because the terminating
534  // flag is on.
535  current->exitCurrentThread();
536  }
537  // do the release in a catch block to ensure we really release this
538  catch (NativeActivation *)
539  {
540  // ok, deactivate this again...this will return the activity because the terminating
541  // flag is on.
542  current->exitCurrentThread();
543 
544  }
545 
546  terminationSem.close();
547 
548  // make sure the root activity is removed by the ActivityManager;
549  ActivityManager::returnRootActivity(rootActivity);
550 
551  // just in case there's still a reference held to this, clear out all object reference fields
552  rootActivity = OREF_NULL;
553  securityManager = OREF_NULL;
554  allActivities = OREF_NULL;
555  defaultEnvironment = OREF_NULL;
556  searchPath = OREF_NULL;
557  searchExtensions = OREF_NULL;
558  localEnvironment = OREF_NULL;
559  commandHandlers = OREF_NULL;
560  requiresFiles = OREF_NULL;
561 
562 
563  // tell the main interpreter controller we're gone.
564  Interpreter::terminateInterpreterInstance(this);
565  return true;
566 }

639 /**
640  * Process interpreter instance options.
641  *
642  * @param options The list of defined options.
643  *
644  * @return True if the options were processed correctly, false otherwise.
645  */
646 bool InterpreterInstance::processOptions(RexxOption *options)
647 {
648  // options are, well, optional...if nothing specified, we're done.
649  if (options == NULL)
650  {
651  return true;
652  }
653  // loop until we get to the last option item
654  while (options->optionName != NULL)
655  {
656  // an initial address environment
657  if (strcmp(options->optionName, INITIAL_ADDRESS_ENVIRONMENT) == 0)
658  {
659  defaultEnvironment = new_string(options->option.value.value_CSTRING);
660  }
661  // application data
662  else if (strcmp(options->optionName, APPLICATION_DATA) == 0)
663  {
664  // this is filled in to the instance context vector
665  context.instanceContext.applicationData = options->option.value.value_POINTER;
666  }
667  // an additional search path
668  else if (strcmp(options->optionName, EXTERNAL_CALL_PATH) == 0)
669  {
670  searchPath = new_string(options->option.value.value_CSTRING);
671  }
672  // additional extensions for processing
673  else if (strcmp(options->optionName, EXTERNAL_CALL_EXTENSIONS) == 0)
674  {
675  const char *extStart = options->option.value.value_CSTRING;
676  const char *extEnd = extStart + strlen(extStart);
677 
678  while (extStart < extEnd)
679  {
680  const char *delim = strchr(extStart, ',');
681  if (delim == NULL)
682  {
683  delim = extEnd;
684  }
685  // make this into a string value and append
686  RexxString *ext = new_string(extStart, delim - extStart);
687  searchExtensions->append(ext);
688 
689  // step past the delimiter and loop
690  extStart = delim + 1;
691  }
692  }
693  // old-style registered exit
694  else if (strcmp(options->optionName, REGISTERED_EXITS) == 0)
695  {
696  RXSYSEXIT *handlers = (RXSYSEXIT *)options->option.value.value_POINTER;
697  // if we have handlers, initialize the array
698  if (handlers != NULL)
699  {
700  // while not at the list ender
701  for (int i = 0; handlers[i].sysexit_code != RXENDLST; i++)
702  {
703  setExitHandler(handlers[i]);
704  }
705  }
706  }
707  // new-style context exit
708  else if (strcmp(options->optionName, DIRECT_EXITS) == 0)
709  {
710  RexxContextExit *handlers = (RexxContextExit *)options->option.value.value_POINTER;
711  // if we have handlers, initialize the array
712  if (handlers != NULL)
713  {
714  for (int i = 0; handlers[i].sysexit_code != RXENDLST; i++)
715  {
716  setExitHandler(handlers[i]);
717  }
718  }
719  }
720  // old-style registered command handler
721  else if (strcmp(options->optionName, REGISTERED_ENVIRONMENTS) == 0)
722  {
723  RexxRegisteredEnvironment *handlers = (RexxRegisteredEnvironment *)options->option.value.value_POINTER;
724  // if we have handlers, initialize the array
725  if (handlers != NULL)
726  {
727  for (int i = 0; handlers[i].registeredName != NULL; i++)
728  {
729  // add the handler to this setup
730  addCommandHandler(handlers[i].name, handlers[i].registeredName);
731  }
732  }
733  }
734  // a direct call command handler
735  else if (strcmp(options->optionName, DIRECT_ENVIRONMENTS) == 0)
736  {
737  RexxContextEnvironment *handlers = (RexxContextEnvironment *)options->option.value.value_POINTER;
738  // if we have handlers, initialize the array
739  if (handlers != NULL)
740  {
741  for (int i = 0; handlers[i].name != NULL && handlers[i].handler != NULL; i++)
742  {
743  // add the handler to this setup
744  addCommandHandler(handlers[i].name, (REXXPFN)handlers[i].handler, HandlerType::DIRECT);
745  }
746  }
747  }
748  // a redirectiong command handler
749  else if (strcmp(options->optionName, REDIRECTING_ENVIRONMENTS) == 0)
750  {
751  RexxRedirectingEnvironment *handlers = (RexxRedirectingEnvironment *)options->option.value.value_POINTER;
752  // if we have handlers, initialize the array
753  if (handlers != NULL)
754  {
755  for (int i = 0; handlers[i].name != NULL && handlers[i].handler != NULL; i++)
756  {
757  // add the handler to this setup
758  addCommandHandler(handlers[i].name, (REXXPFN)handlers[i].handler, HandlerType::REDIRECTING);
759  }
760  }
761  }
762  // a package to load at startup
763  else if (strcmp(options->optionName, LOAD_REQUIRED_LIBRARY) == 0)
764  {

1063 /**
1064  * Resolve a program for intial loading or a subroutine call.
1065  *
1066  * @param _name The target name. This can be fully qualified, or a simple name
1067  * without an extension.
1068  * @param _parentDir The directory of the file of our calling program. The first place
1069  * we'll look is in the same directory as the program.
1070  * @param _parentExtension
1071  * The extension our calling program has. If there is an extension,
1072  * we'll use that version first before trying any of the default
1073  * extensions.
1074  * @param type Either RESOLVE_REQUIRES, which requests an additional extension
1075  * to be tried before the standard search order, or RESOLVE_DEFAULT.
1076  *
1077  * @return A string version of the file name, if found. Returns OREF_NULL if
1078  * the program cannot be found.
1079  */
1080 RexxString* InterpreterInstance::resolveProgramName(RexxString *_name, RexxString *_parentDir, RexxString *_parentExtension, ResolveType type)
1081 {
1082  FileNameBuffer resolvedName;
1083 
1084  const char *name = _name->getStringData();
1085  const char *parentDir = _parentDir == OREF_NULL ? NULL : _parentDir->getStringData();
1086  const char *parentExtension = _parentExtension == OREF_NULL ? NULL : _parentExtension->getStringData();
1087  const char *pathExtension = searchPath == OREF_NULL ? NULL : searchPath->getStringData();
1088 
1089  SysSearchPath searchPath(parentDir, pathExtension);
1090 
1091  // if the file already has an extension, this dramatically reduces the number
1092  // of searches we need to make.
1093  if (SysFileSystem::hasExtension(name))
1094  {
1095  if (SysFileSystem::searchName(name, searchPath.path, NULL, resolvedName))
1096  {
1097  return new_string(resolvedName);
1098  }
1099  return OREF_NULL;
1100  }
1101 
1102  // if we are resolving a REQUIRES file, we first try extension .cls
1103  if (type == RESOLVE_REQUIRES)
1104  {
1105  if (SysFileSystem::searchName(name, searchPath.path, ".cls", resolvedName))
1106  {
1107  return new_string(resolvedName);
1108  }
1109  }
1110 
1111  // if we have a parent extension provided, use that in preference to any default searches
1112  if (parentExtension != NULL)
1113  {
1114  if (SysFileSystem::searchName(name, searchPath.path, parentExtension, resolvedName))
1115  {
1116  return new_string(resolvedName);
1117  }
1118  }
1119 
1120  // ok, now time to try each of the individual extensions along the way.
1121  for (size_t i = 1; i <= searchExtensions->items(); i++)
1122  {
1123  RexxString *ext = (RexxString *)searchExtensions->get(i);
1124 
1125  if (SysFileSystem::searchName(name, searchPath.path, ext->getStringData(), resolvedName))
1126  {
1127  return new_string(resolvedName);
1128  }
1129  }
1130 
1131  // The file may purposefully have no extension.
1132  if (SysFileSystem::searchName(name, searchPath.path, NULL, resolvedName))
1133  {
1134  return new_string(resolvedName);
1135  }
1136 
1137  return OREF_NULL;
1138 }
