141  /**
142  * Skip leading blanks in a string.
143  *
144  * @param String The target string.
145  * @param scanLength
146  * The length of the string segment.
147  */
148  void skipBlanks(const char *&string, size_t &scanLength )
149  {
150  const char *scan = string;
151  size_t length = scanLength;
152 
153  // just skip over any white space characters
154  for (;length > 0; scan++, length--)
155  {
156  if (*scan != ' ' && *scan != '\t')
157  {
158  break;
159  }
160  }
161 
162  // we've either found a non blank or we ran out of string
163  string = scan;
164  scanLength = length;
165  }
166 
167  /**
168  * Skip non-blank characters to the next whitespace char.
169  *
170  * @param String The source string.
171  * @param scanLength
172  * The string length (update on return);
173  */
174  void skipNonBlanks(const char *&string, size_t &scanLength )
175  {
176  const char *scan = string;
177  size_t length = scanLength;
178 
179  for (;length > 0; scan++, length--)
180  {
181  // stop if we find a white space character
182  if (*scan == ' ' || *scan == '\t')
183  {
184  break;
185  }
186  }
187  string = scan;
188  scanLength = length;
189  }
190 
191 
192  /**
193  * Scan forward to the next word in the string.
194  *
195  * @return true if a word was found, false if we've run out out words.
196  */
197  inline bool next()
198  {
199  // NOTE: We leave the word pointer and length untouched until we
200  // have a hit so that after an iteration failure, we can still retrieve
201  // the last word.
202 
203  // if out of string, this is a failure
204  if (scanLength == 0)
205  {
206  return false;
207  }
208 
209  // if we have string left
210  // skip over any blanks
211  skipBlanks(nextPosition, scanLength);
212  // if we ran out of string, there are no more words
213  if (scanLength == 0)
214  {
215  return false;
216  }
217 
218  // save the starting length
219  currentWordLength = scanLength;
220  // save the start of the word
221  currentWord = nextPosition;
222 
223  // skip over the non-blank charactes
224  skipNonBlanks(nextPosition, scanLength);
225  // get the length of the next word
226  currentWordLength -= scanLength;
227  // we have a word
228  return true;
229  }

232  /**
233  * Skip over a given number of words. Returns true if
234  * the count was reached before the end of the string, false
235  * if the end is reached first.
236  *
237  * @param count The count to skip.
238  *
239  * @return true if count words were found.
240  */
241  inline bool skipWords(size_t count)
242  {
243  while (count--)
244  {
245  if (!next())
246  {
247  return false;
248  }
249  }
250  return true;
251  }

328  inline HashCode getStringHash()
329  {
330  if (hashValue == 0) // if we've never generated this, the code is zero
331  {
332  size_t len = getLength();
333 
334  HashCode h = 0;
335  // the hash code is generated from all of the string characters.
336  // we do this in a lazy fashion, since most string objects never need to
337  // calculate a hash code, particularly the long ones.
338  // This hashing algorithm is very similar to that used for Java strings.
339  for (size_t i = 0; i < len; i++)
340  {
341  h = 31 * h + stringData[i];
342  }
343  hashValue = h;
344  }
345  return hashValue;
346  }
