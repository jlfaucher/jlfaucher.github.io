282  * Translate the translated program into a buffer.
283  *
284  * @return The flattened object.
285  */
286 BufferClass *RoutineClass::save()
287 {
288  // detach the source from this routine object before saving
289  Protected<ProgramSource> source = detachSource();
290 
291  Protected<Envelope> envelope = new Envelope;
292  // now pack into an envelope;
293  Protected<BufferClass> result = envelope->pack(this);
294  // reattach the source because we might still execute this
295  attachSource(source);
296  return result;

301  * Save a routine into an externalized buffer form in an RXSTRING.
302  *
303  * @param outBuffer The target output RXSTRING.
304  */
305 void RoutineClass::save(PRXSTRING outBuffer)
306 {
307  ProtectedObject p(this);
308  // flatten the routine
309  Protected<BufferClass> methodBuffer = save();
310  // create a full buffer of the data, plus the information header.
311  ProgramMetaData *data = new (methodBuffer) ProgramMetaData(getLanguageLevel(), methodBuffer);

313  outBuffer->strptr = (char *)data;
314  outBuffer->strlength = data->getDataSize();

328 /**
329  * Save a routine to a target file.
330  *
331  * @param fileName
332  * @param encode Indicates if the file should be saved in binary form or base64 encoded string form.
333  */
334 void RoutineClass::save(const char *fileName, bool encode)
335 {
336  // make sure this is protected from GC during all of this processing
337  ProtectedObject p(this);
338 
339  SysFile target;
340 
341  if (!target.open(fileName, RX_O_CREAT | RX_O_TRUNC | RX_O_WRONLY, RX_S_IREAD | RX_S_IWRITE, RX_SH_DENYRW))
342  {
343  // got an error here
344  reportException(Error_Program_unreadable_output_error, fileName);
345  }
346 
347  // save to a flattened buffer
348  Protected<BufferClass> buffer = save();
349 
350  // create an image header
351  ProgramMetaData metaData(getLanguageLevel(), buffer->getDataLength());
352 
353  // write out the header information
354  metaData.write(target, buffer, encode);
355  target.close();
356 }

430 /**
431  * Create a new routine from REXX code contained in a string or an
432  * array
433  *
434  * @param init_args The array of arguments to the new routine.
435  * @param argCount The argument count.
436  *
437  * @return A new routine object.
438  */
439 RoutineClass *RoutineClass::newRexx(RexxObject **init_args, size_t argCount)
440 {
441  // this class is defined on the object class, but this is actually attached
442  // to a class object instance. Therefore, any use of the this pointer
443  // will be touching the wrong data. Use the classThis pointer for calling
444  // any methods on this object from this method.
445  RexxClass *classThis = (RexxClass *)this;
446 
447  RexxString *programName;
448  Protected<ArrayClass> sourceArray;
449  PackageClass *sourceContext;
450 
451  // parse all of the options
452  processNewExecutableArgs(init_args, argCount, programName, sourceArray, sourceContext);
453 
454  // go create a method from whatever we were given for source.
455  Protected<RoutineClass> newRoutine = LanguageParser::createRoutine(programName, sourceArray, sourceContext);
456 
457  // finish up the object creation. Set the correct instance behavior (this could
458  // be a subclass), check for uninit methods, and finally, send an init message using any
459  // left over arguments.
460  classThis->completeNewObject(newRoutine, init_args, argCount);
461  return newRoutine;

493 /**
494  * Create a routine from an external library source.
495  *
496  * @param name The routine name.
497  *
498  * @return The resolved routine object, or OREF_NULL if unable to load
499  * the routine.
500  */
501 RoutineClass *RoutineClass::loadExternalRoutine(RexxString *name, RexxString *descriptor)
502 {
503  Protected<RexxString> routineName = stringArgument(name, "name");
504  Protected<RexxString> libraryDescriptor = stringArgument(descriptor, "descriptor");
505 
506  // convert external into words
507  Protected<ArrayClass> words = StringUtil::words(libraryDescriptor->getStringData(), libraryDescriptor->getLength());
508  // "LIBRARY libbar [foo]"
509  if (words->size() > 0 && ((RexxString *)(words->get(1)))->strCaselessCompare("LIBRARY"))
510  {
511  RexxString *library = OREF_NULL;
512  // the default entry point name is the internal name
513  RexxString *entry = routineName;
514 
515  // full library with entry name version?
516  if (words->size() == 3)
517  {
518  library = (RexxString *)words->get(2);
519  entry = (RexxString *)words->get(3);
520  }
521  else if (words->size() == 2)
522  {
523  library = (RexxString *)words->get(2);
524  }
525  else // wrong number of tokens
526  {
527  reportException(Error_Translation_bad_external, descriptor);
528  }
529 
530  // create a new native method
531  RoutineClass *routine = PackageManager::loadRoutine(library, entry);
532  return (RoutineClass *)resultOrNil(routine);
533  }
534  else
535  {
536  // unknown external type
537  reportException(Error_Translation_bad_external, descriptor);
538  }
539  return OREF_NULL;
540 }
