100 /**
101  * Expand the contents of a collection to a given bucket
102  * capacity.
103  */
104 void HashCollection::expandContents(size_t capacity )
105 {
106  size_t bucketSize = calculateBucketSize(capacity);
107  Protected<HashContents> newContents = allocateContents(bucketSize, bucketSize * 2);
108  // copy all of the items into the new table
109  contents->reMerge(newContents);
110  // if this is a contents item in the old space, we need to
111  // empty this so any old-to-new table entries in the contents can get updated.
112  if (contents->isOldSpace())
113  {
114  contents->empty();
115  }
116  // replace the contents
117 
118  setField(contents, (HashContents *)newContents);
119 }

122 /**
123  * Ensure that the collection has sufficient space for a
124  * mass incoming addition.
125  *
126  * @param delta The number of entries we wish to add.
127  */
128 void HashCollection::ensureCapacity(size_t delta)
129 {
130  // not enough space? time to expand. We'll go to
131  // the current total capacity plus the delta...or the standard
132  // doubling if the delta is a small value.
133  if (!contents->hasCapacity(delta))
134  {
135  expandContents(contents->capacity() + std::max(delta, contents->capacity()));
136  }
137 }

140 /**
141  * Calculate an optimal bucket size for a given capacity. This
142  * is derived from how Java calculates this. This will generally
143  * allocate more entries than asked for, but will generally
144  * have a lower collision rate than using it directly.
145  *
146  * @param capacity The desired capacity.
147  *
148  * @return The calculated capacity.
149  */
150 size_t HashCollection::calculateBucketSize(size_t capacity)
151 {
152  // if this is a request for a very large table, cap the size.
153  // it is likely we'll never be able to even allocate a bucket that large!
154  if (capacity >= 1 << 30)
155  {
156  return 1 << 30;
157  }
158 
159  if (capacity < MinimumBucketSize)
160  {
161  return MinimumBucketSize;
162  }
163 
164  // we prefer to have a odd number for the bucket size, so add
165  // 1 if we end up with an even number.
166  if ((capacity & 1) == 0)
167  {
168  capacity += 1;
169  }
170 
171  return capacity;
172 }

767 /**
768  * Rehash a collection after a restore.
769  */
770 void HashCollection::reHash()
771 {
772  // make a copy of our contents to get one of the same type and size.
773  Protected<HashContents> newContents = (HashContents *)contents->copy();
774  // empty the new one out
775  newContents->empty();
776  // copy all of the items into the new table
777  contents->reHash(newContents);
778  // make the new contents the active one.
779  setField(contents, newContents);
780 }
