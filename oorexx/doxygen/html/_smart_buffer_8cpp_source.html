110 /**
111  * Copy data into the buffer at the very end.
112  *
113  * @param start The data to be copied.
114  * @param length The length of data to copy.
115  *
116  * @return The offset of the copied data.
117  */
118 size_t SmartBuffer::copyData(void *start, size_t length)
119 {
120  // not enough room to copy this? Get a bigger backing buffer
121  if (space() < length)
122  {
123  setField(buffer, buffer->expand(length));
124  }
125 
126  // copy the data into the buffer
127  buffer->copyData(current, (char *)start, length);
128 
129  // bump the copy location, but return the offset of of where
130  // this data was copied.
131  size_t dataLoc = current;
132  current = current + length;
133  return dataLoc;
134 }
