69 int invokeExitFunction(RexxExitContext *context, const char *name, PEXIT exitInfo)
70 {
71  RXOFNCCAL_PARM *parms = (RXOFNCCAL_PARM *)exitInfo;
72  const char *functionName = parms->rxfnc_name.strptr;
73 
74  if (strcmp(functionName, "TESTGETCONTEXTVARIABLE") == 0)
75  {
76  parms->rxfnc_retc = TestGetContextVariable(context, parms->rxfnc_argc, parms->rxfnc_argv);
77  return RXEXIT_HANDLED;
78  }
79  else if (strcmp(functionName, "TESTSETCONTEXTVARIABLE") == 0)
80  {
81  parms->rxfnc_retc = TestSetContextVariable(context, parms->rxfnc_argc, parms->rxfnc_argv);
82  return RXEXIT_HANDLED;
83  }
84  else if (strcmp(functionName, "TESTDROPCONTEXTVARIABLE") == 0)
85  {
86  parms->rxfnc_retc = TestDropContextVariable(context, parms->rxfnc_argc, parms->rxfnc_argv);
87  return RXEXIT_HANDLED;
88  }
89  else if (strcmp(functionName, "TESTGETALLCONTEXTVARIABLES") == 0)
90  {
91  parms->rxfnc_retc = TestGetAllContextVariables(context, parms->rxfnc_argc, parms->rxfnc_argv);
92  return RXEXIT_HANDLED;
93  }
94  else if (strcmp(functionName, "TESTGETCALLERCONTEXT") == 0)
95  {
96  parms->rxfnc_retc = TestGetCallerContext(context, parms->rxfnc_argc, parms->rxfnc_argv);
97  return RXEXIT_HANDLED;
98  }
99  else if (strcmp(functionName, "TESTSUBCALL") == 0)
100  {
101  // return the status of the flag
102  parms->rxfnc_retc = parms->rxfnc_flags.rxffsub ? context->NewStringFromAsciiz("SUBROUTINE") : context->NewStringFromAsciiz("FUNCTION");
103  return RXEXIT_HANDLED;
104  }
105  else if (strcmp(functionName, "TESTERROR") == 0)
106  {
107  // this should raise an error
108  parms->rxfnc_flags.rxfferr = 1;
109  return RXEXIT_HANDLED;
110  }
111  else if (strcmp(functionName, "TESTNOTOUND") == 0)
112  {
113  // this should raise an error
114  parms->rxfnc_flags.rxffnfnd = 1;
115  return RXEXIT_HANDLED;
116  }
117  else if (strcmp(functionName, "TESTARGUMENTS") == 0 || strcmp(functionName, "TESTARGUMENTS2") == 0)
118  {
119  RexxArrayObject args = context->NewArray(parms->rxfnc_argc);
120  for (size_t i = 0; i < parms->rxfnc_argc; i++)
121  {
122  if (parms->rxfnc_argv[i] != NULLOBJECT)
123  {
124  context->ArrayPut(args, parms->rxfnc_argv[i], i + 1);
125  }
126  }
127  parms->rxfnc_retc = (RexxObjectPtr)context->SendMessage2(args, "TOSTRING", context->String("L"), context->String(","));
128  return RXEXIT_HANDLED;
129  }
130  else
131  {
132  // pass on this
133  return RXEXIT_NOT_HANDLED;
134  }
135 }

170  {
171  // this should raise an error

173  return RXEXIT_HANDLED;

180  }

191 
192  switch (instanceInfo->cmd.action)

200  return RXEXIT_NOT_HANDLED;
201  default: break; // avoid warning: enumeration values not handled in switch
202  }
203 
204  RXCMDHST_PARM *parms = (RXCMDHST_PARM *)exitInfo;
205  // handle commands here...we need to process both the address environment and the commands
206  if (strcmp(parms->rxcmd_address, "FOOBAR") == 0)
207  {
208  strcpy(parms->rxcmd_retc.strptr, "-1");
209  parms->rxcmd_retc.strlength = 2;
210  return RXEXIT_HANDLED;

214  {
215  strcpy(parms->rxcmd_retc.strptr, "0");
216  parms->rxcmd_retc.strlength = 1;
217  return RXEXIT_HANDLED;
218  }
219  else if (strcmp(parms->rxcmd_command.strptr, "ERROR") == 0)
220  {
221  strcpy(parms->rxcmd_retc.strptr, "1");
222  parms->rxcmd_retc.strlength = 1;
223  parms->rxcmd_flags.rxfcerr = 1;
224  return RXEXIT_HANDLED;
225  }
226  // ok, a good address...now do the different commands
227  else if (strcmp(parms->rxcmd_command.strptr, "TRACEON") == 0)
228  {
229  instanceInfo->trc = InstanceInfo::TRACEON;
230  strcpy(parms->rxcmd_retc.strptr, "0");
231  parms->rxcmd_retc.strlength = 1;
232  return RXEXIT_HANDLED;
233  }
234  // ok, a good address...now do the different commands
235  else if (strcmp(parms->rxcmd_command.strptr, "TRACEOFF") == 0)
236  {
237  instanceInfo->trc = InstanceInfo::TRACEOFF;
238  strcpy(parms->rxcmd_retc.strptr, "0");
239  parms->rxcmd_retc.strlength = 1;
240  return RXEXIT_HANDLED;
241  }
242  // ok, a good address...now do the different commands
243  else if (strcmp(parms->rxcmd_command.strptr, "HALT") == 0)
244  {
245  instanceInfo->hlt = InstanceInfo::HALT;
246  strcpy(parms->rxcmd_retc.strptr, "0");
247  parms->rxcmd_retc.strlength = 1;
248  return RXEXIT_HANDLED;
249  }
250  else
251  {
252  // unknown command
253  strcpy(parms->rxcmd_retc.strptr, "-2");
254  parms->rxcmd_retc.strlength = 2;
255  parms->rxcmd_flags.rxfcfail = 1;
256  return RXEXIT_HANDLED;
