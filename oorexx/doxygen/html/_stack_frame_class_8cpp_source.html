81 /**
82  * Constructor for a RexxContext object.
83  *
84  * @param ty The type of frame.
85  * @param n The name of the item at that stack frame instance.
86  * @param e The executable associated with this frame instance.
87  * @param tg The target object, if a message send.
88  * @param a Arguments to the method/routine.
89  * @param t A tracing line.
90  * @param l The frame line position (MAX_SIZE indicates no line available).
91  * @param i The invocation id.
92  * @param c The context object.
93  */
94 StackFrameClass::StackFrameClass(const char *ty, RexxString *n, BaseExecutable *e, RexxObject *tg, ArrayClass *a, RexxString *t, size_t l, uint32_t i, RexxObject *c)
95 {
96  type = ty;
97  name = n;
98  // interpret stack frames don't have a name, but we need to have a valid
99  // return value if requested, so make this a nullstring.
100  if (name == OREF_NULL)
101  {
102  name = GlobalNames::NULLSTRING;
103  }
104  executable = e;
105  target = tg;
106  arguments = a;
107  traceLine = t;
108  line = l;
109  invocation = i; // supplied by RexxActivation
110  context = c; // supplied by RexxActivation
111 }
