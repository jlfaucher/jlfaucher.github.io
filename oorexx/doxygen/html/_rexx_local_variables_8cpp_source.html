55 void RexxLocalVariables::live(size_t liveMark)
56 {
57  // We can only mark if full initialized
58  if (locals != NULL)
59  {
60  RexxVariable **entry;
61  RexxVariable **top;
62 
63  // mark all of the stack entries;
64  for (entry = locals, top = entry + size; entry < top; entry++)
65  {
66  memory_mark(*entry);
67  }
68  }
69  // also mark any created variable dictionary
70  memory_mark(dictionary);
71  memory_mark(owner);
72  memory_mark(objectVariables);
73 }

116 /**
117  * Do a more extensive search for a variable, without creating
118  * one if it doesn't exist. This is only called after we've had
119  * a miss on the variable table.
120  *
121  * @param name The target name.
122  * @param index The target index (can be 0, which requires a search).
123  *
124  * @return The variable object corresponding to the name, or NULL if
125  * not found.
126  */
127 RexxVariable *RexxLocalVariables::findVariable(RexxString *name, size_t index)
128 {
129  // we need to search for this by name, so if we have a variable
130  // dictionary already created, this is a quick search.
131  if (dictionary != OREF_NULL)
132  {
133  // the lookup might be using a non-zero index and failing
134  // because the variable might have been created dynamically (using
135  // VALUE() or interpret, for example). If we have a non-zero
136  // index, store the value from the dictionary into the slot.
137  RexxVariable *variable = dictionary->resolveVariable(name);
138 
139  // if we've had a miss on the local directory but we're auto exposing,
140  // we need to check the object dictionary
141  if (variable == OREF_NULL && autoExpose())
142  {
143  // Important note: If we are auto exposing, then all additional references
144  // need to come from the object variables, creating one if needed, so
145  // we are using getVariable() rather than resolveVariable() here.
146  variable = objectVariables->getVariable(name);
147  // if we found this in the object variables, add to the
148  // local dictionary too
149  if (variable != OREF_NULL)
150  {
151  dictionary->addVariable(name, variable);
152  }
153  }
154 
155  // if we have an index, fill in the cache entry.
156  if (index != 0)
157  {
158  locals[index] = variable;
159  }
160  return variable;
161  }
162  // first dynamic lookup...we might need to create the variable
163  // dictionary.
164  else
165  {
166  // if this is a non-targetted lookup, we don't know the slot
167  // this variable needs to be stored in, or this is a
168  // dynamically accessed variable that may not have a slot. We
169  // might need to create a variable dictionary for this.
170  if (index == 0)
171  {
172  // if we haven't created a variable dictionary yet, scan
173  // the set of variables looking for one we may have created
174  // earlier.
175  for (size_t i = 0; i < size; i++)
176  {
177  RexxVariable *variable = locals[i];
178  // if the slot exists at this position, check the name
179  if (variable != OREF_NULL)
180  {
181  if (name->strCompare(variable->getName()))
182  {
183  return variable;
184  }
185  }
186  }
187  }
188  // we have an index, check the slot directly
189  else
190  {
191  // check the slot...it might be there
192  if (locals[index] != OREF_NULL)
193  {
194  return locals[index];
195  }
196  }
197 
198  // not found in the local table, but this could be an autoexpose situation
199  // we need to check the object dictionary
200  if (autoExpose())
201  {
202  // Important note: If we are auto exposing, then all additional references
203  // need to come from the object variables, creating one if needed, so
204  // we are using getVariable() rather than resolveVariable() here.
205  RexxVariable *variable = objectVariables->getVariable(name);
206  // we did not have a dictionary up to this point, so create it now and
207  // add it.
208  createDictionary();
209  dictionary->addVariable(name, variable);
210  // add this to the slot now
211  if (index != 0)
212  {
213  locals[index] = variable;
214  }
215  return variable;
216  }
217  // a non-zero index with no created variable dictionary means this
218  // variable cannot exist. Just return NULL.
219  }
220  // not found
221  return OREF_NULL;
222 }

240 /**
241  * Create a local variable object of the given name and store
242  * it at the given location.
243  *
244  * @param name The variable name.
245  * @param index The target index position.
246  *
247  * @return The created or resolved variable object.
248  */
249 RexxVariable *RexxLocalVariables::lookupVariable(RexxString *name, size_t index)
250 {
251  // if this is a non-targetted lookup, we don't know the slot
252  // this variable needs to be stored in, or this is a
253  // dynamically accessed variable that may not have a slot. We
254  // might need to create a variable dictionary for this.
255  if (index == 0)
256  {
257  // if we haven't created a variable dictionary yet, scan
258  // the set of variables looking for one we may have created
259  // earlier.
260  if (dictionary == OREF_NULL)
261  {
262  for (size_t i = 0; i < size; i++)
263  {
264  // we're looking for a variable where the names match.
265  RexxVariable *variable = locals[i];
266  if (variable != OREF_NULL)
267  {
268  if (name->strCompare(variable->getName()))
269  {
270  return variable;
271  }
272  }
273  }
274  // go create the dictionary and populate it with our variable set.
275  createDictionary();
276  }
277 
278  // if we're not auto exposing, just lookup the variable which will also
279  // create it.
280  if (!autoExpose())
281  {
282  // retrieve this from the variable dictionary now, which will create
283  // the item we need.
284  return dictionary->getVariable(name);
285  }
286  else
287  {
288  // this might have been previously created. check the dictionary
289  // first, then create if necessary
290  RexxVariable *variable = dictionary->resolveVariable(name);
291  if (variable != OREF_NULL)
292  {
293  return variable;
294  }
295 
296  // get the variable from the object variable dictionary
297  // and add it to the local dictionary.
298  variable = objectVariables->getVariable(name);
299  dictionary->addVariable(name, variable);
300  return variable;
301  }
302  }
303  else
304  {
305  // normal local variable lookup, just create the local variable
306  // and return it.
307  if (!autoExpose())
308  {
309  RexxVariable *variable;
310  // if we've had to create a dictionary for this because of
311  // prior dynamic access, then we need to retrieve the
312  // variable from the dictionary.
313  if (dictionary != OREF_NULL)
314  {
315  variable = dictionary->getVariable(name);
316  }
317  // we've already had a cache miss, so we're creating a variable.
318  else
319  {
320  variable = owner->newLocalVariable(name);
321  }
322  // fill in the cache slot for the next lookup and return
323  // the new variable.
324  locals[index] = variable;
325  return variable;
326  }
327  else
328  {
329  // if we've had to create a dictionary for this because of
330  // prior dynamic access, then we need to retrieve the
331  // variable from the dictionary.
332  if (dictionary != OREF_NULL)
333  {
334  // this might have been previously created. check the dictionary
335  // first, then create if necessary
336  RexxVariable *variable = dictionary->resolveVariable(name);
337  if (variable != OREF_NULL)
338  {
339  return variable;
340  }
341  }
342 
343  // get the variable from the object variable dictionary
344  RexxVariable *variable = objectVariables->getVariable(name);
345  // if we've had to create a dictionary for this because of
346  // prior dynamic access, then we need to retrieve the
347  // variable from the dictionary.
348  if (dictionary != OREF_NULL)
349  {
350  dictionary->addVariable(name, variable);
351  }
352  // fill in the cache slot for the next lookup and return
353  // the new variable.
354  locals[index] = variable;
355  return variable;
356  }
357  }
358 }

374  // dynamically accessed variable that may not have a slot. We
375  // might need to create a variable dictionary for this.
376  if (index == 0)
377  {
378  // if we haven't created a variable dictionary yet, scan
379  // the set of variables looking for one we may have created
380  // earlier.

384  {
385  // we if find a matching variable, return it immediately
386  RexxVariable *variable = locals[i];
387  if (variable != OREF_NULL)
388  {
389  if (name->strCompare(variable->getName()))
390  {
391  return variable;
392  }
393  }
394  }
395 
396  // go create the dictionary and populate it with our variable set.
397  createDictionary();
398  }
399 
400  // if we're not auto exposing, just lookup the variable which will also

403  {
404  // retrieve this from the variable dictionary now, which will create
405  // the item we need.
406 
407  // have the dictionary create this for us
408  return dictionary->getStemVariable(name);
409  }
410  else
411  {
412  // this might have been previously created. check the dictionary
413  // first, then create if necessary
414  RexxVariable *variable = dictionary->resolveVariable(name);
415  if (variable != OREF_NULL)

434

456  return variable;

463  if (dictionary != OREF_NULL)
464  {
465  // this might have been previously created. check the dictionary

488 /**
489  * Update a local variable context with a provided variable object.
490  *
491  * @param variable The variable used for the replacement.
492  */
493 void RexxLocalVariables::updateVariable(RexxVariable *variable)
494 {
495  RexxVariable *oldVariable = OREF_NULL;
496  RexxString *name = variable->getName();
497 
498  // scan the set of variables looking for one we may have created
499  // earlier.
500  for (size_t i = 0; i < size; i++)
501  {
502  oldVariable = locals[i];
503  if (oldVariable != OREF_NULL)
504  {
505  // if we find a match, replace the slot variable at this
506  // location with the new variable object.
507  if (name->strCompare(oldVariable->getName()))
508  {
509  locals[i] = variable;
510 
511  // if we have a dictionary created, replace the entry in that as well.
512  if (dictionary != OREF_NULL)
513  {
514  dictionary->addVariable(name, variable);
515  return;
516  }
517  break;
518  }
519  }
520  }
521 
522 
523  // we didn't find a static one, so this is a completely dynamic
524  // update. Make sure we have a dynamic dictionary and insert
525  // this entry.
526  if (dictionary == OREF_NULL)
527  {
528  createDictionary();
529  }
530  // add the variable to the dictionary
531  dictionary->addVariable(name, variable);
532 }

535 /**
536  * Create a local variable object of the given name and store
537  * it at the given location.
538  *
539  * @param name The variable name.
540  * @param index The target index position.
541  *
542  * @return The created or resolved variable object.
543  */
544 void RexxLocalVariables::aliasVariable(RexxString *name, size_t index, RexxVariable *variable)
545 {
546  // we're going to be putting a variable into the local table using a name
547  // other than the actual variable name. This means we need to get the dictionary
548  // created so we can get this added into the dictionary under the aliased name.
549  // if we don't then the dictionary will get created using the real variable name.
550  // Just requesting the dictionary is sufficient to do this.
551  getDictionary();
552 
553  // see if there is already a variable with this name in the context (which might be
554  // an autoExposed object variable
555  RexxVariable *oldVar = findVariable(name, index);
556 
557  // we can only perform the aliasing if this an unassigned local variable
558  if (oldVar != OREF_NULL && !oldVar->isAliasable())
559  {
560  reportException(Error_Execution_reference_variable_in_use, name);
561  }
562 
563  // if we have a local variable with no value, this would have been created
564  // with USE LOCAL, so just overwrite the existing variable with the aliased one
565 
566 
567  // Put uses the name from the variable, so we repeat that here.
568  // We will most likly have an index since this is only used on the USE ARG
569  // instruction, which tends not to be interpreted.
570  if (index != 0)
571  {
572  // plug the referenced variable into the target slot.
573  locals[index] = variable;
574  }
575 
576  // we already created the dictionary, so we can unconditionally add
577  // this value. Note that we are using the alias name, not the name
578  // from the referenced variable. This will also replace any variable created via
579  // use arg.
580  dictionary->addVariable(name, variable);
581 }
