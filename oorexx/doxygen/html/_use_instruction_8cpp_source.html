54 /**
55  * Handle the assignment for an individual USE arg variable.
56  *
57  * @param arg The argument value passed to this activation.
58  * @param isStrict
59  */
60 void UseVariable::handleArgument(RexxActivation *context, ExpressionStack *stack, RexxObject *argument, size_t argumentPos, bool isStrict)
61 {
62 
63  // get our current variable. We're allowed to skip over variables, so
64  // there might not be anything here.
65  RexxVariableBase *retriever = (RexxVariableBase *)variable;
66  if (variable != OREF_NULL)
67  {
68  // if this is a reference argument, this gets some special handling.
69  // they are also strict by definition
70  if (isOfClass(UseArgVariableRef, variable))
71  {
72  handleReferenceArgument(context, stack, argument, argumentPos);
73  return;
74  }
75  if (argument != OREF_NULL)
76  {
77  context->traceResult(argument); // trace if necessary
78  // assign the value
79  retriever->assign(context, argument);
80  }
81  else
82  {
83  // an omitted argument is only valid if we've marked it as optional
84  // by giving it a default value
85  if (defaultValue != OREF_NULL)
86  {
87  // evaluate the default value now
88  RexxObject *value = defaultValue->evaluate(context, stack);
89  context->traceResult(value); // trace if necessary
90  // assign the value
91  variable->assign(context, value);
92  stack->pop(); // remove the value from the stack
93  }
94  else
95  {
96  // not doing strict checks, revert to old rules and drop the variable.
97  if (!isStrict)
98  {
99  variable->drop(context);
100  }
101  else
102  {
103  if (context->inMethod())
104  {
105  reportException(Error_Incorrect_method_noarg, argumentPos);
106  }
107  else
108  {
109  reportException(Error_Incorrect_call_noarg, context->getCallname(), argumentPos);
110  }
111  }
112  }
113  }
114  }
115 }

118 /**
119  * Handle the hook up between a veriable reference passed
120  * as an argument and a local variable.
121  *
122  * @param context The execution context.
123  * @param stack the current execution stack.
124  * @param argument The argument to process (must be a VariableReference)
125  *
126  * @param argumentPos
127  * The argument position
128  */
129 void UseVariable::handleReferenceArgument(RexxActivation *context, ExpressionStack *stack, RexxObject *argument, size_t argumentPos)
130 {
131  // if we're expecteting a reference, then the argument is required
132  if (argument == OREF_NULL)
133  {
134  reportException(Error_Invalid_argument_no_reference, argumentPos);
135  }
136 
137  // to start, the received object must be a variable reference.
138  if (!isVariableReference(argument))
139  {
140  reportException(Error_Invalid_argument_variable_reference, argumentPos, argument);
141  }
142 
143  UseArgVariableRef *useRef = (UseArgVariableRef *)variable;
144 
145  VariableReference *reference = (VariableReference *)argument;
146 
147  // we must have a match between variable types
148  if (useRef->isStem())
149  {
150  if (!reference->isStem())
151  {
152  reportException(Error_Invalid_argument_variable_reference_stem, argumentPos, reference->getName());
153  }
154  }
155  // this is a simple variable, this cannot be a stem
156  else
157  {
158  if (reference->isStem())
159  {
160  reportException(Error_Invalid_argument_variable_reference_simple, argumentPos, reference->getName());
161  }
162  }
163 
164  // do the aliasing.
165  useRef->aliasVariable(context, reference->getVariable());
166  // and trace the aliasing having taken place
167  context->traceVariableAlias(reference->getName(), useRef->getName());
168 }

172 /**
173  * Initialize a USE instruction instance.
174  *
175  * @param count The count of arguments to process.
176  * @param strict Indicates whether strict argument rules are to be applied.
177  * @param extraAllowed
178  * Indicates extra variables are allowed because ... was
179  * used on the end of the argument list.
180  * @param variable_list
181  * The list of variable objects.
182  * @param defaults The list of defaults to apply to arguments.
183  */
184 RexxInstructionUse::RexxInstructionUse(size_t count, bool strict, bool extraAllowed, QueueClass *variable_list, QueueClass *defaults)
185 {
186  // set the variable count and the option flag
187  variableCount = count;
188  variableSize = extraAllowed; // we might allow an unchecked number of additional arguments
189  minimumRequired = 0; // don't necessarily require any of these.
190  strictChecking = strict; // record if this is the strict form
191 
192  // items are added to the queues in reverse order, so we pop them off and add
193  // them to the end of the list as we go.
194  while (count > 0) // loop through our variable set, adding everything in.
195  {
196  // decrement first, so we store at the correct offset.
197  count--;
198  variables[count].variable = (RexxVariableBase *)variable_list->pop();
199  variables[count].defaultValue = (RexxObject *)defaults->pop();
200 
201  // see if this is the last of the required ones.
202  if (minimumRequired == 0)
203  {
204  // no default value means this is a required argument, this is the min we'll accept.
205  if (variables[count].defaultValue == OREF_NULL)
206  {
207  minimumRequired = count + 1;
208  }
209  }
210  }
211 }

270 /**
271  * Execute a USE instruction
272  *
273  * @param context The current execution context.
274  * @param stack The current evaluation stack.
275  */
276 void RexxInstructionUse::execute(RexxActivation *context, ExpressionStack *stack)
277 {
278  // trace if necessary
279  context->traceInstruction(this);
280 
281  // get the argument information from the context
282  RexxObject **arglist = context->getMethodArgumentList();
283  size_t argcount = context->getMethodArgumentCount();
284 
285  // strict checking means we need to enforce min/max limits
286  if (strictChecking)
287  {
288  // not enough of the required arguments? That's an error
289  if (argcount < minimumRequired)
290  {
291  // this is a pain, but there are different errors for method errors vs. call errors.
292  if (context->inMethod())
293  {
294  reportException(Error_Incorrect_method_minarg, minimumRequired);
295  }
296  else
297  {
298  reportException(Error_Incorrect_call_minarg, context->getCallname(), minimumRequired);
299  }
300  }
301 
302  // potentially too many?
303  if (!variableSize && argcount > variableCount)
304  {
305  if (context->inMethod())
306  {
307  reportException(Error_Incorrect_method_maxarg, variableCount);
308  }
309  else
310  {
311  reportException(Error_Incorrect_call_maxarg, context->getCallname(), variableCount);
312  }
313  }
314  }
315 
316  // now we process each of the variable definitions left-to-right
317  for (size_t i = 0; i < variableCount; i++)
318  {
319  variables[i].handleArgument(context, stack, getArgument(arglist, argcount, i), i+1, strictChecking);
320  }
321  context->pauseInstruction(); // do debug pause if necessary
322 }
