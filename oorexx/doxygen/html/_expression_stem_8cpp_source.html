120 /**
121  * Evaluate a Rexx Stem variable.
122  *
123  * @param context The current execution context
124  * @param stack The current evaluation stack.
125  *
126  * @return The variable value (which will be a Stem object instance.)
127  */
128 RexxObject *RexxStemVariable::evaluate(RexxActivation *context, ExpressionStack *stack )
129 {
130  // look up the name
131  RexxObject *value = context->getLocalStem(stemName, stemIndex);
132  // NOTE: stem accesses do NOT report NOVALUE so that DO OVER,
133  // call-by-reference with a stem and return with a stem does not
134  // trigger a novalue trap unexpectedly
135 
136  // evaluate always pushes on the stack.
137  stack->push(value);
138  context->traceVariable(stemName, value);
139  return value;
140 }

283 /**
284  * Handle procedure expose for a stem variable.
285  *
286  * @param context The current execution context.
287  * @param parent The parent execution context (source of the variable.)
288  */
289 void RexxStemVariable::procedureExpose(RexxActivation *context, RexxActivation *parent)
290 {
291  // get the old variable entry
292  RexxVariable *old_variable = parent->getLocalStemVariable(stemName, stemIndex);
293 
294  // if we have an index (and generally we should because procedure cannot
295  // be interpreted), we just put this variable into the local context.
296  // otherwise, we need to do a dynamic search to update this.
297  if (stemIndex == 0)
298  {
299  context->updateLocalVariable(old_variable);
300  }
301  else
302  {
303  context->putLocalVariable(old_variable, stemIndex);
304  }
305 }
