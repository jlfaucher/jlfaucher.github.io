135 /**
136  * Remove a notification watch from a variable.
137  *
138  * @param informee The activity requesting removal.
139  */
140 void RexxVariable::uninform(Activity *informee)
141 {
142  // remove the entry
143  dependents->remove(informee);
144  // It's probably a coin flip on whether this should
145  // be removed when this becomes empty. This happens
146  // because a method has used GUARD WHEN to wait on a variable.
147  // On the assumption that this object was written with this
148  // sort of multitasking in mind, it is likely that this
149  // variable instance might be waited on again. Therefore,
150  // it is probably prudent to keep the table around to
151  // prevent thrashing. For example, something like a work
152  // queue where one thread is waiting for items to be added
153  // would be an example where you would be constantly adding and
154  // deleting this table.
155 }

172 /**
173  * notify all waiting activities that a variable has been updated.
174  */
175 void RexxVariable::notify()
176 {
177  // if we have a dependents table, iterate through the table tapping
178  // the waiting activities
179  if (dependents != OREF_NULL)
180  {
181  // use an iterator to traverse the table
182  HashContents::TableIterator iterator = dependents->iterator();
183 
184  for (; iterator.isAvailable(); iterator.next())
185  {
186  // copy these methods over any of our own.
187  Activity *activity = (Activity *)iterator.index();
188  activity->guardPost();
189  }
190 
191  // yield control and allow the waiting guard(s) to run too
192  Activity *activity = ActivityManager::currentActivity;
193  activity->yieldControl();
194  }
195 }

204 void RexxVariable::setValue(RexxObject *value)

220 /**
221  * Set a variable to a stem value. This handles all of the
222  * details of stem-to-stem assignment and stem variable
223  * re-initialization.
224  *
225  * @param value The value to set.
226  */
227 void RexxVariable::setStem(RexxObject *value)
228 {
229  // if this is a stem-to-stem assignment, we replace the current variable's
230  // stem object.
231  if (::isStem(value))
232  {
233  set(value);
234  }
235  // assigning a new value to the stem. This creates a new stem object.
236  else
237  {
238  // create a new stem object as value
239  StemClass *stem_table = new StemClass(variableName);
240  set(stem_table); // overlay the reference stem object
241  stem_table->setValue(value); // set the default value
242  }
243 }

257 bool RexxVariable::isAliasable()
258 {
259  // this has to be a local variable
260  if (!isLocal())
261  {
262  return false;
263  }
264 
265  // stems are a bit more complicated. We consider them uninitialized
266  // if they are empty and the default value is also the variable name
267  if (isStem())
268  {
269  StemClass *stem = (StemClass *)variableValue;
270  return stem->isEmpty() && stem->getValue() == variableName;
271  }
272 
273  // local variable, it must be uninitialized
274  return isDropped();
275 }
