209 /**
210  * Expand the contents of a collection to a given bucket
211  * capacity.
212  */
213 void VariableDictionary::expandContents(size_t capacity )
214 {
215  size_t bucketSize = HashCollection::calculateBucketSize(capacity);
216  Protected<StringHashContents> newContents = allocateContents(bucketSize, bucketSize * 2);
217  // copy all of the items into the new table
218  contents->reMerge(newContents);
219  // if this is a contents item in the old space, we need to
220  // empty this so any old-to-new table entries in the contents can get updated.
221  if (contents->isOldSpace())
222  {
223  contents->empty();
224  }
225  // replace the contents
226 
227  setField(contents, newContents);
228 }

231 /**
232  * Ensure that the collection has sufficient space for a
233  * mass incoming addition.
234  *
235  * @param delta The number of entries we wish to add.
236  */
237 void VariableDictionary::ensureCapacity(size_t delta)
238 {
239  // not enough space? time to expand. We'll go to
240  // the current total capacity plus the delta...or the standard
241  // doubling if the delta is a small value.
242  if (!contents->hasCapacity(delta))
243  {
244  expandContents(contents->capacity() + std::max(delta, contents->capacity()));
245  }

278 /**
279  * Perform a deep copy of a variable dictionary. This copies the object,
280  * all variable objects stored in the object, AND any variable dictionaries chained
281  * off of this one in an object's variable set.
282  *
283  * @return A complete chain copy of the object.
284  */
285 VariableDictionary *VariableDictionary::deepCopy()
286 {
287  // make a copy of ourselves first. This also copies the values.
288  Protected<VariableDictionary> newDictionary = (VariableDictionary *)copy();
289  // We might be copying an object with a guard lock, so we need to
290  // clear this out in the new copy.
291  newDictionary->reservingActivity = OREF_NULL;
292  newDictionary->waitingActivities = OREF_NULL;
293  // and propagate this if we're chained
294  if (nextDictionary != OREF_NULL)
295  {
296  newDictionary->setNextDictionary(nextDictionary->deepCopy());
297  }
298 
299  return newDictionary;
300 }

516 /**
517  * Reserve a scope on an object, waiting for completion if this
518  * is already reserved by another activity
519  *
520  * @param activity The activity reserving the scope.
521  */
522 void VariableDictionary::reserve(Activity *activity)
523 {
524  // not reserved at all? This is easy.
525  if (reservingActivity == OREF_NULL)
526  {
527  // Activity objects are automatically safe from
528  // garbage collection and this is something we
529  // want to occur as quickly a possible, so just assign
530  // this value directly without worrying about old-to-new issues.
531  reservingActivity = activity;
532  reserveCount = 1;
533  }
534  // doing this again on the same stack? Just bump the
535  // nesting count. Note that nested activities created via
536  // attach thread will count as being part of the same activity
537  // stack since they are on the same system thread.
538  else if (activity->isSameActivityStack(reservingActivity))
539  {
540  reserveCount++;
541  }
542  // we have an access collision. We need to wait on this one.
543  else
544  {
545  reservingActivity->checkDeadLock(activity);
546  // this one we need to use setField()
547  if (waitingActivities == OREF_NULL)
548  {
549  setField(waitingActivities, new_array());
550  }
551  // add to the end of the queue
552  waitingActivities->append(activity);
553  // ok, now we wait
554  activity->waitReserve(this);
555  }
556 }

559 /**
560  * Release the lock on an object's variable dictionary.
561  *
562  * @param activity The activity owning the lock.
563  */
564 void VariableDictionary::release(Activity *activity)
565 {
566  reserveCount--;
567  // if this is the last reserver on this activity, we just clear everything out
568  if (reserveCount == 0)
569  {
570  // setField() not required here. See comment in reserve()
571  reservingActivity = OREF_NULL;
572  // potential waiters?
573  if (waitingActivities != OREF_NULL && !waitingActivities->isEmpty())
574  {
575  // remove the first item and make it the new reserver
576  reservingActivity = (Activity *)waitingActivities->removeFirst();
577  reserveCount = 1;
578  // wake up the waiting activity.
579  reservingActivity->guardPost();
580  }
581  }
582 }

585 /**
586  * Transfer the variable dictionary lock to another
587  * activity. Occurs when a reply has been issued.
588  *
589  * @param activity The new activity.
590  *
591  * @return true if the reservation could transfer, false if there
592  * is nesting taking place that prevents this from occurring.
593  */
594 bool VariableDictionary::transfer(Activity *activity)
595 {
596  // if there is only a single level of nesting, just change the reserving activity
597  if (reserveCount == 1)
598  {
599  // NOTE: Not using setField
600  reservingActivity = activity;
601  // indicate the transfer is complete
602  return true;
603  }
604  // multiple nesting levels...we perform a release for this activation level,
605  // but this activity still owns the lock.
606  else
607  {
608  release(activity);
609  // could not release.
610  return false;
611  }
612 }

722 /**
723  * Generate a variable retriever to access a variable from a
724  * string name. Used for dynamic access such as the value
725  * function or the variable pool interface.
726  *
727  * @param variable The variable name.
728  *
729  * @return An appropriate variable retriever. Returns NULL if the
730  * name is not a valid variable name.
731  */
732 RexxVariableBase *VariableDictionary::getVariableRetriever(RexxString *variable )
733 {
734  // All variables for symbolic access have uppercase names.
735  variable = variable->upper();
736  // analyze the the string content and decide what sort of variable we have
737  switch (variable->isSymbol())
738  {
739  // totally not a valid variable name
740  case STRING_BAD_VARIABLE:
741  return OREF_NULL;
742 
743  // this is a symbol, but a constant one. Just return the name as
744  // the retriever.
745  case STRING_LITERAL_DOT:
746  case STRING_NUMERIC:
747  return(RexxVariableBase *)variable;
748 
749  // Dot variables retrieve from the environment
750  case STRING_LITERAL:
751  // this is only a dot variable if it begins with a period
752  if (variable->getChar(0) == '.')
753  {
754  return (RexxVariableBase *)new RexxDotVariable(variable->extract(1, variable->getLength() - 1));
755  }
756  // this is a literal symbol not beginning with a period
757  return (RexxVariableBase *)variable;
758 
759  // have a stem name
760  case STRING_STEM:
761  // create a stem retriever with a zero index
762  return(RexxVariableBase *)new RexxStemVariable(variable, 0);
763 
764  // a compound name
765  case STRING_COMPOUND_NAME:
766  // this needs to break the name up into the component parts for retrieva.
767  return(RexxVariableBase *)buildCompoundVariable(variable, false);
768 
769  // just a simple variable
770  case STRING_NAME:
771  return(RexxVariableBase *)new RexxSimpleVariable(variable, 0);
772 
773  // unknown...give back nothing
774  default:
775  return OREF_NULL;

780 /**
781  * Return a retriever for a variable using direct access (i.e.
782  * no substitution in compound variable tails)
783  *
784  * @param variable The full name of the variable. The tail portion of
785  * any compound variables is taken as a literal.
786  *
787  * @return A retriever for the variable type.
788  */
789 RexxVariableBase *VariableDictionary::getDirectVariableRetriever(RexxString *variable)
790 {
791  // we don't perform any uppercasing here...the name is either good, or not.
792  size_t length = variable->getLength();
793  // get the first character
794  char character = variable->getChar(0);
795  bool literal = false;
796 
797  // is this a constant symbol? Remember for later
798  if (character == '.' || (character >= '0' && character <= '9'))
799  {
800  literal = true;
801  }
802 
803  // only process as a potential variable if the name has a good length
804  if (length <= LanguageParser::MAX_SYMBOL_LENGTH && length > 0)
805  {
806  // now we need to scan the name and figure out the type.
807  size_t compound = 0; // no periods yet
808  size_t scan = 0; // start at string beginning
809  size_t nonnumeric = 0; // count of non-numeric characters
810  char last = 0; // no last character
811 
812  while (scan < length)
813  {
814  // a character-by-character scan
815  character = variable->getChar(scan);
816  // hit a period...could be a stem, compound, or just part of a number.
817  if (character == '.')
818  {
819  if (!literal)
820  {
821  // since this is a direct retriever, all variable names with a period
822  // are compound variables...this could a compound variable with a null tail.
823  return(RexxVariableBase *)buildCompoundVariable(variable, true);
824  }
825  else
826  {
827  compound++;
828  }
829  }
830  // may have a special character We also allow valid numeric symbols, which means dealing
831  // with an exponent.
832  else if (!LanguageParser::isSymbolCharacter(character))
833  {
834  // check for the signs first
835  if (character == '+' || character == '-')
836  {
837  // we've tracked periods and non-numeric characters,
838  // so we know if this can be a number. The last character
839  // must also be an E to be accepted.
840  if (compound > 1 || nonnumeric > 1 || last != 'E')
841  {
842  return OREF_NULL;
843  }
844  scan++;
845  // the sign can't be the last character either
846  if (scan >= length)
847  {
848  return OREF_NULL;
849  }
850  // to be valid, we can only have digits from this point
851  while (scan < length)
852  {
853 
854  character = variable->getChar(scan);
855  if (character < '0' || character > '9')
856  {
857  return OREF_NULL;
858  }
859  scan++;
860  }
861  break;
862  }
863  else
864  {
865  // invalid character in a symbol
866  return OREF_NULL;
867  }
868  }
869  // lower case characters fail on a direct lookup
870  else if (LanguageParser::translateChar(character) != character)
871  {
872  return OREF_NULL;
873  }
874  // non-special character...keep count of any non-digit characters
875  else if (character < '0' || character > '9')
876  {
877  nonnumeric++;
878  }
879  // keep track of the last character and continue
880  last = character;
881  scan++;
882  }
883  }
884  else
885  {
886  // either nullstring or too long
887  return OREF_NULL;
888  }
889 
890  // we've screend out all of the invalid cases. This is either just a
891  // literal value or a real simple variable.
892  if (literal)
893  {
894  // literals are their own retrievers.
895  return(RexxVariableBase *)variable;
896  }
897  // a simple variable
898  else
899  {
900  return(RexxVariableBase *)new RexxSimpleVariable(variable, 0);
901  }
902 }

905 /**
906  * Build a retriever for a compound variable (either direct or symbolic)
907  *
908  * @param variable_name
909  * The full name of the variable.
910  * @param direct Indicates whether we build this with direct or symbolic rules.
911  * For direct rules, everything after the first period is
912  * used as a literal tail value. for symbolic rules, we
913  * need to evaluate each of the tail sections.
914  *
915  * @return A retriever for this variable.
916  */
917 RexxVariableBase *VariableDictionary::buildCompoundVariable(RexxString *variable_name, bool direct)
918 {
919  size_t length = variable_name->getLength();
920  size_t position = 0;
921 
922  // ok, we need to scan to the first period. We know we have a
923  // good variable up to the stem name start.
924  while (variable_name->getChar(position) != '.')
925  {
926  position++;
927  length--;
928  }
929 
930  // get the stem part
931  Protected<RexxString> stem = variable_name->extract(0, position + 1);
932 
933  // start decomposing this into the component parts
934  Protected<QueueClass> tails = new_queue();
935  // the first period is part of the stem name.
936  position++;
937  length--;
938 
939  // if we're buildind a direct retriever, there is just one
940  // tail piece that is a literal...and it could be a null string
941  if (direct == true)
942  {
943  // get the remainder as a tail and use that
944  RexxString *tail = variable_name->extract(position, length);
945  tails->push(tail);
946  }
947  // need to scan for component pieces
948  else
949  {
950  size_t endPosition = position + length;
951 
952  while (position < endPosition)
953  {
954  size_t start = position;
955  // find the next period (or the end of the name)
956  while (position < endPosition && variable_name->getChar(position) != '.')
957  {
958  position++; /* step to the next character */
959  }
960  // get the piece as a string
961  RexxString *tail = variable_name->extract(start, position - start);
962 
963  // a null string, or constant symbol (starting with a digit) can be used directly.
964  RexxVariableBase *tailPart;
965  if (tail->getLength() == 0 || (tail->getChar(0) >= '0' && tail->getChar(0) <= '9'))
966  {
967  tailPart = (RexxVariableBase *)tail;
968  }
969  // simple variable, we need a retriever for this
970  else
971  {
972  tailPart = new RexxSimpleVariable(tail, 0);
973  }
974  tails->push(tailPart);
975  position++;
976  }
977 
978  // since we build up the name from pieces and only add dots between sections, add
979  // a null string for the final section to pick up the trailing period
980  if (variable_name->getChar(position - 1) == '.')
981  {
982  tails->push(GlobalNames::NULLSTRING);
983  }
984  }
985 
986  // create and return a new compound
987  return new (tails->items()) RexxCompoundVariable(stem, 0, tails, tails->items());
988 }
