101  */
102 void CPPCode::liveGeneral(MarkReason reason)
103 {
104  // if we're getting ready to save the image, replace the source
105  // package with the global REXX package
106  if (reason == PREPARINGIMAGE)
107  {
108  package = TheRexxPackage;
109  }
110  else if (reason == SAVINGIMAGE)
111  {
112  // if we're saving the image, zero the pointer value out so that this creates
113  // consistently repeatable build.
114  cppEntry = NULL;
115  }

117  {
118  cppEntry = exportedMethods[methodIndex];
119  }
120 
121  memory_mark_general(package);
122 }

125 /**
126  * Run (or call) a CPPMethod.
127  *
128  * @param activity The activity we're running under.
129  * @param method The method to run.
130  * @param receiver The receiver object.
131  * @param messageName
132  * The name used to invoke the message.
133  * @param argPtr The actual arguments.
134  * @param count The argument count.
135  * @param result The returned result.
136  */
137 void CPPCode::run(Activity *activity, MethodClass *method, RexxObject *receiver, RexxString *messageName,
138  RexxObject **argPtr, size_t count, ProtectedObject &result)
139 {
140  InternalActivationFrame frame(activity, messageName, receiver, method, argPtr, count);
141  PCPPM methodEntry = this->cppEntry; /* get the entry point */
142  /* expecting an array? */
143  /* expecting a pointer/count pair? */
144  if (this->argumentCount == A_COUNT)
145  {
146  /* we can pass this right on */
147  result = (receiver->*((PCPPMC1)methodEntry))(argPtr, count);
148  }
149  else
150  {
151  if (count > argumentCount)
152  {
153  reportException(Error_Incorrect_method_maxarg, this->argumentCount);
154  }
155  // This is the temporary list of arguments
156  RexxObject *argument_list[7];
157  if (count < argumentCount) /* need to pad these out? */
158  {
159  // null out the argument list
160  memset(argument_list, 0, sizeof(argument_list));
161  // and copy over the provided argument pointers
162  memcpy(argument_list, argPtr, count * sizeof(RexxObject *));
163  // now switch the argument pointer to the temporary
164  argPtr = &argument_list[0];
165  }
166  // now we make the actual call
167  switch (argumentCount)
168  {
169  case 0: /* zero */
170  result = (receiver->*((PCPPM0)methodEntry))();
171  break;
172 
173  case 1:
174  result = (receiver->*((PCPPM1)methodEntry))(argPtr[0]);
175  break;
176 
177  case 2:
178  result = (receiver->*((PCPPM2)methodEntry))(argPtr[0], argPtr[1]);
179  break;
180 
181  case 3:
182  result = (receiver->*((PCPPM3)methodEntry))(argPtr[0], argPtr[1], argPtr[2]);
183  break;
184 
185  case 4:
186  result = (receiver->*((PCPPM4)methodEntry))(argPtr[0], argPtr[1], argPtr[2], argPtr[3]);
187  break;
188 
189  case 5:
190  result = (receiver->*((PCPPM5)methodEntry))(argPtr[0], argPtr[1], argPtr[2],
191  argPtr[3], argPtr[4]);
192  break;
193 
194  case 6:
195  result = (receiver->*((PCPPM6)methodEntry))(argPtr[0], argPtr[1], argPtr[2],
196  argPtr[3], argPtr[4], argPtr[5]);
197  break;
198 
199  case 7:
200  result = (receiver->*((PCPPM7)methodEntry))(argPtr[0], argPtr[1], argPtr[2],
201  argPtr[3], argPtr[4], argPtr[5], argPtr[6]);
202  break;
203  }
204  }
205 }

269  * Execute an attribute-get operation.
270  *
271  * @param activity The current activity.
272  * @param method The method we're invoking.
273  * @param receiver The receiver object.
274  * @param messageName
275  * The name of the message used to invoke the method.
276  * @param argPtr The pointer to the arguments.
277  * @param count The argument count.
278  * @param result The returned result.
279  */
280 void AttributeGetterCode::run(Activity *activity, MethodClass *method, RexxObject *receiver, RexxString *messageName,
281  RexxObject **argPtr, size_t count, ProtectedObject &result)
282 {
283  // validate the number of arguments
284  if (count > 0)
285  {
286  reportException(Error_Incorrect_method_maxarg, (wholenumber_t)0);
287  }
288  // this is simplier if the method is not guarded
289  if (!method->isGuarded())
290  {
291  result = attribute->getValue(receiver->getObjectVariables(method->getScope()));
292  }
293  else
294  {
295  // get the variable pool and get the guard lock
296  VariableDictionary *objectVariables = receiver->getObjectVariables(method->getScope());
297  objectVariables->reserve(activity);
298  result = attribute->getValue(objectVariables);
299  // and ensure we release this afterwards
300  objectVariables->release(activity);
301  }
302 }

593 /**
594  * Execute a Delegate method forward operation.
595  *
596  * @param activity The current activity.
597  * @param method The method we're invoking.
598  * @param receiver The receiver object.
599  * @param messageName
600  * The name of the message used to invoke the method.
601  * @param argPtr The pointer to the arguments.
602  * @param count The argument count.
603  * @param result The returned result.
604  */
605 void DelegateCode::run(Activity *activity, MethodClass *method, RexxObject *receiver, RexxString *messageName,
606  RexxObject **argPtr, size_t count, ProtectedObject &result)
607 {
608  // get the variable pool and get
609  VariableDictionary *objectVariables = receiver->getObjectVariables(method->getScope());
610  RexxObject *target;
611 
612  // if this is a guarded method, we grab the guard only long enough to
613  // get the target variable. The message send is sent without holding the lock
614  if (method->isGuarded())
615  {
616  objectVariables->reserve(activity);
617  target = attribute->getValue(receiver->getObjectVariables(method->getScope()));
618  // and ensure we release this afterwards
619  objectVariables->release(activity);
620  }
621  else
622  {
623  target = attribute->getValue(receiver->getObjectVariables(method->getScope()));
624  }
625 
626  // and finally, send the message to the resolved target
627  target->sendMessage(messageName, argPtr, count, result);
628 }
