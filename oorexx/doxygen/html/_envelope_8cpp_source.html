102 /**
103  * Potentially copy an object into a buffer and resolve
104  * references to the object to it's buffer location.
105  *
106  * @param newThisVoid
107  * The pointer to the reference to the copied object
108  * owning this reference. If it is necessary to reallocate
109  * the buffer during this mark operation, this pointer
110  * value gets updated.
111  * @param newSelf The current buffer offset for this object.
112  * @param objRefVoid The pointer to the reference getting marked
113  * (will also get updated)
114  */
115 void Envelope::flattenReference(void *newThisVoid, size_t newSelf, void *objRefVoid)
116 {
117  RexxInternalObject **newThis = (RexxInternalObject **)newThisVoid;
118  RexxInternalObject **objRef = (RexxInternalObject **)objRefVoid;
119 
120  RexxInternalObject *obj = *objRef;
121 
122  // see if the object has already been flattened.
123  size_t objOffset = queryObj(obj);
124  // if this object has not been previously flattened, we need to
125  // copy the object into the buffer and add the offset to the table.
126  if (objOffset == 0)
127  {
128  // this is the sart of the buffer
129  char *flattenBuffer = this->bufferStart();
130  // this is the offset to the reference we're working with. If the
131  // buffer needs to reallocate after a copy, we need to be able to
132  // locate this later
133  size_t referenceOffset = (char *)objRef - flattenBuffer;
134 
135  // if this is a proxied object, we need to convert it to a proxy and
136  // copy that object into the buffer and the reference table
137  if (obj->requiresProxyObject())
138  {
139  // get a proxy and make sure it's in our protection table
140  RexxInternalObject *proxyObj = obj->makeProxy(this);
141  saveTable->put(proxyObj, proxyObj);
142 
143  // copy the proxy to the buffer and add it to the dup table
144  // using the original object as the index.
145  objOffset = copyBuffer(proxyObj);
146  // it's not likely, but we might get a dup of the
147  // proxy object too. Add it to our resolution table.
148  associateObject(proxyObj, objOffset);
149  }
150  else
151  {
152 
153  // non-proxied object. This gets copied to the buffer
154  // directly and added to the dup table
155  objOffset = copyBuffer(obj);
156  }
157  // regardless of how we handle this, add an association for the object to the offset
158  associateObject(obj, objOffset);
159  flattenStack->push((RexxInternalObject *)objOffset);
160  // if the buffer reallocated, we need to update the updating object pointer too.
161  char *newBuffer = bufferStart();
162  if (newBuffer != flattenBuffer)
163  {
164  *newThis = (RexxInternalObject *) (newBuffer + newSelf);
165  }
166  // and update the reference with the offset
167  *(RexxInternalObject **)(newBuffer + referenceOffset) = (RexxInternalObject *)objOffset;
168  }
169  else
170  {
171  // no copying means no reallocation...we just replace the
172  // original object reference with the offset value
173  *objRef = (RexxInternalObject *)objOffset;
174  }
175 }

178 /**
179  * Pack an object into an envelope
180  *
181  * @param _receiver The object to be packed.
182  *
183  * @return The buffer containing the flattened object.
184  */
185 BufferClass *Envelope::pack(RexxInternalObject *_receiver)
186 {
187  RexxInternalObject *flattenObj; // flattened object
188  RexxInternalObject *newSelf; // the flattened envelope
189  RexxInternalObject *firstObject; // first object to flatten
190 
191  // NOTE: Envelopes are transient objects and will never appear in the OldSpace
192  // image. We do not need to use OrefSet or setField here.
193 
194  receiver = _receiver;
195  // create a save table to protect any objects (such as proxy
196  // objects) we create during flattening.
197  saveTable = new_identity_table();
198  dupTable = new MapTable(DefaultDupTableSize);
199  buffer = new SmartBuffer(DefaultEnvelopeBuffer);
200  // Allocate a flatten stack
201  flattenStack = new (Memory::LiveStackSize) LiveStack (Memory::LiveStackSize);
202  // push unique terminator onto stack
203  flattenStack->push(OREF_NULL);
204 
205  // First, put a header into the buffer. This is necessary because without
206  // it, the envelope object would be at 0 offset into the buffer, which is not
207  // distinguishable from OREF_NULL when the objects are unpacked from buffer.
208 
209  // the header is just a dummy minimal object instance. We don't bother adding
210  // this to the dup table, as it won't ever be duped.
211  copyBuffer(new RexxObject);
212  // we start the flattening process with the received object
213  firstObject = receiver;
214 
215  currentOffset = copyBuffer(firstObject);
216  // make sure we add this to the dup table in case it's self-referential at any point
217  associateObject(firstObject, currentOffset);
218  // point to the copied one
219  newSelf = (RexxInternalObject *)(bufferStart() + currentOffset);
220 
221  // ok, keep flattening until will find our marker object on the stack
222  newSelf->flatten(this);
223 
224  for (flattenObj = flattenStack->pop(); flattenObj != OREF_NULL; flattenObj = flattenStack->pop())
225  {
226  // the popped object is actually an object offset. We need to store them
227  // that way because the object location can change if the buffer needs to
228  // reallocate. We need to convert this into a real object pointer
229  currentOffset = (size_t)flattenObj;
230  flattenObj = (RexxInternalObject *)(bufferStart() + currentOffset);
231  // and flatten the next object
232  flattenObj->flatten(this);
233  }
234 
235  // now unwrap the smart buffer and fix the length of the real buffer
236  // behind it to the size we've written to it.
237  BufferClass *letter = buffer->getBuffer();
238  letter->setDataLength(buffer->getDataLength());
239  // delete the flatten stack, since that is not allocated from the object heap.
240  delete flattenStack;
241  return letter;
242 }

278 /**
279  * Copy an object into the flattening buffer.
280  *
281  * @param obj The object to copy.
282  *
283  * @return The offset of the object within the buffer.
284  */
285 size_t Envelope::copyBuffer(RexxInternalObject *obj)
286 {
287  // copy the object into the buffer, which might cause the buffer to
288  // resize itself.
289  size_t objOffset = buffer->copyData((void *)obj, obj->getObjectSize());
290  // get a reference to the copied object
291  RexxInternalObject *newObj = (RexxInternalObject *) (buffer->getBuffer()->getData() + objOffset);
292  // if this is a non-primative behaviour, we need to flatten it as well. The
293  // offset is tagged as being a non-primitive behaviour that needs later inflating.
294  if (newObj->behaviour->isNonPrimitive())
295  {
296  void *behavPtr = &newObj->behaviour;
297  flattenReference(&newObj, objOffset, (RexxObject **)behavPtr);
298  }
299  else
300  {
301  // transient classes should never be flattened. This is a problem if we encounter one
302  if (newObj->behaviour->isTransientClass())
303  {
304  reportException(Error_Interpretation);
305  }
306 
307  // just replace the behaviour with its normalized type number. This will be used
308  // to restore it later.
309  newObj->behaviour = newObj->behaviour->getSavedPrimitiveBehaviour();
310  }
311  // if we flattened an object from oldspace, we just copied everything. Make sure
312  // this is no longer marked as oldspace
313  newObj->setNewSpace();
314  // the offset is what we need
315  return objOffset;
316 }

368 void Envelope::addTable(RexxInternalObject *obj)
369 {
370  // The following table will be used by the table_unflatten method.
371  //
372  // Every table that gets unflattened place itself in this table. Once
373  // every object has been unflattened we traverse this table and allow

375  // changed
376 
377  // create the table on first addition
378  if (rehashTable == OREF_NULL)
379  {
380  rehashTable = new_identity_table();
381  }
382 
383  // use put to make sure we only get
384  // a single version of each table
385  rehashTable->put(obj, obj);
