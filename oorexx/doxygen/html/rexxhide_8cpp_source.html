58 int WINAPI WinMain(
59  HINSTANCE hInstance, // handle to current instance
60  HINSTANCE hPrevInstance, // handle to previous instance
61  LPSTR lpCmdLine, // pointer to command line
62  int nCmdShow)
63 {
64  int32_t rc; /* actually running program RC */
65  const char *program_name; /* name to run */
66  CHAR arg_buffer[8192]; /* starting argument buffer */
67  CONSTRXSTRING arguments; /* rexxstart argument */
68  size_t argcount;
69 
70  rc = 0; /* set default return */
71 
72  strcpy(arg_buffer, lpCmdLine);
73  getArguments(&program_name, arg_buffer, &argcount, &arguments);
74 
75  if (program_name == NULL)
76  {
77  /* give a simple error message */
78  MessageBox(NULL, "Syntax: REXXHIDE ProgramName [parameter_1....parameter_n]\n", "Wrong Arguments", MB_OK | MB_ICONHAND);
79  return -1;
80  }
81  else /* real program execution */
82  {
83  RexxInstance *pgmInst;
84  RexxThreadContext *pgmThrdInst;
85  RexxArrayObject rxargs, rxcargs;
86  RexxDirectoryObject dir;
87  RexxObjectPtr result;
88  int i;
89  int argc; // parsed count of arguments
90  PCHAR *argv; // parsed out arguments
91 
92  // parse the arguments into argv/argc format
93  argv = CommandLineToArgvA(lpCmdLine, &argc);
94 
95  RexxCreateInterpreter(&pgmInst, &pgmThrdInst, NULL);
96  // configure the traditional single argument string
97  if ( arguments.strptr != NULL )
98  {
99  rxargs = pgmThrdInst->NewArray(1);
100  pgmThrdInst->ArrayPut(rxargs, pgmThrdInst->NewString(arguments.strptr, arguments.strlength), 1);
101  }
102  else
103  {
104  rxargs = pgmThrdInst->NewArray(0);
105  }
106 
107  // set up the C args into the .local environment
108  dir = (RexxDirectoryObject)pgmThrdInst->GetLocalEnvironment();
109  // lpCmdLine does NOT include the program name, i.e. rexxhide
110  if ( argc > 1 )
111  {
112  rxcargs = pgmThrdInst->NewArray(argc - 1);
113  }
114  else
115  {
116  rxcargs = pgmThrdInst->NewArray(0);
117  }
118 
119  for (i = 1; i < argc; i++)
120  {
121  pgmThrdInst->ArrayPut(rxcargs, pgmThrdInst->NewStringFromAsciiz(argv[i]), i);
122  }
123 
124  pgmThrdInst->DirectoryPut(dir, rxcargs, "SYSCARGS");
125 
126  GlobalFree(argv); // release the parsed arguments
127  // call the interpreter
128  result = pgmThrdInst->CallProgram(program_name, rxargs);
129  // display any error message if there is a condition. if there was an
130  // error, then that will be our return code. we know error code will fit
131  // in an int32_t. This just writes it out to the configured error stream. We
132  // will want to give full information in the popup as well.
133  rc = (int32_t)pgmThrdInst->DisplayCondition();
134  if (rc != 0)
135  {
136  RexxDirectoryObject condition = pgmThrdInst->GetConditionInfo();
137  RexxCondition conditionInfo;
138 
139  pgmThrdInst->DecodeConditionInfo(condition, &conditionInfo);
140  wholenumber_t minorCode = conditionInfo.code - (conditionInfo.rc * 1000);
141 
142  sprintf(arg_buffer, "Error %zd.%1zd running %s line %zd\n\n%s\n%s", conditionInfo.rc, minorCode,
143  pgmThrdInst->StringData(conditionInfo.program), conditionInfo.position,
144  pgmThrdInst->StringData(conditionInfo.errortext), pgmThrdInst->StringData(conditionInfo.message));
145 
146  char title[64];
147  time_t now = time(NULL);
148  struct tm *local = localtime(&now);
149  strftime(title, sizeof(title) - 1, "%F %T Open Object Rexx Execution Error", local);
150 
151  MessageBox(NULL, arg_buffer, title, MB_OK | MB_ICONHAND);
152 
153  pgmInst->Terminate();
154  return -rc; // well, the negation of the error number is the return code
155  }
156  if (result != NULL)
157  {
158  pgmThrdInst->ObjectToInt32(result, &rc);
159  }
160 
161  pgmInst->Terminate();
162  }
163 
164  return rc;
165 }
