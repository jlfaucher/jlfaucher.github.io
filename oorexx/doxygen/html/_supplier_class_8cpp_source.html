249 /**
250  * Retrieve the index of the current position.
251  *
252  * @return The position index.
253  */
254 RexxInternalObject *SupplierClass::index()
255 {
256  // past the end if an error
257  if (position > items->size())
258  {
259  reportException(Error_Incorrect_method_supplier);
260  }
261  // the index array is optional...if we don't have it, just give
262  // the numeric position
263  if (indexes == OREF_NULL)
264  {
265  return new_integer(position);
266  }
267 
268  // already gone past the end of the index array?
269  if (position > indexes->size())
270  {
271  return TheNilObject;
272  }
273  else
274  {
275  // get the current value and return .nil if nothing is there.
276  return resultOrNil(indexes->get(position));
277  }
278 }
