40 /* Class to encapuslate knowledge of various trace settings in an activation. */
41 /* These are frequently constructed by the language parser, but processed by */
42 /* the activation. This removes ownership of this information to a central */
43 /* place. */

53 /**
54  * A class for processing different trace settings as set by the
55  * TRACE command.
56  */
57 class TraceSetting
58 {
59  public:
60 
61  TraceSetting() { }
62 
63  /**
64  * Flag settings within the trace flags.
65  */
66  typedef enum
67  {
68  traceOff, // trace nothing
69  traceNormal, // the default normal trace setting.
70  traceDebug, // interactive trace mode flag
71  traceAll, // trace all instructions
72  traceResults, // trace all results
73  traceIntermediates, // trace all instructions
74  traceCommands, // trace all commands
75  traceLabels, // trace all labels
76  traceErrors, // trace all command errors
77  traceFailures, // trace all command failures
78  pauseInstructions, // pause for instructions (optimization of traceDebug and traceAll)
79  pauseLabels, // pause for labels
80  pauseCommands, // pause for commands,
81  debugToggle, // requesting a debug toggle
82  } TraceFlag;
83 
84  /**
85  * The type for trace flags. We create a few constant versions
86  * of these for simple setting and testing.
87  */
88  typedef FlagSet<TraceFlag, 32> TraceFlags;
89 
90  static const TraceFlags defaultTraceFlags;
91 
92  // now the flag sets for different settings
93  static const TraceFlags traceAllFlags;
94  static const TraceFlags traceResultsFlags;
95  static const TraceFlags traceIntermediatesFlags;
96 
97  inline void clear() { flags.reset(); }
98  inline bool tracingResults() const { return flags[traceResults]; }
99  inline bool tracingIntermediates() const { return flags[traceIntermediates]; }
100  inline bool isDebug() const { return flags[traceDebug]; }
101  inline bool tracingAll() const { return flags[traceAll]; }
102  inline bool tracingErrors() const { return flags[traceErrors]; }
103  inline bool tracingFailures() const { return flags[traceFailures]; }
104  inline bool tracingLabels() const { return flags[traceLabels]; }
105  inline bool tracingCommands() const { return flags[traceCommands]; }
106  inline bool pausingInstructions() const { return flags[pauseInstructions]; }
107  inline bool pausingLabels() const { return flags[pauseLabels]; }
108  inline bool pausingCommands() const { return flags[pauseCommands]; }
109  inline bool isDebugToggle() const { return flags[debugToggle]; }
110  inline bool isTraceOff() const { return flags[traceOff]; }
111 
112  // reset all debug settings
113  inline void resetDebug()
114  {
115  // turn off the debug and all pause flags
116  flags.reset(traceDebug, pauseInstructions, pauseLabels, pauseCommands);
117  }
118  // turn on debug settings
119  inline void setDebug()
120  {
121  // turn on the debug flag and any pause indicators.
122  flags[traceDebug] = true;
123  flags[pauseInstructions] = tracingAll();
124  flags[pauseLabels] = tracingLabels();
125  flags[pauseCommands] = tracingCommands();
126  }
127  // toggle the debug setting
128  inline void toggleDebug()
129  {
130  if (isDebug())
131  {
132  resetDebug();
133  }
134  else
135  {
136  setDebug();
137  }
138  }
139 
140  // turn tracing off
141  inline void setTraceOff()
142  {
143  // reset all of the flags, then set the explicit Off flag
144  flags.reset();
145  flags[traceOff] = true;
146  }
147 
148  // merge the flag settings, keeping existing debug settings
149  inline void merge(const TraceSetting &s)
150  {
151  if (s.isTraceOff())
152  {
153  setTraceOff();
154  }
155  else
156  {
157  // pull in the settings from the other and keep
158  // our debug setting
159  bool debug = isDebug();
160  flags = s.flags;
161  // if we were debug, flip on again to make sure
162  // the pausing flags are set correctly with the new settings.
163  if (debug)
164  {
165  setDebug();
166  }
167  }
168  }
169 
170  // unconditionally set the flag settings
171  inline void set(const TraceSetting &s)
172  {
173  // copy the flags
174  flags = s.flags;
175  // if the debug flag is set, do the set processing again to
176  // make sure the pause flags are in sync with the new bits.
177  if (isDebug())
178  {
179  setDebug();
180  }
181  }
182 
183  /**
184  * Test if we have no explict value set.
185  *
186  * @return True if all flags are off.
187  */
188  inline bool isNoSetting() const
189  {
190  return flags.none();
191  }
192 
193  // set the default trace setting
194  inline void setDefault()
195  {
196  flags.reset();
197  // trace normal really traces failures, but we set the
198  // special flag so we know the setting too.
199  flags = defaultTraceFlags;
200  }
201 
202  // we're tracing all
203  inline void setTraceAll()
204  {
205  flags.reset();
206  flags = traceAllFlags;
207  }
208 
209  // we're tracing just commands
210  inline void setTraceCommands()
211  {
212  flags.reset();
213  // Just the single instruction type here.
214  flags[traceCommands] = true;
215  }
216 
217  // we're tracing just labels
218  inline void setTraceLabels()
219  {
220  flags.reset();
221  // just interested in labels.
222  flags[traceLabels] = true;
223  }
224 
225  // we're tracing just command errors
226  inline void setTraceErrors()
227  {
228  flags.reset();
229  // just errors includes failures.
230  flags[traceErrors] = true;
231  flags[traceFailures] = true;
232  }
233 
234  // we're tracing just command failures
235  inline void setTraceFailures()
236  {
237  flags.reset();
238  // failures, but not errors
239  flags[traceFailures] = true;
240  }
241 
242  // we're tracing just command failures
243  inline void setTraceNormal()
244  {
245  flags.reset();
246  // failures, but not errors, but remember that this is
247  // the Normal version for display
248  flags = defaultTraceFlags;
249  }
250 
251  // we're tracing results
252  inline void setTraceResults()
253  {
254  flags.reset();
255  // results is trace all with results traced also.
256  flags = traceResultsFlags;
257  }
258 
259  // we're tracing everything
260  inline void setTraceIntermediates()
261  {
262  flags.reset();
263  // intermediates is trace results with intermediates traced also.
264  flags = traceIntermediatesFlags;
265  }
266 
267  // flip the debug setting
268  inline void flipDebug()
269  {
270  flags.flip(traceDebug);
271  }
272 
273  // set the debug toggle setting
274  inline void setDebugToggle()
275  {
276  flags.reset();
277  flags[debugToggle] = true;
278  }
279 
280  // set trace to the external tracing form.
281  inline void setExternalTrace()
282  {
283  // The external trace value is Trace ?R
284  setTraceResults();
285  setDebug();
286  }

288  RexxString *toString();
289  bool parseTraceSetting(RexxString *value, char &badOption);
290 
291 
292 protected:
293 
294  TraceFlags flags; // the flag settings associated with the selected trace.
295 };
