138 /**
139  * Create a new package from code contained in a file
140  * or array.
141  *
142  * @param init_args The pointer to the new arguments.
143  * @param argCount The count of arguments.
144  *
145  * @return A new package object.
146  */
147 PackageClass *PackageClass::newRexx(RexxObject **init_args, size_t argCount)
148 {
149  // this class is defined on the object class, but this is actually attached
150  // to a class object instance. Therefore, any use of the this pointer
151  // will be touching the wrong data. Use the classThis pointer for calling
152  // any methods on this object from this method.
153  RexxClass *classThis = (RexxClass *)this;
154 
155  RexxObject *pgmname; // source name
156  RexxObject *programSource; // Array or string object
157  size_t initCount = 0; // count of arguments we pass along
158 
159  Activity *activity = ActivityManager::currentActivity;
160  InterpreterInstance *instance = activity->getInstance();
161 
162  // parse the arguments
163  RexxClass::processNewArgs(init_args, argCount, init_args, initCount, 2, pgmname, &programSource);
164 
165  Protected<PackageClass> package;
166 
167  // get the package name as a string
168  Protected<RexxString> nameString = stringArgument(pgmname, "name");
169  if (programSource == OREF_NULL)
170  {
171  // if no directly provided source, resolve the name in the global context and have the instance
172  // load the file.
173  Protected<RexxString> resolvedName = instance->resolveProgramName(nameString, OREF_NULL, OREF_NULL, RESOLVE_REQUIRES);
174  package = instance->loadRequires(activity, nameString, resolvedName);
175  }
176  // we're creating an in-memory package. We allow a parent context object to be specified
177  // so that the package has access to artifacts of the parent.
178  else
179  {
180  // default parent context is none
181  PackageClass *sourceContext = OREF_NULL;
182 
183  if (initCount != 0)
184  {
185  RexxObject *option;
186  // parse off an additional argument
187  RexxClass::processNewArgs(init_args, initCount, init_args, initCount, 1, option, NULL);
188  // if there are more than 3 options passed, it is possible this one was omitted
189  if (option != OREF_NULL)
190  {
191  if (isOfClass(Method, option) || isOfClass(Routine, option))
192  {
193  sourceContext = ((BaseExecutable *)option)->getPackage();
194  }
195  else if (isOfClass(Package, option))
196  {
197  sourceContext = (PackageClass *)option;
198  }
199  else
200  {
201  reportException(Error_Incorrect_method_argType, IntegerThree, "Method, Routine, or Package object");
202  }
203  }
204  }
205 
206 
207  // validate, and potentially transform, the method source object.
208  Protected<ArrayClass> sourceArray = BaseExecutable::processExecutableSource(programSource, "source");
209 
210  // if not a valid source, give an error
211  if (sourceArray == OREF_NULL)
212  {
213  reportException(Error_Incorrect_method_no_method, "source");
214  }
215 
216  // and create the package
217  package = LanguageParser::createPackage(nameString, sourceArray, sourceContext);
218  // make sure the prolog is run
219  package->runProlog(activity);
220  }
221 
222  // handle Rexx class completion
223  classThis->completeNewObject(package, init_args, initCount);
224  return package;
225 }

510 // extra space required to format a result line. This overhead is
511 //
512 // 8 leading spaces for the line number +
513 // 1 space +
514 // 3 (length of the message prefix +
515 // 1 space +
516 // 2 for an indent +
517 // 2 for the quotes surrounding the value
518 const size_t TRACE_OVERHEAD = 16;
519 
520 // overhead for a traced instruction
521 //
522 // 8 digit line number +
523 // 1 space +
524 // 3 character prefix +
525 // 1 blank
526 const size_t INSTRUCTION_OVERHEAD = 11;
527 
528 // size of a line number
529 const size_t LINENUMBER = 6;
530 
531 // offset of the prefix information
532 const size_t PREFIX_OFFSET = (LINENUMBER + 1);
533 // length of the prefix flag
534 const size_t PREFIX_LENGTH = 3;
535 // amount of indent spacing for results lines
536 const size_t INDENT_SPACING = 2;

539 /**
540  * Format a source line for tracing
541  *
542  * @param activation The activation of the current running code. This can be
543  * null if this is a translation time error.
544  * @param location The source line location.
545  * @param indent The indentation amount to apply to the trace line
546  * @param trace This is a traced line vs. an error line
547  *
548  * @return A formatted trace line, including headers and indentations.
549  */
550 RexxString *PackageClass::traceBack(RexxActivation *activation, SourceLocation &location,
551  size_t indent, bool trace)
552 {
553  char linenumber[11]; /* formatted line number */
554 
555  // format the line number as a string
556  snprintf(linenumber, sizeof(linenumber), "%zu", location.getLineNumber());
557 
558  // get the line from the source string...this can return "" if the source is
559  // not available or this string is somehow out of bounds.
560  RexxString *line = source->extract(location);
561 
562  // not available...we provide some sort of information about what is there, even
563  // if we can't display the source line.
564  if (line == GlobalNames::NULLSTRING)
565  {
566  // old space code means this is part of the interpreter image. Don't include
567  // the package name in the message
568  if (isInternalCode())
569  {
570  line = ActivityManager::currentActivity->buildMessage(Message_Translations_internal_code, new_array((size_t)0));
571  }
572 
573  // if we have an activation (and we should, since the only time we won't would be for a
574  // translation time error...and we have source then), ask it to provide a line describing
575  // the invocation situation
576  if (activation != OREF_NULL)
577  {
578  line = activation->formatSourcelessTraceLine(programName);
579  }
580 
581  // this could be part of the internal code...give a generic message that doesn't identify
582  // the actual package.
583  else if (isInternalCode())
584  {
585  line = ActivityManager::currentActivity->buildMessage(Message_Translations_internal_code, new_array((size_t)0));
586  }
587  else
588  {
589  // generic package message.
590  ArrayClass *args = new_array(programName);
591  ProtectedObject p(args);
592  line = ActivityManager::currentActivity->buildMessage(Message_Translations_no_source_available, args);
593  }
594  }
595 
596  ProtectedObject p(line);
597 
598  // get a raw empty string so we can build this trace up.
599  RexxString *buffer = raw_string(line->getLength() + INSTRUCTION_OVERHEAD + indent * INDENT_SPACING);
600  // blank out the first part
601  buffer->set(0, ' ', INSTRUCTION_OVERHEAD + indent * INDENT_SPACING);
602 
603  //copy in the source line(s)
604  buffer->put(INSTRUCTION_OVERHEAD + indent * INDENT_SPACING, line->getStringData(), line->getLength());
605 
606  // now add in the line number. If the number is two large to add, we just overlay a question mark.
607  // I have never seen that happen!
608  size_t outlength = strlen(linenumber);
609  char *linepointer = linenumber;
610  // if the line number is larger than we can fit in the standard
611  // space, overlay with a question mark. Note that his requires a
612  // program over a million lines long!
613  if (outlength > LINENUMBER)
614  {
615  linepointer += outlength - LINENUMBER;
616  *linepointer = '?';
617  outlength = LINENUMBER;
618  }
619  // copy in the line number
620  buffer->put(LINENUMBER - outlength, linepointer, outlength);
621  // add the traceback prefix, and we're done.
622  buffer->put(PREFIX_OFFSET, "*-*", PREFIX_LENGTH);
623  return buffer;
624 }

676 /**
677  * Merge all of the information from inherited packages
678  * into this lookup context.
679  *
680  * @param source The source object we're merging from.
681  */
682 void PackageClass::mergeRequired(PackageClass *mergeSource)
683 {
684  // handle the directly defined public ones first, followed by any merged from
685  // other sources. This will maintain the proper search order.
686  if (mergeSource->publicRoutines != OREF_NULL)
687  {
688  // first merge attempt? Create our directory...Note that the source
689  // public routines get added to our MERGED public routines
690  if (mergedPublicRoutines == OREF_NULL)
691  {
692  setField(mergedPublicRoutines, new_string_table());
693  }
694 
695  // merge these together
696  mergeSource->publicRoutines->merge(mergedPublicRoutines);
697  }
698 
699  // now add in the ones pulled in from other ::requires. These will not
700  // override any routines already in the merged set.
701  if (mergeSource->mergedPublicRoutines != OREF_NULL)
702  {
703  // first merged attempt? Create our directory
704  if (mergedPublicRoutines == OREF_NULL)
705  {
706  setField(mergedPublicRoutines, new_string_table());
707  }
708  // merge these together. This is a special operation that will only
709  // add new entries to the list, leaving existing ones unchanged.
710  mergeSource->mergedPublicRoutines->merge(mergedPublicRoutines);
711  }
712 
713 
714  // now do the same process for any of the class contexts
715  if (mergeSource->installedPublicClasses != OREF_NULL)
716  {
717  if (mergedPublicClasses == OREF_NULL)
718  {
719  setField(mergedPublicClasses, new_string_table());
720  }
721  // merge these together
722  mergeSource->installedPublicClasses->merge(mergedPublicClasses);
723  }
724 
725  // the merged ones follow again
726  if (mergeSource->mergedPublicClasses != OREF_NULL)
727  {
728  if (mergedPublicClasses == OREF_NULL)
729  {
730  setField(mergedPublicClasses, new_string_table());
731  }
732 
733  // merge these together
734  mergeSource->mergedPublicClasses->merge(mergedPublicClasses);
735  }
736 }

1211  * all ::requires routines, creating all ::class objects, and
1212  * loading all required libraries.
1213  *
1214  * @param activation
1215  */

1217 {

1219  // run into a recursion problem when class init methods are processed
1220  installRequired = false;
1221 
1222  // native packages are processed first. The requires might actually need
1223  // functons loaded by the packages
1224  if (libraries != OREF_NULL)
1225  {
1226  // now loop through the requires items y
1227  size_t count = libraries->items();
1228  for (size_t i = 1; i <= count; i++)
1229  {
1230  // and have it do the installs processing
1231  LibraryDirective *library = (LibraryDirective *)libraries->get(i);
1232  library->install(this, activation);
1233  }
1234  }
1235 
1236  // native methods and routines are lazy resolved on first use, so we don't
1237  // need to process them here.
1238 
1239  // do we have requires to process?
1240  if (requires != OREF_NULL)
1241  {
1242  // record that we're in an installation chain
1243  InstallingPackage installing(activation, programName);
1244 
1245  // now loop through the requires items
1246  size_t count = requires->items();
1247  for (size_t i = 1; i <= count; i++)
1248  {
1249  // and have it do the installs processing. This is a little roundabout, but
1250  // we end up back in our own context while processing this, and the merge

1253  _requires->install(this, activation);
1254  }

1266  {

1270  current_class->install(this, activation);
1271  }

1275  {
1276  /* get the class info */
