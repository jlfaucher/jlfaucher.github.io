52 void setContextVariable(const char *name, const char *value)
53 {
54  SHVBLOCK shvb;
55 
56  memset(&shvb, 0, sizeof(shvb));
57  shvb.shvnext = NULL;
58  shvb.shvname.strptr = name;
59  shvb.shvname.strlength = strlen(name);
60  shvb.shvvalue.strptr = const_cast<char *>(value);
61  shvb.shvvalue.strlength = strlen(value);
62  shvb.shvnamelen = shvb.shvname.strlength;
63  shvb.shvvaluelen = shvb.shvvalue.strlength;
64  shvb.shvcode = RXSHV_SET;
65  shvb.shvret = 0;
66  RexxVariablePool(&shvb);
67 }

104 int RexxEntry TestFunctionExit(int code, int subcode, PEXIT exitInfo)
105 {
106  InstanceInfo *instanceInfo = getApplicationData();
107 
108  switch (instanceInfo->fnc.action)
109  {
110  case InstanceInfo::SKIP:
111  return RXEXIT_NOT_HANDLED;
112  case InstanceInfo::EXIT_ERROR:
113  return RXEXIT_RAISE_ERROR;
114  case InstanceInfo::RAISE:
115  return RXEXIT_RAISE_ERROR;
116  default: break; // avoid warning: enumeration values not handled in switch
117  }
118 
119  RXFNCCAL_PARM *parms = (RXFNCCAL_PARM *)exitInfo;
120  const char *functionName = parms->rxfnc_name;
121  if (strcmp(functionName, "TESTSUBCALL") == 0)
122  {
123  // return the status of the flag
124  if (parms->rxfnc_flags.rxffsub)
125  {
126  strcpy(parms->rxfnc_retc.strptr, "SUBROUTINE");
127  }
128  else
129  {
130  strcpy(parms->rxfnc_retc.strptr, "FUNCTION");
131  }
132  parms->rxfnc_retc.strlength = strlen(parms->rxfnc_retc.strptr);
133  return RXEXIT_HANDLED;
134  }
135  else if (strcmp(functionName, "TESTERROR") == 0)
136  {
137  // this should raise an error
138  parms->rxfnc_flags.rxfferr = 1;
139  return RXEXIT_HANDLED;
140  }
141  else if (strcmp(functionName, "TESTNOTOUND") == 0)
142  {
143  // this should raise an error
144  parms->rxfnc_flags.rxffnfnd = 1;
145  return RXEXIT_HANDLED;
146  }
147  else if (strcmp(functionName, "TESTGETCONTEXTVARIABLE") == 0)
148  {
149  getContextVariable(parms->rxfnc_argv[0].strptr, &parms->rxfnc_retc);
150  return RXEXIT_HANDLED;
151  }
152  else if (strcmp(functionName, "TESTSETCONTEXTVARIABLE") == 0)
153  {
154  setContextVariable(parms->rxfnc_argv[0].strptr, parms->rxfnc_argv[1].strptr);
155  parms->rxfnc_retc.strlength = 0;
156  return RXEXIT_HANDLED;
157  }
158  else if (strcmp(functionName, "TESTDROPCONTEXTVARIABLE") == 0)
159  {
160  dropContextVariable(parms->rxfnc_argv[0].strptr);
161  return RXEXIT_HANDLED;
162  }
163  else
164  {
165  // pass on this
166  return RXEXIT_NOT_HANDLED;
167  }
168 }

204  parms->rxcmd_retc.strlength = 1;
205  parms->rxcmd_flags.rxfcerr = 1;
206  return RXEXIT_HANDLED;
207  }
208  // ok, a good address...now do the different commands
209  else if (strcmp(parms->rxcmd_command.strptr, "TRACEON") == 0)
210  {
211  instanceInfo->trc = InstanceInfo::TRACEON;
212  strcpy(parms->rxcmd_retc.strptr, "0");
213  parms->rxcmd_retc.strlength = 1;
214  return RXEXIT_HANDLED;
215  }

220  strcpy(parms->rxcmd_retc.strptr, "0");
221  parms->rxcmd_retc.strlength = 1;
222  return RXEXIT_HANDLED;
223  }
224  // ok, a good address...now do the different commands
225  else if (strcmp(parms->rxcmd_command.strptr, "HALT") == 0)
226  {
227  instanceInfo->hlt = InstanceInfo::HALT;
228  strcpy(parms->rxcmd_retc.strptr, "0");
