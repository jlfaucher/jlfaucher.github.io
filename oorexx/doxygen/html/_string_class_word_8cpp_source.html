50 /**
51  * Delete a word from a given postion.
52  *
53  * @param position The target word position.
54  * @param plength The number of words to delete.
55  *
56  * @return The string with the range of words deleted.
57  */
58 RexxString *RexxString::delWord(RexxInteger *position, RexxInteger *plength)
59 {
60  size_t wordPos = positionArgument(position, ARG_ONE);
61  // get the number of words to delete...the default is "a very large number"
62  size_t count = optionalLengthArgument(plength, Numerics::MAX_WHOLENUMBER, ARG_TWO);
63 
64  // can't take something from nothing. If we start with a nullstring, we
65  // finish with a null string.
66  if (isNullString())
67  {
68  return GlobalNames::NULLSTRING;
69  }
70  // if the count of words to delete is zero, then return
71  // the target unchanged.
72  if (count == 0)
73  {
74  return this;
75  }
76 
77  // create an iterator for traversing the words
78  WordIterator iterator(this);
79 
80  // to the given word position...if we don't get there,
81  // there is nothing to delete so we can just return the
82  // original string.
83  if (!iterator.skipWords(wordPos))
84  {
85  return this;
86  }
87 
88  // we now know the leading portion of the result string
89  size_t frontLength = iterator.wordPointer() - getStringData();
90 
91  // if we managed to locate the desired number of words, then skip
92  // over the trailing blanks to the next word or the end of the string. Note
93  // that we are positioned at the first word already, so we skip one fewer.
94  if (iterator.skipWords(count - 1))
95  {
96  iterator.skipBlanks();
97  }
98 
99  // we know know the leading and trailing string lengths, so we can allocate
100  // the result string
101  RexxString *retval = raw_string(frontLength + iterator.length());
102  StringBuilder builder(retval);
103 
104  // add on the two string sections
105  builder.append(getStringData(), frontLength);
106  // the iterator knows what the remainder is.
107  iterator.appendRemainder(builder);
108  return retval;
109 }

112 /**
113  * The String space method.
114  *
115  * @param space_count
116  * The number of spaces to insert between words.
117  * @param pad The character to use for the word spacing.
118  *
119  * @return The original string with the words separated by the new space count.
120  */
121 RexxString *RexxString::space(RexxInteger *space_count, RexxString *pad)
122 {
123  size_t spaces = optionalLengthArgument(space_count, 1, ARG_ONE);
124  char padChar = optionalPadArgument(pad, ' ', ARG_TWO);
125 
126  size_t sourceLength = getLength();
127  size_t count = 0;
128  // this is the total size consumed by the words...used to
129  // calculate the result string size
130  size_t wordSize = 0;
131 
132  // get an iterator to count up the number of words and get the sizes
133  WordIterator counter(this);
134 
135  while (counter.next())
136  {
137  count++;
138  wordSize += counter.wordLength();
139  }
140 
141  // no words found? The result is a null string
142  if (count == 0)
143  {
144  return GlobalNames::NULLSTRING;
145  }
146 
147  // count is now the number of inter-word gaps in the result
148  count--;
149 
150  RexxString *retval = raw_string(wordSize + count * spaces);
151  StringBuilder builder(retval);
152 
153  // get a fresh iterator for building the string
154  WordIterator iterator(this);
155 
156  // we know the number of words, so we can just traverse through this
157  while (count--)
158  {
159  iterator.next();
160  // add the current word to the result
161  iterator.append(builder);
162  // add the interword spacing
163  builder.pad(padChar, spaces);
164  }
165 
166  // one last word to append
167  iterator.next();
168  iterator.append(builder);
169  return retval;
170 }
