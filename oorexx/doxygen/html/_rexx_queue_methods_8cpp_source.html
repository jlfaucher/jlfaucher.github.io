47 /**
48  * Retrieve the name of the queue. Raises an exception if the variable is not set.
49  *
50  * @param context The current message context.
51  *
52  * @return a success indicator and the CSTRING version of the queue name.
53  */
54 bool getQueueName(RexxMethodContext *context, CSTRING &name)
55 {
56  // this should be set to a variable
57  RexxObjectPtr queue_name = (RexxStringObject)context->GetObjectVariable("NAMED_QUEUE");
58  // this could be a super class override that never forwarded the INIT message.
59  if (queue_name == NULL)
60  {
61  // raise an exception and return a failure
62  context->RaiseException1(Rexx_Error_Execution_noinit, context->GetSelf());
63  return false;
64  }
65  // return the name
66  name = context->ObjectToStringValue(queue_name);
67  return true;
68 }

92 /********************************************************************************************/
93 /* Rexx_pull_queue */
94 /********************************************************************************************/
95 RexxMethod0(RexxObjectPtr, rexx_pull_queue)
96 {
97  RXSTRING buf; /* pulled line buffer */
98  RexxReturnCode rc; /* pull return code */
99 
100  // the queue name is stored as an object variable, retrieve and convert to
101  // CSTRING form.
102  CSTRING queue_name;
103  if (!getQueueName(context, queue_name))
104  {
105  return NULL;
106  }
107 
108  buf.strptr = NULL; /* ask for a returned buffer */
109  buf.strlength = 0;
110  /* pull a line */
111  rc = RexxPullFromQueue(queue_name, &buf, NULL, RXQUEUE_NOWAIT);
112 
113  if (!rc)
114  { /* get a pulled line? */
115  RexxObjectPtr result = context->NewString(buf.strptr, buf.strlength);
116  if (buf.strptr != OREF_NULL)
117  {
118  RexxFreeMemory(buf.strptr);
119  }
120  return result;
121  }
122  return context->Nil(); /* give back a failure */
123 }

161 /********************************************************************************************/
162 /* add a line to a rexx queue */
163 /********************************************************************************************/
164 wholenumber_t rexx_add_queue(
165  RexxMethodContext *context, // the call context
166  RexxStringObject queue_line, /* line to add */
167  int order ) /* queuing order */
168 {
169  char buffer = 0; // buffer for an empty string
170  CONSTRXSTRING rx_string; // rxstring to push
171  RexxReturnCode rc; // queue return code
172 
173  if (queue_line == NULLOBJECT) /* no line given? */
174  {
175  // just use a null string value
176  MAKERXSTRING(rx_string, &buffer, 0);
177  }
178  else
179  {
180  MAKERXSTRING(rx_string, context->StringData(queue_line), context->StringLength(queue_line));
181  }
182 
183  // the queue name is stored as an object variable, retrieve and convert to
184  // CSTRING form.
185  CSTRING queue_name;
186  if (!getQueueName(context, queue_name))
187  {
188  // this raises an exception, so the return value is irrelevant.
189  return 0;
190  }
191 
192  /* move the line to the queue */
193  rc = RexxAddQueue(queue_name, &rx_string, order);
194  if (rc != 0)
195  {
196  char msg[64];
197  char *reason = (char *)(
198  rc == RXAPI_NORXAPI ? "RXAPI_NORXAPI" :
199  rc == RXAPI_MEMFAIL ? "RXAPI_MEMFAIL" :
200  rc == RXQUEUE_BADQNAME ? "RXQUEUE_BADQNAME" :
201  rc == RXQUEUE_PRIORITY ? "RXQUEUE_PRIORITY" :
202  rc == RXQUEUE_BADWAITFLAG ? "RXQUEUE_BADWAITFLAG" :
203  rc == RXQUEUE_EMPTY ? "RXQUEUE_EMPTY" :
204  rc == RXQUEUE_NOTREG ? "RXQUEUE_NOTREG" :
205  rc == RXQUEUE_ACCESS ? "RXQUEUE_ACCESS" : NULL);
206  if (reason == NULL)
207  {
208  snprintf(msg, sizeof(msg), "SYSTEM QUEUE (reason code %d)", rc);
209  }
210  else
211  {
212  snprintf(msg, sizeof(msg), "SYSTEM QUEUE (%s)", reason);
213  }
214  context->RaiseException1(Rexx_Error_System_service_service, context->NewStringFromAsciiz(msg));
215  }
216  return 0;
217 }
