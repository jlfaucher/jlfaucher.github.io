51 /**
52  * Initialize a GUARD instruction instance.
53  *
54  * @param _expression
55  * Optional guard expression.
56  * @param variable_list
57  * the list of guard variables in the expression.
58  * @param on_off indicates whether this is GUARD ON or GUARD OFF.
59  */
60 RexxInstructionGuard::RexxInstructionGuard(RexxInternalObject *_expression,
61  ArrayClass *variable_list, bool on_off)
62 {
63  expression = _expression;
64  guardOn = on_off;
65  // set a default for this.
66  variableCount = 0;
67  // do we have a guard expression? We need to copy the guard variable
68  // list.
69  if (variable_list != OREF_NULL)
70  {
71  variableCount = variable_list->items();
72  /* loop through the variable list */
73  for (size_t i = 0; i < variableCount; i++)
74  {
75  variables[i] = (RexxVariableBase *)(variable_list->get(i + 1));
76  }
77  }
78 }

110 /**
111  * Execute a GUARD instruction.
112  *
113  * @param context The current execution context.
114  * @param stack The current evaluation stack.
115  */
116 void RexxInstructionGuard::execute(RexxActivation *context, ExpressionStack *stack )
117 {
118  context->traceInstruction(this);
119 
120  // only valid in a method context.
121  if (!context->inMethod())
122  {
123  reportException(Error_Translation_guard_guard);
124  }
125 
126  // non-expression form?
127  else if (expression == OREF_NULL)
128  {
129  // handle the ON/OFF status
130  if (guardOn)
131  {
132  context->guardOn();
133  }
134  else
135  {
136  context->guardOff();
137  }
138  }
139  else
140  {
141  // we need to set a guard watch on all variables used in the GUARD expression
142  for (size_t i = 0; i < variableCount; i++)
143  {
144  variables[i]->setGuard(context);
145  }
146 
147  // ok, we set the GUARD state to the the state we want. Then we evaluate
148  // the expression and if false, we wait for a change in guard stage. Every
149  // time we wait, we release the guard lock, and when we wake up again, we
150  // return to our pre-wait state.
151  if (guardOn)
152  {
153  context->guardOn();
154  }
155  else
156  {
157  context->guardOff();
158  }
159 
160  // initialize the guard SEM
161  context->getActivity()->guardSet();
162  // get the expression result
163  RexxObject *result = expression->evaluate(context, stack);
164  context->traceKeywordResult(GlobalNames::WHEN, result);
165 
166  // do first evaluation without establishing doing any waits
167  if (!result->truthValue(Error_Logical_value_guard))
168  {
169  // ok, we're currently false on the guard expression. We need to loop
170  // until this evaluates to true.
171  do
172  {
173  // clean the expression stack so we don't overrun this doing
174  // multiple checks
175  stack->clear();
176  // now perform the guard wait
177  context->guardWait();
178  // reset our activity guard semaphore.
179  ActivityManager::currentActivity->guardSet();
180  // try the expression again
181  result = expression->evaluate(context, stack);
182  context->traceKeywordResult(GlobalNames::WHEN, result);
183  // and continue until we get a true result
184  } while (!result->truthValue(Error_Logical_value_guard));
185  }
186  // ok, we're good now...remove the variable watch points
187  for (size_t i = 0; i < variableCount; i++)
188  {
189  variables[i]->clearGuard(context);
190  }
191  }
192 }
