57 /**
58  * Construct a statically defined primitive behaviour.
59  * Behaviours are created originally in a table of objects that
60  * are not allocated from object heap memory. There is one
61  * primitive behaviour for every class defined in
62  * PrimitiveClasses.xml. After that, all additional behaviours
63  * are created using copies of one of the primitive behaviours.
64  *
65  * @param newTypenum The primitive type number.
66  * @param operator_methods
67  * The associated operator methods.
68  */
69 RexxBehaviour::RexxBehaviour(ClassTypeCode newTypenum, PCPPM *operator_methods)
70 {
71  // All behaviour objects have a behaviour object too.
72  behaviour = getPrimitiveBehaviour(T_Behaviour);
73  // these are not created via normal means, so we need to hand construct
74  // the header information.
75  header.setObjectSize(sizeof(RexxBehaviour));
76  setClassType(newTypenum);
77  behaviourFlags.reset();
78  methodDictionary = OREF_NULL;
79  operatorMethods = operator_methods;
80  owningClass = OREF_NULL;
81 
82  // if this is an internal class, normalize this so we can
83  // restore this to the correct value if we add additional internal classes.
84  if (newTypenum > T_Last_Exported_Class && newTypenum < T_First_Transient_Class)
85  {
86 
87  behaviourFlags.set(INTERNAL_CLASS);
88  }
89  else if (newTypenum >= T_First_Transient_Class)
90  {
91 
92  behaviourFlags.set(TRANSIENT_CLASS);
93  }
94 }

113 /**
114  * Generalized object marking.
115  *
116  * @param reason The reason for this live marking operation.
117  */
118 void RexxBehaviour::liveGeneral(MarkReason reason)
119 {
120  // special handling if marking during a save image.
121  if (reason == SAVINGIMAGE)
122  {
123  // if non primitive, this will need extra processing during restore.
124  if (isNonPrimitive())
125  {
126  // mark this as needing resolution when restored.
127  setNotResolved();
128  }
129  // even though we re-resolve this on restore, we null this out so what
130  // we create a consistent image build
131  operatorMethods = NULL;
132  }
133  // the other side of the process?
134  else if (reason == RESTORINGIMAGE)
135  {
136  // if we have a non-primitive here on a restore image, we need to fix this up.
137  if (isNonPrimitive())
138  {
139  resolveNonPrimitiveBehaviour();
140  }
141  }
142 
143  memory_mark_general(methodDictionary);
144  memory_mark_general(owningClass);
145 }

207 /**
208  * Copy a behaviour object. This will make copies of all
209  * of the contained tables so that the relevant information
210  * can be changed independently of the original behaviour.
211  *
212  * @return The new behaviour object.
213  */
214 RexxInternalObject *RexxBehaviour::copy()
215 {
216  // first, clone the existing object
217  Protected<RexxBehaviour> newBehaviour = (RexxBehaviour *)clone();
218  // complete the copy process
219  newBehaviour->copyBehaviour();
220  // all copies are non-primitive.
221  newBehaviour->setNonPrimitive();
222  // a copy operation generally means we're subclassing, so revert to the
223  // default operator methods:
224  newBehaviour->operatorMethods = RexxObject::operatorMethods;
225  return newBehaviour;
226 }

258 /**
259  * Define a native kernel method on this behaviour.
260  *
261  * @param name The method name.
262  * @param entryPoint The method entry point
263  * @param arguments The argument definition.
264  *
265  * @return The created method object.
266  */
267 MethodClass *RexxBehaviour::defineMethod(const char *name, PCPPM entryPoint, size_t arguments, const char *entryPointName)
268 {
269  // we're doing this during an image build, so make sure we use the interned string name.
270  RexxString *n = memoryObject.getUpperGlobalName(name);
271  // create a method object using the resolved method pointer.
272  MethodClass *method = new MethodClass(n, CPPCode::resolveExportedMethod(name, entryPoint, arguments, entryPointName));
273  // now add this to the method dictionary, ensuring it is the only method by this name.
274  replaceMethod(n, method);
275  // we need the created method object if adding modifiers after creation.
276  return method;
277 }

498  * Restore a primitive behaviour after an image restore.
499  *
500  * @param saved The behaviour that was stored in the saved image.
501  */
502 void RexxBehaviour::restore(RexxBehaviour * saved)
503 {
504  // set our object type
505  setBehaviour(getPrimitiveBehaviour(T_Behaviour));
506  // fix up the memory management bits, and also turn on
507  // oldspace.
508  setObjectSize(Memory::roundObjectBoundary(sizeof(RexxBehaviour)));
509  setOldSpace();
510 
511  // NOTE: In this situation, we're assigning into the static
512  // behaviour from an oldspace saved version. We don't want to
513  // use setField() to set these right now because memory might
514  // not be completely set up yet.
515 
516  // now pull in the method dictionary from the saved copy.
517  methodDictionary = saved->getMethodDictionary();
518  owningClass = saved->getOwningClass();
519 }

755 /**
756  * Add a full table of methods to a class definition. This is
757  * only used during the initial image build.
758  *
759  * @param newMethods The new methods to add.
760  */
761 RexxObject *RexxBehaviour::defineMethods(StringTable *newMethods)
762 {
763  // loop through the table with an iterator.
764  HashContents::TableIterator iterator = newMethods->iterator();
765 
766  for (; iterator.isAvailable(); iterator.next())
767  {
768  // get the name and the value, then add to this class object
769  RexxString *method_name = (RexxString *)iterator.index();
770  // if this is the Nil object, that's an override. Make it OREF_NULL.
771  MethodClass *method = (MethodClass *)iterator.value();
772  if (method == TheNilObject)
773  {
774  method = OREF_NULL;
775  }
776  // define this method
777  defineMethod(method_name, method);
778  }
779  return OREF_NULL;
780 }
