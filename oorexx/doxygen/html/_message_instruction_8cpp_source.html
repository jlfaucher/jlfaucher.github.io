77 {
78  target = message->target;

80  name = message->messageName;

84  arguments[0] = expression;

145 /**
146  * Execute a message instruction.
147  *
148  * @param context The current execution context.
149  * @param stack The current evaluation stack.
150  */
151 void RexxInstructionMessage::execute(RexxActivation *context, ExpressionStack *stack)
152 {
153  context->traceInstruction(this);
154 
155  // evaluate the target object
156  RexxObject *_target = target->evaluate(context, stack);
157  RexxClass *_super = OREF_NULL;
158 
159  // do we have a superclass override?
160  if (super != OREF_NULL)
161  {
162  // get the superclass target
163  _super = (RexxClass *)super->evaluate(context, stack);
164  // _super an instance of TheClassClass
165  if (!_super->isInstanceOf(TheClassClass))
166  {
167  reportException(Error_Invalid_argument_noclass, "SCOPE", "Class");
168  }
169  // validate the starting scope
170  _target->validateScopeOverride(_super);
171  // we send the message using the stack, which
172  // expects to find the target and the arguments
173  // on the stack, but not the super. We need to
174  // pop this item off after evaluation. Since this
175  // comes either from a variable or the environment, this
176  // is already protected from GC.
177  stack->toss();
178  }
179 
180  // evaluate the arguments first
181  RexxInstruction::evaluateArguments(context, stack, arguments, argumentCount);
182 
183  ProtectedObject result;
184  // issue the send with or without a superclass override
185  if (super == OREF_NULL)
186  {
187  stack->send(name, argumentCount, result);
188  }
189  else
190  {
191  stack->send(name, _super, argumentCount, result);
192  }
193 
194  // for the instruction version, we don't worry about popping
195  // anything off of the evaluation stack...that will be cleared automatically
196  // when we complete.
197 
198  // if this is the double message version, replace the result object
199  // with the target object.
200  if (instructionType == KEYWORD_MESSAGE_DOUBLE)
201  {
202  result = _target;
203  }
204 
205  // if we have a result, trace it and assign it to the variable result.
206  if (!result.isNull())
207  {
208  // trace the message name and result
209  context->traceMessage(name, result);
210  context->setLocalVariable(GlobalNames::RESULT, VARIABLE_RESULT, result);
211  }
212  // for no result, we drop the RESULT variable
213  else
214  {
215  context->dropLocalVariable(GlobalNames::RESULT, VARIABLE_RESULT);
216  }
217 
218  context->pauseInstruction();
219 }
