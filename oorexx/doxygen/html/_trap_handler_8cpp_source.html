106 /**
107  * Determine if a given trap handler can process a given
108  * condition with an ANY condition.
109  *
110  * @param c The condition name.
111  *
112  * @return true if this handler can process the condition, false otherwise.
113  */
114 bool TrapHandler::canHandle(RexxString *c)
115 {
116  // if the trap is CALL ON, there are only a few
117  // conditions we can trap with an ANY
118  if (handler->isType(KEYWORD_CALL_ON) &&
119  (c->strCompare(GlobalNames::SYNTAX) ||
120  c->strCompare(GlobalNames::NOVALUE) ||
121  c->strCompare(GlobalNames::LOSTDIGITS) ||
122  c->strCompare(GlobalNames::NOMETHOD) ||
123  c->strCompare(GlobalNames::NOSTRING)))
124  {
125  return false;
126  }
127  return true;
128 }
