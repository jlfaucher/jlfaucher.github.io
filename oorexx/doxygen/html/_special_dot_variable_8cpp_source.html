118 /**
119  * Evaluate a dot variable symbol.
120  *
121  * @param context The current evaluation context.
122  * @param stack The current evaluation stack.
123  *
124  * @return The dot symbol value.
125  */
126 RexxObject *SpecialDotVariable::evaluate(RexxActivation *context, ExpressionStack *stack )
127 {
128  // evaluate always pushes on the stack.
129  stack->push(variableValue);
130  // trace this if tracing intermediates
131  context->traceSpecialDotVariable(variableName, variableValue);
