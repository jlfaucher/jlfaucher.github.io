51 /**
52  * Get the singleton instance of the local API manager.
53  *
54  * @return The pointer to the created (and initialized) API manager.
55  */
56 LocalAPIManager *LocalAPIManager::getInstance()
57 {
58  Lock lock(messageLock); // make sure we single thread this
59  if (singleInstance == NULL)
60  {
61  // create an intialize this. If this fails, an exception is thrown
62  singleInstance = new LocalAPIManager();
63  // to the process-specific initialization now.
64  singleInstance->initProcess();
65  }
66  else {
67  // if we shut everything down at interpreter termination, reestablish the connections.
68  if (singleInstance->restartRequired)
69  {
70  // resetting this will prevent us from going into conversion loops.
71  singleInstance->restartRequired = false;
72  // we could have been shutdown by a previous interpreter termination.
73  // make sure we have a connection
74  singleInstance->establishServerConnection();
75  }
76  }
77  return singleInstance;
78 }

175  * Perform process-specific client API initialization.
176  */
177 void LocalAPIManager::initProcess()
178 {
179  // we don't need a restart since we're just starting up
180  restartRequired = false;
181 
182  // 1) get our session id and userid information.
183  session = SysProcess::getPid();
184  SysProcess::getUserID(userid);
185 
186  ServiceException *startError = NULL;
187 
188  try
189  {
190  // Initialization steps:
191  // 2) make sure the global environment is started
192  establishServerConnection();
193  } catch (ServiceException *e)
194  {
195  // save this for potential rethrowing after the managers are
196  startError = e;
197  }
198 
199  // 3) initialize the API subsystems
200  registrationManager.initializeLocal(this);
201  macroSpaceManager.initializeLocal(this);
202  queueManager.initializeLocal(this);
203 
204  // if we had an exception trying to connect to rxapi, reraise that exception now.
205  if (startError != NULL)
206  {
207  throw startError;
208  }

212 /**
213  * Ensure that the daemon process managing the API information is initialized and running
214  * before doing any API related activities.
215  */
216 void LocalAPIManager::establishServerConnection()
217 {
218  if (!connectionEstablished)
219  {
220  ClientMessage message(APIManager, CONNECTION_ACTIVE);
221  try
222  {
223  // try to establish a connection to the api server. If this returns without an exception,
224  // we're good to go.
225  message.send();
226 
227  if (message.parameter1 != REXXAPI_VERSION)
228  {
229  throw new ServiceException(API_FAILURE, "Open Object REXX version conflict. Incorrect version of RxAPI server detected");
230  }
231  connectionEstablished = true;
232  return;
233  }
234  catch (ServiceException *e)
235  {
236  // just fall through, but get rid of the exception object first.
237  delete e;
238  }
239 
240  // Unable to connect, so try to start the server process. We're unsynchronized at this point,
241  // so it is possible multiple processes will launch this at the same time. Only one will bind to
242  // the listening port, the others will silently fail.
243  SysLocalAPIManager::startServerProcess();
244 
245  // give this a little time to get launched before trying to connect.
246  SysThread::sleep(50);
247 
248  // now loop multiple times, with a sleep interval, until we finally connect.
249  int count = 200;
250  while (count-- > 0)
251  {
252  try
253  {
254  // try to establish a connection to the api server. If this returns without an exception,
255  // we're good to go.
256  message.send();
257 
258  if (message.parameter1 != REXXAPI_VERSION)
259  {
260  throw new ServiceException(API_FAILURE, "Open Object REXX version conflict. Incorrect version of RxAPI server detected");
261  }
262  connectionEstablished = true;
263  return;
264  }
265  catch (ServiceException *e)
266  {
267  // just fall through.
268  delete e;
269  }
270 
271  // use a minimal sleep time before trying again.
272  // if the server is truly casters up, this could result in a full
273  // second delay on every launch, but that really should not happen.
274  SysThread::sleep(10);
275  }
276  throw new ServiceException(API_FAILURE, "Object REXX API Manager could not be started. This could be due to a version conflict!");
277  }
278 }

305  * Return a connection after use.
306  *
307  * @param connection The returned connection.
308  */
309 void LocalAPIManager::returnConnection(ApiConnection *connection)
310 {
311  // if we've encountered an error, then just delete the connection
312  if (!connection->isClean())
313  {
314  delete connection;
315  return;
316  }
317 
318  {
319  Lock lock(messageLock); // make sure we single thread this
320  if (connections.size() < MAX_CONNECTIONS)
321  {
322  connections.push_back(connection);
323  return;
324  }
325  }
326  // not cachable, make sure this is delete.
327  delete connection;
328 }
