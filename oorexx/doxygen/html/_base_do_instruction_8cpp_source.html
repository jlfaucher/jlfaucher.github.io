102 /**
103  * Common method for handling debug pauses at block boundaries.
104  *
105  * @param context The current activation context.
106  * @param doblock The active deblock (if any).
107  */
108 void RexxBlockInstruction::handleDebugPause(RexxActivation *context, DoBlock *doblock)
109 {
110  // do blocks will only do the debug pause on first execution, so
111  // the context needs to determine if we're at a good point. If this is a
112  // re-execute option, then we need to redo everything from the start.
113  if (context->conditionalPauseInstruction())
114  {
115  // if we have a doblock for this, do termination cleanup
116  if (doblock != OREF_NULL)
117  {
118  this->terminate(context, doblock);
119  }
120  // no block required, just remove the nesting
121  else
122  {
123  context->removeBlockInstruction();
124  }
125  // this makes us the next instruction to be executed
126  context->setNext(this);
127  }
128 }

167 /**
168  * Verify that the name on an END instructon and
169  * the instruction label match.
170  *
171  * @param _end The candidate end statement.
172  * @param parser The LanguageParser context (used for error reporting).
173  */
174 void RexxBaseBlockInstruction::matchLabel(RexxInstructionEnd *_end, LanguageParser *parser)
175 {
176  // get the END name.
177  RexxString *name = _end->endName();
178 
179  // if there a name on the END? If no name, we
180  // will always match up ok.
181  if (name != OREF_NULL)
182  {
183 
184  // get the location for error reporting
185  SourceLocation location = _end->getLocation();
186  // and we include our line number so they know which two entities don't match.
187  size_t lineNum = getLineNumber();
188  // now get my label for comparisons.
189  RexxString *myLabel = getLabel();
190 
191  // if we don't have a label, then the END cannot
192  if (myLabel == OREF_NULL)
193  {
194  // report the error
195  parser->error(Error_Unexpected_end_nocontrol, location, new_array(name, new_integer(lineNum)));
196  }
197  // we both have names, but they mismatch.
198  // NOTE: We deal with interned names here, so a pointer comparison will suffice for identity.
199  else if (name != myLabel)
200  {
201  parser->error(Error_Unexpected_end_control, location, new_array(name, myLabel, new_integer(lineNum)));
202  }
203  }
204 }

262 /**
263  * Base execute method for a DO instruction. This performs all
264  * common execution steps and delegates the continue tests to
265  * the subclasses
266  *
267  * @param context The current execution context.
268  * @param stack The current evaluation stack.
269  */
270 void RexxInstructionBaseLoop::execute(RexxActivation *context, ExpressionStack *stack)
271 {
272  // trace on entry
273  context->traceInstruction(this);
274 
275  // all we do here is create a new doblock and make it active
276  Protected<DoBlock> doblock = new DoBlock(context, this);
277 
278  // perform loop specific initialization before pushing the doblock
279  setup(context, stack, doblock);
280 
281  // set the block to the top of the context stack.
282  context->newBlockInstruction(doblock);
283 
284  // update the iteration counters
285  doblock->newIteration();
286  // now perform the initial iteration checks
287  if (!iterate(context, stack, doblock, true))
288  {
289  // nothing to process, terminate the loop now
290  terminate(context, doblock);
291  }
292  else
293  {
294  // we only set the counter variable if we are entering the loop body
295  doblock->setCounter(context);
296  }
297 
298  // handle a debug pause that might cause re-execution
299  handleDebugPause(context, OREF_NULL);
300 }

303 /**
304  * Base re-execute method for do loops. This does all common
305  * logic, then delegates the continue steps to the subclasses.
306  *
307  * @param context The current execution context.
308  * @param stack The current evaluation stack.
309  * @param doblock The doblock associated with this loop instance.
310  */
311 void RexxInstructionBaseLoop::reExecute(RexxActivation *context, ExpressionStack *stack, DoBlock *doblock)
312 {
313  // change control to the top of the loop
314  context->setNext(nextInstruction);
315  context->traceInstruction(this);
316  context->indent();
317 
318  // update the iteration counters
319  doblock->newIteration();
320 
321  // now perform the loop iteration checkes now...if we're good, we just return
322  if (iterate(context, stack, doblock, false))
323  {
324  doblock->setCounter(context);
325  // we're all good.
326  return;
327  }
328 
329  // we need to terminate this loop
330  endLoop(context);
331 }
