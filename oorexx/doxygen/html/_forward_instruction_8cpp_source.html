122 /**
123  * Execute a FORWARD instruction.
124  *
125  * @param context The current execution context.
126  * @param stack The current evaluation stack.
127  */
128 void RexxInstructionForward::execute(RexxActivation *context, ExpressionStack *stack)
129 {
130  context->traceInstruction(this);
131  // only allowed in method contexts.
132  if (!context->inMethod())
133  {
134  reportException(Error_Execution_forward);
135  }
136 
137  // intialize each of the pieces and process what has been specified on the command.
138  // the context figures out what has been overridden from the current context, so
139  // we only set the pieces we've been given
140  RexxObject *_target = OREF_NULL;
141  RexxObject *_message = OREF_NULL;
142  RexxClass *_superClass = OREF_NULL;
143  RexxObject **_arguments = OREF_NULL;
144  size_t count = 0;
145 
146  // sent to a different object target?
147  if (target != OREF_NULL)
148  {
149  _target = target->evaluate(context, stack);
150  context->traceKeywordResult(GlobalNames::TO, _target);
151  }
152 
153  // sending a different message?
154  if (message != OREF_NULL)
155  {
156  // we need this as a string value...and in upper case.
157  RexxObject *temp = message->evaluate(context, stack);
158  // needs tracing too
159  context->traceKeywordResult(GlobalNames::MESSAGE, temp);
160  _message = temp->requestString()->upper();
161  // leave this on the stack.
162  stack->push(_message);
163  }
164 
165  // have a superclass override?
166  if (superClass != OREF_NULL)
167  {
168  _superClass = (RexxClass *)superClass->evaluate(context, stack);
169  if (_superClass != OREF_NULL )
170  {
171  if (!_superClass->isInstanceOf(TheClassClass))
172  {
173  reportException(Error_Invalid_argument_noclass, "SCOPE", "Class");
174  }
175  }
176  context->traceKeywordResult(GlobalNames::CLASS, _superClass);
177  }
178 
179  // overriding the arguments?
180  if (arguments != OREF_NULL)
181  {
182  // we need to evaluate this argument, then get as an array
183  RexxObject *temp = arguments->evaluate(context, stack);
184  context->traceKeywordResult(GlobalNames::ARGUMENTS, temp);
185  ArrayClass *argArray = temp->requestArray();
186  // protect this on the stack too
187  stack->push(argArray);
188  // make sure we got an acceptable array back.
189  if (argArray == TheNilObject || argArray->isMultiDimensional())
190  {
191  reportException(Error_Execution_forward_arguments);
192  }
193  // get the size...
194  count = argArray->size();
195  // now we need to find the last real argument in this array
196  if (count != 0 && argArray->get(count) == OREF_NULL)
197  {
198  count--;
199  while (count > 0)
200  {
201  // done if we find a non-null argument
202  if (argArray->get(count) != OREF_NULL)
203  {
204  break;
205  }
206  count--;
207  }
208  }
209  // point to the data in the argument array.
210  _arguments = argArray->messageArgs();
211  }
212 
213  // have we been overridden via the ARRAY keyword? We only have
214  // one of ARGUMENTS or ARRAY.
215  if (array != OREF_NULL)
216  {
217  // this is an array of expressions, so we need to evaluate all of these
218  count = array->size();
219  for (size_t i = 1; i <= count; i++)
220  {
221  RexxObject *argElement = (RexxObject *)array->get(i);
222  // a real argument? */
223  if (argElement != OREF_NULL)
224  {
225  // trace each of these as arguments
226  RexxObject *arg = argElement->evaluate(context, stack);
227  // trace each of these as arguments
228  context->traceArgument(arg);
229  }
230  else
231  {
232  // just push a null reference for the missing ones
233  stack->push(OREF_NULL);
234  context->traceArgument(GlobalNames::NULLSTRING);
235  }
236  }
237  context->traceKeywordResult(GlobalNames::ARRAY, array);
238  // note that we evaluated this one last, so that other
239  // values pushed on the stack will not interfere with the arguments.
240  _arguments = stack->arguments(count);
241  }
242  // now have the context forward this
243  ProtectedObject result = context->forward(_target, (RexxString *)_message, _superClass, _arguments, count, continueExecution);
244  // if we continued, then we may need to set the RESULT variable in the current context.
245  if (continueExecution)
246  {
247  // if we have a result, then we need to trace this and set the result variable
248  if (!result.isNull())
249  {
250  context->traceResult(result);
251  context->setLocalVariable(GlobalNames::RESULT, VARIABLE_RESULT, result);
252  }
253  // ne result returned, so we drop the RESULT variable
254  else
255  {
256  context->dropLocalVariable(GlobalNames::RESULT, VARIABLE_RESULT);
257  }
258  // and finally, pause
259  context->pauseInstruction();
260  }
261 }
