50 /**
51  * String method for performing a BITAND operation.
52  *
53  * @param string2 The string to AND with (optional)
54  * @param pad The optional pad character.
55  *
56  * @return The ANDed string.
57  */
58 RexxString *RexxString::bitAnd(RexxString *string2, RexxString *pad)
59 {
60  // the other string is optional...the operation can be performed using just
61  // pad characters.
62  string2 = optionalStringArgument(string2, GlobalNames::NULLSTRING, ARG_ONE);
63 
64  size_t string2Len = string2->getLength();
65  // get the pad character...the default is 'ff'x.
66  char padChar = optionalPadArgument(pad, (char)0xff, ARG_TWO);
67 
68  size_t string1Len = getLength();
69 
70  size_t minLength;
71  size_t maxLength;
72  const char *padString;
73  const char *source;
74 
75  // we adjust for whichever string is longer
76  if (string1Len <= string2Len)
77  {
78  minLength = string1Len;
79  maxLength = string2Len;
80  padString = string2->getStringData();
81  source = getStringData();
82  }
83  else
84  {
85  minLength = string2Len;
86  maxLength = string1Len;
87  padString = getStringData();
88  source = string2->getStringData();
89  }
90 
91  // calculate the length of padding required
92  size_t padLength = maxLength - minLength;
93 
94  // get a result string the length of the longer string.
95  RexxString *retval = raw_string(maxLength);
96  char *target = retval->getWritableData();
97 
98  // copy in the longer string
99  memcpy(target, padString, maxLength);
100 
101  // now peform the AND operation between the two strings
102 // avoid warning: writing 1 byte into a region of size 0 [-Wstringop-overflow=]
103 // (or similar) in gcc 12 and above, due to our RexxString char stringData[4]
104 #ifdef HAVE_PRAGMA_GCC_STRINGOPOVERFLOW
105 #pragma GCC diagnostic push
106 #pragma GCC diagnostic ignored "-Wstringop-overflow"
107 #endif
108  while (minLength--)
109  {
110  *target = *target & *source++;
111  target++;
112  }
113 
114  // now do the end part with the pad character
115  while (padLength--)
116  {
117  *target = *target & padChar;
118  target++;
119  }
120 #ifdef HAVE_PRAGMA_GCC_STRINGOPOVERFLOW
121 #pragma GCC diagnostic pop
122 #endif
123 
124  return retval;
125 }
