194 // a convenience macro for initializing instruction/expression objects.
195 // for many of the instructions, they have a variable size array
196 // that are initialized in reverse order using items pulled from a
197 // provided queue. This simplifies this initialization process and
198 // ensures it is done correctly. NOTE: This decrements the count
199 // variable so make sure you use the one from the argument list!!!!!!
200 
201 #define initializeObjectArray(count, array, type, queue) \
202 { \
203  while (count > 0) \
204  { \
205  array[--count] = (type *)queue->pop(); \
206  } \
207 }
