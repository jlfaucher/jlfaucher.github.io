184 /**
185  *
186  * Create a supplier for a directory, including the results of all
187  * of the SETMETHOD methods as values
188  *
189  * @return An appropriate supplier object.
190  */
191 SupplierClass *DirectoryClass::supplier()
192 {
193  // get the supplier for the base collection.
194  Protected<SupplierClass> supplier = contents->supplier();
195 
196  // do we have a method table? We need to include these also, which
197  // requires running each of the methods to obtain the value.
198  if (methodTable != OREF_NULL)
199  {
200  Protected<ArrayClass> indexes = new_array(methodTable->items());
201  Protected<ArrayClass> values = new_array(methodTable->items());
202 
203  HashContents::TableIterator iterator = methodTable->iterator();
204 
205  for (; iterator.isAvailable(); iterator.next())
206  {
207  RexxString *name = (RexxString *)iterator.index();
208  MethodClass *method = (MethodClass *)iterator.value();
209 
210  ProtectedObject v;
211  // run the method, using the directory as the receiver
212  method->run(ActivityManager::currentActivity, this, name, NULL, 0, v);
213 
214  indexes->append(name);
215  values->append(v);
216  }
217  // append the method table part to the existing supplier
218  supplier->append(values, indexes);
219  }
220  return supplier;
221 }

309 {

311  RexxInternalObject *oldVal = get(entryName);
312

325 /**
326  * Retrieve an item from a directory.
327  *
328  * @param _index The target index.
329  *
330  * @return The result value, if available.
331  */
332 RexxInternalObject *DirectoryClass::get(RexxInternalObject *index)
333 {
334  // first from the contents
335  RexxInternalObject *result = contents->get(index);
336  if (result == OREF_NULL)
337  {
338  // then try the method table
339  result = methodTableValue(index);
340  // now try the unknown method if we still don't have anything.
341  if (result == OREF_NULL)
342  {
343  result = unknownValue(index);
344  }
345  }
346  return result;
347 }

385 /**
386  * Retrieve an index for a given item. Which index is returned
387  * is indeterminate.
388  *
389  * @param target The target object.
390  *
391  * @return The index for the target object NULL
392  */
393 RexxInternalObject *DirectoryClass::getIndex(RexxInternalObject *target)
394 {
395  // retrieve this from the hash table
396  RexxInternalObject *result = contents->getIndex(target);
397  // not found, check the other tables
398  if (result == OREF_NULL)
399  {
400  // rats, we might need to do this the hard way
401  if (methodTable != OREF_NULL)
402  {
403  // This is a serious pain. We need to run the method table
404  // calling all of the methods to see if we get a return value
405  // that matches our target object.
406 
407  HashContents::TableIterator iterator = methodTable->iterator();
408 
409  for (; iterator.isAvailable(); iterator.next())
410  {
411  // we need to run each method, looking for a value that matches
412  RexxString *name = (RexxString *)iterator.index();
413  MethodClass *method = (MethodClass *)iterator.value();
414  ProtectedObject v;
415  method->run(ActivityManager::currentActivity, this, name, NULL, 0, v);
416  // got a match?
417  if (target->equalValue(v))
418  {
419  return name; // the name is the index
420  }
421  }
422  }
423  }
424  return result;
425 }

472 /**
473  * Set a method of executable code into a directory object.
474  *
475  * @param entryname The name this is set under
476  * @param methodobj The associated method object.
477  *
478  * @return Nothing.
479  */
480 RexxInternalObject *DirectoryClass::setMethodRexx(RexxString *name, MethodClass *methodobj)
481 {
482  Protected<RexxString> entryname = stringArgument(name, "index")->upper();
483  if (methodobj != OREF_NULL)
484  {
485  // make sure we have a method object for this. The scope is .nil to indicate object scope.
486  Protected<MethodClass> method = MethodClass::newMethodObject(entryname, methodobj, (RexxClass *)TheNilObject, "method");
487 
488  // the unknown method? We keep that in a special place
489  if (entryname->strCompare(GlobalNames::UNKNOWN))
490  {
491  setField(unknownMethod, method);
492  }
493  else
494  {
495  // create the table if this is the first addition
496  if (methodTable == OREF_NULL)
497  {
498  setField(methodTable, new_string_table());
499  }
500  // and add the method to the table
501  methodTable->put(method, entryname);
502  }
503  }
504  // no method object given, this is a removal
505  else
506  {
507  // if unknown, remove this from the special place.
508  if (entryname->strCompare(GlobalNames::UNKNOWN))
509  {
510  clearField(unknownMethod);
511  }
512  else
513  {
514  // remove from the method table if we have one
515  if (methodTable != OREF_NULL)
516  {
517  methodTable->remove(entryname);
518  }
519  }
520  }
521 
522  // since setting a method will also override any contents, we need to
523  // ensure the contents don't have this either.
524  contents->remove(entryname);
525  return OREF_NULL;
526 }
