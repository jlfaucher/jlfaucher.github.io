89 /**
90  * Initialize the free chains, either at construction time or
91  * after an empty() operation.
92  */
93 void ListContents::initializeFreeChain()
94 {
95 
96  // this is an empty bucket
97  itemCount = 0;
98 
99  // we keep the available items on a chain, so
100  // chain up the items into a free chain
101  freeChain = 0;
102 
103  for (ItemLink i = freeChain; i < totalSize; i++)
104  {
105  entries[i].next = i + 1;
106  }
107  // make sure the last item ends the chain. Note, we
108  // don't bother double linking the free chain since we
109  // only remove from the front.
110  entries[totalSize - 1].next = NoMore;
111 }

114 /**
115  * We're merging a list contents into this collection after an
116  * expansion. We need to take care to keep the same index
117  * values for each of the inserted items. We cleare out all
118  * slot positions, then rebuild the chains using the same
119  * positions as the original. Once the merge is complete, then
120  * we rescan the table for free entries and rebuild the table.
121  */
122 void ListContents::prepareForMerge()
123 {
124  // no first or last items.
125  firstItem = NoMore;
126  lastItem = NoMore;
127 
128  // this is an empty bucket
129  itemCount = 0;
130 
131  // we keep the available items on a chain, so
132  // chain up the items into a free chain
133  freeChain = NoMore;
134 
135  // now clear all of the entries out
136  for (ItemLink i = 0; i < totalSize; i++)
137  {
138  clearEntry(i);
139  }
140 }
141 
142 
143 /**
144  * We've completed the merge operation. Now we need to up the
145  * unused slots and build a free chain.
146  */
147 void ListContents::completeMerge()
148 {
149  // we keep the available items on a chain, so
150  // chain up the items into a free chain
151  freeChain = NoMore;
152 
153  // ok, scan from the end of the table to the beginning, so we
154  // build the free chain with the smallest indexes in the front
155  for (ItemLink i = totalSize; i > 0; i--)
156  {
157  ItemLink index = i - 1;
158 
159  // if this slot is empty
160  if (isAvailable(index))
161  {
162  // put this back on the free chain
163  returnToFreeChain(index);
164  }
165  }
166 }

218 /**
219  * Merge the list maintained in this contents object into
220  * a target one after an expansion has occurred. Note that this
221  * is a little complicated, because we need to maintain the
222  * relationship between items are their current indexes.
223  *
224  * @param target The target contents.
225  */
226 void ListContents::mergeInto(ListContents *target)
227 {
228  // NOTE: This assumes the target contents item is at least
229  // as large as this one.
230 
231  // tell the target we're going to do a merge
232  target->prepareForMerge();
233 
234  // run the chain appending each item on to the target
235  for (ItemLink position = firstItem; position != NoMore; position = nextEntry(position))
236  {
237  // this version of append will use the same index value for the append.
238  target->append(position, entryValue(position));
239  }
240 
241  // now the target needs to rebuild the free chains from the unused
242  // slots.
243  target->completeMerge();
244 }

247 /**
248  * Allocate a slot for a new value and fill it in.
249  *
250  * @param value The value we're inserting.
251  *
252  * @return The index position for the inserted item.
253  */
254 ListContents::ItemLink ListContents::allocateSlot(RexxInternalObject *value)
255 {
256  // this is a nice central place to handle bumping the count
257  itemCount++;
258 
259  // get an item off of the chain and set the value
260  ItemLink newItem = freeChain;
261 
262  // belt-and-braces...this should not occur...but give a logic
263  // error if it does occur, since that indicates something bad has
264  // occurred.
265  if (newItem == NoMore)
266  {
267  Interpreter::logicError("Attempt to add an object to a full List contents");
268  }
269 
270  freeChain = nextEntry(newItem);
271 
272  setValue(newItem, value);
273  clearLinks(newItem);
274  return newItem;
275 }

278 /**
279  * Insert an item at the end of the existing list.
280  *
281  * @param newItem The new item we're inserting.
282  */
283 void ListContents::insertAtEnd(ListContents::ItemLink newItem)
284 {
285  // first insertion into this list?
286  if (lastItem == NoMore)
287  {
288  // make this the first and last, and leave all of its links
289  // as terminators
290  firstItem = newItem;
291  lastItem = newItem;
292  }
293  // insert after this item
294  else
295  {
296  insertAfter(newItem, lastItem);
297  }
298 }

324 /**
325  * Insert an item after a given list item.
326  *
327  * @param newItem The new item we're inserting.
328  * @param insertItem The item this is inserted after.
329  */
330 void ListContents::insertAfter(ListContents::ItemLink newItem, ListContents::ItemLink insertItem)
331 {
332  // the new item gets the next item of our predecessor
333  setNext(newItem, nextEntry(insertItem));
334  // set the next and previous of these two to point at each other
335  setNext(insertItem, newItem);
336  setPrevious(newItem, insertItem);
337 
338  // do we have a following item?...if not, we're the new last item
339  if (nextEntry(newItem) == NoMore)
340  {
341  lastItem = newItem;
342  }
343  // need to update the item after us to point back to us
344  else
345  {
346  setPrevious(nextEntry(newItem), newItem);
347  }
348 }

366  if (previousEntry(newItem) == NoMore)
367  {

370  // need to update the item after us to point back to us

378 /**
379  * Add a value to the list at a given index position.
380  *
381  * @param value The value to add.
382  * @param index The target index postion. NoMore indicates add to the end.
383  *
384  * @return The index position of the new item.
385  */
386 ListContents::ItemLink ListContents::insert(RexxInternalObject *value, ListContents::ItemLink index)
387 {
388  ItemLink newItem = allocateSlot(value);
389  // if we got a .nil index for the insertion, this is at the beginning.
390  if (index == AtEnd)
391  {
392  insertAtEnd(newItem);
393  }
394  else if (index == AtBeginning)
395  {
396  insertAtFront(newItem);
397  }
398  else
399  {
400  // this is inserted after the index position
401  insertAfter(newItem, index);
402  }
403 
404  // return the new index
405  return newItem;
406 }

478 /**
479  * Remove an item postion from the chain and return
480  * the location to the free chain.
481  *
482  * @param item The item position to remove.
483  */
484 void ListContents::removeItem(ListContents::ItemLink item)
485 {
486  // we have one fewer item now.
487  itemCount--;
488 
489  // handle being just the only item first, since it
490  // simplifies some things by removing that possibility first
491  if (item == firstItem)
492  {
493  // first and last, this is easy.
494  if (item == lastItem)
495  {
496  firstItem = NoMore;
497  lastItem = NoMore;
498  }
499  // removing the first item, promote our
500  // successor to the front
501  else
502  {
503  firstItem = nextEntry(item);
504  setPrevious(firstItem, NoMore);
505  }
506 
507  }
508  // not the first, could be the last
509  else if (item == lastItem)
510  {
511  lastItem = previousEntry(item);
512  setNext(lastItem, NoMore);
513  }
514  // have both a previous and next,
515  else
516  {
517  // update the next item first
518  setPrevious(nextEntry(item), previousEntry(item));
519  // and the reverse for our previous item
520  setNext(previousEntry(item), nextEntry(item));
521  }
522 
523  // put this back on the free chain
524  returnToFreeChain(item);
525 }

703 /**
704  * Empty the list of all contents.
705  */
706 void ListContents::empty()
707 {
708  // clear all of the entries so we handle old-to-new properly.
709  for (ItemLink position = firstItem; position != NoMore;)
710  {
711  // get the next link before clearing
712  ItemLink next = nextEntry(position);
713  clearEntry(position);
714  position = next;
715  }
716 
717  itemCount = 0;
718  firstItem = NoMore;
719  lastItem = NoMore;
720 
721  // reset the free chains
722  initializeFreeChain();
723 }

817 /**
818  * Not really part of normal list operation, but classes
819  * maintain their subclasses as a list of weak references
820  * so that subclasses can get garbage collected when no
821  * longer needed. When the class needs to iterate over
822  * the subclasses, it needs to resolve which weak references
823  * are still valid. This method will check the weak
824  * references and remove any stale entries. Then it
825  * will return an array of the dereferenced objects.
826  *
827  * @return An array of the dereferenced objects.
828  */
829 ArrayClass *ListContents::weakReferenceArray()
830 {
831  // this is a little tricky. We allocate the result array
832  // before we process the weak references. If we prune
833  // the stale references first and then allocate an array,
834  // we might hit a GC window that could create more stale
835  // references.
836 
837  // make this large enough to hold what is currently there.
838  // this could be more than we need, but that's fine. The initial size is zero.
839  Protected<ArrayClass> result = new (0, items()) ArrayClass;
840 
841  ItemLink position = firstItem;
