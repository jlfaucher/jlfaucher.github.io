514 /******************************************************************************/
515 /* various expression terminator sets */
516 /******************************************************************************/
517 #define TERM_EOC 0x00000001u // terminate on end of clause
518 #define TERM_RIGHT 0x00000002u // terminate on left paren
519 #define TERM_SQRIGHT 0x00000004u // terminate on left square bracket
520 #define TERM_TO 0x00000008u // terminate on TO keyword
521 #define TERM_BY 0x00000010u // terminate on BY keyword
522 #define TERM_FOR 0x00000020u // terminate on FOR keyword
523 #define TERM_WHILE 0x00000040u // terminate on WHILE/UNTIL keywords
524 #define TERM_WITH 0x00000100u // terminate on WITH keyword
525 #define TERM_THEN 0x00000200u // terminate on THEN keyword
526 #define TERM_KEYWORD 0x10000000u // perform keyword terminator checks
527  // terminate on controlled DO keywords
528 #define TERM_CONTROL (TERM_KEYWORD | TERM_TO | TERM_BY | TERM_FOR | TERM_WHILE)
529  // terminate on DO conditionals
530 #define TERM_COND (TERM_KEYWORD | TERM_WHILE)
531  // terminators for an OVER condition
532 #define TERM_OVER (TERM_KEYWORD | TERM_FOR | TERM_WHILE)
533  // terminate on a THEN keyword
534 #define TERM_IF (TERM_KEYWORD | TERM_THEN)
