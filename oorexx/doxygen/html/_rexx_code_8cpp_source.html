162 /**
163  * Call a unit of Rexx code as a routine or top-level call.
164  *
165  * @param activity The activity this is running on.
166  * @param routine The routine object this code is attached to.
167  * @param routineName The name of the call.
168  * @param argPtr Pointer to the call arguments.
169  * @param argcount The number of arguments being passed.
170  * @param calltype The type of call (used in parse source)
171  * @param environment
172  * The initial address environment.
173  * @param context The call context indicator (program, routine, internalcall, etc.)
174  * @param result A protected object for passing the return
175  * value back.
176  */
177 void RexxCode::call(Activity *activity, RoutineClass *routine, RexxString *routineName,
178  RexxObject**argPtr, size_t argcount, RexxString *calltype, RexxString *environment,
179  ActivationContext context, ProtectedObject &result)
180 {
181  // check the stack space before proceeding
182  activity->checkStackSpace();
183  // create a new activity to run this and add to the stack.
184  RexxActivation *newacta = ActivityManager::newActivation(activity, routine, this, calltype, environment, context);
185  activity->pushStackFrame(newacta);
186  // have the activation run this code. The return result is passed back through result.
187  newacta->run(OREF_NULL, routineName, argPtr, argcount, OREF_NULL, result);
188 }

193  *
194  * @param activity The activity this is running on.

196  * @param receiver The target object for the message.

204  RexxString *msgname, RexxObject**argPtr, size_t argcount, ProtectedObject &result)
205 {
206  // create a new activation object and push it on the top of the stack.

209  // run the method. The result is returned via the ProtectedObject reference.
210  newacta->run(receiver, msgname, argPtr, argcount, OREF_NULL, result);

267 /**
268  * Add an instruction to the end of this code object.
269  *
270  * @param i The instruction to add
271  * @param m The maximum stack size required by this instruction.
272  * @param v The size of the variables frame.
273  */
274 void RexxCode::addInstruction(RexxInstruction *i, size_t m, size_t v)
275 {
276  // first instruction? just set the chain head
277  if (start == OREF_NULL)
278  {
279  start = i;
280  // we have a single instruction, so that is the bounds of the source
281  location.setLocation(i->getLocation());
282  }
283  // run the chain and add this to the end
284  else
285  {
286  RexxInstruction *current = start;
287  while (!current->isLast())
288  {
289  current = current->next();
290  }
291 
292  current->setNext(i);
293  // update the bounds of the source location.
294  location.setLocation(i->getLocation());
295  }
296 
297  // the max stack needs to be the largest value required by any of the
298  // instructions. This is managed by the parser as the constants are
299  // processed, so just take the latest value.
300  maxStack = m + MINIMUM_STACK_FRAME;
301  // we also need to variable dictionary frame size.
302  vdictSize = v;
303 }
