130 /**
131  * Execute a PARSE instruction.
132  *
133  * @param context The current execution context.
134  * @param stack The current evaluation stack.
135  */
136 void RexxInstructionParse::execute(RexxActivation *context, ExpressionStack *stack )
137 {
138  // trace if necessary
139  context->traceInstruction(this);
140 
141  // this is our parsing target, which holds the current parsing information.
142  RexxTarget target;
143  // we only have multiple argument lists if this is PARSE ARG
144  bool multiple = false;
145 
146  // default different pieces of source data information
147  RexxObject *value = GlobalNames::NULLSTRING;
148  RexxObject **argList = OREF_NULL;
149  size_t argCount = 0;
150 
151  switch (stringSource)
152  {
153  // PULL or PARSE PULL instruction
154  case SUBKEY_PULL:
155  // pull a line from the queue and push on the stack for safekeeping
156  value = context->pullInput();
157  stack->push(value);
158  context->traceKeywordResult(GlobalNames::PULL, value);
159  break;
160 
161  // PARSE LINEIN
162  case SUBKEY_LINEIN:
163  // read a line from the console stream
164  value = context->lineIn();
165  stack->push(value);
166  context->traceKeywordResult(GlobalNames::LINEIN, value);
167  break;
168 
169  // ARG OR PARSE ARG
170  case SUBKEY_ARG:
171  // we have multiple values
172  multiple = true;
173  // get the argument list information
174  argList = context->getMethodArgumentList();
175  argCount = context->getMethodArgumentCount();
176  break;
177 
178  // PARSE SOURCE
179  case SUBKEY_SOURCE:
180  // we're using the source string
181  value = context->sourceString();
182  stack->push(value);
183  context->traceKeywordResult(GlobalNames::SOURCE, value);
184  break;
185 
186  // PARSE VERSION
187  case SUBKEY_VERSION:
188  // get the version string
189  value = Interpreter::getVersionString();
190  stack->push(value);
191  context->traceKeywordResult(GlobalNames::VERSION_STRING, value);
192  break;
193 
194  // PARSE VAR
195  case SUBKEY_VAR:
196  // our expression is a variable retriever which we can
197  // evaluate to get the value.
198  value = expression->evaluate(context, stack);
199  context->traceKeywordResult(GlobalNames::VAR, value);
200  // Evaluate always pushes the object on to the evaluation stack.
201  // We don't need to do anthing extra to protect this.
202  break;
203 
204  // PARSE VALUE expr WITH
205  case SUBKEY_VALUE:
206  // we should have an expression, but if not, use a NULLSTRING
207  if (expression != OREF_NULL)
208  {
209  value = expression->evaluate(context, stack);
210  }
211  // PARSE VALUE WITH template actually is legal. Not useful, but legal :-)
212  else
213  {
214  value = GlobalNames::NULLSTRING;
215  }
216  context->traceKeywordResult(GlobalNames::VALUE, value);
217  // the expression version is still on the evalation stack.
218  break;
219 
220  // invalid PARSE subkey (should really never happen)
221  default:
222  reportException(Error_Interpretation_switch, "PULL/PARSE subkey", stringSource);
223  break;
224  }
225 
226  // create the parse target
227  target.init(value, argList, argCount, parseFlags, multiple, context, stack);
228 
229  // now loop through the triggers, have each perform its configured operation.
230  for (size_t i = 0; i < triggerCount; i++)
231  {
232  // a NULL trigger marks the boundary between comman delimited template
233  // sections. For PARSE ARG, this will advance to the next argument. For
234  // all other sources, this will just use "" as the string value.
235  ParseTrigger *trigger = triggers[i];
236  if (trigger == OREF_NULL)
237  {
238  target.next(context);
239  }
240  else
241  {
242  // each trigger handles both movement and variable assignment.
243  trigger->parse(context, stack, &target);
244  }
245  }
246 
247  // and the ubiquitous debug pause.
248  context->pauseInstruction();
249 }
