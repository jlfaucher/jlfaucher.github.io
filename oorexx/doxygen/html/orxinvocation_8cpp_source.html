382 RexxMethod3(RexxObjectPtr, callInstanceProgram,
383  CSELF, self,
384  CSTRING, program,
385  OPTIONAL_RexxArrayObject, args)
386 {
387  InstanceInfo *instanceInfo = (InstanceInfo *)self;
388  instanceInfo->programName = program;
389 
390  if (args == NULLOBJECT)
391  {
392  instanceInfo->argCount = 0;
393  }
394  else
395  {
396  instanceInfo->argCount = context->ArraySize(args);
397  for (size_t i = 0; i < instanceInfo->argCount; i++)
398  {
399  RexxObjectPtr arg = context->ArrayAt(args, i + 1);
400  if (arg != NULLOBJECT)
401  {
402  instanceInfo->arguments[i] = context->CString(arg);
403  }
404  else
405  {
406  instanceInfo->arguments[i] = NULL;
407  }
408  }
409  }
410 
411  invokeProgram(instanceInfo);
412  if (instanceInfo->rc != 0)
413  {
414  return context->Nil();
415  }
416  else
417  {
418  return context->String(instanceInfo->returnResult);
419  }
420 }
