120 // a thread that reads ERROR or OUTPUT data from the command pipe
121 class ReaderThread : public SysThread
122 {
123 
124 public:
125  inline ReaderThread() : SysThread(),
126  pipe(0), pipeBuffer(NULL), dataLength(0), error(0) { }
127  inline ~ReaderThread()
128  {
129  if (pipeBuffer != NULL)
130  {
131  free(pipeBuffer);
132  pipeBuffer = NULL;
133  }
134  terminate();
135  }
136 
137  void start(HANDLE _pipe)
138  {
139  pipe = _pipe;
140  SysThread::createThread();
141  }
142 
143  virtual void dispatch()
144  {
145  DWORD cntRead;
146  size_t bufferSize = READ_SIZE; // current size of our buffer
147  pipeBuffer = (char *)malloc(READ_SIZE); // allocate an initial buffer
148  if (pipeBuffer == NULL)
149  {
150  // use the Windows error code
151  error = ERROR_NOT_ENOUGH_MEMORY;
152  return;
153  }
154  for (;;)
155  {
156  if (!ReadFile(pipe, pipeBuffer + dataLength, (DWORD)(bufferSize - dataLength), &cntRead, NULL) || cntRead == 0)
157  {
158  break;
159  }
160 
161  dataLength += cntRead;
162  // have we hit the end of the buffer?
163  if (dataLength >= bufferSize)
164  {
165  // increase the buffer by another increment
166  bufferSize += READ_SIZE;
167  char *largerBuffer = (char *)realloc(pipeBuffer, bufferSize);
168  if (largerBuffer == NULL)
169  {
170  // use the Windows error code
171  error = ERROR_NOT_ENOUGH_MEMORY;
172  return;
173  }
174  pipeBuffer = largerBuffer;
175  }
176  }
177  CloseHandle(pipe);
178  }
179 
180  HANDLE pipe; // the pipe we read the data from
181  char *pipeBuffer; // initally errorBuffer = firstBuffer
182  size_t dataLength; // the length of data we've read
183  int error; // and error that resulted.
184 };

213 /* Handle "SET XX=YYY" command in same process */
214 bool sys_process_set(RexxExitContext *context, const char *command, const char * cmd, RexxObjectPtr &rc)
215 {
216  rc = NULLOBJECT;
217  const char * eqsign;
218  const char * st;
219  char name[256];
220  char value[4096];
221  eqsign = strchr(cmd, '=');
222  if (!eqsign)
223  {
224  return false;
225  }
226 
227  st = &cmd[4];
228  while ((*st) && (*st == ' '))
229  {
230  st++;
231  }
232  if (st == eqsign)
233  {
234  return false;
235  }
236  strncpy(name, st, eqsign-st);
237  name[eqsign-st]='\0';
238 
239  if (ExpandEnvironmentStrings(eqsign+1, value, 4095) && SetEnvironmentVariable(name,value))
240  {
241  rc = context->False(); // just return a zero
242  }
243  else
244  {
245  context->RaiseCondition("ERROR", context->String(command), NULLOBJECT, context->WholeNumberToObject(GetLastError()));
246  }
247  return true;
248 }

272 /* Handle "CD XXX" command in same process */
273 bool sys_process_cd(RexxExitContext *context, const char *command, const char * cmd, RexxObjectPtr &res)
274 {
275  const char * st;
276  int rc, error;
277  res = NULLOBJECT;
278 
279  st = &cmd[3];
280  while ((*st) && (*st == ' '))
281  {
282  st++;
283  }
284  if (!*st)
285  {
286  return false;
287  }
288 
289  if ((strlen(st) == 2) && (st[1] == ':'))
290  {
291  int drive = toupper( *st ) - 'A' + 1;
292  if (drive < 1 || drive > 26)
293  {
294  // an out-of-range drive will make _chdrive() MSVC run-time library fail
295  rc = -1;
296  error = 15; // The system cannot find the drive specified.
297  }
298  else
299  {
300  rc = _chdrive(drive);
301  error = GetLastError();
302  }
303 
304  }
305  else
306  {
307  AutoFree unquoted = unquote(st);
308  if (unquoted == NULL)
309  {
310  return false;
311  }
312  rc = _chdir(unquoted);
313  error = GetLastError();
314  }
315  if (rc != 0)
316  {
317  context->RaiseCondition("ERROR", context->String(command), NULLOBJECT, context->WholeNumberToObject(error));
318  }
319  else
320  {
321  res = context->False();
322  }
323 
324  return true;
325 }

328  SECURITY_ATTRIBUTES* sAttr, // pointer to the security attr.
329  CSTRING name, // name of the pipe
330  RexxExitContext* context) // context for errors

361 /*-----------------------------------------------------------------------------
362  | Name: sysCommandNT |
363  | |
364  | Arguments: cmd - Command to be executed |
365  | |
366  | Returned: rc - Return Code |
367  | Note: if CreateProcess fails return GetLastError code |
368  | else return rc from executed command |
369  | |
370  | Notes: Handles processing of a system command on a Windows NT system |
371  | |
372  ----------------------------------------------------------------------------*/
373 bool sysCommandNT(RexxExitContext *context,
374  const char *command,
375  const char *cmdstring_ptr,
376  bool direct,
377  RexxObjectPtr &result,
378  RexxIORedirectorContext *ioContext)
379 {
380  DWORD rc;
381  STARTUPINFO siStartInfo; // process startup info
382  PROCESS_INFORMATION piProcInfo; // returned process info
383  char ctitle[256];
384  DWORD creationFlags;
385  BOOL titleChanged;
386 
387  logical_t redirIn = ioContext->IsInputRedirected(); // Input redirected?
388  logical_t redirOut = ioContext->IsOutputRedirected(); // Output redirected?
389  logical_t redirErr = ioContext->IsErrorRedirected(); // Error redirected?
390  logical_t combo = ioContext->AreOutputAndErrorSameTarget(); // Err/Out combined?
391 
392  InputWriterThread inputThread; // used if need to write input
393  ReaderThread outputThread; // separate thread if we need to read OUTPUT
394  ReaderThread errorThread; // separate thread if we need to read ERROR
395 
396  BOOL readOK = false;
397  // handles for pipes if needed
398  HANDLE inPipeW, outPipeR, errPipeR;
399 
400  // security attributes for pipes
401  SECURITY_ATTRIBUTES secAttr;
402  secAttr.nLength = sizeof(SECURITY_ATTRIBUTES);
403  // Set the bInheritHandle flag so pipe handles are inherited.
404  secAttr.bInheritHandle = TRUE;
405  secAttr.lpSecurityDescriptor = NULL;
406 
407  ZeroMemory(&siStartInfo, sizeof(siStartInfo));
408  ZeroMemory(&piProcInfo, sizeof(piProcInfo));
409  /****************************************************************************/
410  /* Invoke the system command handler to execute the command */
411  /****************************************************************************/
412  siStartInfo.cb = sizeof(siStartInfo);
413  // change the following if redirecting I/O
414  siStartInfo.hStdInput = GetStdHandle(STD_INPUT_HANDLE);
415  siStartInfo.hStdOutput = GetStdHandle(STD_OUTPUT_HANDLE);
416  siStartInfo.hStdError = GetStdHandle(STD_ERROR_HANDLE);
417  titleChanged = GetConsoleTitle(ctitle, 255) != 0;
418  siStartInfo.lpTitle = (LPSTR)cmdstring_ptr;
419  // if creation flag CREATE_NEW_PROCESS_GROUP is specified, CreateProcess() will
420  // implicitly call SetConsoleCtrlHandler(NULL,TRUE) which causes the created
421  // process to ignore Ctrl+C signals, as reported in [bugs:#1328]
422  // (for details see MSDN CreateProcess() and SetConsoleCtrlHandler())
423  creationFlags = GetPriorityClass(GetCurrentProcess()); // | CREATE_NEW_PROCESS_GROUP;
424  if (!siStartInfo.hStdInput && !siStartInfo.hStdOutput && !titleChanged) /* is REXXHIDE running without console */
425  {
426  if (!direct)
427  {
428  siStartInfo.wShowWindow = SHOWWINDOWFLAGS;
429  siStartInfo.dwFlags |= STARTF_USESHOWWINDOW;
430  }
431  else if (SystemInterpreter::explicitConsole)
432  {
433  creationFlags |= CREATE_NEW_CONSOLE; /* new console if CMD ord COMMAND was specified */
434  }
435  }
436  else /* just use standard handles if we are running in a console */
437  {
438  if (direct)
439  {
440  siStartInfo.dwFlags = STARTF_USESTDHANDLES; /* no SW_HIDE for direct commands */
441  }
442  else
443  {
444  siStartInfo.dwFlags = STARTF_USESHOWWINDOW | STARTF_USESTDHANDLES;
445  }
446  siStartInfo.wShowWindow = SHOWWINDOWFLAGS;
447  }
448 
449  if (ioContext->IsRedirectionRequested()) {
450  // create the pipes needed to implement the redirection
451  if (redirIn)
452  {
453  makePipe(&siStartInfo.hStdInput, &inPipeW, &secAttr, "Input", context);
454  }
455 
456  if (redirOut)
457  {
458  makePipe(&outPipeR, &siStartInfo.hStdOutput, &secAttr, "Output", context);
459  }
460 
461  if (redirErr)
462  {
463  if (combo)
464  { // send output and error to the same stream
465  siStartInfo.hStdError = siStartInfo.hStdOutput;
466  redirErr = false;
467  }
468  else
469  {
470  makePipe(&errPipeR, &siStartInfo.hStdError, &secAttr, "Error", context);
471  }
472  }
473  }
474 
475  if (CreateProcess(NULL, // address of module name
476  (LPSTR)cmdstring_ptr,// address of command line
477  NULL, // address of process security attrs
478  NULL, // address of thread security attrs
479  true, // new process inherits handles?
480  creationFlags, // creation flags
481  NULL, // address of new environment block
482  NULL, // address of current directory name
483  &siStartInfo, // address of STARTUPINFO
484  &piProcInfo)) // address of PROCESS_INFORMATION
485  {
486  // CreateProcess succeeded, change the title if needed
487  if (titleChanged)
488  {
489  SetConsoleTitle(siStartInfo.lpTitle);
490  }
491 
492  // if Input is redirected, write the input data to the input pipe
493  if (redirIn)
494  {
495  ioContext->ReadInputBuffer(&inputThread.inputBuffer, &inputThread.bufferLength);
496  // only start the thread if we have real data
497  if (inputThread.inputBuffer != NULL)
498  {
499  // we need the pipe handle to do this
500  inputThread.start(inPipeW);
501  }
502  }
503 
504  if (redirErr)
505  {
506  // we start a separate thread to read ERROR data from the error pipe
507  errorThread.start(errPipeR);
508  }
509 
510 
511  if (redirOut)
512  {
513  // we start a separate thread to read OUTPUT data from the stdout pipe
514  outputThread.start(outPipeR);
515  }
516 
517 
518  SystemInterpreter::exceptionHostProcess = piProcInfo.hProcess;
519  SystemInterpreter::exceptionHostProcessId = piProcInfo.dwProcessId;
520 
521  if (WAIT_FAILED != WaitForSingleObject ( piProcInfo.hProcess, INFINITE ) )
522  {
523  // Completed ok, get termination rc
524  GetExitCodeProcess(piProcInfo.hProcess, &rc);
525  // do we have input cleanup to perform?
526  if (redirIn)
527  {
528  // the process has returned, but the input thread may be stuck
529  // on a write for data that was never used. We need to close the
530  // pipe to force it to complete
531  inputThread.shutdown();
532  // close the process end of the pipe
533  CloseHandle(siStartInfo.hStdInput);
534  // wait for everything to complete
535  inputThread.waitForTermination();
536  // the INPUT thread may have encountered an error .. raise it now
537  if (inputThread.error != 0)
538  {
539  ErrorRedirection(context, inputThread.error);
540  return false;
541  }
542  }
543 
544  // did we start an ERROR thread?
545  if (redirOut)
546  {
547  CloseHandle(siStartInfo.hStdOutput); // close the handle so readFile will stop
548  // wait for the ERROR thread to finish
549  outputThread.waitForTermination();
550  if (outputThread.dataLength > 0)
551  { // return what the ERROR thread read from its pipe
552  ioContext->WriteOutputBuffer(outputThread.pipeBuffer, outputThread.dataLength);
553  }
554  // the OUTPUT thread may have encountered an error .. raise it now
555  if (outputThread.error != 0)
556  {
557  ErrorRedirection(context, outputThread.error);
558  return false;
559  }
560  }
561 
562  // did we start an ERROR thread?
563  if (redirErr)
564  {
565  CloseHandle(siStartInfo.hStdError); // close the handle so readFile will stop
566  // wait for the ERROR thread to finish
567  errorThread.waitForTermination();
568  if (errorThread.dataLength > 0)
569  { // return what the ERROR thread read from its pipe
570  ioContext->WriteErrorBuffer(errorThread.pipeBuffer, errorThread.dataLength);
571  }
572  // the ERROR thread may have encountered an error .. raise it now
573  if (errorThread.error != 0)
574  {
575  ErrorRedirection(context, errorThread.error);
576  return false;
577  }
578  }
579  }
580  else
581  {
582  rc = GetLastError(); // Bad termination, get error code
583  context->RaiseCondition("FAILURE", context->String(command), NULLOBJECT, context->WholeNumberToObject(rc));
584  result = NULLOBJECT;
585  return true;
586  }
587 
588  // The new process must be detached so it will be discarded
589  // automatically after execution. The thread must be closed first
590  if (titleChanged)
591  {
592  SetConsoleTitle(ctitle);
593  }
594  CloseHandle(piProcInfo.hThread);
595  CloseHandle(piProcInfo.hProcess);
596  }
597  else
598  {
599  // return this as a failure for now ... the caller might try this again
600  // later
601  return false;
602  }
603 
604  SystemInterpreter::exceptionHostProcess = NULL;
605  SystemInterpreter::exceptionHostProcessId = 0;
606 
607  if (rc != 0)
608  {
609  context->RaiseCondition("ERROR", context->String(command), NULLOBJECT, context->Int32ToObject(rc));
610  result = NULLOBJECT;
611  return true;
612  }
613  // this is a zero return
614  result = context->False();
615  return true;
616 }

620 /******************************************************************************/
621 /* Name: sys_command */
622 /* */
623 /* Arguments: cmd - Command to be executed */
624 /* */
625 /* Returned: rc - Return Code */
626 /* Note: if non-zero rc from DosExecPgm return DosExecPgm rc */
627 /* else if non-zero termination code from system return code */
628 /* else return rc from executed command */
629 /* */
630 /* Notes: Handles processing of a system command. Finds location of */
631 /* system command handler using the COMSPEC environment variable */
632 /* and invokes the system specific routine which invokes the */
633 /* command handler with the command to be executed */
634 /* */
635 /******************************************************************************/
636 RexxObjectPtr RexxEntry systemCommandHandler(RexxExitContext *context,
637  RexxStringObject address,
638  RexxStringObject command,
639  RexxIORedirectorContext *ioContext)
640 {
641  // address the command information
642  const char *cmd = context->StringData(command);
643  const char *cl_opt = " /c "; // The "/c" opt for system commandd handler
644  const char *interncmd;
645  RexxObjectPtr result = NULLOBJECT;
646 
647  // is this directed to the no-shell "path" environment?
648  if (stricmp(context->StringData(address), "path") == 0)
649  {
650  if (!sysCommandNT(context, cmd, cmd, true, result, ioContext))
651  {
652  // rc from GetLastError() would be 2, ERROR_FILE_NOT_FOUND
653  // but for consistency with cmd.exe rc for an unknown command
654  // we always set this to 1
655  context->RaiseCondition("FAILURE", context->String(cmd), NULLOBJECT, context->WholeNumberToObject(1));
656  return NULLOBJECT;
657  }
658  return result;
659  }
660 
661  // Remove the "quiet sign" if present
662  if (cmd[0] == '@')
663  {
664  interncmd = cmd + 1;
665  }
666  else
667  {
668  interncmd = cmd;
669  }
670 
671  /* Check for redirection symbols, ignore them when enclosed in double
672  * quotes. If there are more than 2 double qoutes, cmd.exe strips off the
673  * first and last. To preserve what the user sent to us, we count the
674  * double quotes, and, if more than two, add a double quote to the front and
675  * end of the string.
676  */
677  size_t quoteCount = 0;
678  bool noDirectInvoc = false;
679  bool inQuotes = false;
680  size_t i;
681 
682  for (i = 0; i < strlen(interncmd); i++)
683  {
684  if (interncmd[i] == '"')
685  {
686  inQuotes = !inQuotes;
687  quoteCount++;
688  }
689  else
690  {
691  /* if we're in the unquoted part and the current character is one of */
692  /* the redirection characters or the & for multiple commands then we */
693  /* will no longer try to invoke the command directly */
694  if (!inQuotes && (strchr("<>|&", interncmd[i]) != NULL))
695  {
696  noDirectInvoc = true;
697  if ( quoteCount > 2 )
698  {
699  break;
700  }
701  }
702  }
703  }
704 
705  i = 0; // reset to zero for next usage
706 
707  // scan for the first non-whitespace character
708  size_t j = 0;
709  while (interncmd[j] == ' ')
710  {
711  j++;
712  }
713 
714  if (!noDirectInvoc)
715  {
716  char tmp[8];
717  strncpy(tmp, &interncmd[j], 4);
718  tmp[4] = '\0';
719 
720  if (!stricmp("set ",tmp))
721  {
722  if (sys_process_set(context, cmd, &interncmd[j], result))
723  {
724  return result;
725  }
726  }
727  else
728  {
729  strncpy(tmp, &interncmd[j], 3);
730  tmp[3] = '\0';
731  if (!stricmp("cd ",tmp))
732  {
733  if (sys_process_cd(context, cmd, &interncmd[j], result))
734  {
735  return result;
736  }
737  }
738  else
739  { // Check if the command is to change drive
740  if ((tmp[1] == ':') && ((tmp[2] == ' ') || (!tmp[2])))
741  {
742  int rc, error;
743 
744  int drive = toupper( tmp[0] ) - 'A' + 1;
745  if (drive < 1 || drive > 26)
746  {
747  // an out-of-range drive will make _chdrive() MSVC run-time library fail
748  rc = -1;
749  error = 15; // The system cannot find the drive specified.
750  }
751  else
752  {
753  rc = _chdrive(drive);
754  error = GetLastError();
755  }
756  if (rc != 0)
757  {
758  context->RaiseCondition("ERROR", command, NULLOBJECT, context->WholeNumberToObject(error));
759  return NULLOBJECT;
760  }
761  else
762  {
763  // 0 result.
764  return context->False();
765  }
766  }
767  else
768  {
769  // Check if a START command is specified, if so do not
770  // invoke the command directly.
771  strncpy(tmp, &interncmd[j], 6);
772  tmp[6] = '\0';
773  noDirectInvoc = stricmp("start ",tmp) == 0;
774  }
775  }
776  }
777  }
778 
779  const char *sys_cmd_handler; // Pointer to system cmd handler
780 
781  // Determine the system command interpreter. This could be the full path
782  // name if COMSPEC is set, or a default name if it isn't. We no longer
783  // suport Windows 95, so the default will be cmd.exe. But, it is still
784  // possible for people to set COMSPEC to command.com, so we could end up
785  // with command.com
786  if ( (sys_cmd_handler = getenv(COMSPEC)) == NULL )
787  {
788  sys_cmd_handler = CMDDEFNAME;
789  }
790 
791  // Determine the maximum possible buffer size needed to pass the final
792  // command to sysCommandNT().
793  size_t maxBufferSize = strlen(sys_cmd_handler) + 1
794  + strlen(cl_opt)
795  + strlen(&interncmd[j])
796  + 2 // Two possible extra quotes
797  + 1; // Terminating null
798 
799  char cmdstring[CMDBUFSIZE]; // Default static buffer.
800  char *cmdstring_ptr = cmdstring; // Will point to static buffer.
801 
802  if ( maxBufferSize > CMDBUFSIZE )
803  {
804  // Allocate dynamic memory and set cmdstring_ptr to point to it.
805  cmdstring_ptr = (char *)LocalAlloc(LPTR, maxBufferSize);
806  if ( cmdstring_ptr == NULL )
807  {
808  context->RaiseException1(Rexx_Error_System_resources_user_defined,
809  context->String("Failed to allocate memory"));
810  return NULLOBJECT;
811  }
812  }
813  else
814  {
815  // We want maxBufferSize to relect the actual size of the buffer we are
816  // using so that we can test the return from SearchPath()
817  maxBufferSize = CMDBUFSIZE;
818  }
819 
820  SystemInterpreter::exceptionConsole = false;
821  SystemInterpreter::explicitConsole = false;
822 
823  // Check whether or not the command to invoke is cmd.exe or command.com
824  _strupr(strcpy(cmdstring_ptr, &interncmd[j]));
825  bool searchFile = strstr(cmdstring_ptr, "CMD") != NULL;
826 
827  if (searchFile)
828  {
829  if (cmdstring_ptr[0] == '\"')
830  {
831  cmdstring_ptr++;
832  while (cmdstring_ptr[i] && (cmdstring_ptr[i] != '\"'))
833  {
834  i++;
835  }
836  cmdstring_ptr[i]='\0';
837  }
838  else if (cmdstring_ptr[0] == '\'')
839  {
840  cmdstring_ptr++;
841  while (cmdstring_ptr[i] && (cmdstring_ptr[i] != '\''))
842  {
843  i++;
844  }
845  cmdstring_ptr[i]='\0';
846  }
847  else
848  {
849  while (cmdstring_ptr[i] && (cmdstring_ptr[i] != ' '))
850  {
851  i++;
852  }
853  cmdstring_ptr[i]='\0';
854  }
855 
856  LPSTR filepart;
857  uint32_t count = SearchPath(NULL, cmdstring_ptr, ".EXE", (uint32_t)(maxBufferSize - 1), cmdstring, &filepart);
858  bool fileFound = count != 0 && count <= maxBufferSize;
859 
860  // Set pointer back again to cmd buffer (might have been increased)
861  cmdstring_ptr = cmdstring;
862 
863  if (fileFound && !stricmp(sys_cmd_handler, cmdstring_ptr))
864  {
865  SystemInterpreter::exceptionConsole = true;
866  SystemInterpreter::explicitConsole = true;
867  }
868  }
869 
870  // First check whether we can run the command directly as a program. (There
871  // can be no file redirection when not using cmd.exe or command.com)
872  // N.B. ADDRESS ... WITH also implies redirection
873  if (SystemInterpreter::explicitConsole || (!noDirectInvoc && !ioContext->IsRedirectionRequested()))
874  {
875  // Invoke this directly. If we fail, we fall through and try again.
876  if (sysCommandNT(context, cmd, &interncmd[j], true, result, ioContext))
877  {
878  if ( cmdstring_ptr != cmdstring )
879  {
880  // Not pointing to static buffer so we need to free it.
881  LocalFree(cmdstring_ptr);
882  }
883  return result;
884  }
885  }
886 
887  // We couldn't invoke the command directly, or we tried and failed. So,
888  // pass the command to cmd.exe or command.com.
889 
890 
891  // Start the command buffer with the system cmd handler
892  strcpy(cmdstring_ptr,sys_cmd_handler);
893 
894  // Check whether or not the user specified the /k option. If so do not use
895  // the /c option. The /k option can only be specified as the first
896  // argument, and if used, keeps the command handler process open after the
897  // command has finished. Normally the /c option would be usee to close the
898  // command handler process when the command is finished..
899  if (!( (strlen(interncmd) > j+1) && (interncmd[j] == '/')
900  && ((interncmd[j+1] == 'k') || (interncmd[j+1] == 'K'))
901  && ((interncmd[j+2] == ' ') || (interncmd[j+2] == '\0')) ))
902  {
903  strcat(cmdstring_ptr,cl_opt);
904  }
905  else
906  {
907  SystemInterpreter::exceptionConsole = true;
908  strcat(cmdstring_ptr," ");
909  }
910 
911  // Add cmd to be executed, possibly quoting it to preserve embedded quotes.
912  if ( quoteCount> 2 )
913  {
914  strcat(cmdstring_ptr,"\"");
915  strcat(cmdstring_ptr,interncmd);
916  strcat(cmdstring_ptr,"\"");
917  }
918  else
919  {
920  strcat(cmdstring_ptr,interncmd);
921  }
922 
923  // Invoke the command
924  if (!sysCommandNT(context, cmd, cmdstring_ptr, false, result, ioContext))
925  {
926  // Failed, get error code and return
927  context->RaiseCondition("FAILURE", context->String(cmd), NULLOBJECT, context->WholeNumberToObject(GetLastError()));
928 
929  if ( cmdstring_ptr != cmdstring )
930  {
931  // Not pointing to static buffer so we need to free it.
932  LocalFree(cmdstring_ptr);
933  }
934  return NULLOBJECT;
935  }
936 
937  if ( cmdstring_ptr != cmdstring )
938  {
939  // Not pointing to static buffer so we need to free it.
940  LocalFree(cmdstring_ptr);
941  }
942  SystemInterpreter::exceptionConsole = false;
943  return result;
944 }
