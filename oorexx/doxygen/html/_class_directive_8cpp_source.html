157 /**
158  * Do install-time processing of the ::requires directive. This
159  * will resolve the directive and merge all of the public information
160  * from the resolved file into this program context.
161  *
162  * @param package The package doing the install (used to resolve class references)
163  * @param activation The activation we're running under for the install.
164  */
165 void ClassDirective::install(PackageClass *package, RexxActivation *activation)
166 {
167  RexxClass *metaclass = OREF_NULL;
168  RexxClass *subclass = TheObjectClass;
169 
170  // make this the current line for the error context
171  activation->setCurrent(this);
172 
173  if (metaclassName != OREF_NULL)
174  {
175  // resolve the class.. This must be locatable in the
176  // context of the package.
177  metaclass = metaclassName->lookup(package);
178  if (metaclass == OREF_NULL)
179  {
180  reportException(Error_Execution_nometaclass, metaclassName->getName());
181  }
182  }
183 
184  // do we have an explicit subclass
185  if (subclassName != OREF_NULL)
186  {
187  // resolve the explicit subclass
188  subclass = subclassName->lookup(package);
189  if (subclass == OREF_NULL)
190  {
191  reportException(Error_Execution_noclass, subclassName->getName());
192  }
193  }
194 
195  // create the class object using the appropriate mechanism
196 
197  // creating a mixing
198  if (isMixinClass())
199  {
200  classObject = subclass->mixinClass(package, idName, metaclass, classMethods);
201  }
202  // creating a direct subclass
203  else
204  {
205  classObject = subclass->subclass(package, idName, metaclass, classMethods);
206  }
207  // add the class to the directory, which also protects it from GC for the
208  // subsequent steps
209  package->addInstalledClass(publicName, classObject, isPublic());
210 
211  // using multiple inheritance? Process each one in turn.
212  if (inheritsClasses != OREF_NULL)
213  {
214  // now handle the multiple inheritance issues
215  size_t count = inheritsClasses->items();
216 
217  for (size_t i = 1; i <= count; i++)
218  {
219  // get the next mixin class name and resolve...again, this
220  // is required.
221  ClassResolver *inheritsResolver = (ClassResolver *)inheritsClasses->get(i);
222  RexxClass *mixin = inheritsResolver->lookup(package);
223  if (mixin == OREF_NULL)
224  {
225  reportException(Error_Execution_noclass, inheritsResolver->getName());
226  }
227 
228  ProtectedObject result;
229  // inherit from the mixin
230  classObject->sendMessage(GlobalNames::INHERIT, mixin, result);
231  }
232  }
233 
234  // have instance methods to add? have the class define them
235  if (instanceMethods != OREF_NULL)
236  {
237  classObject->defineMethods(instanceMethods);
238  }
239 
240  // set the annotations if we have any
241  if (annotations != OREF_NULL)
242  {
243  classObject->setAnnotations(annotations);
244  }
245 
246  // if we had the abstract keyword specified, mark the class as abstract.
247  if (isAbstract())
248  {
249  classObject->makeAbstract();
250  }
251 }

254 /**
255  * Resolve any dynamically generated ::CONSTANT directives for the class.
256  */
257 void ClassDirective::resolveConstants(PackageClass *package, Activity *activity)
258 {
259  // no dynamic constants? This is easy.
260  if (constantInitializer == OREF_NULL)
261  {
262  return;
263  }
264 
265  // this was created without a package originally, so attach the package to it.
266  constantInitializer->setPackageObject(package);
267 
268 
269  // In order to install, we need to call something. We manage this by
270  // creating a dummy stub routine that we can call to force things to install
271  Protected<MethodClass> code = new MethodClass(GlobalNames::CONSTANT_DIRECTIVE, constantInitializer);
272  // this is given the class object as a scope so that super will get set correctly.
273  code->setScope(classObject);
274  ProtectedObject dummy;
275  // run this as a method against the target class object.
276  code->run(activity, classObject, GlobalNames::CONSTANT_DIRECTIVE, NULL, 0, dummy);
277 }

292 /**
293  * Check if a dependency this class has is based off of a
294  * class co-located in the same class package.
295  *
296  * @param name The class name.
297  * @param classDirectives
298  * The global local classes list.
299  */
300 void ClassDirective::checkDependency(ClassResolver *classReference, StringTable *classDirectives)
301 {
302  if (classReference != OREF_NULL && !classReference->isQualified())
303  {
304  RexxString *name = classReference->getName();
305 
306  // if this is in install?
307  if (classDirectives->hasIndex(name))
308  {
309  if (dependencies == OREF_NULL)
310  {
311  dependencies = new_string_table();
312  }
313  // add to our pending list
314  dependencies->setEntry(name, name);
315  }
316  }
317 }

320 /**
321  * Check our class dependencies against the locally defined class
322  * list to develop a cross dependency list.
323  *
324  * @param classDirectives
325  * The global set of defined classes in this package.
326  */
327 void ClassDirective::addDependencies(StringTable *classDirectives)
328 {
329  // now for each of our dependent classes, if this is defined locally, we
330  // add an entry to our dependency list to aid the class ordering
331 
332  checkDependency(metaclassName, classDirectives);
333  checkDependency(subclassName, classDirectives);
334  // process each inherits item the same way
335  if (inheritsClasses != OREF_NULL)
336  {
337  size_t count = inheritsClasses->items();
338 
339  for (size_t i = 1; i <= count; i++)
340  {
341  ClassResolver *inheritsName = (ClassResolver *)inheritsClasses->get(i);
342  checkDependency(inheritsName, classDirectives);
343  }
344  }
345 }
