207 /**
208  * Process UNKNOWN messages for a variable reference object,
209  * forwarding them to the referenced variable value.
210  *
211  * @param message The message target.
212  * @param arguments The message arguments.
213  *
214  * @return The message result.
215  */
216 RexxObject *VariableReference::unknownRexx(RexxString *message, ArrayClass *arguments)
217 {
218  Protected<RexxString> messageName = stringArgument(message, ARG_ONE);
219  Protected<ArrayClass> argumentList = arrayArgument(arguments, ARG_TWO);
220 
221  ProtectedObject result;
222  // get the resolved variable value and forward this to the value
223  return variable->getResolvedValue()->sendMessage(messageName, argumentList, result);
224 }

229  * an optimized bypass for the Object default method that can
230  * bypass creating an array for the arguments and sending the

232  * through the integer unknown method, this is a big
233  * optimization.

243  // just send this as a message indirect variable value
244  variable->getResolvedValue()->messageSend(messageName, arguments, count, result);
