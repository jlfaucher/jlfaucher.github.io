176 /**
177  * The Rexx stub for the Queue PUT method. replaces the Array
178  * version because it has slightly different semantics for index
179  * validation. The only valid indexes are those within range
180  * and the size is not adjusted for a put out of bounds.
181  *
182  * @param arguments The array of all arguments sent to the
183  * method (variable arguments allowed
184  * here...the first argument is the item being
185  * added, all other arguments are the index).
186  * @param argCount The number of arguments in the method call.
187  *
188  * @return Always return nothing.
189  */
190 RexxObject *QueueClass::putRexx(RexxObject *value, RexxObject *index)
191 {
192  requiredArgument(value, ARG_ONE);
193  // make sure we have an index specified before trying to decode this.
194  requiredArgument(index, ARG_TWO);
195 
196  // Validate the index argument, but don't allow expansion.
197  size_t position;
198 
199  if (!validateIndex(&index, 1, ARG_TWO, IndexAccess, position))
200  {
201  reportException(Error_Incorrect_method_index, index);
202  }
203 
204  // we can only update assigned items, so make sure this is within bounds.
205  checkInsertIndex(position);
206 
207  // set the new value and return nothing
208  put(value, position);
209  return OREF_NULL;
210 }

298 /**
299  * Create an instance of the queue class and fill it with objects.
300  *
301  * @param args The argument pointer.
302  * @param argCount The count of arguments.
303  *
304  * @return An instance of the queue class populated with objects.
305  */
306 QueueClass *QueueClass::ofRexx(RexxObject **args, size_t argCount)
307 {
308  // create a queue object.
309  Protected<QueueClass> newQueue = (QueueClass *)QueueClass::newRexx(NULL, 0);
310 
311  for (size_t i = 0; i < argCount; i++)
312  {
313  // add each item, but ommitted arguments are an error.
314  RexxObject *item = args[i];
315  if (item == OREF_NULL)
316  {
317  reportException(Error_Incorrect_method_noarg, i + 1);
318  }
319  newQueue->append(item);
320  }
321  return newQueue;
322 }
