120 /**
121  * Evaluate a simple variable in an expression.
122  *
123  * @param context The current execution context.
124  * @param stack The current evaluation stack.
125  *
126  * @return The variable value (also pushed on stack)
127  */
128 RexxObject *RexxSimpleVariable::evaluate(RexxActivation *context, ExpressionStack *stack)
129 {
130  // look up the variable
131  RexxVariable *variable = context->getLocalVariable(variableName, index);
132  RexxObject *value = variable->getVariableValue();
133  // we need to do novalue checks.
134  if (value == OREF_NULL)
135  {
136  // try the various novalue mechanisms
137  value = context->handleNovalueEvent(variableName, variableName, variable);
138  }
139  // stack, trace, and return
140  stack->push(value);
141 
142  context->traceVariable(variableName, value);
143  return value;
144 }
