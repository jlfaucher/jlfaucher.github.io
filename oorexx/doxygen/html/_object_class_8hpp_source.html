183 /**
184  * create first base level class dummy virtual function to force
185  * the virtual function table to a specific location. Different
186  * compilers place the virtual function table pointer at
187  * different locations. By having virtual methods defined on
188  * the very first class level, we can get things where we expect
189  * to find it.
190  */
191 class RexxVirtualBase
192 {
193  protected:
194  virtual ~RexxVirtualBase() {; }
195  virtual void baseVirtual() {;}
196 
197  public:
198  // the following need to be defined at the base virtual level. When
199  // an exception is thrown from within an object constructor, the destructors
200  // unwind and the constructed object just ends up with a virtual base
201  // vft. If the garbage collector sees this, it will crash unless these
202  // are defined at this level.
203  virtual void live(size_t) {;}
204  virtual void liveGeneral(MarkReason reason) {;}
205  virtual void flatten(Envelope *) {;}
206  virtual RexxInternalObject* unflatten(Envelope *) { return (RexxInternalObject *)this; };
207 };
