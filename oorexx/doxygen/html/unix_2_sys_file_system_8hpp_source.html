199 class AutoClose
200 {
201  public:
202  AutoClose() : value(-1)
203  { };
204  AutoClose(int fd) : value(fd)
205  { }
206  ~AutoClose()
207  {
208  close(false);
209  }
210  AutoClose & operator=(int fd)
211  {
212  close(false);
213  value = fd;
214  return *this;
215  }
216  operator int() const
217  {
218  return value;
219  }
220  int operator==(int fd)
221  {
222  return value == fd;
223  }
224  int close(bool returnError = true)
225  {
226  int closeStatus = 0;
227  if (returnError)
228  {
229  if (value >= 0)
230  {
231  closeStatus = ::close(value);
232  }
233  }
234  else
235  {
236  if (value >= 0)
237  {
238  int backup = errno;
239  ::close(value);
240  errno = backup;
241  }
242  }
243  value = -1;
244  return closeStatus;
245  }
246  private:
247  int value; // >= 0 if opened
248 };
