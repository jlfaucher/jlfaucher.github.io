160 /**
161  * Return a new token object, with information appropriately
162  * filled in
163  *
164  * @param classId The token class identifier.
165  * @param subclass The token subclass
166  * @param value The (optional) string value.
167  * @param l The token location.
168  *
169  * @return A filled in token object.
170  */
171 RexxToken *RexxClause::newToken(TokenClass classId, TokenSubclass subclass, RexxString *value, SourceLocation &l)
172 {
173 
174  //. do we need to extend our cache?
175  if (free > size)
176  {
177  // make sure the token array is large enough for the additional token's we're adding.
178  tokens->ensureSpace(tokens->size() + EXTEND_SIZE);
179 
180  for (size_t i = 0; i < EXTEND_SIZE; i++)
181  {
182  tokens->append(new RexxToken());
183  }
184  size += EXTEND_SIZE;
185  }
186 
187  // get the first free token and call the in-memory constructor for this
188  /* get the first free token */
189  RexxToken *token = (RexxToken *)tokens->get(free++);
190  // construct the token and return
191  ::new ((void *)token) RexxToken(classId, l, subclass, value);
192  return token;
193 }
