131 /**
132  * Convert a trigger value to an unsigned integer with
133  * appropriate error reporting.
134  *
135  * @param context The current execution context.
136  * @param stack The current evaluation stack.
137  *
138  * @return The converted integer value.
139  */
140 size_t ParseTrigger::integerTrigger(RexxActivation *context, ExpressionStack *stack)
141 {
142  // if we have a value that requires evaluation, get the value and trace.
143  RexxObject *trigger = value->evaluate(context, stack);
144  context->traceResult(trigger);
145 
146  // NOTE: We leave this on the stack to protect from GC until after we convert.
147 
148  size_t result;
149  // try to convert to an unsigned number...report an error if this failed.
150  if (!trigger->requestUnsignedNumber(result, number_digits()))
151  {
152  reportException(Error_Invalid_whole_number_parse, trigger);
153  }
154  // once we have converted this value, we're done with it. It no longer
155  // requires protection.
156  stack->pop();
157  return result;
158 }

182 /**
183  * Apply a parsing trigger against a parsing context.
184  *
185  * @param context The current execution context.
186  * @param stack The current evaluation stack.
187  * @param target The current parsing context.
188  */
189 void ParseTrigger::parse(RexxActivation *context, ExpressionStack *stack,
190  RexxTarget *target )
191 {
192  // perform the trigger operaitons
193  switch (triggerType)
194  {
195  // move to the end of the pattern
196  case TRIGGER_END:
197  target->moveToEnd();
198  break;
199 
200  // a positive relative movement.
201  case TRIGGER_PLUS:
202  target->forward(integerTrigger(context, stack));
203  break;
204 
205  // negative relative movement
206  case TRIGGER_MINUS:
207  target->backward(integerTrigger(context, stack));
208  break;
209 
210  // positive relative movement with length semantics (>n)
211  case TRIGGER_PLUS_LENGTH:
212  target->forwardLength(integerTrigger(context, stack));
213  break;
214 
215  // negative relative movement with length semantics (<n)
216  case TRIGGER_MINUS_LENGTH:
217  target->backwardLength(integerTrigger(context, stack));
218  break;
219 
220  // absolute string positioning
221  case TRIGGER_ABSOLUTE:
222  target->absolute(integerTrigger(context, stack));
223  break;
224 
225  // string search
226  case TRIGGER_STRING:
227  target->search(stringTrigger(context, stack));
228  // the string trigger was protected on the stack. Remove it now
229  stack->pop();
230  break;
231 
232  // caseless string search
233  case TRIGGER_MIXED:
234  target->caselessSearch(stringTrigger(context, stack));
235  // the string trigger was protected on the stack. Remove it now
236  stack->pop();
237  break;
238 
239  // invalid PARSE trigger type (should really never happen)
240  default:
241  reportException(Error_Interpretation_switch, "PARSE trigger type", triggerType);
242  break;
243  }
244 
245  // if we are tracing, we need to display each assignment result. We
246  // have two copies of this loop, one optimized for untraced operation and
247  // a second one for traced operation.
248  if (context->tracingResults())
249  {
250  for (size_t i = 0; i < variableCount; i++)
251  {
252  RexxString *variableValue;
253 
254  // if this is the last variable on the list, this gets the
255  // remainder of the parsing segment. Otherwise, we just
256  // extract the next blank delimited word.
257  if (i + 1 == variableCount)
258  {
259  variableValue = target->remainder();
260  }
261  else
262  {
263  variableValue = target->getWord();
264  }
265  // needs protecting
266  ProtectedObject p(variableValue);
267  // get the next variable
268  RexxVariableBase *variable = variables[i];
269  // the '.' dummy placeholder shows up as a NULL value in the list.
270  // the dummy placeholder has a special trace form.
271  if (variable != OREF_NULL)
272  {
273  // NOTE: The different variable types handle their own assignment tracing
274  variable->assign(context, variableValue);
275  // if only tracing results and not intermediates, then we need to
276  // trace this value explicitly.
277  if (!context->tracingIntermediates())
278  {
279  context->traceResult(variableValue);
280  }
281  }
282  // this is the dummy variable
283  else
284  {
285  context->traceIntermediate(variableValue, RexxActivation::TRACE_PREFIX_DUMMY);
286  }
287  }
288  }
289  // not tracing...this version is a bit more optimized.
290  else
291  {
292  for (size_t i = 0; i < variableCount; i++)
293  {
294  // get the next retriever
295  RexxVariableBase *variable = variables[i];
296  // if we have a real variable (not a .), extract the string piece and assign.
297  if (variable != OREF_NULL)
298  {
299  RexxObject *variableValue;
300  // if this is the last variable in the list, grab the remainder,
301  // otherwise, we need to parse word off.
302  if (i + 1 == variableCount)
303  {
304  variableValue = target->remainder();
305  }
306  else
307  {
308  variableValue = target->getWord();
309  }
310  // needs protecting if the assignment is a compound var or a message
311  // target.
312  ProtectedObject p(variableValue);
313  // do the assignment */
314  variable->assign(context, variableValue);
315  }
316  // dummy variable, we just skip the assignment
317  else
318  {
319  // we need to figure out if we're skipping a word, or skipping everything.
320  if (i + 1 == variableCount)
321  {
322  target->skipRemainder();
323  }
324  else
325  {
326  target->skipWord();
327  }
328  }
329  }
330  }
331 }
