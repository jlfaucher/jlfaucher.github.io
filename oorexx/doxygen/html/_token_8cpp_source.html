89  * (+=, -=, etc.).

91  * @param source The source for the original operator token.

93 void RexxToken::checkAssignment(LanguageParser *parser, RexxString *newValue)
94 {
95  // check if the next character is a special assignment shortcut
96  if (parser->nextSpecial('=', tokenLocation))
97  {
98  // this is a special type, which uses the same subtype.
99  classId = TOKEN_ASSIGNMENT;
100  // this is the new string value of the token
101  stringValue = newValue;
102  }

106 /**
107  * Determine a Token's operator precedence.
108  *
109  * @return A numeric ranking for operator characters.
110  */
111 int RexxToken::precedence()
112 {
113  // the subclass determines what type of operator this is.
114  switch (subclass)
115  {
116  default:
117  return 0; // this is the bottom of the heap
118  break;
119 
120  case OPERATOR_OR:
121  case OPERATOR_XOR:
122  return 1; // various OR types are next
123  break;
124 
125  case OPERATOR_AND:
126  return 2; // AND operator ahead of ORs
127  break;
128 
129  case OPERATOR_EQUAL: // comparisons are all together
130  case OPERATOR_BACKSLASH_EQUAL:
131  case OPERATOR_GREATERTHAN:
132  case OPERATOR_BACKSLASH_GREATERTHAN:
133  case OPERATOR_LESSTHAN:
134  case OPERATOR_BACKSLASH_LESSTHAN:
135  case OPERATOR_GREATERTHAN_EQUAL:
136  case OPERATOR_LESSTHAN_EQUAL:
137  case OPERATOR_STRICT_EQUAL:
138  case OPERATOR_STRICT_BACKSLASH_EQUAL:
139  case OPERATOR_STRICT_GREATERTHAN:
140  case OPERATOR_STRICT_BACKSLASH_GREATERTHAN:
141  case OPERATOR_STRICT_LESSTHAN:
142  case OPERATOR_STRICT_BACKSLASH_LESSTHAN:
143  case OPERATOR_STRICT_GREATERTHAN_EQUAL:
144  case OPERATOR_STRICT_LESSTHAN_EQUAL:
145  case OPERATOR_LESSTHAN_GREATERTHAN:
146  case OPERATOR_GREATERTHAN_LESSTHAN:
147  return 3;
148  break;
149 
150  case OPERATOR_ABUTTAL: // concatentates
151  case OPERATOR_CONCATENATE:
152  case OPERATOR_BLANK:
153  return 4;
154  break;
155 
156  case OPERATOR_PLUS: // plus and minus
157  case OPERATOR_SUBTRACT:
158  return 5;
159  break;
160 
161  case OPERATOR_MULTIPLY: // multiply and divide versions
162  case OPERATOR_DIVIDE:
163  case OPERATOR_INTDIV:
164  case OPERATOR_REMAINDER:
165  return 6;
166  break;
167 
168  case OPERATOR_POWER:
169  return 7; // almost the top of the heap
170  break;
171 
172  case OPERATOR_BACKSLASH:
173  return 8; // NOT is the top honcho
174  break;
175  }
176 }

179 /**
180  * Test if a token qualifies as a terminator in the current
181  * parsing context.
182  *
183  * @param terminators
184  * The list of terminators.
185  *
186  * @return true if this is a terminator, false if it doesn't match
187  * one of the terminator classes.
188  */
189 bool RexxToken::isTerminator(int terminators)
190 {
191  // process based on terminator class
192  switch (classId)
193  {
194  case TOKEN_EOC: // end-of-clause is always a terminator
195  {
196  return true;
197  }
198  case TOKEN_COMMA: // commas are also always subexpression terminators
199  {
200  return true;
201  }
202  case TOKEN_RIGHT: // found a right paren
203  {
204  if (terminators&TERM_RIGHT)
205  {
206  return true;
207  }
208  break;
209  }
210  case TOKEN_SQRIGHT: // closing square bracket?
211  {
212  if (terminators&TERM_SQRIGHT)
213  {
214  return true;
215  }
216  break;
217  }
218  case TOKEN_SYMBOL: // the token is a symbol...need to check on keyword terminators
219  {
220  // keyword terminators all set a special keyword flag. We only check
221  // symbols that are simple variables.
222  if (terminators&TERM_KEYWORD && isSimpleVariable())
223  {
224  // map the keyword token to a key word code. This are generally
225  // keyword options on DO/LOOP, although THEN and WHEN are also terminators
226  switch (subKeyword())
227  {
228  case SUBKEY_TO:
229  {
230  if (terminators&TERM_TO)
231  {
232  return true;
233  }
234  break;
235  }
236  case SUBKEY_BY:
237  {
238  if (terminators&TERM_BY)
239  {
240  return true;
241  }
242  break;
243  }
244  case SUBKEY_FOR:
245  {
246  if (terminators&TERM_FOR)
247  {
248  return true;
249  }
250  break;
251  }
252  case SUBKEY_WHILE: // a single terminator type picks up both
253  case SUBKEY_UNTIL: // while and until
254  {
255  if (terminators&TERM_WHILE)
256  {
257  return true;
258  }
259  break;
260  }
261  case SUBKEY_WITH: // WITH keyword in PARSE value
262  {
263  if (terminators&TERM_WITH)
264  {
265  return true;
266  }
267  break;
268  }
269  case SUBKEY_THEN: // THEN subkeyword from IF or WHEN
270  {
271  if (terminators&TERM_THEN)
272  {
273  return true;
274  }
275  break;
276  }
277  default: // not a terminator type
278  break;
279  }
280  }
281  }
282  default:
283  break;
284  }
285  return false; // no terminator found
286 }

300 /**
301  * Get a display value for a token. For all symbols,
302  * literals, and operators, this is just the string value
303  * already set in the token. For others, we generate
304  * a string object automatically.
305  *
306  * @return The string display value of the token.
307  */
308 RexxString *RexxToken::displayValue()
309 {
310  // get the token string value. Not all tokens have an assigned
311  // string value, so we generate it here for the error.
312  if (stringValue != OREF_NULL)
313  {
314  return stringValue;
315  }
316 
317  // some tokens don't directly have a string value...we can provide one here
318  switch (classId)
319  {
320  // blank operator
321  case TOKEN_BLANK:
322  return new_string(" ", 1);
323 
324  // end of clause...just use a semicolon, even though it might be
325  // a linend.
326  case TOKEN_EOC:
327  return new_string(";", 1);
328 
329  // comma
330  case TOKEN_COMMA:
331  return new_string(",", 1);
332 
333  case TOKEN_LEFT:
334  return new_string("(", 1);
335 
336  case TOKEN_RIGHT:
337  return new_string(")", 1);
338 
339  case TOKEN_SQLEFT:
340  return new_string("[", 1);
341 
342  case TOKEN_SQRIGHT:
343  return new_string("]", 1);
344 
345  case TOKEN_COLON:
346  return new_string(":", 1);
347 
348  case TOKEN_TILDE:
349  return new_string("~", 1);
350 
351  case TOKEN_DTILDE:
352  return new_string("~~", 2);
353 
354  case TOKEN_DCOLON:
355  return new_string("::", 2);
356 
357  // token we don't have an answer for...just use a null string
358  default:
359  return (RexxString *)GlobalNames::NULLSTRING;
360  }
