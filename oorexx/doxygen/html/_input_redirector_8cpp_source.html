57 /**
58  * A dispatcher for trapping and handling conditions and errors from
59  * method calls used in redirection. This will ensure the errors get
60  * propagated to the base native activation for the command call.
61  */
62 class RedirectionDispatcher : public ConditionTrappingDispatcher
63 {
64 public:
65  inline RedirectionDispatcher(TrapInvoker &t) : ConditionTrappingDispatcher(t)
66  {
67 
68  }
69  virtual ~RedirectionDispatcher() { ; }
70 
71  virtual void handleError(DirectoryClass *conditionObj)
72  {
73  // we're only interested in NOTREADY or SYNTAX conditions. We swallow the
74  // NOTREADY's, and propagate the SYNTAX conditions
75  RexxString *conditionName = (RexxString *)conditionObj->get(GlobalNames::CONDITION);
76  if (conditionName->strCompare(GlobalNames::SYNTAX))
77  {
78  // set this in the base command callout activation so the error will get raised
79  // when the command completes.
80  ((NativeActivation *)(activation->getPreviousStackFrame()))->setConditionInfo(conditionObj);
81  }
82 
83  // now take care of normal trapping processes.
84  ConditionTrappingDispatcher::handleError(conditionObj);
85  }
86 };

89 /**
90  * Read all of the input data as a single buffer.
91  *
92  * @param data The returned pointer to the data
93  * @param length The returned data length.
94  */
95 void InputRedirector::readBuffered(NativeActivation *context, const char *&data, size_t &length)
96 {
97  // ok, called a second time, we can return this data again.
98  if (inputBuffer != OREF_NULL)
99  {
100  data = inputBuffer->getStringData();
101  length = inputBuffer->getLength();
102  return;
103  }
104 
105  // create a new mutable buffer instance to write into
106  inputBuffer = new MutableBuffer(defaultBufferSize, defaultBufferSize);
107  // now process all of the lines
108  for (;;)
109  {
110  // read the next line and break if we have nothing
111  RexxString *newLine = read(context);
112  if (newLine == OREF_NULL)
113  {
114  break;
115  }
116  inputBuffer->append(newLine);
117  inputBuffer->append(SysFileSystem::EOL_Marker);
118  }
119  data = inputBuffer->getStringData();
120  length = inputBuffer->getLength();
121 }

176 /**
177  * Perform initial setup on this input source.
178  */
179 void StemInputSource::init()
180 {
181  // get the stem.0 value
182  RexxObject *count = stem->getElement(size_t(0));
183  // if completely not there, then we'll just treat this like a replace
184  if (count == OREF_NULL)
185  {
186  // this is an error for input
187  reportException(Error_Execution_missing_stem_array_index, stem->getName());
188  }
189  // the stem.0 value must be an integer size. We'll start writing at the next position
190  else
191  {
192  // a valid whole number index
193  if (!count->unsignedNumberValue(arraySize, Numerics::ARGUMENT_DIGITS))
194  {
195  reportException(Error_Invalid_whole_number_stem_array_index, stem->getName(), count);
196  }
197  }
198  // we always start at element 1
199  index = 1;
200 }

203 /**
204  * Read the next line from the stem source
205  *
206  * @param context The NativeActivation context for the command handler, which
207  * will handler error/condition traps for this callback.
208  */
209 RexxString *StemInputSource::read(NativeActivation *context)
210 {
211  // if we've reached the end, done reading, so return nothing.
212  if (index > arraySize)
213  {
214  lastValue = OREF_NULL;
215  return OREF_NULL;
216  }
217  // get the next element
218  RexxObject *value = stem->getFullElement(index++);
219  // this is not supposed to be a sparse array, but for safety,
220  // we'll turn this into a null string
221  if (value == OREF_NULL)
222  {
223  return GlobalNames::NULLSTRING;
224  }
225  // Conceptually, everything in here is supposed to be a string.
226  // However, if we unalaterally assume it is a string, they very bad things
227  // happen. Additionally, this value might be an integer or numberstring object
228  // pretending to be a string, so we need to force this to be a real string.
229 
230  // Additionally, we keep a reference to this in case it is a newly
231  // allocated object that could be garbage collected unless we hold on
232  // to a reference
233  lastValue = value->requestString();
234 
235  return lastValue;
236 }

307 /**
308  * Read the next line from the stem source
309  *
310  * @param context The NativeActivation context for the command handler, which
311  * will handler error/condition traps for this callback.
312  */
313 RexxString *StreamObjectInputSource::read(NativeActivation *context)
314 {
315  // if we previously hit the end, then return nothing
316  if (hitEnd)
317  {
318  lastValue = OREF_NULL;
319  return OREF_NULL;
320  }
321 
322  // read a line from the stream
323  ProtectedObject result;
324 
325  // invoke LINEIN with trap protection
326  LineinInvoker invoker(stream, lastValue);
327  RedirectionDispatcher dispatcher(invoker);
328 
329  // invoke the method with appropriate trapping
330  context->getActivity()->run(dispatcher);
331 
332  // this terminated due to a trapped condition (likely NOTREADY)
333  // treat this as an eof
334  if (dispatcher.conditionOccurred())
335  {
336  hitEnd = true;
337  return OREF_NULL;
338  }
339 
340  // return the read string
341  return lastValue;
342 }

400  * Perform initialization of a stream input source

402 void StreamInputSource::init()
403 {
404  // create an instance of the stream class for the given name
405  RexxClass *streamClass = TheRexxPackage->findClass(GlobalNames::STREAM);
406  ProtectedObject result;
407  stream = streamClass->sendMessage(GlobalNames::NEW, name, result);
408 
409  // open for input only

411  // the open should return ready
