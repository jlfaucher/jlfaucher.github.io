368  * be executed or not.
369  *
370  * @param context The current execution context.
371  * @param stack The current evaluation stack.
372  * @param doblock The doblock for this instruction instance.
373  * @param first True if this is the first iteration, false for
374  * reExecute iterations.

379 bool RexxInstructionDoOverUntil::iterate(RexxActivation *context, ExpressionStack *stack, DoBlock *doblock, bool first)
380 {

382  if (first)

386 
387  return !whileLoop.checkUntil(context, stack) && doblock->checkOver(context, stack);
