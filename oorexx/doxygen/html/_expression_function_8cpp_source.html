171 /**
172  * Evaluate a function call.
173  *
174  * @param context The current execution context.
175  * @param stack the current evaluation stack.
176  *
177  * @return The function return value.
178  */
179 RexxObject *RexxExpressionFunction::evaluate(RexxActivation *context, ExpressionStack *stack )
180 {
181  // save the top of the stack for popping values off later.
182  size_t stacktop = stack->location();
183 
184  // evaluate the arguments first
185  RexxInstruction::evaluateArguments(context, stack, arguments, argumentCount);
186 
187  ProtectedObject result; // returned result
188 
189  // do we have a resolved external routine to call?
190  if (externalTarget != OREF_NULL)
191  {
192  context->externalCall(functionName, externalTarget, stack->arguments(argumentCount), argumentCount, GlobalNames::FUNCTION, result);
193  }
194 
195  // if we resolved to an internal label, call that now
196  else if (target != OREF_NULL)
197  {
198  context->internalCall(functionName, target, stack->arguments(argumentCount), argumentCount, result);
199  }
200  // if this was resolved to a builtin, call directly
201  else if (builtinIndex != NO_BUILTIN)
202  {
203  result = (*(LanguageParser::builtinTable[builtinIndex]))(context, argumentCount, stack);
204  }
205  else
206  {
207  // this is a potentially resolved external target
208  RoutineClass *resolvedTarget = OREF_NULL;
209 
210  context->externalCall(resolvedTarget, functionName, stack->arguments(argumentCount), argumentCount, GlobalNames::FUNCTION, result);
211 
212  // this potentially resolved a target that will allow us
213  // to fast path the next call
214  setField(externalTarget, resolvedTarget);
215  }
216 
217  // functions must have a return result
218  if (result.isNull())
219  {
220  reportException(Error_Function_no_data_function, functionName);
221  }
222 
223  // remove the arguments from the stack and push our result
224  stack->setTop(stacktop);
225  stack->push(result);
226 
227  // trace if needed and return the result
228  context->traceFunction(functionName, result);
229  return result;
230 }
