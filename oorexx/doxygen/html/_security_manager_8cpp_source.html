141 /**
142  * Do the actual invocation of the security manager.
143  *
144  * @param methodName The method name to invoke.
145  * @param arguments The arguments to the specific method.
146  *
147  * @return true if the security manager overrode this, false otherwise.
148  */
149 bool SecurityManager::callSecurityManager(RexxString *methodName, DirectoryClass *arguments)
150 {
151  ProtectedObject result;
152  // invoke the manager
153  RexxObject *resultObj = manager->sendMessage(methodName, arguments, result);
154  // a result is required
155  if (resultObj == OREF_NULL)
156  {
157  reportException(Error_No_result_object_message, methodName);
158  }
159  return resultObj->truthValue(Error_Logical_value_authorization);
160 }

233 /**
234  * Check for permission to call an external command
235  *
236  * @param activity The activity we're running on
237  * @param address
238  * @param command
239  * @param result The returned result.
240  * @param condition
241  *
242  * @return true if the security manager handled this call, false otherwise.
243  */
244 bool SecurityManager::checkCommand(Activity *activity, RexxString *address, RexxString *command, ProtectedObject &result, ProtectedObject &condition)
245 {
246  // no method here
247  if (manager == OREF_NULL)
248  {
249  return false;
250  }
251  DirectoryClass *securityArgs = new_directory();
252  ProtectedObject p(securityArgs);
253  // add the command and the accress target
254  securityArgs->put(command, GlobalNames::COMMAND);
255  securityArgs->put(address, GlobalNames::ADDRESS);
256  // if the manager handled this, we need to decode the return stuff
257  if (callSecurityManager(GlobalNames::COMMAND, securityArgs))
258  {
259  result = securityArgs->get(GlobalNames::RC);
260  // if no return code received, use a zero code
261  if (result.isNull())
262  {
263  result = IntegerZero;
264  }
265  // failure indicated? Need to raise a failure condition
266  if (securityArgs->get(GlobalNames::FAILURE) != OREF_NULL)
267  {
268  // raise the condition when things are done
269  condition = activity->createConditionObject(GlobalNames::FAILURE, result, command, OREF_NULL, OREF_NULL);
270  }
271  // same for an error condition
272  else if (securityArgs->get(GlobalNames::ERRORNAME) != OREF_NULL)
273  {
274  // raise the condition when things are done
275  condition = activity->createConditionObject(GlobalNames::ERRORNAME, result, command, OREF_NULL, OREF_NULL);
276  }
277  return true;
278  }
279 
280  return false; // not handled
281 }

312 /**
313  * Check for requires file access
314  *
315  * @param name The name of the stream.
316  *
317  * @return The actual requires file name that should be loaded. A
318  * return of OREF_NULL means access to this is not permitted.
319  */
320 RexxString *SecurityManager::checkRequiresAccess(RexxString *name, RexxObject *&securityManager)
321 {
322  // just return the same name if no manager object set.
323  if (manager == OREF_NULL)
324  {
325  return name;
326  }
327 
328  DirectoryClass *securityArgs = new_directory();
329  ProtectedObject p(securityArgs);
330 
331  /* add the program name */
332  securityArgs->put(name, GlobalNames::NAME);
333  /* did manager handle this? */
334  if (callSecurityManager(GlobalNames::REQUIRES, securityArgs))
335  {
336  // retrieve any security manager that the security manager wants us to use for
337  // a new file.
338  RexxObject *secObject = (RexxObject *)securityArgs->get(GlobalNames::SECURITYMANAGER);
339  if (secObject != OREF_NULL && secObject != TheNilObject)
340  {
341  securityManager = secObject;
342  }
343  // the name can be replaced by the security manager
344  return (RexxString *)securityArgs->get(GlobalNames::NAME);
345  }
346  // not handled, return the name unchanged
347  return name;
348 }
