182 /**
183  * The main entry point for spawned activities.
184  */
185 void Activity::runThread()
186 {
187  // establish the stack base pointer for control stack full detection.
188  // the stack base is the lowest address of the stack. The stack grows from the
189  // highest address downward. When it hits the line in the sand we've drawn,
190  // we raise a control stack full error.
191  stackLimit = currentThread.getStackBase() + errorRecoveryStack;
192 
193  for (;;)
194  {
195  // save the actitivation level in case there's an error unwind for an unhandled
196  // exception;
197  size_t activityLevel = 0;
198  // the thread might have terminated for a control stack issue
199  // so make sure checking is turned back on before trying to run
200  // anything
201  stackcheck = true;
202 
203  try
204  {
205  // wait for permission to run, then figure out what action
206  // we've been asked to take.
207  waitForRunPermission();
208  // told to exit. Most likely we were in the thread pool
209  // and the interpreer is shutting down
210  if (exit)
211  {
212  break;
213  }
214 
215  // we need to have the kernel lock before we can really start working
216  requestAccess();
217  activityLevel = getActivationLevel();
218 
219  // if we have a dispatch message set, send it send message to kick everything off
220  if (dispatchMessage != OREF_NULL)
221  {
222  MessageDispatcher dispatcher(dispatchMessage);
223  run(dispatcher);
224  }
225  else
226  {
227 
228  // this is a reply message start, just dispatch the Rexx code
229  topStackFrame->dispatch();
230  }
231  }
232  catch (ActivityException) // we've had a termination event, raise an error
233  {
234  // it's possible that we might not have the kernel lock when
235  // control returns to here. Make sure we have it.
236  if (ActivityManager::currentActivity != this)
237  {
238  requestAccess();
239  }
240  // process any error, including notifiying any listeners that this occurred.
241  error();
242  }
243 
244  // make sure we get restored to the same base activation level.
245  restoreActivationLevel(activityLevel);
246 
247  // this is a good place for checking if there are pending uninit objects.
248  memoryObject.checkUninitQueue();
249 
250  // cast off any items related to our initial dispatch.
251  dispatchMessage = OREF_NULL;
252 
253  // no longer an active activity
254  deactivate();
255 
256  // make sure we clean up any mutexes we hold
257  cleanupMutexes();
258 
259  // reset our semaphores
260  runSem.reset();
261  guardSem.reset();
262 
263  // try to pool this. If the ActivityManager doesn't take it,
264  // we go into termination mode
265  if (!instance->poolActivity(this))
266  {
267  releaseAccess();
268  break;
269  }
270  // release the kernel lock and go wait for more work
271  releaseAccess();
272  }
273  // tell the activity manager we're going away
274  ActivityManager::activityEnded(this);
275 }

341 /**
342  * Initialize an Activity object.
343  *
344  * @param createThread
345  * Indicates whether we're going to be running on the
346  * current thread, or creating a new thread.
347  */
348 Activity::Activity(GlobalProtectedObject &p, bool createThread)
349 {
350  // we need to protect this from garbage collection while constructing.
351  // unfortunately, we can't use ProtectedObject because that requires an
352  // active activity, which we can't guarantee at this point.
353  p = this;
354 
355  // globally clear the object because we could be reusing the
356  // object storage
357  clearObject();
358  // we need a stack that activations can use
359  activations = new_internalstack(ACT_STACK_SIZE);
360  // The framestack creates space for expression stacks and local variable tables
361  frameStack.init();
362  // an activity has a couple of semaphores it uses to synchronize execution.
363  runSem.create();
364  guardSem.create();
365  activationStackSize = ACT_STACK_SIZE;
366  // stack checking is enabled by default
367  stackcheck = true;
368  // the activity keeps a pointer to a set of numeric settings that
369  // it can swap to global settings when it becomes the
370  // active activity.
371  numericSettings = Numerics::getDefaultSettings();
372  // generate a fresh random seed number
373  generateRandomNumberSeed();
374 
375  // when loading requires, we need to keep track of which ones are
376  // being activively loaded on this thread to sort out reference cycles.
377  requiresTable = new_string_table();
378  // create a stack frame for this running context
379  createNewActivationStack();
380 
381  // if we have been asked to create a new thread, we create the
382  // system thread instance here
383  if (createThread)
384  {
385  // we need to make sure this is cleared, since we use this
386  // to wait for dispatch at thread start up.
387  runSem.reset();
388  // we need to enter this thread already marked as active, otherwise
389  // the main thread might shut us down before we get a chance to perform
390  // whatever function we're getting asked to run.
391  activate();
392 
393  // we create the new thread with the same size stack as the parent thread
394  currentThread.create(this, currentThread.getStackSize());
395  }
396  // we are creating an activity that represents the thread
397  // we're currently executing on.
398  else
399  {
400  // run on the current thread
401  currentThread.useCurrentThread();
402  // reset the stack base for this thread.
403  // establish the stack base pointer for control stack full detection.
404  stackLimit = currentThread.getStackBase() + errorRecoveryStack;
405  }
406 }

435 /**
436  * Generate a fresh random number seed.
437  */
438 void Activity::generateRandomNumberSeed()
439 {
440  // we use our own random number generator, but it's perfectly
441  // to use the C library one to generate a random initial seed.
442  randomSeed = 0;
443  // rand uses a static seed, so we have to set a
444  // reasonably random starting seed. Using both the time and
445  // clock makes things a little more random, and the process
446  // id and thread id lessens the chance that we get a repeat.
447  srand((int)time(NULL) + (int)clock() + SysProcess::getPid() + (intptr_t)currentThread.getThreadID());
448 
449  // the random number implementations vary on how large the
450  // values are, but the are guaranteed to be at least 16-bit
451  // quanties. We'll compose the inital seed by shifting and xoring
452  // 4 values generated by rand().
453  for (size_t i = 0; i < 4; i++)
454  {
455  randomSeed = randomSeed << 16 ^ rand();
456  }
457 }

578 /**
579  * Raise a condition, with potential trapping.
580  * Also checks for potential ::OPTIONS condition overrides.
581  *
582  * @param condition The condition name.
583  * @param rc The rc value
584  * @param description
585  * The description value.
586  * @param additional the exception additional information.
587  * @param result The condition result info.
588  *
589  * @return true if this was trapped via CALL ON, false for untrapped
590  * conditions.
591  */
592 bool Activity::raiseCondition(RexxString *condition, RexxObject *rc, RexxObject *description, RexxObject *additional, RexxObject *result)
593 {
594  // if we have ::OPTIONS condition SYNTAX set on the package, we
595  // raise a SYNTAX error instead of raising the condition
596  RexxActivation *activation = getCurrentRexxFrame();
597  if (activation != OREF_NULL)
598  {
599  if (activation->isErrorSyntaxEnabled() && condition->strCompare(GlobalNames::ERRORNAME))
600  {
601  reportException(Error_Execution_error_syntax, description, result);
602  }
603  else if (activation->isFailureSyntaxEnabled() && condition->strCompare(GlobalNames::FAILURE))
604  {
605  reportException(Error_Execution_failure_syntax, description, result);
606  }
607  else if (activation->isLostdigitsSyntaxEnabled() && condition->strCompare(GlobalNames::LOSTDIGITS))
608  {
609  reportException(Error_Execution_lostdigits_syntax, description);
610  }
611  else if (activation->isNostringSyntaxEnabled() && condition->strCompare(GlobalNames::NOSTRING))
612  {
613  reportException(Error_Execution_nostring_syntax, description);
614  }
615  else if (activation->isNotreadySyntaxEnabled() && condition->strCompare(GlobalNames::NOTREADY))
616  {
617  reportException(Error_Execution_notready_syntax, description);
618  }
619  }
620 
621  // check the context first to see if this will be trapped. Creating
622  // the condition object is pretty expensive, so we want to avoid
623  // creating this if we'll just end up throwing it away.
624  if (!checkCondition(condition))
625  {
626  return false;
627  }
628 
629  // just create a condition object and process the traps.
630  DirectoryClass *conditionObj = createConditionObject(condition, rc, description, additional, result);
631  return raiseCondition(conditionObj);
632 }

635 /**
636  * Test if a condition will be trapped before creating a
637  * condition object for it.
638  *
639  * @param condition
640  * The name of the condition being raised
641  *
642  * @return true if this will be trapped, false otherwise.
643  */
644 bool Activity::checkCondition(RexxString *condition)
645 {
646  // unwind the stack frame calling trap until we reach the first real Rexx activation
647  for (ActivationBase *activation = getTopStackFrame();
648  !activation->isStackBase();
649  activation = activation->getPreviousStackFrame())
650  {
651  // see if there is an activation interested in trapping this.
652  if (activation->willTrap(condition))
653  {
654  return true;
655  }
656 
657  // for a normal condition, we stop checking at the first Rexx activation.
658  if (isOfClass(Activation, activation))
659  {
660  return false;
661  }
662  }
663 
664  // reached the bottom of the stack and nothing handled this
665  return false;
666 }

669 /**
670  * Process condition trapping for a condition or syntax
671  * error.
672  *
673  * @param conditionObj
674  * The condition object that describes the condition.
675  *
676  * @return true if this was trapped, false otherwise. If trapped
677  * via a SIGNAL ON, this will NOT return to here.
678  */
679 bool Activity::raiseCondition(DirectoryClass *conditionObj)
680 {
681  // the condition has not been handled yet.
682  bool handled = false;
683  RexxString *condition = (RexxString *)conditionObj->get(GlobalNames::CONDITION);
684 
685  // unwind the stack frame calling trap until we reach the first real Rexx activation
686  for (ActivationBase *activation = getTopStackFrame();
687  !activation->isStackBase();
688  activation = activation->getPreviousStackFrame())
689  {
690  // see if there is an activation interested in trapping this.
691  handled = activation->trap(condition, conditionObj);
692  // for a normal condition, we stop checking at the first Rexx activation.
693  if (handled || isOfClass(Activation, activation))
694  {
695  break;
696  }
697  }
698 
699  // Control will not return here if the condition was trapped via
700  // SIGNAL ON SYNTAX. For CALL ON conditions, handled will be
701  // true if a trap is pending.
702 
703  return handled;
704 }

909  * Raise an exception on the current activity.
910  *
911  * @param errcode The syntax error code.
912  * @param description
913  * The associated description string.
914  * @param additional The message substitution parameters.
915  * @param result The message result.
916  */
917 void Activity::raiseException(RexxErrorCodes errcode, RexxString *description, ArrayClass *additional, RexxObject *result)
918 {
919  // during error processing, we need to request the string value of message
920  // substitution objects. It is possible that the string process will also
921  // cause a syntax error, resulting in a recursion loop. We snip that off here,
922  // by disallowing a nested error condition.
923  if (requestingString)
924  {
925  throw RecursiveStringError;
926  }
927 
928  // get the top-most stack frame and also the top Rexx frame
929  ActivationBase *topFrame = getTopStackFrame();
930  RexxActivation *activation = getCurrentRexxFrame();
931  // if we're raised within a real Rexx context, we need to deal with forwarded
932  // frames
933  if (topFrame == activation)
934  {
935  // unwind the stack until we find a frame that is not forwarded
936  while (activation != OREF_NULL && activation->isForwarded())
937  {
938  // terminate and remove this stack frame
939  popStackFrame(activation);
940  // grab the new current frame
941  activation = getCurrentRexxFrame();
942  }
943  }
944  else
945  {
946  activation = NULL; // raised from a native context, don't add to stack trace
947  }

950  conditionobj = createExceptionObject(errcode, description, additional, result);
951 
952  // and go raise as a condition
953  if (!raiseCondition(conditionobj))
954  {
955  // we're raising a syntax error and this was not trapped by the
956  // top condition. We start propagating this condition until we either
957  // find a frame that traps this or we run out of frames.
958  // fill in the propagation status
959  conditionobj->put(TheTrueObject, GlobalNames::PROPAGATED);
960  // if we have an Rexx context to work with, unwind to that point,
961  if (activation != OREF_NULL)
962  {
963  // unwind the frame to this point
964  unwindToFrame(activation);
965  popStackFrame(activation); // remove it from the stack
966  }
967  // go propagate
968  raisePropagate(conditionobj);
969  }

993  // are th minor code. The major code is above that.
994  wholenumber_t primary = (errcode / 1000) * 1000;
995

1022  {
1023  additional = new_array((size_t)0);
1024  }
1025 
1026  // add this in
1027  exobj->put(additional, GlobalNames::ADDITIONAL);
1028

1030  if (primary != errcode)
1031  {
1032  // build the message and add to the condition object
1033  RexxString *message = buildMessage(errcode, additional);
1034  exobj->put(message, GlobalNames::MESSAGE);
1035  }
1036  else
1037  {
1038  // make this explicitly .nil
1039  exobj->put(TheNilObject, GlobalNames::MESSAGE);
1040  }

1043  if (description == OREF_NULL)
1044  {
1045  // use an explicit null string
1046  exobj->put(GlobalNames::NULLSTRING, GlobalNames::DESCRIPTION);

1065  exobj->put(TheFalseObject, GlobalNames::PROPAGATED);

1232 /**
1233  * Perform any required message substitutions on the secondary
1234  * error message.
1235  *
1236  * @param message The error text. Substitution parameters are marked
1237  * inside the messages.
1238  * @param additional The array of values to substitute.
1239  *
1240  * @return The formatted string.
1241  */
1242 RexxString* Activity::messageSubstitution(RexxString *message, ArrayClass *additional)
1243 {
1244  size_t substitutions = additional->size();
1245  // build this up into a mutable buffer.
1246  Protected<MutableBuffer> newMessage = new MutableBuffer();
1247 
1248  const char *messageData = message->getStringData();
1249  size_t searchOffset = 0;
1250 
1251  for (size_t i = 1; i <= substitutions; i++)
1252  {
1253  // search for the substibution value
1254  size_t subposition = message->pos(GlobalNames::AND, searchOffset);
1255  // no more '&' markers found in the message, we're done building.
1256  if (subposition == 0)
1257  {
1258  break;
1259  }
1260 
1261  // append the next message section to the buffer
1262  newMessage->append(messageData + searchOffset, subposition - searchOffset - 1);
1263  // this will be where we start searching for the next one.
1264  searchOffset = subposition + 1;
1265 
1266  // get the character following the '&'. This should be a numeric
1267  // substitution number. We only support digits 1-9.
1268  // NOTE: pos returns the position origin 1, but getChar()
1269  // is origin 0. This really picks up the next character.
1270  size_t selector = message->getChar(subposition);
1271  RexxString *stringVal = GlobalNames::NULLSTRING;
1272  // if this a bad selector, substitute anyway, but give a generic marker
1273  if (selector < '0' || selector > '9')
1274  {
1275  stringVal = new_string("<BAD MESSAGE>");
1276  }
1277  else
1278  {
1279  // now get the value from the substitutions array
1280  selector -= '0';
1281  // still in range? Convert to a string value and add to the message
1282  if (selector <= substitutions)
1283  {
1284  RexxInternalObject *value = additional->get(selector);
1285  if (value != OREF_NULL)
1286  {
1287  // do this carefully, we need to try to avoid recursion errors
1288  requestingString = true;
1289  stackcheck = false;
1290  // save the actitivation level in case there's an error unwind for an unhandled
1291  // exception;
1292  size_t activityLevel = getActivationLevel();
1293  // do this under a try block to intercept problems
1294  try
1295  {
1296  stringVal = value->stringValue();
1297  }
1298  catch (ActivityException)
1299  {
1300  // we use the default object name if there are any problems
1301  stringVal = value->defaultName();
1302  }
1303 
1304  // make sure we get restored to the same base activation level.
1305  restoreActivationLevel(activityLevel);
1306  // returning to normal operations.
1307  requestingString = false;
1308  stackcheck = true;
1309  }
1310  }
1311  }
1312  // add on the message substitutions
1313  newMessage->append(stringVal);
1314  }
1315  // append the remainder of the message to the buffer and convert to a string.
1316  newMessage->append(messageData + searchOffset, message->getLength() - searchOffset);
1317  return newMessage->makeString();
1318 }
