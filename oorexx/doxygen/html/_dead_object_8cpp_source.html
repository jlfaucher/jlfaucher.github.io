65 DeadObject *DeadObjectPool::findBestFit(size_t length)
66 /******************************************************************************/
67 /* Function: Find an object that is the best fit for a requested length. */
68 /* To get the best fit, we either want an object that fits well enough that */
69 /* we don't need to create a trailing dead object from the remainder, OR */
70 /* we return the largest possible object so that we can create the largest */
71 /* possible remainder object. */
72 /******************************************************************************/
73 {
74  DeadObject *newObject = anchor.next;
75  DeadObject *largest = NULL;
76  size_t largestSize = 0;
77  size_t deadLength = newObject->getObjectSize();
78 
79  for (; deadLength != 0; deadLength = newObject->getObjectSize()) {
80  if (deadLength >= length) {
81  /* if within an allocation unit of the request size (which should */
82  /* be common, since we round to a higher boundary for */
83  /* large allocations), use this block */
84  if ((deadLength - length) < Memory::VeryLargeObjectGrain) {
85  newObject->remove();
86  logHit();
87  return newObject;
88  }
89  /* keep track of the largest block. We want to */
90  /* suballocate that if we can't find a good fit. */
91  else if (deadLength > largestSize) {
92  largestSize = deadLength;
93  largest = newObject;
94  }
95  }
96  newObject = newObject->next;
97  }
98  /* we didn't find a close fit, so use the largest that will */
99  /* work, if we found one. */
100  if (largest != NULL) {
101  logHit();
102  largest->remove();
103  }
104  else {
105  logMiss();
106  }
107  return largest;
108 }

111 DeadObject *DeadObjectPool::findSmallestFit(size_t minSize)
112 /******************************************************************************/
113 /* Function: Find the smallest object in the pool that will hold the length. */
114 /******************************************************************************/
115 {
116  DeadObject *newObject = anchor.next;
117  DeadObject *smallest = NULL;
118  size_t smallestSize = Memory::MaximumObjectSize;
119 
120  while (newObject->isReal()) {
121  size_t deadLength = newObject->getObjectSize();
122  /* does this fit the request size? */
123  if (deadLength >= minSize && deadLength < smallestSize) {
124  /* remember this for the end. */
125  smallestSize = deadLength;
126  smallest = newObject;
127  /* did we get an exact fit? No need to look at any */
128  /* others then. */
129  if (deadLength == minSize) {
130  break;
131  }
132  }
133  newObject = newObject->next;
134  }
135  /* if we found a fit, remove the block and return it. */
136  if (smallest != NULL) {
137  smallest->remove();
138  logHit();
139  }
140  else {
141  logMiss();
142  }
143  return smallest;
144 }

147 void DeadObjectPool::addSortedBySize(DeadObject *obj)
148 /******************************************************************************/
149 /* Function: Add an object to the pool sorted by size (ascending order). */
150 /******************************************************************************/
151 {
152 // checkObjectOverlap(obj);
153 // checkObjectGrain(obj);
154  /* we start with the first element in the chain as the */
155  /* insertion point. If the chain is empty, we'll terminate the */
156  /* loop immediately and use the anchor as the insertion point, */
157  /* which gives us the result we want. */
158  DeadObject *insertPoint = anchor.next;
159  size_t size = obj->getObjectSize();
160 
161  while (insertPoint->isReal()) {
162  /* if the current block is larger than the one we're */
163  /* inserting, we've found our spot. */
164  if (insertPoint->getObjectSize() >= size) {
165  break;
166  }
167  insertPoint = insertPoint->next;
168  }
169  /* insert this at the given point */
170  insertPoint->insertBefore(obj);
171 }
