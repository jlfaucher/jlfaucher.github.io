50 /**
51  * Initialize an END IF instruction.
52  *
53  * @param _parent The parent IF instruction.
54  */
55 RexxInstructionEndIf::RexxInstructionEndIf(RexxInstructionIf *_parent)
56 {
57  // set the default type...this can change based on additional events.
58  setType(KEYWORD_ENDTHEN);
59  // tell the parent IF that we exist
60  parent = _parent;
61  parent->setEndInstruction(this);
62 
63  // now check to see what our parent instruction is and change
64  // our type based on the parent type.
65 
66  // actually following an ELSE? We're really an END ELSE.
67  if (parent->instructionType == KEYWORD_ELSE)
68  {
69  setType(KEYWORD_ENDELSE);
70  }
71  // is our IF really a WHEN? Again, change to match.
72  else if (parent->instructionType == KEYWORD_WHENTHEN)
73  {
74  setType(KEYWORD_ENDWHEN);
75  }
76 }

126 /**
127  * Execute an END IF instuction.
128  *
129  * @param context The current execution context.
130  * @param stack The current evaluation stack.
131  */
132 void RexxInstructionEndIf::execute(RexxActivation *context, ExpressionStack *stack)
133 {
134  // end of a WHEN block? This means we've executed a WHEN clause of a SELECT,
135  // which terminates the SELECT. The next instruction is the one after the closing
136  // END.
137  if (this->instructionType == KEYWORD_ENDWHEN)
138  {
139  // remove the select block and reset the indent
140  context->terminateBlockInstruction();
141  // branch to the END of the SELECT.
142  context->setNext(else_end->nextInstruction);
143  }
144  else
145  {
146  // We indent twice at the THEN, so we need to remove both of these.
147  context->unindentTwice();
148 
149  // There are three situation here 1) We're following a THEN clause
150  // with no ELSE, 2) we're following an ELSE, or 3) we're following
151  // an THEN with an ELSE. For 1&2, we automatically fall through
152  // to the next instruction. For 3, we need to jump to the instruction
153  // following the else.
154  if (else_end != OREF_NULL)
155  {
156  context->setNext(else_end->nextInstruction);
157  }
158  }
159 }
