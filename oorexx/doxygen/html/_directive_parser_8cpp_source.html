61 /**
62  * Parse a directive instruction.
63  */
64 void LanguageParser::nextDirective()
65 {
66  // we are in a context where everthing we check is expected to be a
67  // directive.
68 
69  // step to the next clause
70  nextClause();
71 
72  // if we've hit the end, we're done.
73  if (noClauseAvailable())
74  {
75  return;
76  }
77 
78  RexxToken *token = nextReal();
79  // we expect anything found here to be the start of a directive
80  if (!token->isType(TOKEN_DCOLON))
81  {
82  syntaxError(Error_Translation_bad_directive);
83  }
84 
85  // and this needs to be followed by a symbol keyword
86  token = nextReal();
87  if (!token->isSymbol())
88  {
89  syntaxError(Error_Symbol_expected_directive);
90  }
91 
92  // and process each of the named directives
93  switch (token->keyDirective())
94  {
95  // define a class
96  case DIRECTIVE_CLASS:
97  classDirective();
98  break;
99 
100  // create a method
101  case DIRECTIVE_METHOD:
102  methodDirective();
103  break;
104 
105  // create a routine
106  case DIRECTIVE_ROUTINE:
107  routineDirective();
108  break;
109 
110  // require a package file or library
111  case DIRECTIVE_REQUIRES:
112  requiresDirective();
113  break;
114 
115  // define an attribute
116  case DIRECTIVE_ATTRIBUTE:
117  attributeDirective();
118  break;
119 
120  // define a constant
121  case DIRECTIVE_CONSTANT:
122  constantDirective();
123  break;
124 
125  // define package-wide options
126  case DIRECTIVE_OPTIONS:
127  optionsDirective();
128  break;
129 
130  // create package named resources
131  case DIRECTIVE_RESOURCE:
132  resourceDirective();
133  break;
134 
135  // create package annotations
136  case DIRECTIVE_ANNOTATE:
137  annotateDirective();
138  break;
139 
140  // an unknown directive
141  default:
142  syntaxError(Error_Translation_bad_directive);
143  break;
144  }
145 }

148 /**
149  * Verify that no code follows a directive except for more
150  * directive instructions.
151  *
152  * @param errorCode The error code to issue if this condition was violated.
153  */
154 void LanguageParser::checkDirective(RexxErrorCodes errorCode)
155 {
156  // save the clause location so we can reset for errors
157  SourceLocation location = clauseLocation;
158 
159  // step to the next clause...if there is one, it must
160  // be a directive.
161  nextClause();
162  if (clauseAvailable())
163  {
164  // The first token of the next instruction must be a ::
165  RexxToken *token = nextReal();
166 
167  if (!token->isType(TOKEN_DCOLON))
168  {
169  syntaxError(errorCode);
170  }
171  // back up and push the clause back
172  firstToken();
173  reclaimClause();
174  }
175  // this resets the current clause location so that any errors on the current
176  // clause detected after the clause check reports this on the correct line
177  // number
178  clauseLocation = location;
179 }

182 /**
183  * Test if a directive is followed by a body of Rexx code
184  * instead of another directive or the end of the source.
185  *
186  * @return True if there is a non-directive clause following the current
187  * clause.
188  */
189 bool LanguageParser::hasBody()
190 {
191  // assume there's no body here
192  bool result = false;
193 
194  // if we have anything to look at, see if it is a directive or not.
195  nextClause();
196  if (clauseAvailable())
197  {
198  // we have a clause, now check if this is a directive or not
199  RexxToken *token = nextReal();
200  // not a "::", not a directive, which means we have real code to deal with
201  result = !token->isType(TOKEN_DCOLON);
202  // reset this clause entirely so we can start parsing for real.
203  firstToken();
204  reclaimClause();
205  }
206  return result;
207 }

278 /**
279  * Parse a class reference for a ::CLASS directive. This
280  * could be a subclass, mixinclass, metaclass, or inherits
281  * class name.
282  *
283  * @param error The error to raise for nothing there.
284  *
285  * @return A ClassResolver to look up the class reference.
286  */
287 ClassResolver *LanguageParser::parseClassReference(RexxErrorCodes error)
288 {
289  // this is a required string or symbol value
290  RexxToken *token = nextReal();
291  if (!token->isSymbolOrLiteral())
292  {
293  syntaxError(error, token);
294  }
295 
296  // the class specification can be a literal, a symbol, or a
297  // symbol:symbol namespace specification. If this is a string,
298  // we're done
299  if (token->isLiteral())
300  {
301  return new ClassResolver(OREF_NULL, commonString(token->upperValue()));
302  }
303  else
304  {
305  // this is a symbol form. Get the name, which might be either
306  // a namespace name or the class name. We need to parse a little
307  // further to determine.
308  RexxString *name = token->value();
309  token = nextReal();
310 
311  // if this is not a colon, put it back and return the single
312  // name form of resolver
313  if (!token->isType(TOKEN_COLON))
314  {
315  previousToken();
316  return new ClassResolver(OREF_NULL, name);
317  }
318 
319  // ok, this should be of the form symbol:symbol.
320  token = nextReal();
321  if (!token->isSymbol())
322  {
323  syntaxError(Error_Symbol_expected_namespace_class);
324  }
325  // this is a qualified class reference
326  return new ClassResolver(name, token->value());
327  }
328 }

331 /**
332  * Process a ::CLASS directive for a source file.
333  */
334 void LanguageParser::classDirective()
335 {
336  // first token is the name, which must be a symbol or string name
337  RexxToken *token = nextReal();
338  if (!token->isSymbolOrLiteral())
339  {
340  syntaxError(Error_Symbol_or_string_class);
341  }
342 
343  // get the class name
344  RexxString *name = token->value();
345  // and we export this name in uppercase
346 
347  RexxString *public_name = commonString(name->upper());
348  // check for a duplicate class
349  if (isDuplicateClass(public_name))
350  {
351  syntaxError(Error_Translation_duplicate_class);
352  }
353 
354  // create a class directive and add this to the dependency list
355  activeClass = new ClassDirective(name, public_name, clause);
356 
357  // if we have any computed constants associated with this class, we need to
358  // keep some special information about the constant expressions.
359 
360  // a table of variables...starting with the special variables we allocated space for.
361  // we use the initial "clean" version, then save it after each parse constant associated
362  // with the class.
363  constantVariables = OREF_NULL;
364  // also start with the default index counter
365  constantVariableIndex = RexxLocalVariables::FIRST_VARIABLE_INDEX;
366  // and also the largest stack needed to evaluate these.
367  constantMaxStack = 0;
368  // add this to our directives list.
369  addClassDirective(public_name, activeClass);
370 
371  // we're using default scope.
372  AccessFlag accessFlag = DEFAULT_ACCESS_SCOPE;
373 
374  // now we have a bunch of option keywords to handle, which can
375  // only be specified once each.
376  for (;;)
377  {
378  // real simple class definition.
379  token = nextReal();
380  if (token->isEndOfClause())
381  {
382  break;
383  }
384  // all options are symbols
385  else if (!token->isSymbol())
386  {
387  syntaxError(Error_Invalid_subkeyword_class, token);
388  }
389  else
390  {
391  // directive sub keywords are also table based
392  DirectiveSubKeyword type = token->subDirective();
393  switch (type)
394  {
395  // ::CLASS name METACLASS metaclass
396  case SUBDIRECTIVE_METACLASS:
397  // can't be a duplicate keyword
398  if (activeClass->getMetaClass() != OREF_NULL)
399  {
400  syntaxError(Error_Invalid_subkeyword_class, token);
401  }
402 
403  // parse off the class name and set in the directive
404  activeClass->setMetaClass(parseClassReference(Error_Symbol_or_string_metaclass));
405  break;
406 
407  // ::CLASS name PUBLIC
408  case SUBDIRECTIVE_PUBLIC:
409  // have we already seen an ACCESS flag? This is an error
410  if (accessFlag != DEFAULT_ACCESS_SCOPE)
411  {
412  syntaxError(Error_Invalid_subkeyword_class, token);
413  }
414  accessFlag = PUBLIC_SCOPE;
415  // set the access in the active class.
416  activeClass->setPublic();
417  break;
418 
419  // ::CLASS name PRIVATE
420  case SUBDIRECTIVE_PRIVATE:
421  if (accessFlag != DEFAULT_ACCESS_SCOPE)
422  {
423  syntaxError(Error_Invalid_subkeyword_class, token);
424  }
425  accessFlag = PRIVATE_SCOPE;
426 
427  // don't need to set anything in the directive...this is the default
428  break;
429 
430  // ::CLASS name SUBCLASS sub
431  case SUBDIRECTIVE_SUBCLASS:
432  // If we have a subclass set already, this is an error
433  if (activeClass->getSubClass() != OREF_NULL)
434  {
435  syntaxError(Error_Invalid_subkeyword_class, token);
436  }
437 
438  // parse off the class name and set in the directive
439  activeClass->setSubClass(parseClassReference(Error_Symbol_or_string_subclass));
440  break;
441 
442  // ::CLASS name MIXINCLASS mclass
443  case SUBDIRECTIVE_MIXINCLASS:
444  // If we have a subclass set already, this is an error
445  // NOTE: setting a mixin class also defines the subclass...
446  if (activeClass->getSubClass() != OREF_NULL)
447  {
448  syntaxError(Error_Invalid_subkeyword_class, token);
449  }
450 
451  // parse off the class name and set in the directive
452  activeClass->setMixinClass(parseClassReference(Error_Symbol_or_string_mixinclass));
453  break;
454 
455  // ::CLASS name INHERIT classes
456  case SUBDIRECTIVE_INHERIT:
457  // all tokens after the keyword will be consumed by the INHERIT keyword.
458  token = nextReal();
459  if (token->isEndOfClause())
460  {
461  syntaxError(Error_Symbol_or_string_inherit, token);
462  }
463 
464  while (!token->isEndOfClause())
465  {
466  // backup for the parsing
467  previousToken();
468 
469  activeClass->addInherits(parseClassReference(Error_Symbol_or_string_inherit));
470  token = nextReal();
471  }
472  // step back a token for final completion checks
473  previousToken();
474  break;
475 
476  // ::CLASS name ABSTRACT
477  case SUBDIRECTIVE_ABSTRACT:
478  // already been specified? this is an error
479  if (activeClass->isAbstract())
480  {
481  syntaxError(Error_Invalid_subkeyword_class, token);
482  }
483 
484  // mark this as abstract
485  activeClass->setAbstract();
486  break;
487 
488  // invalid keyword
489  default:
490  syntaxError(Error_Invalid_subkeyword_class, token);
491  break;
492  }
493  }
494  }
495 }

498 /**
499  * check for a duplicate method.
500  *
501  * @param name The name to check.
502  * @param classMethod
503  * Indicates whether this is a check for a CLASS or INSTANCE method.
504  * @param errorMsg
505  * The error code to use if there is a duplicate.
506  */
507 void LanguageParser::checkDuplicateMethod(RexxString *name, bool classMethod, RexxErrorCodes errorMsg)
508 {
509  // no previous ::CLASS directive?
510  if (activeClass == OREF_NULL)
511  {
512  // cannot create unattached class methods.
513  if (classMethod)
514  {
515  syntaxError(Error_Translation_missing_class);
516  }
517  // duplicate method name?
518  if (unattachedMethods->entry(name) != OREF_NULL)
519  {
520  syntaxError(errorMsg);
521  }
522  }
523  else
524  { /* add the method to the active class*/
525  // adding the method to the active class
526  if (activeClass->checkDuplicateMethod(name, classMethod))
527  {
528  syntaxError(errorMsg);
529  }
530  }
531 }

626 /**
627  * Process a ::METHOD directive in a source file.
628  */
629 void LanguageParser::methodDirective()
630 {
631  // set default modifiers
632  AccessFlag accessFlag = DEFAULT_ACCESS_SCOPE;
633  ProtectedFlag protectedFlag = DEFAULT_PROTECTION;
634  GuardFlag guardFlag = DEFAULT_GUARD;
635  // other attributes of methods
636  bool isClass = false;
637  bool isAttribute = false;
638  bool isAbstract = false;
639  RexxString *externalname = OREF_NULL; // not an external method yet
640  RexxString *delegateName = OREF_NULL; // no delegate target yet
641 
642  // method name must be a symbol or string
643  RexxToken *token = nextReal();
644  if (!token->isSymbolOrLiteral())
645  {
646  syntaxError(Error_Symbol_or_string_method, token);
647  }
648 
649  // this is the method internal name
650  RexxString *name = token->value();
651  // this is the look up name
652  RexxString *internalname = commonString(name->upper());
653 
654  // now process any additional option keywords
655  for (;;)
656  {
657  // finished on EOC
658  token = nextReal();
659  if (token->isEndOfClause())
660  {
661  break;
662  }
663  // option keywords must be symbols
664  else if (!token->isSymbol())
665  {
666  syntaxError(Error_Invalid_subkeyword_method, token);
667  }
668  else
669  {
670  // potential option keyword
671  switch (token->subDirective())
672  {
673  // ::METHOD name CLASS
674  case SUBDIRECTIVE_CLASS:
675  // no dups
676  if (isClass)
677  {
678  syntaxError(Error_Invalid_subkeyword_method, token);
679  }
680  isClass = true;
681  break;
682 
683  // ::METHOD name EXTERNAL extname
684  case SUBDIRECTIVE_EXTERNAL:
685  // no dup on external and abstract is mutually exclusive
686  if (externalname != OREF_NULL || delegateName != OREF_NULL || isAbstract)
687  {
688  syntaxError(Error_Invalid_subkeyword_method, token);
689  }
690  token = nextReal();
691  // external name must be a string value
692  if (!token->isLiteral())
693  {
694  syntaxError(Error_Symbol_or_string_external, token);
695  }
696  // this will be parsed at install time
697  externalname = token->value();
698  break;
699 
700  // ::METHOD name PRIVATE
701  case SUBDIRECTIVE_PRIVATE:
702  // has an access flag already been specified?
703  if (accessFlag != DEFAULT_ACCESS_SCOPE)
704  {
705  syntaxError(Error_Invalid_subkeyword_method, token);
706  }
707  accessFlag = PRIVATE_SCOPE;
708  break;
709 
710  // ::METHOD name PACKAGE
711  case SUBDIRECTIVE_PACKAGE:
712  // has an access flag already been specified?
713  if (accessFlag != DEFAULT_ACCESS_SCOPE)
714  {
715  syntaxError(Error_Invalid_subkeyword_method, token);
716  }
717  accessFlag = PACKAGE_SCOPE;
718  break;
719 
720  // ::METHOD name PUBLIC
721  case SUBDIRECTIVE_PUBLIC:
722  // has an access flag already been specified?
723  if (accessFlag != DEFAULT_ACCESS_SCOPE)
724  {
725  syntaxError(Error_Invalid_subkeyword_method, token);
726  }
727 
728  accessFlag = PUBLIC_SCOPE;
729  break;
730 
731  // ::METHOD name PROTECTED
732  case SUBDIRECTIVE_PROTECTED:
733  // had a protection flag specified already?
734  if (protectedFlag != DEFAULT_PROTECTION)
735  {
736  syntaxError(Error_Invalid_subkeyword_method, token);
737  }
738  protectedFlag = PROTECTED_METHOD;
739  break;
740 
741  // ::METHOD name UNPROTECTED
742  case SUBDIRECTIVE_UNPROTECTED:
743  // had a protection flag specified already?
744  if (protectedFlag != DEFAULT_PROTECTION)
745  {
746  syntaxError(Error_Invalid_subkeyword_method, token);
747  }
748  protectedFlag = UNPROTECTED_METHOD;
749  break;
750 
751  // ::METHOD name UNGUARDED
752  case SUBDIRECTIVE_UNGUARDED:
753  // already had a guard specification?
754  if (guardFlag != DEFAULT_GUARD)
755  {
756  syntaxError(Error_Invalid_subkeyword_method, token);
757  }
758 
759  guardFlag = UNGUARDED_METHOD;
760  break;
761 
762  // ::METHOD name GUARDED
763  case SUBDIRECTIVE_GUARDED:
764  // already had a guard specification?
765  if (guardFlag != DEFAULT_GUARD)
766  {
767  syntaxError(Error_Invalid_subkeyword_method, token);
768  }
769 
770  guardFlag = GUARDED_METHOD;
771  break;
772 
773  // ::METHOD name ATTRIBUTE
774  case SUBDIRECTIVE_ATTRIBUTE:
775  // check for duplicates
776  if (isAttribute)
777  {
778  syntaxError(Error_Invalid_subkeyword_method, token);
779  }
780 
781  isAttribute = true;
782  break;
783 
784  // ::METHOD name ABSTRACT
785  case SUBDIRECTIVE_ABSTRACT:
786  // can't have dups or external name or be a delegate
787  if (isAbstract || externalname != OREF_NULL || delegateName != OREF_NULL)
788  {
789  syntaxError(Error_Invalid_subkeyword_method, token);
790  }
791 
792  isAbstract = true;
793  break;
794 
795  // ::METHOD name DELEGATE property
796  case SUBDIRECTIVE_DELEGATE:
797  // no dup on external and abstract is mutually exclusive
798  if (externalname != OREF_NULL || delegateName != OREF_NULL || isAbstract)
799  {
800  syntaxError(Error_Invalid_subkeyword_method, token);
801  }
802 
803  token = nextReal();
804  // delegate name must be a symbol
805  if (!token->isSymbol())
806  {
807  syntaxError(Error_Symbol_expected_delegate, token);
808  }
809  // get the name of the attribute used for the forwarding
810  delegateName = token->value();
811  break;
812 
813  // something invalid
814  default:
815  syntaxError(Error_Invalid_subkeyword_method, token);
816  break;
817  }
818  }
819  }
820 
821  // go check for a duplicate and validate the use of the CLASS modifier
822  checkDuplicateMethod(internalname, isClass, Error_Translation_duplicate_method);
823 
824  Protected<MethodClass> _method;
825 
826  // handle delegates first. A delegate method can also be an attribute, which really
827  // just means we produce two delegate methods
828  if (delegateName != OREF_NULL)
829  {
830  // now get a variable retriever to get the property
831  RexxVariableBase *retriever = getRetriever(delegateName);
832 
833  // cannot have code following an method with the delegate keyword
834  checkDirective(Error_Translation_delegate_method);
835 
836  if (isAttribute)
837  {
838  // now get this as the setter method.
839  RexxString *setterName = commonString(internalname->concatWithCstring("="));
840  // need to check for duplicates on that too
841  checkDuplicateMethod(setterName, isClass, Error_Translation_duplicate_method);
842  // create the method pair and quit.
843  createDelegateMethod(setterName, retriever, isClass, accessFlag, protectedFlag, guardFlag, true);
844  }
845  // create the method pair and quit.
846  createDelegateMethod(internalname, retriever, isClass, accessFlag, protectedFlag, guardFlag, isAttribute);
847  return;
848  }
849  // is this an attribute method?
850  else if (isAttribute)
851  {
852  // now get this as the setter method.
853  RexxString *setterName = commonString(internalname->concatWithCstring("="));
854  // need to check for duplicates on that too
855  checkDuplicateMethod(setterName, isClass, Error_Translation_duplicate_method);
856 
857  // cannot have code following an method with the attribute keyword
858  checkDirective(Error_Translation_attribute_method);
859  // this might be externally defined setters and getters.
860  if (externalname != OREF_NULL)
861  {
862  RexxString *l = OREF_NULL;
863  RexxString *p = OREF_NULL;
864  decodeExternalMethod(internalname, externalname, l, p);
865  Protected<RexxString> library = l;
866  Protected<RexxString> procedure = p;
867  Protected<RexxString> getName = procedure->concatToCstring("GET");
868  Protected<RexxString> setName = procedure->concatToCstring("SET");
869  // now create both getter and setting methods from the information.
870  _method = createNativeMethod(internalname, library, getName);
871  _method->setAttributes(accessFlag, protectedFlag, guardFlag);
872  // mark this as an attribute method
873  _method->setAttribute();
874  // add to the compilation
875  addMethod(internalname, _method, isClass);
876 
877  _method = createNativeMethod(setterName, library, setName);
878  _method->setAttributes(accessFlag, protectedFlag, guardFlag);
879  // add to the compilation
880  addMethod(setterName, _method, isClass);
881  }
882  // an abstract method as an attribute
883  else if (isAbstract)
884  {
885  // create the method pair and quit.
886  createAbstractMethod(internalname, isClass, accessFlag, protectedFlag, guardFlag, true);
887  createAbstractMethod(setterName, isClass, accessFlag, protectedFlag, guardFlag, true);
888  }
889  else
890  {
891  // now get a variable retriever to get the property
892  RexxVariableBase *retriever = getRetriever(name);
893 
894  // create the method pair and quit.
895  createAttributeGetterMethod(internalname, retriever, isClass, accessFlag, protectedFlag, guardFlag);
896  createAttributeSetterMethod(setterName, retriever, isClass, accessFlag, protectedFlag, guardFlag);
897  }
898  return;
899  }
900  // abstract method?
901  else if (isAbstract)
902  {
903  // check that there is no code following this method.
904  checkDirective(Error_Translation_abstract_method);
905  // this uses a special code block
906  Protected<BaseCode> code = new AbstractCode();
907  _method = new MethodClass(name, code);
908  // make sure the method is marked abstract
909  _method->setAbstract();
910  }
911  // regular Rexx code method?
912  else if (externalname == OREF_NULL)
913  {
914  // NOTE: It is necessary to translate the block and protect the code
915  // before allocating the MethodClass object. The new operator allocates the
916  // the object first, then evaluates the constructor arguments after the allocation.
917  // Since the translateBlock() call will allocate a lot of new objects before returning,
918  // there's a high probability that the method object can get garbage collected before
919  // there is any opportunity to protect the object.
920  Protected<RexxCode> code = translateBlock();
921 
922  // go do the next block of code
923  _method = new MethodClass(name, code);
924  }
925  // external method
926  else
927  {
928  RexxString *l = OREF_NULL;
929  RexxString *p = OREF_NULL;
930  decodeExternalMethod(internalname, externalname, l, p);
931  Protected<RexxString> library = l;
932  Protected<RexxString> procedure = p;
933 
934  // check that there this is only followed by other directives.
935  checkDirective(Error_Translation_external_method);
936  // and make this into a method object.
937  _method = createNativeMethod(name, library, procedure);
938  }
939  _method->setAttributes(accessFlag, protectedFlag, guardFlag);
940  // add to the compilation
941  addMethod(internalname, _method, isClass);
942 }

1412  ProtectedFlag protectedFlag = DEFAULT_PROTECTION;
1413  GuardFlag guardFlag = DEFAULT_GUARD;
1414  // by default, we create both methods for the attribute.
1415  AttributeType style = ATTRIBUTE_BOTH;
1416  bool isClass = false; // default is an instance method

1422  if (!token->isSymbolOrLiteral())
1423  {
1424  syntaxError(Error_Symbol_or_string_attribute, token);
1425  }
1426 
1427  // get the attribute name and the internal value that we create the method names with
1428  RexxString *name = token->value();
1429  RexxString *internalname = commonString(name->upper());
1430  RexxString *externalname = OREF_NULL;
1431 
1432  // process options
1433  for (;;)
1434  {
1435  // if last token we're out of here.
1436  token = nextReal();
1437  if (token->isEndOfClause())
1438  {
1439  break;
1440  }
1441  // all options must be symbols
1442  else if (!token->isSymbol())
1443  {
1444  syntaxError(Error_Invalid_subkeyword_attribute, token);
1445  }
1446  else
1447  {
1448  switch (token->subDirective())
1449  {
1450  // GET define this as a attribute get method
1451  case SUBDIRECTIVE_GET:
1452  // only one per customer
1453  if (style != ATTRIBUTE_BOTH)
1454  {
1455  syntaxError(Error_Invalid_subkeyword_attribute, token);
1456  }
1457  style = ATTRIBUTE_GET;
1458  break;
1459 
1460  // SET create an attribute assignment method
1461  case SUBDIRECTIVE_SET:
1462  // only one of GET/SET allowed
1463  if (style != ATTRIBUTE_BOTH)
1464  {
1465  syntaxError(Error_Invalid_subkeyword_attribute, token);
1466  }
1467  style = ATTRIBUTE_SET;
1468  break;
1469 
1470 
1471  // ::ATTRIBUTE name CLASS creating class methods
1472  case SUBDIRECTIVE_CLASS:
1473  // no dups allowed
1474  if (isClass)
1475  {
1476  syntaxError(Error_Invalid_subkeyword_attribute, token);
1477  }
1478  isClass = true;
1479  break;
1480 
1481  // private access?
1482  case SUBDIRECTIVE_PRIVATE:
1483  // can have just one of PUBLIC or PRIVATE
1484  if (accessFlag != DEFAULT_ACCESS_SCOPE)
1485  {
1486  syntaxError(Error_Invalid_subkeyword_attribute, token);
1487  }
1488  accessFlag = PRIVATE_SCOPE;
1489  break;
1490 
1491  // define with public access (the default)
1492  case SUBDIRECTIVE_PUBLIC:
1493  // must be first access specifier
1494  if (accessFlag != DEFAULT_ACCESS_SCOPE)
1495  {
1496  syntaxError(Error_Invalid_subkeyword_attribute, token);
1497  }
1498  accessFlag = PUBLIC_SCOPE;
1499  break;

1502  case SUBDIRECTIVE_PACKAGE:
1503  // must be first access specifier
1504  if (accessFlag != DEFAULT_ACCESS_SCOPE)
1505  {
1506  syntaxError(Error_Invalid_subkeyword_attribute, token);
1507  }
1508  accessFlag = PACKAGE_SCOPE;
1509  break;
1510 
1511  // ::METHOD name PROTECTED
1512  case SUBDIRECTIVE_PROTECTED:
1513  // only one of PROTECTED UNPROTECTED
1514  if (protectedFlag != DEFAULT_PROTECTION)
1515  {
1516  syntaxError(Error_Invalid_subkeyword_attribute, token);
1517  }
1518  protectedFlag = PROTECTED_METHOD;
1519  break;
1520

1522  case SUBDIRECTIVE_UNPROTECTED:
1523  // only one of PROTECTED UNPROTECTED
1524  if (protectedFlag != DEFAULT_PROTECTION)
1525  {
1526  syntaxError(Error_Invalid_subkeyword_attribute, token);
1527  }
1528  protectedFlag = UNPROTECTED_METHOD;
1529  break;

1532  case SUBDIRECTIVE_UNGUARDED:
1533  if (guardFlag != DEFAULT_GUARD)
1534  {
1535  syntaxError(Error_Invalid_subkeyword_attribute, token);
1536  }
1537  guardFlag = UNGUARDED_METHOD;
1538  break;
1539 
1540  // guarded access to a method (the default)
1541  case SUBDIRECTIVE_GUARDED:
1542  if (guardFlag != DEFAULT_GUARD)
1543  {
1544  syntaxError(Error_Invalid_subkeyword_attribute, token);
1545  }
1546  guardFlag = GUARDED_METHOD;
1547  break;
1548 
1549  // external attributes?
1550  case SUBDIRECTIVE_EXTERNAL:
1551  // can't be abstract and external
