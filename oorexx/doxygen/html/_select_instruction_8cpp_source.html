125 /**
126  * Execute a SELECT instruction.
127  *
128  * @param context The current execution context.
129  * @param stack The current evaluation stack.
130  */
131 void RexxInstructionSelect::execute(RexxActivation *context, ExpressionStack *stack)
132 {
133  context->traceInstruction(this);
134 
135  // create an active DO block, which marks that we have an active SELECT
136  // in case someone tries do SIGNAL into the middle of the instruction.
137  DoBlock *doblock = new DoBlock (context, this);
138  // set the block to the top of the context stack.
139  context->newBlockInstruction(doblock);
140  // Debug pause requires a conditional pause that terminates the block construct
141  // if we've been asked to re-execute.
142  if (context->conditionalPauseInstruction())
143  {
144  this->terminate(context, doblock);
145  }
146 }

175 /**
176  * Match an END instruction up with this SELECT.
177  *
178  * @param partner The partner END
179  * @param parser The language parser environment (used for error reporting)
180  */
181 void RexxInstructionSelect::matchEnd(RexxInstructionEnd *partner, LanguageParser *parser)
182 {
183  // get some location information for error reporting
184  SourceLocation endLocation = partner->getLocation();
185  size_t lineNum = getLineNumber();
186 
187  // ok, we need to match up the names. If the END has a label, then it must
188  // match a label on the SELECT.
189  RexxString *name = partner->endName();
190  if (name != OREF_NULL)
191  {
192  // One error if we don't had a label, a different error if the
193  // labels don't match
194  RexxString *myLabel = getLabel();
195  if (myLabel == OREF_NULL)
196  {
197  parser->error(Error_Unexpected_end_select_nolabel, endLocation, new_array(name, new_integer(lineNum)));
198  }
199  else if (name != myLabel)
200  {
201  parser->error(Error_Unexpected_end_select, endLocation, new_array(name, myLabel, new_integer(lineNum)));
202  }
203  }
204 
205  // record the END instruction, then do some additional validity checks
206  end = partner;
207 
208  // if we don't have any WHEN clauses, this is an error
209  size_t whenCount = whenList->items();
210  if (whenCount == 0)
211  {
212  parser->error(Error_When_expected_when, getLocation(), new_array(new_integer(lineNum)));
213  }
214 
215  // now link up each of the WHEN clauses with the END
216  while (whenCount--)
217  {
218  // pull the next item from the queue
219  RexxInstruction *when = (RexxInstruction *)whenList->pull();
220  if (when->isType(KEYWORD_WHEN))
221  {
222  // hook up with the partner END instruction
223  ((RexxInstructionIf *)when)->fixWhen((RexxInstructionEndIf *)partner);
224  }
225  // this is a select case version
226  else
227  {
228  // hook up with the partner END instruction
229  ((RexxInstructionCaseWhen *)when)->fixWhen((RexxInstructionEndIf *)partner);
230  }
231  }
232 
233  // the when list is empty, we can scrap it now
234  whenList = OREF_NULL;
235 
236  // do we have an OTHERWISE block?
237  if (otherwise != OREF_NULL)
238  {
239  // for the END terminator on an OTHERWISE, we need to see if this
240  // select has a label. If it does, this needs special handling.
241  if (getLabel() == OREF_NULL)
242  {
243  partner->setStyle(OTHERWISE_BLOCK);
244  }
245  else
246  {
247  partner->setStyle(LABELED_OTHERWISE_BLOCK);
248  }
249  }
250  else
251  {
252  // the SELECT style will raise an error if hit, since it means
253  // there is no OTHERWISE clause. This doesn't matter if there
254  // is a label or not.
255  partner->setStyle(SELECT_BLOCK);
256  }
257 }

356  * Execute a SELECT CASE instruction.
357  *
358  * @param context The current execution context.
359  * @param stack The current evaluation stack.
360  */
361 void RexxInstructionSelectCase::execute(RexxActivation *context, ExpressionStack *stack)
362 {
363  context->traceInstruction(this);
364 
365  // create an active DO block, which marks that we have an active SELECT
366  // in case someone tries do SIGNAL into the middle of the instruction.
367  DoBlock *doblock = new DoBlock (context, this);
368 
369  // evaluate the CASE instruction and store in the doblock so the WHEN
370  // instructions can retrieve it.
371  RexxObject *caseResult = caseExpr->evaluate(context, stack);
372  context->traceKeywordResult(GlobalNames::CASE, caseResult);
373 
374  // set the block to the top of the context stack.
375  context->newBlockInstruction(doblock);
376 
377  doblock->setCase(caseResult);
378 
379  // Debug pause requires a conditional pause that terminates the block construct
380  // if we've been asked to re-execute.
381  if (context->conditionalPauseInstruction())
382  {
383  terminate(context, doblock);
384  }
