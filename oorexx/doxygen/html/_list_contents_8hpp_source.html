132  // copy an entry contents into another entry
133  inline void copyEntry(ItemLink target, ItemLink source)
134  {
135  // copy all of the information (NOTE: we need to use setField() for this)
136  setValue(target, entryValue(source));
137  entries[target].next = entries[source].next;
138  entries[target].previous = entries[source].next;

142  // remove a non-anchor entry from a hash chain
143  inline void closeChain(ItemLink position, ItemLink previous)
144  {
145  if (previous != NoMore)
146  {
147  entries[previous].next = entries[position].next;
148  }
149  if (entries[position].next != NoMore)
150  {
151  entries[entries[position].next].previous = previous;
152  }
153 
154  // move the removed item back to the free chain and
155  // clear it out
156  returnToFreeChain(position);
157  }
158 
159  // return an entry to the free chain
160  inline void returnToFreeChain(ItemLink position)
161  {
162  // clear the entry, then place at the head of the
163  // free chain.
164  clearEntry(position);
165  // only use a single link for the free chain
166  entries[position].next = freeChain;
167  freeChain = position;
168  }
