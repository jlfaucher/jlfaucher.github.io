194 /**
195  * Remove a session reference cookie from the chain.
196  *
197  * @param s The cookie to remove.
198  */
199 void RegistrationData::removeSessionReference(SessionCookie *s)
200 {
201  if (s == references)
202  {
203  references = s->next;
204  }
205  else
206  {
207  SessionCookie * current = references;
208  while (current != NULL)
209  {
210  if (current->next == s)
211  {
212  current->next = s->next;
213  break;
214  }
215  current = current->next;
216  }
217  }
218  delete s;
219 }

223 // Add a library registration item to the table.
224 // Message arguments have the following meanings:
225 //
226 // parameter1 -- registration type
227 // nameArg -- name of the registered object
228 void RegistrationTable::registerLibraryCallback(ServiceMessage &message)
229 {
230  ServiceRegistrationData *regData = (ServiceRegistrationData *)message.getMessageData();
231  // get the argument names
232  const char *name = message.nameArg;
233  const char *module = regData->moduleName;
234  // We don't know whether we're being called through RXSUBCOM or not,
235  // but if we are, there's a certain complication: registering and
236  // dropping will be called with a different session ID as each
237  // subcommand runs in a separate process. But as RXSUBCOM REGISTER
238  // will always call us with DROP_ANY, we can fix this by setting the
239  // session ID to zero upon registration, and later when dropping.
240  SessionID session = regData->dropAuthority == DROP_ANY ? 0 : message.session;
241 
242  RegistrationData *callback = locate(name, module);
243  // update the reference counts to make sure drops don't
244  // clear things out for other processes.
245  if (callback != NULL)
246  {
247  callback->addSessionReference(session);
248  message.setResult(DUPLICATE_REGISTRATION);
249  }
250  else
251  {
252  callback = new RegistrationData(name, module, session, regData);
253 
254  // add to the chain
255  callback->next = firstLibrary;
256  firstLibrary = callback;
257 
258  if (locate(name, session) != NULL)
259  {
260  message.setResult(DUPLICATE_REGISTRATION);
261  }
262  else
263  {
264  message.setResult(REGISTRATION_COMPLETED);
265  }
266  }
267 
268  // make sure the data message buffer is not passed back.
269  message.freeMessageData();
270 }

318 // General query by name only. Can return either form of registraction.
319 // Message arguments have the following meanings:
320 //
321 // parameter1 -- registration type
322 // nameArg -- The registration name
323 void RegistrationTable::queryCallback(ServiceMessage &message)
324 {
325  // get the argument name (local copy only)
326  const char *name = message.nameArg;
327 
328  // now check the exe version first.
329  RegistrationData *callback = locate(name, message.session);
330  // not found? try a library version
331  if (callback == NULL || callback->owner != message.session)
332  {
333  callback = locate(firstLibrary, name);
334  }
335  // copy the data into the buffer if we found one
336  if (callback != NULL)
337  {
338  ServiceRegistrationData *regData = (ServiceRegistrationData *)message.allocateMessageData(sizeof(ServiceRegistrationData));
339  // copy the registration information
340  callback->getRegistrationData(*regData);
341  message.setResult(CALLBACK_EXISTS);
342  }
343  else
344  {
345  message.setResult(CALLBACK_NOT_FOUND);
346  }
347 }

356 {

419 // Drop a callback by name only.
420 // Message arguments have the following meanings:
421 //
422 // parameter1 -- registration type
423 // nameArg -- The registration name
424 void RegistrationTable::dropCallback(ServiceMessage &message)
425 {
426  // get the argument name (local copy only)
427  const char *name = message.nameArg;
428  RegistrationData **anchor = &firstEntryPoint;
429 
430  // now check the exe version first.
431  RegistrationData *callback = locate(name, message.session);
432  // not found? try a library version
433  if (callback == NULL)
434  {
435  callback = locate(firstLibrary, name);
436  anchor = &firstLibrary;
437  }
438  if (callback != NULL)
439  {
440  // an attempt to drop by somebody other than the owner?
441  if (callback->dropAuthority == OWNER_ONLY && callback->owner != message.session)
442  {
443  message.setResult(DROP_NOT_AUTHORIZED);
444  }
445  else
446  {
447  // remove this session reference.
448  callback->removeSessionReference(callback->dropAuthority == DROP_ANY ? 0 : message.session);
449  // still referenced by other processes?
450  if (callback->hasReferences())
451  {
452  // the callback has more references, but docs say we
453  // don't return DUPLICATE_REGISTRATION here
454  message.setResult(CALLBACK_DROPPED);
455  }
456  else
457  {
458  remove(anchor, callback);
459  delete callback;
460  message.setResult(CALLBACK_DROPPED);
461  }
462  }
463  }
464  else
465  {
466  message.setResult(CALLBACK_NOT_FOUND);
467  }
468  // make sure the data message buffer is not passed back.
469  message.freeMessageData();
470 }

542  *

544  */
545 RegistrationData *RegistrationTable::locate(RegistrationData *anchor, const char *name)
546 {
547  RegistrationData *current = anchor;
548  RegistrationData *previous = NULL;
549 
550  while (current != NULL) /* while more registrations */
551  {
552  // find the one we want?
553  if (current->matches(name))
554  {
555  return current;
556  }
557  previous = current; /* remember this block */
558  current = current->next; /* step to the next block */
559  }
560  return NULL;

564  * search for a name-only registration and remove it from
565  * the chain.
566  *
567  * @param anchor The chain anchor position.
568  * @param block The block to locate.
569  */
570 void RegistrationTable::remove(RegistrationData **anchor, RegistrationData *block)
571 {
572  RegistrationData *current = *anchor;
573  RegistrationData *previous = NULL;
574 
575  while (current != NULL) /* while more registrations */
576  {
577  // find the one we want?
578  if (current == block)
579  {
580  // is this the first one?
581  if (previous == NULL)
582  {
583  // update the anchor position
584  *anchor = current->next;
585  }
586  else
587  {
588  // close the chain and get out of here
589  previous->next = current->next;
590  }
591  return;
592  }
593  previous = current; /* remember this block */
594  current = current->next; /* step to the next block */
595  }

676 /**
677  * Reorder the registration blocks so that we put the
678  * most recently referenced registrations at the front
679  * of the queue.
680  *
681  * @param anchor The chain anchor.
682  * @param current The block we're reordering.
683  * @param previous The previous block in the chain (can be NULL if this
684  * item is already at the head of the chain).

686 void RegistrationTable::reorderBlocks(RegistrationData *& anchor, RegistrationData *current, RegistrationData *previous)
687 {
688  if (previous != NULL) // if we have a predecessor
689  {
690  // rearrange to get "most recently used" behavior
691  previous->next = current->next;
692  current->next = anchor;
693  anchor = current;
694  }
