52 /**
53  * Set up for execution of a FOR loop.
54  *
55  * @param context The current execution context.
56  * @param stack The current evaluation stack.
57  * @param doblock The context doblock useds to store loop state data.
58  */
59 void ForLoop::setup(RexxActivation *context,
60  ExpressionStack *stack, DoBlock *doblock, bool forKeyword)
61 {
62  // we might not have anything here, but we need to set
63  // a marker in the doblock so we know not to use this
64  if (forCount == OREF_NULL)
65  {
66  // set this to a negative value to indicate not to use this.
67  doblock->setFor(SIZE_MAX);
68  return;
69  }
70 
71  // get the expression value. This must be a whole number, so we need to
72  // validate and convert now.
73  wholenumber_t count = 0;
74  RexxObject *result = forCount->evaluate(context, stack);
75 
76  context->traceKeywordResult(GlobalNames::FOR, result);
77 
78  // if this is an integer value already and we're at the default digits setting,
79  // we should be able to use this directly.
80  if (isInteger(result) && context->digits() >= Numerics::DEFAULT_DIGITS)
81  {
82  // get the value directly and trace
83  count = ((RexxInteger *)result)->getValue();
84  }
85  else
86  {
87  // first get the string version and and request a string version
88  Protected<NumberString> strResult = result->requestString()->numberString();
89  // non-numeric value, this is an error
90  if (strResult == OREF_NULL)
91  {
92  reportException(forKeyword ? Error_Invalid_whole_number_for : Error_Invalid_whole_number_repeat, result);
93  }
94  // force rounding
95  Protected<RexxObject> rounded = strResult->callOperatorMethod(OPERATOR_PLUS, OREF_NULL);
96  // now convert the rounded value to an integer, if possible
97  if (!rounded->requestNumber(count, number_digits()))
98  {
99  // use original object in the error report.
100  reportException(forKeyword ? Error_Invalid_whole_number_for : Error_Invalid_whole_number_repeat, result);
101  }
102  }
103 
104  // This must be a non-negative value.
105  if (count < 0)
106  {
107  reportException(forKeyword ? Error_Invalid_whole_number_for : Error_Invalid_whole_number_repeat, result);
108  }
109 
110  // set this value in the doblock
111  doblock->setFor((size_t)count);
112 }

115 /**
116  * Set up for execution of a controlled loop.
117  *
118  * @param context The current execution context.
119  * @param stack The current evaluation stack.
120  * @param doblock The context doblock useds to store loop state data.
121  */
122 void ControlledLoop::setup( RexxActivation *context,
123  ExpressionStack *stack, DoBlock *doblock)
124 {
125  // evaluate the initial expression
126  RexxObject *_initial = initial->evaluate(context, stack);
127 
128  // force rounding by adding zero to this
129  _initial = _initial->callOperatorMethod(OPERATOR_PLUS, OREF_NULL);
130  // now process each of the expressions. the expressions
131  // array allows us to process these in the order they were specified on
132  // the instruction
133  for (size_t i = 0; i < 3 && expressions[i] != 0; i++)
134  {
135  switch (expressions[i])
136  {
137  // The TO expression;
138  case EXP_TO:
139  {
140  // get the too value and round...which has the side effect
141  // of also validating that this is a valid numeric.
142  Protected<RexxObject> result = to->evaluate(context, stack);
143  context->traceKeywordResult(GlobalNames::TO, result);
144 
145  // prefix + is like adding zero
146  result = result->callOperatorMethod(OPERATOR_PLUS, OREF_NULL);
147 
148  // if the result is a string, see if we can convert this to
149  // an integer value. This is very common in loops, and can
150  // save us a lot of processing on each loop iteration.
151  RexxObject *temp = result->integerValue(number_digits());
152  if (temp != TheNilObject)
153  {
154  result = temp;
155  }
156  // this value gets saved in the doblock as state data we can reuse.
157  doblock->setTo(result);
158  break;
159  }
160 
161  // BY expression
162  case EXP_BY:
163  {
164  // get the expression value and round
165  Protected<RexxObject> result = by->evaluate(context, stack);
166  context->traceKeywordResult(GlobalNames::BY, result);
167  result = result->callOperatorMethod(OPERATOR_PLUS, OREF_NULL);
168  // this gets saved in the doblock
169  doblock->setBy(result);
170 
171  // now we need to check if this is a negative value so set know how to
172  // compare.
173  if (result->callOperatorMethod(OPERATOR_LESSTHAN, IntegerZero) == TheTrueObject)
174  {
175  // we're counting down, so check less than for termination
176  doblock->setCompare(OPERATOR_LESSTHAN);
177  }
178  else
179  {
180  // counting up...compare greater than
181  doblock->setCompare(OPERATOR_GREATERTHAN);
182  }
183  break;
184  }
185 
186  // FOR expression...does a binary count
187  // our superclass can handle that one.
188  case EXP_FOR:
189  ForLoop::setup(context, stack, doblock, true);
190  break;
191  }
192  }
193  // if the loop did not specify a BY expression, we use a default increment of 1.
194  // and we are counting upward, so use a greater than comparison
195  if (by == OREF_NULL)
196  {
197  doblock->setBy(IntegerOne);
198  doblock->setCompare(OPERATOR_GREATERTHAN);
199  }
200  // also disable the FOR count if that was not specified.
201  if (forCount == OREF_NULL)
202  {
203  // set this to a negative value to indicate not to use this.
204  doblock->setFor(SIZE_MAX);
205  }
206 
207  // set the control variable in the doblock (it will use it on subsequent passes).
208  doblock->setControl(control);
209  // do the initial assignment
210  control->assign(context, _initial);
211 }

214 /**
215  * Set up for execution of a DO OVER loop.
216  *
217  * @param context The current execution context.
218  * @param stack The current evaluation stack.
219  * @param doblock The context doblock useds to store loop state data.
220  */
221 void OverLoop::setup( RexxActivation *context,
222  ExpressionStack *stack, DoBlock *doblock)
223 {
224  // evaluate the array target
225  RexxObject* result = target->evaluate(context, stack);
226  // anchor immediately to protect from GC
227  doblock->setTo(result);
228 
229  context->traceKeywordResult(GlobalNames::OVER, result);
230  // if this is already an array item, request the non-sparse version
231 
232  ArrayClass *array;
233  if (isArray(result))
234  {
235  array = ((ArrayClass *)result)->makeArray();
236  }
237  else
238  {
239  // some other type of collection, use the less direct means
240  // of requesting an array
241  array = result->requestArray();
242  // raise an error if this did not convert ok, or we got
243  // back something other than a real Rexx array.
244  if (!isArray(array))
245  {
246  reportException(Error_Execution_noarray, result);
247  }
248  }
249 
250  // we use the TO field to store the array, and the for
251  // counter is our index position.
252  doblock->setTo(array);
253  doblock->setOverIndex(1);
254  // and don't forget the variable (which of course, I DID forget!)
255  doblock->setControl(control);
256 }

259 /**
260  * Evaluate a WHILE condition, return true if the condition
261  * should continue and false if the WHILE condition failed.
262  *
263  * @param context The current execution context.
264  * @param stack The current evaluation stack.
265  *
266  * @return true if the WHILE condition passes, false if it fails.
267  */
268 bool WhileUntilLoop::checkWhile(RexxActivation *context, ExpressionStack *stack )
269 {
270  // evaluate the condition and trace
271  RexxObject *result = conditional->evaluate(context, stack);
272  context->traceKeywordResult(GlobalNames::WHILE, result);
273 
274  // most comparisons return either true or false directly, so we
275  // can optimize this check. WHILE conditions are more likely to
276  // evaluate to true, so we'll check that first
277  if (result == TheTrueObject)
278  {
279  return true;
280  }
281  else if (result == TheFalseObject)
282  {
283  return false;
284  }
285  // This is some sort of computed boolean, so we need to do a real
286  // validation on this
287  return result->truthValue(Error_Logical_value_while);
288 }

327  *
328  * @param context The current execution context.
329  * @param stack The current evaluation stack.
330  * @param doblock The context doblock useds to store loop state data.
331  */
332 void WithLoop::setup( RexxActivation *context,
333  ExpressionStack *stack, DoBlock *doblock)
334 {
335  // evaluate the supplier provider
336  RexxObject* result = supplierSource->evaluate(context, stack);
337 
338  context->traceKeywordResult(GlobalNames::WITH, result);
339  // Now send this expression result the supplier message
340  // to get a supplier instance.
341  ProtectedObject p;
342 
343  SupplierClass *supplier = (SupplierClass *)result->sendMessage(GlobalNames::SUPPLIER, p);
344 
345  if (supplier == OREF_NULL || !isOfClassType(Supplier, supplier))
346  {
347  reportException(Error_Execution_no_supplier, result);
348  }
349 
350  // Store this instance in the doblock
351  doblock->setSupplier(supplier);
352 }

355 /**
356  * Handle the loop iteration for a WITH loop.
357  *
358  * @param context The current execution context.
359  * @param stack The evaluation stack.
360  * @param doblock The do block context.
361  * @param first The first iteration indicator.
362  *
363  * @return true if the loop should iterate, false otherwise.
364  */
365 bool WithLoop::checkIteration(RexxActivation *context, ExpressionStack *stack, DoBlock *doblock, bool first)
366 {
367  // get the supplier object from the doblock
368  SupplierClass *supplier = (SupplierClass *)doblock->getSupplier();
369 
370  // make sure we step to next one on each iteration...this is
371  // why the loop variation exists!
372  if (!first)
373  {
374  supplier->loopNext();
375  }
376 
377  // have we reached the end? Then stop the loop
378  if (!supplier->loopAvailable())
379  {
380  return false;
381  }
382 
383  // we must have one of the variable items, but need not have both.
384  if (indexVar != OREF_NULL)
385  {
386  RexxObject *index = supplier->loopIndex();
387  // the control variable gets set immediately, and we trace this
388  // increment result
389  indexVar->assign(context, index);
390  }
391 
392  // we must have one of the variable items, but need not have both.
393  if (itemVar != OREF_NULL)
394  {
395  RexxObject *item = supplier->loopItem();
396  // the control variable gets set immediately, and we trace this
397  // increment result
398  itemVar->assign(context, item);
399  }
400 
401  // this is a good iteration
402  return true;
403 }
