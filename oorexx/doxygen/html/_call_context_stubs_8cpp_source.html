428 void RexxEntry ReadInput(RexxIORedirectorContext *c, CSTRING *data, size_t *length)
429 {
430  ApiContext context(c);
431 
432  // set the default
433  *data = NULL;
434  *length = 0;
435  try
436  {
437  CommandIOContext *ioContext = ((RedirectorContext *)c)->ioContext;
438 
439  // The command handler does not get passed an operable context if
440  // this is not an ADDRESS WITH variant. If that's the case, we return
441  // nothing if this is called.
442  if (ioContext == OREF_NULL)
443  {
444  return;
445  }
446  // request the next input line. This will be NULL if we've reached the end.
447  // Note that the string object is anchored by the ioContext, so
448  // we don't need to add this to the context local reference table
449  RexxString *nextLine = ioContext->readInput(context.context);
450  if (nextLine != OREF_NULL)
451  {
452  *data = (CSTRING)nextLine->getStringData();
453  *length = nextLine->getLength();
454  }
455  }
456  catch (NativeActivation *)
457  {
458  }
459  return;
460 }

463 void RexxEntry ReadInputBuffer(RexxIORedirectorContext *c, CSTRING *data, size_t *length)
464 {
