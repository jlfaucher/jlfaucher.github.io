97 /**
98  * Verify that a function has received all of its required
99  * arguments, and did not receive extras.
100  *
101  * @param argcount The argument count.
102  * @param min The min count for this function.
103  * @param max The max count for this expression.
104  * @param function The function name.
105  */
106 void ExpressionStack::expandArgs(size_t argcount, size_t min, size_t max, const char *function )
107 {
108  // handle checks for min and max argument counts
109  if (argcount < min)
110  {
111  reportException(Error_Incorrect_call_minarg, function, min);
112  }
113  else if (argcount > max)
114  {
115  reportException(Error_Incorrect_call_maxarg, function, max);
116  }
117  else
118  {
119  // now check all of the arguments up to the min count
120  // to verify none of them have been omitted.
121  RexxInternalObject **current = pointer(argcount - 1);
122  for (size_t i = min; i; i--)
123  {
124  // an omitted value?
125  if (*current++ == OREF_NULL)
126  {
127  reportException(Error_Incorrect_call_noarg, function, min - i + 1);
128  }
129  }
130  }
131 }

134 /**
135  * Retrieve an object from the expression stack and potentially
136  * convert it into a string argument.
137  *
138  * @param position The argument position.
139  *
140  * @return The argument String value.
141  */
142 RexxString *ExpressionStack::requiredStringArg(size_t position)
143 {
144  // get the argument from the stack. If this is a true
145  // string, just return directly.
146  RexxInternalObject *argument = peek(position);
147  if (isString(argument))
148  {
149  return (RexxString *)argument;
150  }
151  // get the string form, raising a NOSTRING condition if necessary
152  RexxString *newStr = argument->requestString();
153  // we replace the original stack object with the string value
154  replace(position, newStr);
155  return newStr;
156 }

192  * Retrieve an object from the expression stack and potentially
193  * convert it into an integer argument.
194  *
195  * @param position The argument position.
196  * @param argcount The argument count (used for error reporting)
197  * @param function The function name.
198  *
199  * @return The argument as an integer object.
200  */
201 RexxInteger *ExpressionStack::requiredIntegerArg(size_t position,
202  size_t argcount, const char *function)
203 {
204  // if the argument is an integer already, this is a quick return.
205  RexxInternalObject *argument = peek(position);
206  if (isInteger(argument))
207  {
208  return (RexxInteger *)argument;
209  }
210 
211  // convert to an integer value, give an error if it did not convert.
212  wholenumber_t numberValue;
213  if (!argument->requestNumber(numberValue, Numerics::ARGUMENT_DIGITS))
214  {
215  reportException(Error_Incorrect_call_whole, function, argcount - position, (RexxObject *)argument);
216  }
217 
218  // replace the slot with the new value
219  RexxInteger *newInt = new_integer(numberValue);
220  replace(position, newInt);
221  return newInt;
222 }
