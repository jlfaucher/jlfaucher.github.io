241 {
242  requiredArgument(other, ARG_ONE);
243

245  // pretending to be the String class. We mask this here by making all of them

249  {

251  }
252  else
253  {
254  // this is a direct identity compare
255  return booleanObject(this == other);
256  }

548 /**
549  * Inherit the instance methods from another class definition.
550  * This is not a true inherit operation where the class is part
551  * of the hierarchy. This directly grabs the instance methods
552  * defined directly by the other class and merges them into the
553  * class method dictionary. This is a special method that is
554  * only done during the initial image build.
555  *
556  * @param newMethods The new methods to add.
557  */
558 void RexxClass::inheritInstanceMethods(RexxClass *source)
559 {
560  MethodDictionary *sourceMethods = source->instanceMethodDictionary;
561 
562  // we want all methods to be "inherited" with this scope
563  sourceMethods->setMethodScope(this);
564 
565  // loop through the table with an iterator.
566  HashContents::TableIterator iterator = sourceMethods->iterator();
567 
568  for (; iterator.isAvailable(); iterator.next())
569  {
570  // get the name and the value, then add to this class object
571  RexxString *methodName = (RexxString *)iterator.index();
572  // if this is the Nil object, that's an override. Make it OREF_NULL.
573  MethodClass *method = (MethodClass *)iterator.value();
574  // we just add this to the instance method dictionary for when we
575  // build the behaviour
576  instanceMethodDictionary->addMethod(methodName, method);
577  }
578 
579  // now update the instance behaviour from our superclasses and
580  // merge in the new methods.
581  instanceBehaviour->setMethodDictionary(OREF_NULL);
582  createInstanceBehaviour(instanceBehaviour);
583 
584  // see if we have an uninit method defined now that this is done
585  checkUninit();
586 }

651 /**
652  * Initialize a base Rexx class.
653  */
654 void RexxClass::buildFinalClassBehaviour()
655 {
656  // get a copy of the class instance
657  // behaviour mdict before the merge
658  // with OBJECT. This unmerged mdict
659  // is kept in this class's
660  // class_instance_mdict field.
661  instanceMethodDictionary = getInstanceBehaviourDictionary();
662  // now clear the instance behaviour and start building anew
663  instanceBehaviour->clearMethodDictionary();
664  // Add OBJECT to the behaviour scope table
665  instanceBehaviour->addScope(TheObjectClass);
666  // if this is not the object class, then we need to add object
667  // to the different behaviours before we merge in our methods
668  if (this != TheObjectClass)
669  {
670  // add to the instance behaviour scope
671  instanceBehaviour->addScope(TheObjectClass);
672  // merge the class instance behaviour with object.
673  instanceBehaviour->merge(TheObjectBehaviour);
674  }
675 
676  // now add in the methods defined for this class
677  instanceBehaviour->mergeMethodDictionary(instanceMethodDictionary);
678 
679  // add this class to the scope table
680  instanceBehaviour->addScope(this);
681  // now we do the same thing with the class behaviour
682  classMethodDictionary = getBehaviourDictionary();
683  // now add the scope levels to this class behaviour
684  // If this isn't OBJECT put OBJECT in first
685  if (this != TheObjectClass)
686  {
687  behaviour->addScope(TheObjectClass);
688  }
689  // if this is OBJECT - merge the object instance
690  // methods with the object class methods
691  else
692  {
693  behaviour->merge(TheObjectBehaviour);
694  // and put them into the class mdict
695  // so all the classes will inherit
696  classMethodDictionary = getBehaviourDictionary();
697  }
698  // The merge of the mdict's is order specific. By processing OBJECT
699  // first then CLASS and then the rest of the subclassable classes
700  // the mdict's will be set up correctly.
701  behaviour->merge(TheClassBehaviour);
702  // if this isn't CLASS put CLASS in next
703  if (this != TheClassClass)
704  {
705  behaviour->addScope(TheClassClass);
706  }
707  // now add this class to the scope
708  behaviour->addScope(this);
709 
710  // that's the behaviour information...now fill in other state data.
711 
712  // All primitive methods have TheClassClass as the meta class.
713  metaClass = TheClassClass;
714 
715  // The Baseclass for non-mixin classes is self
716  baseClass = this;
717  // as is the instance superclasses list
718  superClasses = new_array();
719  // create the subclasses list
720  subClasses = new_list();
721  // if this is not the object class, we have superclass information to add
722  if (this != TheObjectClass)
723  {
724  // The instance superclasses for all except OBJECT is OBJECT
725  superClasses->addLast(TheObjectClass);
726  // and for OBJECT we need to add all the other classes as
727  // subclasses except for the pseudo classes of integer and number string.
728  if (this != TheIntegerClass && this != TheNumberStringClass)
729  {
730  TheObjectClass->addSubClass(this);
731  }
732  }
733  // and point the instance behaviour back to this class
734  instanceBehaviour->setOwningClass(this);
735  // and the class behaviour to CLASS
736  behaviour->setOwningClass(TheClassClass);
737  // these are primitive classes
738  setPrimitive();
739 
740  // check to see if we have an uninit methods.
741  checkUninit();
742 
743  // if this is the CLASS class, make it a meta class.
744  if (this == TheClassClass)
745  {
746  setMetaClass();
747  }
748 }

751 /**
752  * Initialize a base Rexx class that inherits from a primitive
753  * class other than Object.
754  *
755  * @param superClass The immediate superclass of the created
756  * class.
757  */
758 void RexxClass::buildFinalClassBehaviour(RexxClass *superClass)
759 {
760  // get a copy of the class instance behaviour mdict before the merge
761  // with OBJECT. This unmerged mdict is kept in this class's
762  // class_instance_mdict field.
763  setField(instanceMethodDictionary, getInstanceBehaviourDictionary());
764 
765  // set up the superclass/subclass relationships
766  setField(superClasses, new_array(superClass));
767  // create the subclasses list
768  setField(subClasses, new_list());
769  // and add this as a subclass to our superclass
770  superClass->addSubClass(this);
771 
772  // create the merged method dictionary for the instancebehavior
773  // and update all of the scopes.
774  mergeBehaviour(instanceBehaviour);
775 
776  // get a copy of the class behaviour mdict before the merge with the
777  // CLASS instance behaviour. This unmerged mdict is kept in the
778  // class_mdict field
779  setField(classMethodDictionary, getBehaviourDictionary());
780  // The merge of the mdict's is order specific. By processing OBJECT
781  // first then CLASS and then the rest of the subclassable classes
782  // the mdict's will be set up correctly. In this way merging the CLASS
783  // behaviour will only be the CLASS instance methods when OBJECT is

1089 /**
1090  * Create the class behaviour (the behavior for the class object
1091  * itself). This merges all of the information from our
1092  * superclasses with the information defined at this class
1093  * scope.
1094  *
1095  * @param target_class_behaviour
1096  * The behaviour object we're going to build into.
1097  */
1098 void RexxClass::createClassBehaviour(RexxBehaviour *target_class_behaviour)
1099 {
1100  // we are going to call each of our superclasses, start from the last to the
1101  // first asking them to merge their information. The last superclass should be
1102  // Object, the first will be our immediate superclass.
1103  for (size_t index = superClasses->items(); index > 0; index--)
1104  {
1105  RexxClass *superclass = (RexxClass *)superClasses->get(index);
1106  // if there is a superclass and this hasn't been added into this
1107  // behaviour yet, ask it to merge its information into this. We
1108  // can have dups when mixin classes are involved, since we inherit the
1109  // same baseClass chain from each mixin.
1110  if (!target_class_behaviour->hasScope(superclass))
1111  {
1112  superclass->createClassBehaviour(target_class_behaviour);
1113  }
1114  }
1115 
1116  // now see if we need to merge our information into this behaviour (we likely do)
1117  if (!target_class_behaviour->hasScope(this))
1118  {
1119  // Object is a special case, since it is top dog.
1120  if (TheObjectClass != this)
1121  {
1122  // add whichever metaclasses have not been added yet
1123  if (!target_class_behaviour->hasScope(metaClass))
1124  {
1125  // merge in the meta class instance method dictionary into our method dictionary.
1126  // this also merges the scopes.
1127  metaClass->mergeInstanceBehaviour(target_class_behaviour);
1128  }
1129  }
1130 
1131  // Merge this class mdict with the target behaviour class mdict
1132  target_class_behaviour->mergeMethodDictionary(classMethodDictionary);
1133 
1134  // update the target behaviour scopes with this class, if necessary.
1135  target_class_behaviour->addScope(this);
1136  }

1290  if (isRexxDefined())
1291  {
1292  reportException(Error_Execution_rexx_defined_class);
1293  }

1296  requiredArgument(mixin_class, "mixin class");
1297

1299  if (!mixin_class->isInstanceOf(TheClassClass) || !mixin_class->isMixinClass())
1300  {

1302  }
1303

1305  if (this == mixin_class )
1306  {
1307  reportException(Error_Execution_recursive_inherit, this, mixin_class);
1308  }
1309

1311  if (behaviour->hasScope(mixin_class))
