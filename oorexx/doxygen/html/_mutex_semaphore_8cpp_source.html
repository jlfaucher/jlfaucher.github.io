151 /**
152  * Acquire the mutex.
153  * @param t An optional timeout in seconds, either String or TimeSpan.
154  * Returns immediately if the timeout is zero,
155  * waits indefinitely if it's negative.
156  * Default is -1.
157  *
158  * @return true if this was successfully acquired,
159  * false if we timed out.
160  */
161 RexxObject* MutexSemaphoreClass::request(RexxObject *t)
162 {
163  wholenumber_t timeout;
164 
165  if (t == OREF_NULL)
166  {
167  timeout = -1;
168  }
169  else
170  {
171  RexxObject *seconds;
172 
173  // we allow timeout to be a TimeSpan or a String object
174  RexxClass *TimeSpan = TheRexxPackage->findClass(GlobalNames::TIMESPAN);
175  if (t->isInstanceOf(TimeSpan))
176  {
177  ProtectedObject p;
178  seconds = t->sendMessage(GlobalNames::TOTALSECONDS, p);
179  }
180  else
181  {
182  seconds = t;
183  }
184  double secs = floatingPointArgument(seconds, "timeout");
185  // wait() is uint32, so we can wait for 4294967 seconds at most
186  if (secs >= 0 && secs <= 4294967)
187  {
188  // we need milliseconds (the product will always fit a unint32)
189  timeout = (wholenumber_t)(secs * 1000.0);
190  }
191  else
192  {
193  // negative or too large, we just wait forever
194  timeout = -1;
195  }
196  }
197 
198  bool acquired;
199  if (timeout < 0)
200  {
201  UnsafeBlock releaser;
202  acquired = semaphore.request();
203  }
204  else if (timeout == 0)
205  {
206  acquired = semaphore.requestImmediate();
207  }
208  else
209  {
210  UnsafeBlock releaser;
211 
212  acquired = semaphore.request((uint32_t)timeout);
213  }
214 
215  // only do the nesting thing if we got the lock
216  if (acquired)
217  {
218  handleNesting();
219  }
220 
221  return booleanObject(acquired);
222 }

225 /**
226  * Release the mutex semaphore
227  *
228  * @return true if this was successfully released, false if we were not the semaphore owners.
229  */
230 RexxObject* MutexSemaphoreClass::release()
231 {
232  // don't even attempt this if this isn't held.
233  if (nestCount == 0)
234  {
235  return TheFalseObject;
236  }
237 
238  // if this thread really owns this mutex, this will succeed
239  bool released = semaphore.release();
240  if (released)
241  {
242  nestCount--;
243  if (nestCount == 0)
244  {
245  // tell the current activity this is no longer needed
246  ActivityManager::currentActivity->removeMutex(this);
247  }
248  }
249  return booleanObject(released);
250 }
