64 /**
65  * Base constructor for an instruction execution object.
66  *
67  * @param clause The clause this was created from (gives location information).
68  * @param type The instruction keyword type. Note that the type
69  * does not necessarily map to a keyword name. There
70  * are special purpose instructions that do not correspond
71  * to specific keywords, and some keyword instructions
72  * may have more that one execution object tailored
73  * to specific subfunctions.
74  */
75 RexxInstruction::RexxInstruction(RexxClause *clause, InstructionKeyword type)
76 {
77  instructionType = type;
78  // for instructions that are generated as part of other instructions (for example,
79  // internal branching instructions for an IF instruction), we don't have a clause
80  // to provide location information. Just zero the location.
81  if (clause != OREF_NULL)
82  {
83  instructionLocation = clause->getLocation();
84  }
85  else
86  {
87  instructionLocation.setStart(0, 0);
88  }
89 }

135 /**
136  * Common method for evaluating arrays of arguments for
137  * the different instruction or expression types that
138  * use arguments (CALL, message, functions, etc.)
139  *
140  * @param context The current execution context.
141  * @param stack The current evaluation stack
142  * @param argArray The pointer to the array of argument expressions.
143  * @param argCount The number of argument expressions.
144  */
145 void RexxInstruction::evaluateArguments(RexxActivation *context, ExpressionStack *stack, RexxInternalObject **argArray, size_t argCount)
146 {
147  // evaluate all of the arguments
148  for (size_t i = 0; i < argCount; i++)
149  {
150  // real argument expression
151  if (argArray[i] != OREF_NULL)
152  {
153  // evaluate the expression (and the argument is left on the stack)
154  RexxObject *result = argArray[i]->evaluate(context, stack);
155  context->traceArgument(result);
156  }
157  // omitted argument. Push a null value and trace as a null string
158  else
159  {
160  stack->push(OREF_NULL);
161  context->traceArgument(GlobalNames::NULLSTRING);
162  }
163  }
164 }

212 /**
213  * Common method for evaluating the single expression
214  * and tracing the result for subclasses of this
215  * class.
216  *
217  * @param context The current execution context.
218  * @param stack The current evaluation stack.
219  *
220  * @return The expression result, or OREF_NULL if there is no
221  * expression to evaluate.
222  */
223 RexxObject *RexxInstructionExpression::evaluateExpression(RexxActivation *context, ExpressionStack *stack)
224 {
225  // if we have an expression value, evaluate it.
226  if (expression != OREF_NULL)
227  {
228  // evaluate this
229  RexxObject *result = expression->evaluate(context, stack);
230  context->traceResult(result);
231  return result;
232  }
233  // no expression, no result
234  return OREF_NULL;
235 }

238 /**
239  * Common method for evaluating the single expression
240  * and tracing the result for subclasses of this
241  * class. The result is forced to string form.
242  *
243  * @param context The current execution context.
244  * @param stack The current evaluation stack.
245  *
246  * @return The expression result, or OREF_NULL if there is no
247  * expression to evaluate.
248  */
249 RexxString *RexxInstructionExpression::evaluateStringExpression(RexxActivation *context, ExpressionStack *stack)
250 {
251  // if we have an expression value, evaluate it.
252  if (expression != OREF_NULL)
253  {
254  // evaluate this
255  RexxObject *result = expression->evaluate(context, stack);
256  // force to string form, trace, and return the string version
257  RexxString *stringResult = result->requestString();
258  // protect on the stack
259  stack->push(stringResult);
260  context->traceResult(stringResult);
261  return stringResult;
262  }
263  // the string expression is required here, so return a NULL string if no
264  // expression. We still need to trace that.
265  else
266  {
267  context->traceResult(GlobalNames::NULLSTRING);
268  return GlobalNames::NULLSTRING;
269  }
270 }
