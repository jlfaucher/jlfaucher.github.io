61 /**
62  * Execute an OPTIONS instruction. NOTE: Currently, there
63  * are no OPTIONS that we support on this.
64  *
65  * @param context The current execution context.
66  * @param stack The current evaluation stack.
67  */
68 void RexxInstructionOptions::execute(RexxActivation *context, ExpressionStack *stack)
69 {
70  // even though we don't support anything, we still trace the evaluation and
71  // raise any errors if they occure.
72  context->traceInstruction(this);
73 
74  // trace, and get as a string.
75  RexxString *stringValue = evaluateStringExpression(context, stack);
76 
77 // NOTE: processing is currently disabled, but if options are reenabled, the
78 // processing framework is here.
79 #ifdef _DEBUG
80  // break up into an array of words
81  ArrayClass *words = stringValue->subWords(OREF_NULL, OREF_NULL);
82  size_t wordCount = words->size();
83 
84  for (size_t i = 1; i <= wordCount ;i++)
85  {
86  RexxString *word = (RexxString *)words->get(i);
87  if (word->strCaselessCompare("DUMPMEMORY"))
88  {
89  memoryObject.dumpImageStats();
90  }
91  }
92 #endif
93  context->pauseInstruction();
94 }
