53 /**
54  * Base class for I/O redirectors.
55  */
56 class InputRedirector : public RexxInternalObject
57 {
58  public:
59  // Note, we never directly create instances of this
60  // class so we don't need new, live, etc. These must
61  // be implemented by the concrete classes
62  inline InputRedirector() : inputBuffer(OREF_NULL) { ; }
63  inline InputRedirector(RESTORETYPE restoreType) { ; };
64 
65  virtual void init () { ; }
66  virtual RexxString *read(NativeActivation *context) { return OREF_NULL; }
67  virtual void readBuffered(NativeActivation *context, const char *&data, size_t &length);
68  virtual void cleanup() { ; }
69  virtual RedirectionType::Enum type() { return RedirectionType::NONE; }
70  virtual RexxObject *target() { return OREF_NULL; }
71 
72 protected:
73  MutableBuffer *inputBuffer; // buffered version of the input
74 
75  // a reasonable initial buffer size
76  static const size_t defaultBufferSize = 4096;
77 };
