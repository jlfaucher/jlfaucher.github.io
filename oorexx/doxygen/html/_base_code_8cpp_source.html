153 /**
154  * Default class resolution...which only looks in the environment
155  * or .local.
156  *
157  * @param className The target class name.
158  *
159  * @return The resolved class object, or OREF_NULL if this is not known.
160  */
161 RexxClass *BaseCode::findClass(RexxString *className)
162 {
163  RexxClass *classObject;
164  RexxObject *t = OREF_NULL; // required for the findClass call
165 
166  // if we have a package set, search the packge first
167  if (package != OREF_NULL)
168  {
169  classObject = package->findClass(className, t);
170  if (classObject != OREF_NULL)
171  {
172  return classObject;
173  }
174  }
175 
176  // the interpreter class handles the default lookups
177  return Interpreter::findClass(className);
178 }
