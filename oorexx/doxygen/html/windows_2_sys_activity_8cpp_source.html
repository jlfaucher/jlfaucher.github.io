124 /**
125  * Return the pointer to the base of the current stack.
126  * This is used for checking recursion overflows.
127  *
128  * @return The character pointer for the stack base.
129  */
130 char* SysActivity::getStackBase()
131 {
132 // this is the ideal solution, but it's only available with Windows 8 and above.
133 #if 0
134  ULONG_PTR stackBottom;
135  ULONG_PTR stackTop;
136 
137  GetCurrentThreadStackLimits(&stackBottom, &stackTop);
138  return (char *)stackBottom;
139 #else
140  NT_TIB *tib = (NT_TIB *)NtCurrentTeb();
141  char *highLimit = (char *)tib->StackBase;
142 
143  // the StackLimit field in the tib is the current committed lowbound of the
144  // stack, not the actual bound that has been reserved for the stack. To get that,
145  // we need to use VirtualQuery() to find the beginning of the memory segment
146  // used for the stack. Since the beginning of the stack is probably uncommitted
147  // memory, don't panic if the value obtained from VirtualQuery() is pointing to
148  // non-addressible memory when viewed in the debugger.
149  MEMORY_BASIC_INFORMATION mbi;
150  VirtualQuery(tib->StackLimit, &mbi, sizeof(mbi));
151  char *lowLimit = (char *)mbi.AllocationBase;
152  return lowLimit;
153 #endif
154 }
