52 /**
53  * Default virtual method for handling a run() methods on
54  * an activity dispatcher.
55  */
56 void RexxStartDispatcher::run()
57 {
58  ProtectedSet savedObjects;
59 
60  // set default return values
61  rc = 0;
62  retcode = 0;
63 
64  RexxString *name = GlobalNames::NULLSTRING; // name of the invoked program
65  RexxString *fullname = name; // default the fulllength name to the simple name
66 
67  // if we've been given an actual name, get the string version of it
68  if (programName != NULL)
69  {
70  name = new_string(programName);
71  savedObjects.add(name);
72  }
73 
74  // get an array version of the arguments and protect
75  ArrayClass *new_arglist = new_array(argcount);
76  savedObjects.add(new_arglist);
77 
78  // for compatibility reasons, if this is a command invocation and there is a leading blank
79  // on the only argument, then remove that leading blank from the argument
80  if (calltype == RXCOMMAND && argcount == 1 && arglist[0].strlength > 1 && arglist[0].strptr != NULL && arglist[0].strptr[0] == ' ')
81  {
82  new_arglist->put(new_string(arglist[0].strptr + 1, arglist[0].strlength - 1), 1);
83  }
84  // we need to create an array argument list from the RXSTRINGs
85  else
86  {
87  for (size_t i = 0; i < argcount; i++)
88  {
89  // only add real arguments
90  if (arglist[i].strptr != NULL)
91  {
92  new_arglist->put(new_string(arglist[i]), i + 1);
93  }
94  }
95  }
96 
97  RexxString *source_calltype;
98 
99  // now get the calltype as a character string to used in
100  // the source string. .
101  switch (calltype)
102  {
103  case RXCOMMAND:
104  source_calltype = GlobalNames::COMMAND;
105  break;
106 
107  case RXFUNCTION:
108  source_calltype = GlobalNames::FUNCTION;
109  break;
110 
111  case RXSUBROUTINE:
112  source_calltype = GlobalNames::SUBROUTINE;
113  break;
114 
115  // if not specified, call it a COMMAND.
116  default:
117  source_calltype = GlobalNames::COMMAND;
118  break;
119  }
120 
121  Protected<RoutineClass> program;
122 
123  // if not an instore request, we load this from a file.
124  if (instore == NULL)
125  {
126  fullname = activity->resolveProgramName(name, OREF_NULL, OREF_NULL, RESOLVE_DEFAULT);
127  if (fullname == OREF_NULL)
128  {
129  reportException(Error_Program_unreadable_notfound, name);
130  }
131  savedObjects.add(fullname);
132  program = LanguageParser::createProgramFromFile(fullname);
133  }
134  // we either need to parse the instore source or restore from a
135  // previous image.
136  else
137  {
138  program = LanguageParser::processInstore(instore, name);
139  if (program.isNull())
140  {
141  reportException(Error_Program_unreadable_name, name);
142  }
143  }
144 
145  RexxString *initial_address = activity->getInstance()->getDefaultEnvironment();
146  // actually need to run this?
147  if (!program.isNull())
148  {
149  ProtectedObject program_result;
150  // call the program
151  program->runProgram(activity, source_calltype, initial_address, new_arglist->messageArgs(), argcount, program_result);
152 
153  // provided for a return result (that's optional)
154  if (result != NULL)
155  {
156  // actually have a result to return?
157  if (!program_result.isNull())
158  {
159  // force to a string value
160  program_result = program_result->stringValue();
161  // copy this into the result RXSTRING
162  ((RexxString *)program_result)->copyToRxstring(*result);
163  }
164  // nothing to return
165  else
166  {
167  MAKERXSTRING(*result, NULL, 0);
168  }
169  }
170 
171  // if we have a return result and it is an integer value, return that as a return code.
172  if (!program_result.isNull())
173  {
174  wholenumber_t return_code;
175 
176  if (program_result->numberValue(return_code) && return_code <= SHRT_MAX && return_code >= SHRT_MIN)
177  {
178  retcode = (short)return_code;
179  }
180  }
181  }
182 }

185 /**
186  * Default handler for any error conditions. This just sets the
187  * condition information in the dispatch unit.
188  *
189  * @param c The condition information for the error.
190  */
191 void RexxStartDispatcher::handleError(wholenumber_t r, DirectoryClass *c)
192 {
193  // use the base error handling and set our return code to the negated error code.
194  ActivityDispatcher::handleError(-r, c);
195  retcode = (short)rc;
196  // in case there are errors trying to display
197  activation->enableConditionTraps();
198  // process the error to display the error message.
199  activity->error(activation, conditionData);
200 }

224  * Run a routine for a thread context API call.
225  */
226 void CallProgramDispatcher::run()
227 {
228  RexxString *targetName = new_string(program);
229  //we are resolving from a program name
230  RexxString *name = activity->resolveProgramName(targetName, OREF_NULL, OREF_NULL, RESOLVE_DEFAULT);
231  if (name == OREF_NULL)
232  {
233  reportException(Error_Program_unreadable_notfound, targetName);
234  }
235  ProtectedObject p(name);
236  // create a routine from this file
237  Protected<RoutineClass> routine = LanguageParser::createProgramFromFile(name);
238 
239  if (arguments != OREF_NULL)
240  {
241  // use the provided name for the call name
242  routine->runProgram(activity, arguments->messageArgs(), arguments->messageArgCount(), result);
243  }
244  else
245  {
246  // we use a null string for the name when things are called directly
247  routine->runProgram(activity, NULL, 0, result);
248  }
