545 /**
546  * Add an integer to another object.
547  *
548  * @param other The argument object.
549  *
550  * @return The addition result.
551  */
552 RexxObject *RexxInteger::plus(RexxInteger *other)
553 {
554  // if we want to do this with binary math, our operand must be valid
555  // under the current numeric digits
556  if (Numerics::isValid(value, number_digits()))
557  {
558  // if this is an prefix plus, we just return this object as the result
559  if (other == OREF_NULL)
560  {
561  return this;
562  }
563  // binary operation
564  else
565  {
566  // to calculate the sum with binary math, also the second operand
567  // must be a RexxInteger that is valid under the current numeric digits
568  if (isInteger(other) && Numerics::isValid(other->value, number_digits()))
569  {
570  // neither in the 32-bit, nor in the 64-bit case, will the sum
571  // overflow wholenumber_t: for 32-bit, wholenumber_t accepts up to
572  // 2^31-1 = 2147483647, which is more than twice as large as the
573  // maximum of 999999999 for a RexxInteger
574  // for the 64-bit case, wholenumber_t accepts up to
575  // 2^63 -1 = 9223372036854775807 which is also more than twice as
576  // large as 999999999999999999, the maximum for a RexxInteger
577  wholenumber_t result = value + other->value;
578 
579  // though no wholenumber_t overflow is possible, the sum may
580  // still be too large for a RexxInteger under current numeric digits
581  if (Numerics::isValid(result, number_digits()))
582  {
583  return new_integer(result);
584  }
585  }
586  }
587  }
588 
589  // we will have to forward to NumberString::plus
590  return integer_forward(plus, other);
591 }

643 /**
644  * Multiply an integer by another object.
645  *
646  * @param other The other object.
647  *
648  * @return The multiplication result.
649  */
650 RexxObject *RexxInteger::multiply(RexxInteger *other)
651 {
652  // we'll try to multiply with binary math if both factors
653  // are RexxIntegers that are valid under the current numeric digits
654  if (Numerics::isValid(value, number_digits()) &&
655  other != OREF_NULL && isInteger(other))
656  {
657  wholenumber_t multiplier = other->getValue();
658  wholenumber_t result;
659  if (Numerics::isValid(multiplier, number_digits()))
660  {
661  // check if we can cut this short
662  switch(multiplier)
663  {
664  case 0:
665  // n * 0 is 0, always
666  return IntegerZero;
667  case 1:
668  // n * 1 is n, always
669  return this;
670  case -1:
671  // we just negate
672  return new_integer(-value);
673  case -2: case 2:
674  // we'll evaluate n * 2 by bit shifting
675  // neither in the 32-bit, nor in the 64-bit case, will the shift
676  // overflow wholenumber_t: for 32-bit, wholenumber_t accepts up to
677  // 2^31-1 = 2147483647, which is more than twice as large as the
678  // maximum of 999999999 for a RexxInteger
679  // for the 64-bit case, wholenumber_t accepts up to
680  // 2^63 -1 = 9223372036854775807 which is also more than twice as
681  // large as 999999999999999999, the maximum for a RexxInteger
682  result = value << 1;
683 
684  // though no wholenumber_t overflow is possible, the shift may
685  // still be too large for a RexxInteger under current numeric digits
686  if (Numerics::isValid(result, number_digits()))
687  {
688  // for a multiplier of -2, result will be negative
689  return new_integer(multiplier == -2 ? -result : result);
690  }
691  break;
692  }
693 
694  // the product should be a valid integer under the current numeric
695  // digits; if we know it won't fit, there's no need to multiply
696  // we can estimate this: multiplying an m-bit number with an n-bit
697  // number yields a product of either (m + n - 1) or (m + n) bits
698  // we test (m + n - 1) <= 30 (for 32-bit) and 60 (for 64-bit),
699  // which means (m + n) <= 31 (32-bit) and <= 61 (64-bit), which in
700  // turn makes sure there will be no wholenumber_t overflow
701  if (length_in_bits(value) + length_in_bits(multiplier) - 1 <= Numerics::maxBitsForDigits(number_digits()))
702  {
703  result = value * multiplier;
704  // the result may still be slightly too large; need to check
705  if (Numerics::isValid(result, number_digits()))
706  {
707  return new_integer(result);
708  }
709  }
710  }
711  }
712 
713  // we will have to forward to NumberString::multiply
714  return integer_forward(multiply, other);
715 }

718 /**
719  * Divide an integer object by another object.
720  *
721  * @param other The other object.
722  *
723  * @return The divide result
724  */
725 RexxObject *RexxInteger::divide(RexxInteger *other)
726 {
727  // we'll try to divide with binary math if both dividend and divisor
728  // are RexxIntegers that are valid under the current numeric digits
729  if (Numerics::isValid(value, number_digits()) &&
730  other != OREF_NULL && isInteger(other))
731  {
732  wholenumber_t divisor = other->getValue();
733  if (Numerics::isValid(divisor, number_digits()))
734  {
735  // this may be an integer division, which we can do in binary math
736  // if so, there's no need to check for the size of the resulting quotient
737  switch(divisor)
738  {
739  case 0:
740  break; // let NumberString::divide handle this
741  case -1:
742  return new_integer(-value);
743  case 1:
744  return this;
745  case -2:
746  case 2:
747  // if even, dividing by 2 (or -2) is easy
748  if (!(value & 1))
749  {
750  return new_integer(value / divisor);
751  }
752  break;
753  case -4:
754  case 4:
755  // if multiple of four, dividing is easy
756  if (!(value & 3))
757  {
758  return new_integer(value / divisor);
759  }

891 /**
892  * Integer power operation.
893  *
894  * @param other The power exponent
895  *
896  * @return the exponent result.
897  */
898 RexxObject *RexxInteger::power(RexxObject *other)
899 {
900  // we'll try to do this with binary math if both base and power
901  // are RexxIntegers that are valid under the current numeric digits
902  if (Numerics::isValid(value, number_digits()) &&
903  other != OREF_NULL && isInteger(other))
904  {
905  wholenumber_t power = ((RexxInteger *)other)->getValue();
906  if (Numerics::isValid(power, number_digits()))
907  {
908  // handle some common bases
909  switch (value)
910  {
911  case 0:
912  // base 0 returns integers for positive powers only
913  // 0 ^ n is 0 for n >= 1
914  if (power >= 1)
915  {
916  return integerZero;
917  }
918  // handle power = 0 case and other powers later
919  break;
920 
921  case 1:
922  // base 1 returns integers with both positive and negative powers
923  // 1 ^ n is 1, always
924  return integerOne;
925 
926  case -1:
927  // base -1 returns integers with both positive and negative powers
928  // (-1) ^ n is -1 if n is odd, and +1 if n is even
929  return (power & 1) ? integerMinusOne : integerOne;
930 
931  case 2: case -2:
932  // bases 2 and -2 return integers for positive powers only
933  // we'll evaluate 2 ^ n by bit shifting
934  // the number of bits to shift left must be less than the
935  // maximum bits allowed under the current numeric digits
936  // 2 ** 59 = 576460752303423488 is maximum for 64-bits
937  // 2 ** 29 = 536870912 is maximum for 32-bits
938  // base -2 results will be negative if power is odd
939  if (power >= 0 && power < Numerics::maxBitsForDigits(number_digits()))
940  {
941  // no wholenumber_t overflow possible
942  wholenumber_t result = (wholenumber_t)1 << power;
943 
944  // base -2 results will be negative if power is odd
945  return (value == -2 && power & 1) ? new_integer(-result) : new_integer(result);
946  }
947  // can't use RexxInteger, have to fall back to NumberString
948  return integer_forward(power, other);
949 
950  case 4: case -4:
951  case 8: case -8:
952  case 16: case -16:
953  case 32: case -32:
954  case 64: case -64:
955  case 128: case -128:
956  case 256: case -256:
957  // similar to above base 2 case
958  wholenumber_t baseFactor;
959  baseFactor = length_in_bits(value) - 1;
960  if (power >= 0 && power * baseFactor < Numerics::maxBitsForDigits(number_digits()))
961  {
962  // no wholenumber_t overflow possible
963  wholenumber_t result = (wholenumber_t)1 << (power * baseFactor);
964 
965  // result for negative bases will be negative if power is odd
966  return (value < 0 && power & 1) ? new_integer(-result) : new_integer(result);
967  }
968  // can't use RexxInteger, have to fall back to NumberString
969  return integer_forward(power, other);
970 
971  case 10:
972  case -10:
973  // bases 10 and -10 return integers for positive powers only
974  // we'll evaluate 10 ^ n by lookup (n less than REXXINTEGER_DIGITS)
975  // to keep the result valid under the current numeric digits,
976  // n must be also less than the current numeric digits
977  if (power >= 0 &&
978  power < Numerics::REXXINTEGER_DIGITS &&
979  power < number_digits())
980  {
981  // we abuse our table of maximum values for specified digits
982  // e. g. 10^9 = 999 999 999 (maximum for numeric digits 9) plus 1
983  wholenumber_t result = Numerics::maxValueForDigits(power) + 1;
984  // base -10 results will be negative if power is odd
985  return (value == -10 && power & 1) ? new_integer(-result) : new_integer(result);
986  }
987  // can't use RexxInteger, have to fall back to NumberString
988  return integer_forward(power, other);
989  }
990 
991  // handle some common powers
992  switch (power)
993  {
994  case 0:
995  // n ^ 0 is 1, even 0 ^ 0 is defined as 1 in ooRexx
996  return integerOne;
997 
998  case 1:
999  // n ^ 1 is n, of course
1000  return this;
1001 
1002  case 2:
1003  // we'll evaluate n ^ 2 as n * n
1004  // the result should be a valid integer under the current numeric
1005  // digits; if we know it won't fit, there's no need to multiply
1006  // we can estimate this: squaring an n-bit number yields a result
1007  // of either (2n - 1) or (2n) bits
1008  // we test (2n - 1) <= 30 (for 32-bit) and 60 (for 64-bit),
1009  // which means (2n) <= 31 (32-bit) and <= 61 (64-bit), which in
1010  // turn makes sure there will be no wholenumber_t overflow
1011  if (length_in_bits(value) * 2 - 1 <= Numerics::maxBitsForDigits(number_digits()))
1012  {
1013  wholenumber_t result;
1014  result = value * value;
1015  // the result may still be slightly too large; need to check
1016  if (Numerics::isValid(result, number_digits()))
1017  {
1018  return new_integer(result);
1019  }
1020  }
1021  // we cannot use NumberString::multiply to finish this
1022  // return integer_forward(multiply, this);
1023  // because "*" and "**" behave differently as documented
1024  // "**" will remove trailing zeros as if divided by 1
1025  // "*" will keep any trailing zeros
1026  // there's a test case in EXPONENT.testGroup which shows this:
1027  // ::method "test_25"
1028  // numeric digits 18
1029  // self~assertSame(12345678900000 ** 2, 1.5241578750190521E+26)
1030  // Expected: [[1.52415787501905210E+26]
1031  // Actual: [[1.5241578750190521E+26],
1032  // another example to reproduce this issue is
1033  // a = 123000; say a ** 2 a * a -> 1.5129E+10 1.51290000E+10
1034  // we instead forward to NumberString::power
1035  return integer_forward(power, other);
1036  }
1037 
1038  // at this point, with bases 0, 1, 2, and powers 0, 1, 2 checked,
1039  // we know that (no matter what the current numeric digits setting is),
1040  // a RexxInteger cannot handle
1041  // - negative powers,
1042  // - any power larger than 18 (32-bit) or 37 (64-bit), and
1043  // - any base larger than +/-999 (32-bit) or +/-999999 (64-bit)
1044  if (power < 0 || power > RexxIntegerMaxPower ||
1045  std::abs(value) > RexxIntegerMaxBase)
1046  {
1047  return integer_forward(power, other);
1048  }
1049 
1050  // no common base or power .. try to do the full calculation
1051  // the result should be a valid integer under the current numeric digits
1052  // if we know it won't fit, there's no need to try
1053  // we can estimate the result size: if base has b bits, the result
1054  // will require between (b * power - b + 1) and (b * power) bits
1055  wholenumber_t maxBits = Numerics::maxBitsForDigits(number_digits());
1056  if ((length_in_bits(value) - 1) * power + 1 <= maxBits)
1057  {
1058  wholenumber_t base = value;
1059  wholenumber_t result = 1;
1060  bool overflow = 0;
1061 
1062  // https://en.wikipedia.org/wiki/Exponentiation_by_squaring#Basic_method
1063  while (power > 1)
1064  {
1065  if (power & 1)
1066  {
1067  // make sure that result * base will not overflow
1068  if (length_in_bits(result) + length_in_bits(base) - 1 > maxBits)
1069  {
1070  overflow = 1;
1071  break;
1072  }
1073  result *= base;
1074  power = (power - 1) / 2;
1075  }
1076  else
1077  {
1078  power /= 2;
1079  }
1080  // make sure that base * base will not overflow
1081  if (length_in_bits(base) * 2 - 1 > maxBits)
1082  {
1083  overflow = 1;
1084  break;
1085  }
1086  base *= base;
1087  }
1088  // check that we didn't break out of the loop due to overflow
1089  if (!overflow &&
1090  // and make sure that result * base will not overflow
1091  length_in_bits(result) + length_in_bits(base) - 1 <= maxBits)
1092  {
1093  result *= base;
1094  // our result may still be slightly too large; need to check
1095  if (Numerics::isValid(result, number_digits()))
1096  {
1097  return new_integer(result);
1098  }
1099  }
1100  }
1101  }
1102  }
1103 
1104  // else we fall back using NumberString's power()
1105  return integer_forward(power, other);
1106 }
