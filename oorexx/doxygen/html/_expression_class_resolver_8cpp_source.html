116 /**
117  * Evaluate a dot variable symbol.
118  *
119  * @param context The current evaluation context.
120  * @param stack The current evaluation stack.
121  *
122  * @return The dot symbol value.
123  */
124 RexxObject *ClassResolver::evaluate(RexxActivation *context, ExpressionStack *stack )
125 {
126  PackageClass *package = context->getPackage();
127  // do the lookup, including raising error messages for failures. We only
128  // get called for evaluate because this is a qualified dot-variable lookup.
129  // Error messages will have been issued if this did not resolve.
130  RexxClass *resolvedClass = lookup(package);
131 
132  // evaluate always pushes on the stack.
133  stack->push(resolvedClass);
134  // trace this if tracing intermediates
135  context->traceClassResolution(namespaceName, className, resolvedClass);
136  return resolvedClass;
137 }

140 /**
141  * Resolve a class reference, including handling namespace
142  * qualification. This raises errors for any failure to locate
143  * a class.
144  *
145  * @param package The the source package.
146  *
147  * @return A resolved class object. Raises errors for any failures.
148  */
149 RexxClass *ClassResolver::lookup(PackageClass *package)
150 {
151  // if this is not qualified by namespace, do the full class search.
152  if (namespaceName == OREF_NULL)
153  {
154  RexxObject *t = OREF_NULL; // required for the findClass call
155 
156  RexxClass *resolvedClass = package->findClass(className, t);
157 
158  // We found something, but it must be a real class object
159  if (resolvedClass != OREF_NULL && !resolvedClass->isInstanceOf(TheClassClass))
160  {
161  reportException(Error_Translation_bad_class, className);
162  }
163  return resolvedClass;
164  }
165  // namespace qualified
166  else
167  {
168  PackageClass *namespacePackage = package->findNamespace(namespaceName);
169  if (namespacePackage == OREF_NULL)
170  {
171  reportException(Error_Execution_no_namespace, namespaceName, package->getProgramName());
172  }
173  // we only look for public classes in the namespaces
174  RexxClass *resolvedClass = namespacePackage->findPublicClass(className);
175  // we give a specific error for this one
176  if (resolvedClass == OREF_NULL)
177  {
178  reportException(Error_Execution_no_namespace_class, className, namespaceName);
179  }
180 
181  // We found something, but it must be a real class object
182  if (resolvedClass != OREF_NULL && !resolvedClass->isInstanceOf(TheClassClass))
183  {
184  reportException(Error_Translation_bad_class, className);
185  }
186  return resolvedClass;
187  }
188 }
