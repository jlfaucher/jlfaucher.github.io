152 /**
153  * Wait for a synchronization object to be in the signaled state.
154  *
155  * Any thread that creates windows must process messages. A thread that
156  * calls WaitForSingelObject with an infinite timeout risks deadlocking the
157  * system. MS's solution for this is to use MsgWaitForMultipleObjects to
158  * wait on the object, or a new message arriving in the message queue. Some
159  * threads create windows indirectly, an example is COM with CoInitialize.
160  * Since we can't know if the current thread has a message queue that needs
161  * processing, we use MsgWaitForMultipleObjects.
162  *
163  * However, with the introduction of the C++ native API in ooRexx 4.0.0, it
164  * became possible for external native libraries to attach a thread with an
165  * active window procedure to the interpreter. If a wait is done on that
166  * thread, here in waitHandle(), PeekMessage() causes non-queued messages to be
167  * dispatched, and the window procedure is reentered. This can cause the Rexx
168  * program to hang. In addition, in the one known extension where this problem
169  * happens, ooDialog, the messages need to be passed to the dialog manager
170  * rather than dispatched directly to the window.
171  *
172  * For this special case, the thread can explicity ask, through
173  * RexxSetProcessMessages(), that messages are *not* processed during this
174  * wait. Thread local storage is used to keep track of a flag signalling this
175  * case on a per-thread basis.
176  *
177  * Note that MsgWaitForMultipleObjects only returns if a new message is
178  * placed in the queue. PeekMessage alters the state of all messages in
179  * the queue so that they are no longer 'new.' Once PeekMessage is called,
180  * all the messages on the queue need to be processed.
181  */
182 inline bool waitHandle(HANDLE s, bool bypassMessageLoop)
183 {
184  // If no message loop is flagged, then use WaitForSingleobject()
185  if (bypassMessageLoop)
186  {
187  WaitForSingleObject(s, INFINITE);
188  return true;
189  }
190 
191  // If already signaled, return.
192  if (WaitForSingleObject(s, 0) == WAIT_OBJECT_0)
193  {
194  return true;
195  }
196 
197  MSG msg = { 0 };
198 
199  do
200  {
201  while (PeekMessage(&msg, NULL, 0, 0, PM_REMOVE))
202  {
203  TranslateMessage(&msg);
204  DispatchMessage(&msg);
205 
206  // Check to see if signaled.
207  if (WaitForSingleObject(s, 0) == WAIT_OBJECT_0)
208  {
209  return true;
210  }
211  }
212  }
213  while (MsgWaitForMultipleObjects(1, &s, FALSE, INFINITE, QS_ALLINPUT) == WAIT_OBJECT_0 + 1);
214  return true;
215 }

218 /**
219  * Wait for a synchronization object to be in the signaled state.
220  *
221  * Any thread that creates windows must process messages. A thread that
222  * calls WaitForSingelObject with an infinite timeout risks deadlocking the
223  * system. MS's solution for this is to use MsgWaitForMultipleObjects to
224  * wait on the object, or a new message arriving in the message queue. Some
225  * threads create windows indirectly, an example is COM with CoInitialize.
226  * Since we can't know if the current thread has a message queue that needs
227  * processing, we use MsgWaitForMultipleObjects.
228  *
229  * However, with the introduction of the C++ native API in ooRexx 4.0.0, it
230  * became possible for external native libraries to attach a thread with an
231  * active window procedure to the interpreter. If a wait is done on that
232  * thread, here in waitHandle(), PeekMessage() causes non-queued messages to be
233  * dispatched, and the window procedure is reentered. This can cause the Rexx
234  * program to hang. In addition, in the one known extension where this problem
235  * happens, ooDialog, the messages need to be passed to the dialog manager
236  * rather than dispatched directly to the window.
237  *
238  * For this special case, the thread can explicity ask, through
239  * RexxSetProcessMessages(), that messages are *not* processed during this
240  * wait. Thread local storage is used to keep track of a flag signalling this
241  * case on a per-thread basis.
242  *
243  * Note that MsgWaitForMultipleObjects only returns if a new message is
244  * placed in the queue. PeekMessage alters the state of all messages in
245  * the queue so that they are no longer 'new.' Once PeekMessage is called,
246  * all the messages on the queue need to be processed.
247  */
248 inline bool waitHandle(HANDLE s, uint32_t timeOut, bool bypassMessageLoop)
249 {
250  // If no message loop is flagged, then use WaitForSingleobject()
251  if (bypassMessageLoop)
252  {
253  return WaitForSingleObject(s, timeOut) == WAIT_OBJECT_0;
254  }
255 
256  // If already signaled, return.
257  if (WaitForSingleObject(s, 0) == WAIT_OBJECT_0)
258  {
259  return true;
260  }
261 
262  MSG msg = { 0 };
263 
264  // calculate our target stop time
265  ULONGLONG stopTime = GetTickCount64() + timeOut;
266 
267  while (true)
268  {
269  while (PeekMessage(&msg, NULL, 0, 0, PM_REMOVE))
270  {
271  TranslateMessage(&msg);
272  DispatchMessage(&msg);
273 
274  // Check to see if signaled.
275  if (WaitForSingleObject(s, 0) == WAIT_OBJECT_0)
276  {
277  return true;
278  }
279  }
280 
281  // now
282  ULONGLONG currentTime = GetTickCount64();
283  // this is a time out situation
284  if (currentTime > stopTime)
285  {
286  return false;
287  }
288  timeOut = (uint32_t)(stopTime - currentTime);
289  // now wait for some action on the objects
290  DWORD rc = MsgWaitForMultipleObjects(1, &s, FALSE, timeOut, QS_ALLINPUT);
291  // we got what we want for our semaphore
292  if (rc == WAIT_OBJECT_0)
293  {
294  return true;
295  }
296  // if not for the message queue, we have either a timeout or a failure,
297  // either is a false return
298  else if (rc != WAIT_OBJECT_0 + 1)
299  {
300  return false;
301  }
302  }
303  return true;
304 }
