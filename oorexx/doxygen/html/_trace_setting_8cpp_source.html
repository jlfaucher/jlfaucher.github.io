55 /**
56  * Format an encoded trace setting back into human readable form.
57  *
58  * @param setting The source setting.
59  *
60  * @return The string representation of the trace setting.
61  */
62 RexxString *TraceSetting::toString()
63 {
64  char setting[3];
65  setting[0] = '\0';
66  int index = 0;
67 
68  // do we have a debug setting?
69  // add a '?' prefix to the formatted setting
70  if (isDebug())
71  {
72  setting[index++] = '?';
73  }
74  // we need to decode the flags to generate the actual setting.
75 
76  if (flags[traceOff])
77  {
78  setting[index++] = 'O';
79  }
80  // The following must be done in this order because N and E both
81  // trace failures.
82  else if (flags[traceNormal])
83  {
84  setting[index++] = 'N';
85  }
86  else if (flags[traceErrors])
87  {
88  setting[index++] = 'E';
89  }
90  else if (flags[traceFailures])
91  {
92  setting[index++] = 'F';
93  }
94  // these also must be done in this order...
95  // each option also includes the setting below
96  else if (flags[traceIntermediates])
97  {
98  setting[index++] = 'I';
99  }
100  else if (flags[traceResults])
101  {
102  setting[index++] = 'R';
103  }
104  // ALL includes Label and Command, so it comes before those
105  else if (flags[traceAll])
106  {
107  setting[index++] = 'A';
108  }
109  else if (flags[traceLabels])
110  {
111  setting[index++] = 'L';
112  }
113  else if (flags[traceCommands])
114  {
115  setting[index++] = 'C';
116  }
117 
118  // null terminate, then return as a string
119  setting[index] = '\0';
120  return new_string(setting);
121 }

124 /**
125  * Parse a trace setting value into a decoded setting
126  * and the RexxActivation debug flag set to allow
127  * new trace settings to be processed more quickly.
128  *
129  * @param value The string source of the trace setting.
130  * @param newSetting The returned setting in binary form.
131  * @param debugFlags The debug flag representation of the trace setting.
132  */
133 bool TraceSetting::parseTraceSetting(RexxString *value, char &badOption)
134 {
135  bool turnDebugOn = false;
136 
137  size_t length = value->getLength();
138 
139  // null string? This just sets tracing to default.
140  if (length == 0)
141  {
142  setTraceNormal();
143  return true;
144  }
145 
146  // turn off entirely so we know if we have just a debug toggle request.
147  clear();
148 
149  // scan the characters. We only recognize the first characters of
150  // words, but this can also have a prefix.
151  for (size_t pos = 0; pos < length; pos++)
152  {
153  switch (Utilities::toUpper(value->getChar(pos)))
154  {
155  // Toggle the debug character...we can have any number of these, we
156  // only perform an operation if we have an odd number of them.
157  case '?':
158  // flip the debug setting. We figure out what to do once
159  // we've processed the whole string.
160  turnDebugOn = !turnDebugOn;
161  continue;
162 
163  // TRACE ALL
164  case 'A':
165  setTraceAll();
166  break;
167 
168  // TRACE COMMANDS
169  case 'C':
170  setTraceCommands();
171  break;
172 
173  // TRACE LABELS
174  case 'L':
175  setTraceLabels();
176  break;
177 
178  // TRACE ERRORS
179  case 'E':
180  setTraceErrors();
181  break;
182 
183  // TRACE FAILURES
184  case 'F':
185  setTraceFailures();
186  break;
187 
188  // TRACE NORMAL
189  case 'N':
190  // default setting is the same as failure
191  setTraceNormal();
192  break;
193 
194  // TRACE OFF
195  case 'O':
196  setTraceOff();
197  break;
198 
199  // TRACE RESULTS
200  case 'R':
201  setTraceResults();
202  break;
203 
204  // TRACE INTERMEDIATES
205  case 'I':
206  setTraceIntermediates();
207  break;
208 
209  // unknown trace setting
210  default:
211  // each context handles its own error reporting, so give back the
212  // information needed for the message.
213  badOption = value->getChar(pos);
214  return false;
215  break;
216  }
217  // we break out of the loop if we get here. Situations that
218  // need additional parsing do a continue.
219  break;
220  }
221 
222  // we need to somehow set the debug setting. If there is no other
223  // trace setting, this is a debug toggle. Otherwise, turn on the debug flag in
224  // the setting.
225  if (turnDebugOn)
226  {
227  if (isNoSetting())
228  {
229  setDebugToggle();
230  }
231  // trace OFF is special...it unconditionally turns off debug, so
232  // don't set the debug flag on if we have OFF.
233  else if (!isTraceOff())
234  {
235  // this turns on special optimization flags also.
236  setDebug();
237  }
238  }
239  return true;
240 }
