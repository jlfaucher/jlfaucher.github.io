121 /**
122  * Execute an addres instruction.
123  *
124  * @param context The current program execution context.
125  * @param stack The current evaluation stack.
126  */
127 void RexxInstructionAddress::execute(RexxActivation *context, ExpressionStack *stack )
128 {
129  // tracing is handled seperately in each section because the command form
130  // needs to trace if "Trace Commands" is in effect.
131 
132  // Nothing specified is just an address toggle.. this is simple.
133  if (environment == OREF_NULL && dynamicAddress == OREF_NULL)
134  {
135  // trace if necessary
136  context->traceInstruction(this);
137  context->toggleAddress();
138  context->pauseInstruction();
139  }
140  // have a static address name? We could also have a command to issue.
141  else if (environment != OREF_NULL)
142  {
143  // Is this the command form? Evaluate and issue to the target environment
144  if (command != OREF_NULL)
145  {
146  // this also traces if TRACE COMMANDS is in effect.
147  context->traceCommand(this);
148  // evaluate the command expression
149  RexxObject *result = command->evaluate(context, stack);
150  // this must be a string
151  RexxString *_command = result->requestString();
152  // protect this
153  stack->push(_command);
154  // are we tracing commands?
155  if (context->tracingCommands())
156  {
157  // trace the full command result
158  context->traceResultValue(_command);
159  }
160  // validate the address name using system rules
161  SystemInterpreter::validateAddressName(environment);
162  // and execute the command
163  context->command(environment, _command, getIOConfig());
164  }
165  // we're just changing the current address target
166  else
167  {
168  context->traceInstruction(this);
169  // validate this environment name
170  SystemInterpreter::validateAddressName(environment);
171  // and make that the current address
172  context->setAddress(environment, getIOConfig());
173  context->pauseInstruction();
174  }
175  }
176  // ADDRESS VALUE form
177  else
178  {
179  context->traceInstruction(this);
180  // evaluate
181  RexxObject *result = dynamicAddress->evaluate(context, stack);
182  RexxString *_address = result->requestString();
183  // protect this
184  stack->push(_address);
185  context->traceResult(_address);
186  // validate this using system rules, then set the new address
187  SystemInterpreter::validateAddressName(_address);
188  context->setAddress(_address, getIOConfig());
189  context->pauseInstruction();
190  }
191 }
