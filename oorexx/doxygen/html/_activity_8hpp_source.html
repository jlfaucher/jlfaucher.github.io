269  inline bool waitForDispatch()
270  {
271  waitingOnSemaphore = true;
272  bool timedOut = false;
273  // we wait for a small interval, then post the semaphore
274  // ourselves if we time out so the dispatcher knows we're no longer
275  // waiting.
276  if (!runSem.wait(MaxDispatchWait))
277  {
278  // our return indicates we timed out and are still in the wait queue
279  timedOut = true;
280  postDispatch();
281  }
282  waitingOnSemaphore = false;
283  return timedOut;
284  }
