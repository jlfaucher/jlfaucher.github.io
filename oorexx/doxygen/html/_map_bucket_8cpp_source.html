64  * Construct a MapBucket item of the given size.
65  *
66  * @param entries The total number of entries in the object.

68 MapBucket::MapBucket(size_t entries)
69 {
70  totalSize = entries;
71

73  // an even value here, so we steal a slot from the overflow area
74  bucketSize = entries / 2;
75  if (bucketSize % 2 == 0)
76  {
77  bucketSize++;
78  }
79  freeItem = entries - 1; // this is our first free overflow item
80  itemCount = 0; // nothing in this

114 /**
115  * Remove an element from the table.
116  *
117  * @param index The index of the item to remove.
118  *
119  * @return The value stored at that table, or 0 if nothing was
120  * found.
121  */
122 size_t MapBucket::remove(RexxInternalObject *index)
123 {
124  // get the slot position to start searching.
125  MapLink position = hashIndex(index);
126 
127  // if there is nothing in the slot postion, we do not
128  // have this entry.
129  if (entries[position].isAvailable())
130  {
131  return 0;
132  }
133 
134 
135  MapLink previous = NoLink;
136  // no run the elements chained off of this slot position looking for
137  // the target one.
138  do
139  {
140  // got a hit? We need to fix up the chains to remove this entry
141  if (entries[position].isIndex(index))
142  {
143  // we found our item, so decrement our count.
144  itemCount--;
145  // save the return value
146  size_t removed = entries[position].value;
147  // get the next pointer and see what we have
148  MapLink next = entries[position].next;
149 
150  // this is the last entry in the slot chain.
151  if (next == NoMore)
152  {
153  entries[position].clear();
154  // is this the first one in the chain. Nothing left
155  // to do but return
156  if (previous == NoLink)
157  {
158  return removed;
159  }
160 
161  // The free position is the highest available free slot. If
162  // this has been removed for a higher position (likely), make this
163  // the new free position.
164  if (position > freeItem)
165  {
166  freeItem = position;
167  }
168  // chop off the tail position
169  entries[previous].next = NoMore;
170  }
171  // our item has trailing items. We need to close up the chain.
172  else
173  {
174  // close up the link
175  entries[position].copyElement(entries[next]);
176  // clean the removed link
177  entries[next].clear();
178  // fix up the search position for a new item.
179  if (freeItem < next)
180  {
181  freeItem = next;
182  }
183  }
184  return removed;
185  }
186  // remember the previous and step
187  previous = position;
188  position = entries[position].next;
189 
190  } while (position != NoMore);
191 
192  // the target item not found
193  return 0;
194 }

233 /**
234  * Add an entry to the table, replacing any existing
235  * entries.
236  *
237  * @param value The value to put into the table.
238  * @param index The index object for the value.
239  *
240  * @return An indicator that we were unable to add because we're
241  * full.
242  */
243 bool MapBucket::put(size_t value, RexxInternalObject *index)
244 {
245  // indicate an addition failure if we're out of room
246  if (isFull())
247  {
248  return false;
249  }
250 
251  // calculate the hash slot where we add this.
252  MapLink position = hashIndex(index);
253 
254  // if this is available, then set the value and return
255  if (entries[position].isAvailable())
256  {
257  // one more item in the chain.
258  entries[position].set(index, value);
259  itemCount++;
260  return true;
261  }
262 
263  MapLink previous = NoLink;
264 
265  // this slot is in use. We either A) have this entry already, in
266  // which case we replace the value, or B) need to add another entry to the end
267  do
268  {
269  // if we got a hit, just change the slot position
270  if (entries[position].isIndex(index))
271  {
272  // everything worked, we have room.
273  entries[position].setValue(value);
274  return true;
275  }
276  previous = position;
277  position = entries[position].next;
278  } while (position != NoMore);
279 
280  // append a new value.
281  return append(value, index, previous);
282 }

285 /**
286  * Add a link to a slot position chain.
287  *
288  * @param value The value to add.
289  * @param index The index value.
290  * @param position The position of the last link in the chain.
291  *
292  * @return true if we added this successfully, false if we had
293  * an overflow condition.
294  */
295 bool MapBucket::append(size_t value, RexxInternalObject * index, MapLink position)
296 {
297  // the free position is "generally" an empty slot, at least until
298  // we start removing things. We'll need to scan for a real empty one
299  for (MapLink over = freeItem; over >= bucketSize; over--)
300  {
301  // found an empty slot, insert the value here
302  if (entries[over].isAvailable())
303  {
304  // set the value and set the next link to nothing
305  entries[over].set(index, value);
306  entries[over].next = NoMore;
307  // append this to the bucket position chain.
308  entries[position].next = over;
309  // we start searching for a new slot here the next time.
310  freeItem = over - 1;
311  // we have another item here.
312  itemCount++;
313  // this worked fine
314  return true;
315  }
316  }
317  // no free slots available...which should not really happen because
318  // we checked to see if we were full first.
319  return false;
320 }

341 void MapBucket::merge(MapBucket *other)
342 {
343  for (size_t i = 0; i < totalSize; i++)
344  {
345  // if we have a real entry, add these values to the target bucket.
346  // this will rehash and redistrubute into the larger table.

355 /**
356  * Locate the slot position for an index.
357  *
358  * @param index The index object.
359  *
360  * @return The position of the item, or NoLink if this is not
361  * found.
362  */
363 MapBucket::MapLink MapBucket::locate(RexxInternalObject *index)
364 {
365  // get the slot position
366  MapLink position = hashIndex(index);
367 
368  // nothing in that slot, we don't have this
369  if (entries[position].isAvailable())
370  {
371  return NoLink;
372  }
373 
374  // loop looking for a match
375  do
376  {
377  // got a match, we can return this now.
378  if (entries[position].isIndex(index))
379  {
380  return position;
381  }
382  position = entries[position].next;
383  } while (position != NoMore);
384  // no entry found
385  return NoLink;
386 }

412  * Decrement the value associated with a key. If the value goes
413  * to zero, the key is removed.
414  *
415  * @param key The target key.
416  */
417 void MapBucket::decrement(RexxInternalObject *key)
418 {
419  // locate the item
420  MapLink position = locate(key);
421 
422  // bail if
423  if (position == NoLink)
424  {
425  return;
426  }
427 
428  // if this is already zero, remove the item
429  if (entries[position].value == 0)
430  {
431  remove(key);
432  return;
433  }
434 
435  // decrement the value
436  entries[position].value--;
437 
438  // and test again to see if we went to zero

440  {
441  remove(key);
442  }
443 }
