142 /**
143  * Evaluate a function call.
144  *
145  * @param context The current execution context.
146  * @param stack the current evaluation stack.
147  *
148  * @return The function return value.
149  */
150 RexxObject *QualifiedFunction::evaluate(RexxActivation *context, ExpressionStack *stack)
151 {
152  // save the top of the stack for popping values off later.
153  size_t stacktop = stack->location();
154 
155  // evaluate the arguments first
156  RexxInstruction::evaluateArguments(context, stack, arguments, argumentCount);
157 
158  ProtectedObject result; // returned result
159 
160  // get the current package from the context
161  PackageClass *package = context->getPackage();
162 
163  // we must be able to find the named namespace
164  PackageClass *namespacePackage = package->findNamespace(namespaceName);
165  if (namespacePackage == OREF_NULL)
166  {
167  reportException(Error_Execution_no_namespace, namespaceName, package->getProgramName());
168  }
169 
170  // we only look for public routines in the namespaces
171  RoutineClass *resolvedRoutine = namespacePackage->findPublicRoutine(functionName);
172  // we give a specific error for this one
173  if (resolvedRoutine == OREF_NULL)
174  {
175  reportException(Error_Routine_not_found_namespace, functionName, namespaceName);
176  }
177 
178  // call the resolved function
179  resolvedRoutine->call(context->getActivity(), functionName, stack->arguments(argumentCount), argumentCount, GlobalNames::FUNCTION, OREF_NULL, EXTERNALCALL, result);
180 
181  // functions must have a return result
182  if (result.isNull())
183  {
184  reportException(Error_Function_no_data_function, functionName);
185  }
186 
187  // remove the arguments from the stack and push our result
188  stack->setTop(stacktop);
189  stack->push(result);
190 
191  // trace if needed and return the result
192  context->traceFunction(functionName, result);
193  return result;
194 }
