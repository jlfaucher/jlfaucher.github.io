119  static inline void relinquish(Activity *activity)
120  {
121  // if we have waiting activities, then let one of them
122  // in next.
123  if (hasWaiters())
124  {
125  addWaitingActivity(activity, true);
126  }
127  }
128  // give up control, but only if the time slice requires it.
129  static inline void relinquishIfNeeded(Activity *activity)
130  {
131  // if we have waiting activities, then let one of them
132  // in next.
133  if (hasWaiters())
134  {
135  // if we have API calls trying to get in, then don't bother checking the time stamp.
136  if (hasApiWaiters())
137  {
138  addWaitingActivity(activity, true);
139  }
140  else
141  {
142  // check the time that we've have been holding the lock and release it
143  // if we've crossed the threshold,
144  uint64_t timeNow = SysThread::getMillisecondTicks();
145  if (timeNow - lastLockTime > timeSliceLength)
146  {
147  addWaitingActivity(activity, true);
148  }
149  }
150  }
151  }
