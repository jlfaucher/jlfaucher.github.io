415  static inline bool isSymbolCharacter(unsigned int ch)
416  {
417  // The anding is necessary to keep characters > 0x7F from being
418  // treated as negative numbers and returning bogus values.
419  // the test for less that 256 is necessary because we have
420  // a magic "not-a-character" marker that's out of range.
421  return ch < 256 && characterTable[ch & 0xff] != 0;
422  }
423 
424  static inline int translateChar(unsigned int ch)
425  {
426  // The anding is necessary to keep characters > 0x7F from being
427  // treated as negative numbers and returning bogus values. This
428  // assumes were're working with a good character already.
429  return characterTable[ch & 0xff];
430  }
