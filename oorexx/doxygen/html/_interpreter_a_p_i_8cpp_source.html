98 /******************************************************************************/
99 /* Name: RexxMain */
100 /* */
101 /* Arguments: argcount - Number of args in arglist */
102 /* arglist - Array of args (array of RXSTRINGs) */
103 /* programname - REXX program to run */
104 /* instore - Instore array (array of 2 RXSTRINGs) */
105 /* envname - Initial cmd environment */
106 /* calltype - How the program is called */
107 /* exits - Array of system exit names (array of RXSTRINGs) */
108 /* */
109 /* Returned: result - Result returned from program */
110 /* rc - Return code from program */
111 /* */
112 /* Notes: Primary path into Object REXX. Makes sure Object REXX is up */
113 /* and runs the requested program. */
114 /* */
115 /******************************************************************************/
116 int REXXENTRY RexxStart(
117  size_t argcount, /* Number of args in arglist */
118  PCONSTRXSTRING arglist, /* Array of args */
119  const char *programname, /* REXX program to run */
120  PRXSTRING instore, /* Instore array */
121  const char *envname, /* Initial cmd environment */
122  int calltype, /* How the program is called */
123  PRXSYSEXIT exits, /* Array of system exit names */
124  short *retcode, /* Integer form of result */
125  PRXSTRING result) /* Result returned from program */
126 {
127  if (calltype == RXCOMMAND && argcount == 1 &&
128  arglist[0].strptr != NULL &&
129  arglist[0].strlength == 3 &&
130  StringUtil::caselessCompare(arglist[0].strptr, "//T", 3) == 0)
131  {
132  TranslateDispatcher arguments;
133  arguments.programName = programname;
134  arguments.instore = instore;
135  // this just translates and gives the error, potentially returning
136  // the instore image
137  arguments.outputName = NULL;
138  arguments.encode = false;
139  // go run this program
140  arguments.invoke(exits, envname);
141 
142  return (int)arguments.rc; /* return the error code (negated) */
143  }
144 
145 
146  // this is the dispatcher that handles the actual
147  // interpreter call. This gets all of the RexxStart arguments, then
148  // gets dispatched on the other side of the interpreter boundary
149  RexxStartDispatcher arguments;
150  /* copy all of the arguments into */
151  /* the info control block, which is */
152  /* passed across the kernel boundary */
153  /* into the real RexxStart method */
154  /* this is a real execution */
155  arguments.argcount = argcount;
156  arguments.arglist = arglist;
157  arguments.programName = programname;
158  arguments.instore = instore;
159  arguments.calltype = calltype;
160  arguments.retcode = 0;
161  arguments.result = result;
162 
163  // go run this program
164  arguments.invoke(exits, envname);
165  *retcode = arguments.retcode;
166 
167  // terminate and clean up the interpreter runtime. This only works
168  // if there are no active instances
169  Interpreter::terminateInterpreter();
170 
171  return (int)arguments.rc; /* return the error code (negated) */
172 }

178  *
179  * @param inFile The input source file.

185 RexxReturnCode REXXENTRY RexxCompileProgram(const char *inFile, const char *outFile, PRXSYSEXIT exits, bool encode)
186 {

193  arguments.encode = encode;
194 
195  // go run this program

197 
198  // terminate and clean up the interpreter runtime. This only works
199  // if there are no active instances

201 
202  return (RexxReturnCode)arguments.rc; /* return the error code (negated) */
