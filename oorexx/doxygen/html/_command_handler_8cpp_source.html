88 /**
89  * Call a command handler
90  *
91  * @param activity The current activity.
92  * @param activation The top-most activation.
93  * @param address The address name.
94  * @param command The command string.
95  * @param result The returned RC value.
96  * @param condition A potential returned condition object.
97  */
98 void CommandHandler::call(Activity *activity, RexxActivation *activation, RexxString *address, RexxString *command, ProtectedObject &result, ProtectedObject &condition, CommandIOContext *ioContext)
99 {
100  if (type == HandlerType::REGISTERED_NAME)
101  {
102  // not all handlers support I/O redirection. Give an error
103  // if an attempt is made here
104  if (ioContext != OREF_NULL)
105  {
106  reportException(Error_Execution_address_redirection_not_supported, address);
107  }
108 
109  CommandHandlerDispatcher dispatcher(activity, entryPoint, command);
110 
111  // run this and give back the return code
112  activity->run(dispatcher);
113  dispatcher.complete(command, result, condition);
114  }
115  // new style command handler
116  else if (type == HandlerType::DIRECT)
117  {
118  // not all handlers support I/O redirection. Give an error
119  // if an attempt is made here
120  if (ioContext != OREF_NULL)
121  {
122  reportException(Error_Execution_address_redirection_not_supported, address);
123  }
124 
125  ContextCommandHandlerDispatcher dispatcher(entryPoint, address, command, result, condition);
126 
127  // run this and give back the return code
128  activity->run(dispatcher);
129  }
130  // a command handler that supports I/O redirection. This is
131  // the only one that uses the I/O context.
132  else if (type == HandlerType::REDIRECTING)
133  {
134  RedirectingCommandHandlerDispatcher dispatcher(entryPoint, address, command, result, condition, ioContext);
135 
136  // if we got an io context back, it is time to initialize everthing. This is the very
137  // last point before we exit the interpreter.
138  if (ioContext != OREF_NULL)
139  {
140  ioContext->init();
141  }
142  // run this and give back the return code
143  activity->run(dispatcher);
144 
145  // and also perform clean up as soon as we get control back
146  if (ioContext != OREF_NULL)
147  {
148  ioContext->cleanup();
149  }
150  }
151 }

178 /**
179  * Do post-callout processing of a command dispatch. This
180  * code runs after re-entering the interpreter, so all
181  * interpreter facilities are available.
182  *
183  * @param result The return RC result.
184  * @param condition A potential condition return.
185  */
186 void CommandHandlerDispatcher::complete(RexxString *command, ProtectedObject &result, ProtectedObject &condition)
187 {
188  // did we get a numeric return code? Turn into an Integer object.
189  if (sbrc != 0)
190  {
191  result = new_integer(sbrc);
192  }
193  // maybe we got a string value back?
194  else if (!RXNULLSTRING(retstr))
195  {
196  // make into a string value and try to convert to an integer (not an error
197  // if it doesn't convert)
198  result = new_string(retstr.strptr, retstr.strlength);
199  // try to get this as a numeric value
200  result->numberValue(sbrc);
201  // handle any buffer reallocation
202  if (retstr.strptr != default_return_buffer)
203  {
204  SystemInterpreter::releaseResultMemory(retstr.strptr);
205  }
206  }
207  // default return code is zero
208  else
209  {
210  result = IntegerZero;
211  }
212 
213  // Check error flags from subcom handler and if needed, stick condition
214  // into result array.
215  if (flags & (unsigned short)RXSUBCOM_FAILURE)
216  {
217  // raise the condition when things are done
218  condition = activity->createConditionObject(GlobalNames::FAILURE, result, command, OREF_NULL, OREF_NULL);
219  }
220  else if (flags & (unsigned short)RXSUBCOM_ERROR)
221  {
222  // raise the condition when things are done
223  condition = activity->createConditionObject(GlobalNames::ERRORNAME, result, command, OREF_NULL, OREF_NULL);
224  }
225 }

244 /**
245  * Default handler for any error conditions. This just sets the
246  * condition information in the dispatch unit.
247  *
248  * @param c The condition information for the error.
249  */
250 void ContextCommandHandlerDispatcher::handleError(DirectoryClass *c)
251 {
252  // this only gets added if there is a condition
253  // NB: This is called by the native activation after re-entering the
254  // kernel code, so this has full access to kernel calls.
255  if (c != OREF_NULL)
256  {
257  // check to see if this is an error or failure situation
258  RexxString *conditionName = (RexxString *)c->get(GlobalNames::CONDITION);
259  // if this is not a syntax error, this is likely an error
260  // or failure condition, and the return value is taken from
261  // the condition object.
262  if (!conditionName->strCompare(GlobalNames::SYNTAX))
263  {
264  // just set the condition now...additional processing will be
265  // done on return to the command issuer
266  condition = c;
267  }
268  else
269  {
270  // raise this as a normal error by default
271  activation->checkConditions();
272  }
273  }
274 }
