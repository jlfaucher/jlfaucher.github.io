71 /**
72  * Start the rxapi daemon process.
73  */
74 void SysLocalAPIManager::startServerProcess()
75 {
76 #define RXAPI "rxapi"
77 #define DOTDOT_BIN_RXAPI "../bin/" RXAPI
78 #define DOT_RXAPI "./" RXAPI
79  char apiExeName[] = RXAPI;
80  char *apiExeArg[2];
81  apiExeArg[0] = apiExeName;
82  apiExeArg[1] = NULL;
83 
84  if (signal(SIGPIPE, SIG_IGN) == SIG_ERR)
85  {
86  return;
87  }
88 
89 
90  pid_t pid = fork();
91  if (pid < 0)
92  {
93  throw new ServiceException(API_FAILURE, "Unable to start API server");
94  }
95 
96  if (pid != 0)
97  {
98  // we are the parent process
99  return;
100  }
101  // if we get here we are the child process
102 
103  // become the session leader
104  setsid();
105 
106  // housekeeping - chdir to the root subdir and close all open files
107  int ignore = chdir("/");
108  for(int i = 0; i < 1024; i++)
109  {
110  close(i);
111  }
112 
113  // we will make multiple attempts at locating rxapi, first with a full path.
114  // while on Windows the rxapi executable is located in the same directory
115  // as the rexxapi library, on Unix these two are in different paths.
116  // typically the rexxapi library is in install-path/lib/ or in
117  // install-path/lib64/ and the rxapi executable is in install-path/bin/
118  // so we try to locate rxapi with ../bin/rxapi relative to the library
119  // location.
120  AutoFree fullExeName = NULL;
121  const char *installLocation = SysProcess::getLibraryLocation();
122  if (installLocation != NULL)
123  {
124  // the library location includes the trailing "/" character
125  size_t commandSize = strlen(installLocation) + strlen(DOTDOT_BIN_RXAPI) + 1;
126 
127  fullExeName = (char *)malloc(commandSize);
128  // the path might contain blanks, so we'll need to enclose the
129  // command name in quotes
130  snprintf(fullExeName, commandSize, "%s%s", installLocation, DOTDOT_BIN_RXAPI);
131  execvp(fullExeName, apiExeArg);
132  }
133 
134  // next we use the unqualified rxapi name and try to locate it on $PATH
135  execvp(RXAPI, apiExeArg);
136 
137  // did this still fail? Last attempt, try to load this from the current directory
138  execvp(DOT_RXAPI, apiExeArg);
139 
140  // still no luck? This is a launch failure. Because we are the forked process,
141  // we need to exit immediately.
142  exit(1);
143 }

146 /**
147  * Check to see if we've inherited a session queue from a calling process. This shows
148  * up as an environment variable value.
149  *
150  * @param sessionQueue
151  * The returned session queue handle, if it exists.
152  *
153  * @return true if the session queue is inherited, false if a new once needs to be created.
154  */
155 bool SysLocalAPIManager::getActiveSessionQueue(QueueHandle &sessionQueue)
156 {
157  // check to see if we have an env variable set...if we do we
158  // inherit from our parent session
159  char *envbuffer = getenv("RXQUEUESESSION");
160  if (envbuffer != NULL)
161  {
162  sscanf(envbuffer, "%p", (void **)&sessionQueue);
163  return true;
164  }
165  return false;
166 }
