64 /**
65  * Initialize a MethodDictionary.
66  *
67  * @param capacity The desired capacity.
68  */
69 MethodDictionary::MethodDictionary(size_t capacity) : StringHashCollection(capacity)
70 {
71  ProtectedObject p(this);
72  // Method dictionaries don't get created until we're defining/adding
73  //. methods to a class. We can assume that once created, we'll need most of the
74  // items (except maybe the instance methods).
75  instanceMethods = OREF_NULL;
76  // this keeps our main lookup order, defined by the inheritance order of the
77  // superclasses and mixins.
78  scopeList = new_array(DefaultScopeListSize);
79  // this defines the lookup order from each class in the hierarchy.
80  // because mixins can get inserted in the order, this is not
81  // necessarily determined by straight superclass order. This is really
82  // a bug because it can screw up super calls if a mixin slips a method in and
83  // does not support the same signatures. Unfortunately, a lot of code has been
84  // written using the broken behaviour and it would be difficult to change this
85  // at this point. This table allows us to determine the superclass and perform
86  // superclass method lookup the respects this order.
87  scopeOrders = new_identity_table();
88 }

158 /**
159  * Add a method to the main method dictionary
160  *
161  * @param methodName The target method name.
162  * @param method The method object to add.
163  */
164 void MethodDictionary::addMethod(RexxString *methodName, MethodClass *method)
165 {
166  // if there is no method, we're removing this method.
167  // write .nil to the table
168  if (method == OREF_NULL || method == TheNilObject)
169  {
170  addFront(TheNilObject, methodName);
171  }
172  else
173  {
174  // when merging method dictionaries, we want to avoid having
175  // duplicate methods. Scan all methods with the same name to
176  // see if we have an existing method with the same scope.
177  HashContents::IndexIterator iterator = contents->iterator(methodName);
178 
179  for (; iterator.isAvailable(); iterator.next())
180  {
181  MethodClass *tableMethod = (MethodClass *)iterator.value();
182  // if the scopes are the same, then we are replacing this
183  // method in the table.
184  if (tableMethod->isScope(method->getScope()))
185  {
186  iterator.replace(method);
187  return;
188  }
189  }
190 
191  // this is a new method for this index...add it to the front
192  addFront(method, methodName);
193  }
194 }

216  * Overlay a collection of methods on top of this dictionary.
217  *
218  * @param source The source method dictionary.
219  * @param scope The scope the copied methods will have in the target dictionary.
220  */
221 void MethodDictionary::replaceMethods(MethodDictionary *source, RexxClass *scope)
222 {
223  // use an iterator to traverse the table
224  HashContents::TableIterator iterator = source->iterator();
225 
226  for (; iterator.isAvailable(); iterator.next())
227  {
228  // copy these methods over any of our own.
229  MethodClass *method = (MethodClass *)iterator.value();
230  RexxString *name = (RexxString *)iterator.index();

232  {
233  method = method->newScope(scope);
234  }
235  replaceMethod(name, method);
236  }

355  * Reverse a SETMETHOD call on an object.
356  *
357  * @param name The name of the method to remove.
358  */
359 void MethodDictionary::removeInstanceMethod(RexxString *name)
360 {
361  // This is only possible if we've had prior calls...
362  if (instanceMethods != OREF_NULL)
363  {
364  // When we add a method to the behaviour dynamically, we add it
365  // to both the instance methods and the regular methods. The
366  // instance methods one allows us to track what we've added, the
367  // method dictionary version is just for the lookups.
368  if (instanceMethods->remove(name) != OREF_NULL)
369  {
370  remove(name);
371  }
372  }
373 }

376 /**
377  * Add an instance method to the dictionary.
378  *
379  * @param name The name of the method.
380  * @param method The target method object.
381  */
382 void MethodDictionary::addInstanceMethod(RexxString *name, MethodClass *method)
383 {
384  // this could be our first one (rather likely, actually)
385  if (instanceMethods == OREF_NULL)
386  {
387  setField(instanceMethods, new_string_table());
388  }
389  else
390  {
391  // if we've already added one of these, remove
392  // if from the main dictionary also
393  if (instanceMethods->hasIndex(name))
394  {
395  remove(name);
396  }
397  }
398 
399  // add this to the front of the search order in the main dictionary
400  addFront(method, name);
401  // and also add to the instance dictionary, replacing any
402  // existing method.
403  instanceMethods->put(method, name);
404 }

427 /**
428  * Locate a method from the method dictionary with a
429  * specific starting scope.
430  *
431  * @param name The name of the target method.
432  * @param startScope The starting scope value.
433  */
434 MethodClass *MethodDictionary::findSuperMethod(RexxString *name, RexxClass *startScope)
435 {
436  // get the list of scopes "visible" from this starting scope.
437  ArrayClass *scopes = (ArrayClass *)scopeOrders->get(startScope);
438  // shouldn't happen, but worth checking for
439  if (scopes == OREF_NULL)
440  {
441  return OREF_NULL;
442  }
443 
444  // do we have a list to search through? Now search the matching methods
445  // for one with a scope in this list.
446  // use an interator that does not create a new object array.
447  HashContents::IndexIterator iterator = contents->iterator(name);
448 
449  for (; iterator.isAvailable(); iterator.next())
450  {
451  MethodClass *method = (MethodClass *)iterator.value();
452  // we might have .nil in here as well as method objects.
453  if (method != (MethodClass *)TheNilObject)
454  {
455  RexxClass *methodScope = method->getScope();
456 
457  // if this method has a scope that's in the allowed list, return it.
458  // NOTE: the scopes list does not include the scope itself, so we test
459  // separately. The starting scope is frequently the target anyway, so
460  // this is generally more efficient
461  if (methodScope == startScope || scopes->hasIdentityItem(methodScope))
462  {
463  return method;
464  }
465  }
466  }
467  // nothing found
468  return OREF_NULL;
469 }

499 /**
500  * Get all methods with the target scope, or all methods
501  * if the scope is omitted.
502  *
503  * @param scope The target scope (can be NULL)
504  *
505  * @return A supplier to iterate over the requested methods.
506  */
507 SupplierClass *MethodDictionary::getMethods(RexxClass *scope)
508 {
509  // manually count first, then traverse again to fill the supplier arrays
510  size_t count = 0;
511 
512  // use an iterator to traverse the table
513  HashContents::TableIterator iterator = contents->iterator();
514 
515  for (; iterator.isAvailable(); iterator.next())
516  {
517  MethodClass *method = (MethodClass *)iterator.value();
518  // we're only interested in the real method objects.
519  if (isMethod(method))
520  {
521  // only count if this is a scope match
522  if (scope == OREF_NULL || method->isScope(scope))
523  {
524  count++;
525  }
526  }
527  }
528 
529  ArrayClass *names = new_array(count);
530  ArrayClass *methods = new_array(count);
531 
532  count = 1;
533 
534 
535  // traverse again, filling in the methods
536  iterator = contents->iterator();
537 
538  for (; iterator.isAvailable(); iterator.next())
539  {
540  MethodClass *method = (MethodClass *)iterator.value();
541  // we're only interested in the real method objects.
542  if (isMethod(method))
543  {
544  // only count if this is a scope match
545  if (scope == OREF_NULL || method->isScope(scope))
546  {
547  names->put(iterator.index(), count);
548  methods->put(method, count);
549  count++;
550  }
551  }
552  }
553 
554  // and return as a supplier
555  return (SupplierClass *)new_supplier(methods, names);
556 }

559 /**
560  * Retrieve the immediate super scope item from the
561  * behaviour class list. This determines what value is
562  * used for the super variable on method calls.
563  *
564  * @param scope The scope we're resolving.
565  *
566  * @return The immediate superclass value (return .nil for the Object
567  * class).
568  */
569 RexxClass *MethodDictionary::resolveSuperScope(RexxClass *scope)
570 {
571  ArrayClass *scopes = (ArrayClass *)scopeOrders->get(scope);
572  // should not happen, give a belt-and-braces return
573  if (scopes == OREF_NULL)
574  {
575  return (RexxClass *)TheNilObject;
576  }
577 
578  // Object will have an empty array here...return .nil for that
579  if (scopes->isEmpty())
580  {
581  return (RexxClass *)TheNilObject;
582  }
583 
584  // return the last item of the array, which is the immediate superscope.
585  return (RexxClass *)scopes->getLastItem();

589 /**
590  * Add a scope to the scope table. This is done unconditionall
591  *
592  * @param scope The new scope to add.
593  */
594 void MethodDictionary::addScope(RexxClass *scope)
595 {
596  // only add a scope to the list if this is not already here.
597  // this merges things together in the right order.
598  if (!scopeOrders->hasIndex(scope))
599  {
600  // we take a snapshot of the current scope. This is used both to
601  // determine the superscope and filter scoped method lookups. Note that
602  // the added scope class is NOT part of the lookup
603  ArrayClass *orders = (ArrayClass *)scopeList->copy();
604  scopeOrders->put(orders, scope);
605 
606  // append this to the scope list for this behaviour. This is
607  // our total class lookup scope and will be attached to the class
608  // object once the behaviour is fully constructed.
609  scopeList->append(scope);
610  }
611 }

620 void MethodDictionary::mergeMethods(MethodDictionary *target)
