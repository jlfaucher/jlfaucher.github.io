153 /**
154  * Runtime execution of a static Call instruction.
155  *
156  * @param context The current program execution context.
157  * @param stack The current context expression stack.
158  */
159 void RexxInstructionCall::execute(RexxActivation *context, ExpressionStack *stack)
160 {
161  // perform a stack space check here.
162  ActivityManager::currentActivity->checkStackSpace();
163  context->traceInstruction(this);
164 
165  // evaluate the arguments first
166  RexxInstruction::evaluateArguments(context, stack, arguments, argumentCount);
167 
168  ProtectedObject result; // returned result
169 
170  // do we have a resolved external routine to call?
171  if (externalTarget != OREF_NULL)
172  {
173  context->externalCall(targetName, externalTarget, stack->arguments(argumentCount), argumentCount, GlobalNames::SUBROUTINE, result);
174  }
175 
176  // if this has not resolved to an internal call, this is set to NULL
177  else if (targetInstruction != OREF_NULL)
178  {
179  context->internalCall(targetName, targetInstruction, stack->arguments(argumentCount), argumentCount, result);
180  }
181  // if this was resolved to a builtin, call directly
182  else if (builtinIndex != NO_BUILTIN)
183  {
184  result = (*(LanguageParser::builtinTable[builtinIndex]))(context, argumentCount, stack);
185 
186  }
187  // an external call...this is handled elsewhere.
188  else
189  {
190  // this is a potentially resolved external target
191  RoutineClass *resolvedTarget = OREF_NULL;
192 
193  context->externalCall(resolvedTarget, targetName, stack->arguments(argumentCount), argumentCount, GlobalNames::SUBROUTINE, result);
194 
195  // this potentially resolved a target that will allow us
196  // to fast path the next call
197  setField(externalTarget, resolvedTarget);
198  }
199 
200  // did we get a result returned? We need to either set or drop
201  // the result variable and potentially trace this.
202  if (!result.isNull())
203  {
204  context->setLocalVariable(GlobalNames::RESULT, VARIABLE_RESULT, result);
205  context->traceResult(result);
206  }
207  else
208  {
209  context->dropLocalVariable(GlobalNames::RESULT, VARIABLE_RESULT);
210  }
211  // and finally the debut pause.
212  context->pauseInstruction();
213 }

280 /**
281  * Execute a call to a dynamic call target.
282  *
283  * @param context The current program execution context.
284  * @param stack The current context evaluation stack.
285  */
286 void RexxInstructionDynamicCall::execute(RexxActivation *context, ExpressionStack *stack)
287 {
288  // perform a stack space check here.
289  ActivityManager::currentActivity->checkStackSpace();
290  context->traceInstruction(this);
291 
292  // NB: This leaves this on the stack...that's fine, because
293  // it protects the expression
294  RexxObject *evaluatedTarget = dynamicName->evaluate(context, stack);
295  // this needs to be in string form, and protected
296  Protected<RexxString> targetName = evaluatedTarget->requestString();
297  context->traceResult(targetName);
298 
299  // evaluate the arguments first
300  RexxInstruction::evaluateArguments(context, stack, arguments, argumentCount);
301 
302  // see if we can find an internal label target
303  RexxInstruction *targetInstruction = OREF_NULL;
304  // see if the context has a matching label (case sensitive lookup)
305  StringTable *labels = context->getLabels();
306  if (labels != OREF_NULL)
307  {
308  targetInstruction = (RexxInstruction *)(labels->get(targetName));
309  }
310 
311  ProtectedObject result; // returned result
312 
313  // if this has not resolved to an internal call, this is still NULL
314  if (targetInstruction != OREF_NULL)
315  {
316  context->internalCall(targetName, targetInstruction, stack->arguments(argumentCount), argumentCount, result);
317  }
318  // builtin checks come next.
319  else
320  {
321  // map the name to a builtin index code. If we get a hit, call this now.
322  BuiltinCode builtinIndex = RexxToken::resolveBuiltin(targetName);
323  if (builtinIndex != NO_BUILTIN)
324  {
325  result = (*(LanguageParser::builtinTable[builtinIndex]))(context, argumentCount, stack);
326  }
327  // an external call...this is handled elsewhere.
328  else
329  {
330  // we need to provide the variable, but we don't cache the result in this case
331  RoutineClass *resolvedRoutine = OREF_NULL;
332 
333  context->externalCall(resolvedRoutine, targetName, stack->arguments(argumentCount), argumentCount, GlobalNames::SUBROUTINE, result);
334  }
335  }
336 
337  // did we get a result returned? We need to either set or drop
338  // the result variable and potentially trace this.
339  if (!result.isNull())
340  {
341  context->setLocalVariable(GlobalNames::RESULT, VARIABLE_RESULT, result);
342  context->traceResult(result);
343  }
344  else
345  {
346  context->dropLocalVariable(GlobalNames::RESULT, VARIABLE_RESULT);
347  }
348  // and finally the debug pause.
349  context->pauseInstruction();
350 }

424  * Execute a call to a dynamic call target.
425  *
426  * @param context The current program execution context.
427  * @param stack The current context evaluation stack.
428  */
429 void RexxInstructionQualifiedCall::execute(RexxActivation *context, ExpressionStack *stack)
430 {
431  // perform a stack space check here.
432  ActivityManager::currentActivity->checkStackSpace();
433  context->traceInstruction(this);
434 
435  // evaluate the arguments first
436  RexxInstruction::evaluateArguments(context, stack, arguments, argumentCount);
437 
438  ProtectedObject result; // returned result
439 
440  // get the current package from the context
441  PackageClass *package = context->getPackage();
442 
443  // we must be able to find the named namespace
444  PackageClass *namespacePackage = package->findNamespace(namespaceName);
445  if (namespacePackage == OREF_NULL)
446  {
447  reportException(Error_Execution_no_namespace, namespaceName, package->getProgramName());
448  }
449 
450  // we only look for public routines in the namespaces
451  RoutineClass *resolvedRoutine = namespacePackage->findPublicRoutine(routineName);
452  // we give a specific error for this one

454  {
455  reportException(Error_Routine_not_found_namespace, routineName, namespaceName);
456  }
457 
458  // call the resolved function
459  resolvedRoutine->call(context->getActivity(), routineName, stack->arguments(argumentCount), argumentCount, GlobalNames::SUBROUTINE, OREF_NULL, EXTERNALCALL, result);
460 
461 
462  // did we get a result returned? We need to either set or drop
463  // the result variable and potentially trace this.
464  if (!result.isNull())
465  {
466  context->setLocalVariable(GlobalNames::RESULT, VARIABLE_RESULT, result);

562 /**
563  * Execute a CALL ON/OFF instruction. This either
564  * activates or deactivates the call trap. Calling of
565  * a target only happens when a trap is activated.
566  *
567  * @param context The current program context.
568  * @param stack The current context evaluation stack.
569  */
570 void RexxInstructionCallOn::execute(RexxActivation *context, ExpressionStack *stack)
571 {
572  // do trace stuff.
573  context->traceInstruction(this);
574 
575  // if we do not have a target name set, this is a CALL OFF instruction. Just
576  // disable the trap.
577  if (targetName != OREF_NULL)
578  {
579  // wax on...
580  context->trapOn(conditionName, this, false);
581  }
582  else
583  {
584  // wax off...
585  context->trapOff(conditionName, false);
586  }

590 /**
591  * Process a trapped condition.
592  *
593  * @param context The trapping context.
594  * @param conditionObj
595  * The condition object associated with the condition.
596  */
597 void RexxInstructionCallOn::trap(RexxActivation *context, DirectoryClass *conditionObj)
598 {
599  ProtectedObject result;
600 
601  // Call ONs do not completely disable a trap, but they do put it into a
602  // delayed state while the trap is executing.
603  context->trapDelay(conditionName);
604 
605  // have a resolved target label?
606  if (targetInstruction != OREF_NULL)
607  {
608  // this is handled by the activation context
609  context->internalCallTrap(targetName, targetInstruction, conditionObj, result);
610  }
611  // match for a builtin function? A little strange, but allowed. This will most
612  // likely give an error, since we call with no arguments
613  else if (builtinIndex != NO_BUILTIN)
614  {
615  // arguments to builtins are passed on the stack
616  context->getStack()->push(conditionObj);
617 
618  (*(LanguageParser::builtinTable[builtinIndex]))(context, 1, context->getStack());
619  }
620  // this is an external call.
621  else
622  {
623  // we need to provide the variable, but we don't cache the result in this case
624  RoutineClass *resolvedRoutine = OREF_NULL;
625 
626  context->externalCall(resolvedRoutine, targetName, (RexxObject **)&conditionObj, 1, GlobalNames::SUBROUTINE, result);
627  }
628 
629  // NOTE: Any result object is ignored for a CALL ON trap
630 
631  // finally, restore the trap condition
632  context->trapUndelay(conditionName);
633 }
