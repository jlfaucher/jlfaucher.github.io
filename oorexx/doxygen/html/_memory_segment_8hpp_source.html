343  inline RexxInternalObject *allocateObject(size_t allocationLength)
344  {
345  DeadObject *newObject;
346  size_t targetPool;
347  size_t realLength;
348 
349  /* Calculate the dead chain. Note that if this is larger than */
350  /* the largest subpool, the for() loop test below will fail, */
351  /* causing this to drop down to the large block allocation */
352  /* logic. This eliminates an additional check against the */
353  /* large size. */
354  targetPool = lengthToDeadPool(allocationLength);
355 
356  if (targetPool < DeadPools)
357  {
358 
359  /* pick up the last successful one */
360  size_t currentDead = lastUsedSubpool[targetPool];
361  /* loop through the small pool chains looking for a block. */
362  /* We only go up to the largest blocks as a last resort to */
363  /* reduce the fragmentation. */
364  while (currentDead < DeadPools)
365  {
366  /* See if the chain has an object. Once we get an */
367  /* object, we return this directly. We accept over */
368  /* allocations when then come from the subpool chain. */
369  /* Since this is such a heavily hit path, we don't want */
370  /* to absorb the overhead of attempting to split the */
371  /* blocks. For the majority of over allocations, we */
372  /* can't split anyway. When we do split, the result is */
373  /* a very small fragment. */
374  newObject = subpools[currentDead].getFirstSingle();
375  if (newObject != OREF_NULL)
376  {
377  /* Record the success. Next time around, */
378  /* allocations will come directly here. */
379  lastUsedSubpool[targetPool] = currentDead;
380  /* we have a block. Now see if we got this from a */
381  /* higher level chain and have enough room to */
382  /* subdivide into a small block. */
383  /* Convert this from a dead object into a real one of the */
384  /* given size. */
385  return (RexxInternalObject *)newObject;
386  }
387 
388  currentDead++;
389 
390  while (currentDead < DeadPools)
391  {
392  if (lastUsedSubpool[currentDead] < DeadPools)
393  {
394  // this pool might be redirected already, so
395  // pick up the index of where it's redirected to.
396  currentDead = lastUsedSubpool[currentDead];
397  lastUsedSubpool[targetPool] = currentDead;
398  break;
399  }
400  currentDead++;
401  }
402  }
403  /* we've gone all the way to the end without finding */
404  /* anything. Cause the next allocation to skip directly to */
405  /* the large chain. */
406  lastUsedSubpool[targetPool] = DeadPools;
407  }
408  /* Nope, go through the LARGEDEAD object looking for the 1st */
409  /* one we can use either our object is too big for all the */
410  /* small chains, or the small chains are depleted.... */
411  /* Go through the LARGEDEAD object looking for the 1st */
412  /* one we can use either our object is too big for all the */
413  /* small chains, or the small chains are depleted.... */
414  newObject = largeDead.findFit(allocationLength, &realLength);
415  if (newObject != NULL)
416  { /* did we find an object? */
417  size_t deadLength = realLength - allocationLength;
418  /* remainder too small or this is a very large request */
419  /* is the remainder two small to reuse? */
420  if (deadLength < Memory::MinimumObjectSize)
421  {
422  /* Convert this from a dead object into a real one of the */
423  /* given size. */
424  return (RexxInternalObject *)newObject;
425  }
426  else
427  {
428  /* potentially split this object into a smaller unit so we */
429  /* can reuse the remainder. */
430  return splitNormalDeadObject(newObject, allocationLength, deadLength);
431  }
432  }
433  return OREF_NULL;
434  }

481  {
482  /* we need to keep all of these sizes as ObjectGrain multiples, */
483  /* so round it down...the allocation will get all of the extra. */
484  /* deadLength = rounddown(deadLength, ObjectGrain);
485  allocationLength is rounded, deadLength might be
486  an ungrained object size from old tokenized format */
487 
488  /* Yes, so pull new object out of the front of the dead */
489  /* object, adjust the size of the Dead object. We want */
490  /* to use the front rather than the back so that if we */
491  /* hit the need to split a segment because of low memory */
492  /* conditions, we increase the probability that we'll be */
493  /* able to use the end of the segment. */
494  DeadObject *largeObject = (DeadObject *)(((char *)object) + allocationLength);
495  /* if the length is larger than the biggest subpool we */
496  /* maintain, we add this to the large block list. */
497  if (deadLength > LargestSubpool)
498  {
499  /* ideally, we'd like to add this sorted by size, but */
500  /* this is called so frequently, attempting to sort */
501  /* degrades performance by about 10%. */
502  largeDead.add(new (largeObject) DeadObject(deadLength));
503  }
504  else
505  {
506  /* calculate the dead chain */
507  /* and add that to the appropriate chain */
508  size_t deadChain = lengthToDeadPool(deadLength);
509  subpools[deadChain].addSingle(new (largeObject) DeadObject(deadLength));
510  /* we can mark this subpool as having items again */
511  lastUsedSubpool[deadChain] = deadChain;
512  }
513  /* Convert this from a dead object into a real one of the */
514  /* given size. */
515  ((RexxInternalObject *)object)->setObjectSize(allocationLength);
516  return (RexxInternalObject *)object;
517  }

521  size_t lastUsedSubpool[DeadPools + 1];/* a look-aside index to tell us what pool to use for a given size */
