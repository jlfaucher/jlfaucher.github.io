121  * Construct a HashContent item of the given size.
122  *
123  * @param entries The total number of entries in the object.
124  */
125 HashContents::HashContents(size_t entries, size_t total)
126 {
127  // clear the entire object for safety
128  clearObject();
129 
130  // this is the total number of slots in the table. The
131  // optimal bucket size should already have been calculated
132  bucketSize = entries;
133  // this is the total size of the bucket
134  totalSize = total;
135 
136  // initialize the free chains
137  initializeFreeChain();
138 }

141 /**
142  * Initialize the free chains, either at construction time or
143  * after an empty() operation.
144  */
145 void HashContents::initializeFreeChain()
146 {
147  // initialize the next link in all bucket slots to NoMore.
148  for (ItemLink i = 0; i < bucketSize; i++)
149  {
150  entries[i].next = NoMore;
151  }
152 
153  // this is an empty bucket
154  itemCount = 0;
155 
156  // we keep the available items on a chain, so
157  // chain up the items into a free chain
158  freeChain = bucketSize;
159 
160  for (ItemLink i = freeChain; i < totalSize; i++)
161  {
162  entries[i].next = i + 1;
163  }
164  // make sure the last item ends the chain
165  entries[totalSize - 1].next = NoMore;
166 }

216 /**
217  * Put an item into the hashtable. Default behavior is
218  * to replace any existing items.
219  *
220  * @param value The value to insert.
221  * @param _index The index to insert under.
222  *
223  * @return true if was inserted ok, false if we need to expand the
224  * table.
225  */
226 void HashContents::put(RexxInternalObject *value, RexxInternalObject *index)
227 {
228  // NOTE: This depends on the caller to make sure there is
229  // room in the table for this item!
230 
231  // calculate the bucket position
232  ItemLink position = hashIndex(index);
233 
234  // if the hash slot is empty, we can just fill this in right here
235  if (isAvailable(position))
236  {
237  setEntry(position, value, index);
238  // new item, so bump the count
239  itemCount++;
240  return;
241  }
242 
243  ItemLink previous = NoLink;
244 
245  // this slot is in use. We either A) have this entry already, in
246  // which case we replace the value, or B) need to add another entry to the end
247  do
248  {
249  // if we got a hit, just change the slot position
250  if (isIndex(position, index))
251  {
252  // everything worked, we have room. No additional items added to
253  // this table.
254  setValue(position, value);
255  return;
256  }
257 
258  // continue running the chain to the end of the list.
259  previous = position;
260  position = entries[position].next;
261  } while (position != NoMore);
262 
263  // append a new value.
264  append(value, index, previous);
265 }

268 /**
269  * Add a link to a slot position chain.
270  *
271  * @param value The value to add.
272  * @param index The index value.
273  * @param position The position of the last link in the chain.
274  */
275 void HashContents::append(RexxInternalObject *value, RexxInternalObject * index, ItemLink position)
276 {
277  // we keep all of the free items in a chain, so we can just pull
278  // the first entry off of the chain. In theory, we've already checked
279  // that we're not full, so we should find one there.
280 
281  ItemLink newEntry = freeChain;
282 
283  // belt-and-braces...this should not occur...but give a logic
284  // error if it does occur, since that indicates something bad has
285  // occurred.
286  if (newEntry == NoMore)
287  {
288  Interpreter::logicError("Attempt to add an object to a full Hash table");
289  }
290 
291  // close up the chain
292  freeChain = entries[newEntry].next;
293 
294  // set the entry
295  setEntry(newEntry, value, index);
296  // add this to the end of the chain
297  setNext(position, newEntry);
298  setNext(newEntry, NoMore);
299 
300  // we have a new item in the list.
301  itemCount++;
302 }

350 RexxInternalObject *HashContents::remove(RexxInternalObject *index)
351 {
352  ItemLink position;
353  ItemLink previous;
354 
355  // go find the matching entry. We have nothing to do if not there.
356  if (!locateEntry(index, position, previous))
357  {
358  return OREF_NULL;
359  }
360 
361  // save the original value
362  RexxInternalObject *removed = entryValue(position);
363  // remove the chain link and return the value
364  removeChainLink(position, previous);
365  return removed;

369 /**
370  * Remove a link from the chain, taking into consideration
371  * where the link occurs. This will add a new link to
372  * the free chain.
373  *
374  * @param position The position we're removing.
375  * @param previous The previous chain entry (NoLink value if this is the head of the chain).
376  */
377 void HashContents::removeChainLink(ItemLink &position, ItemLink previous)
378 {
379  // reduce the item count for this removal.
380  itemCount--;
381 
382  // are we removing the first item in the chain? We can't move this
383  // to the free list because this is the anchor for the hash postion.
384  // we need to either promote the next item to the front of the chain
385  // or clear this item out if this is also the end of the chain
386  if (previous == NoLink)
387  {
388  ItemLink next = nextEntry(position);
389  // just the single item on the chain? reset to empty state and we're done.
390  if (next == NoMore)
391  {
392  clearEntry(position);
393  return;
394  }
395  // copy the next item to the head of the list
396  copyEntry(position, next);
397  // return the next entry to the free chain
398  returnToFreeChain(next);
399  }
400  else
401  {
402  // close up the chain, and and also update the
403  // previous position if we still need to run the chain
404  closeChain(position, previous);
405  position = nextEntry(previous);
406  }
407 }

410 /**
411  * Locate an entry in the table.
412  *
413  * @param index The target entry index.
414  * @param position The return position if located.
415  * @param previous The previous link if located. Has the value NoLink if
416  * the located item is the chain anchor.
417  *
418  * @return true if the item is located, false for a failure
419  */
420 bool HashContents::locateEntry(RexxInternalObject *index, ItemLink &position, ItemLink &previous)
421 {
422  // get the bucket position to start searching
423  position = hashIndex(index);
424  previous = NoLink;
425 
426  // ok, run the chain searching for an index match.
427  while (position != NoMore && isInUse(position))
428  {
429  // have a match? return to the caller. All of the position
430  // stuff should be set now
431  if (isIndex(position, index))
432  {
433  return true;
434  }
435  // remember the previous position (needed for chain updates)
436  previous = position;
437  // step to the next link in the chain
438  position = nextEntry(position);
439  }
440 
441  // hit the end of the chain, we don't have this item
442  position = NoMore;
443  return false;
444 }

454 void HashContents::nextMatch(RexxInternalObject *index, ItemLink &position)

457  if (position == NoMore)

463  position = nextEntry(position);
464 
465  // keep looping until the end of this chain
466  while (position != NoMore)

468  // if this is a match, we're done
469  if (isIndex(position, index))

471  return;
472  }
473  // step to the next position
474  position = nextEntry(position);
475  }
476 
477  // hit the end of the chain, we don't have this item
478  // (position also marks this as the end)
479  return;

483 /**
484  * Iterate to the next occupied entry of the list.
485  *
486  * @param position The starting position.
487  * @param nextBucket The next bucket to step to once this chain is used up.
488  */
489 void HashContents::iterateNext(ItemLink &position, ItemLink &nextBucket)
490 {
491  // have a good current position...step to the next chain item.
492  if (position != NoMore)
493  {
494  // if we still have a chain, then we can just return this position as the next one
495  position = nextEntry(position);
496  if (position != NoMore)
497  {
498  return;
499  }
500  }
501 
502  // look for another bucket with an active chain.
503  while (nextBucket < bucketSize)
504  {
505  position = nextBucket++;
506  // if this bucket position is active, we've found our match
507  if (isInUse(position))
508  {
509  return;
510  }
511  }
512  // make sure this is marked as a failure
513  position = NoMore;
514 }

517 /**
518  * Safely remove an item from the table and advance the
519  * iterator position to the next location.
520  *
521  * @param position The current position.
522  * @param nextBucket The nextBucket we advance to.
523  */
524 void HashContents::iterateNextAndRemove(ItemLink &position, ItemLink &nextBucket)
525 {
526  // a remove operation is special if the we're at the head of the chain,
527  // since a remove operation will move another item. into the slot, if
528  // there is one.
529  if (isBucketPosition(position))
530  {
531  // if this is the only item in this bucket, we can just
532  // advance normally, then remove the item.
533  if (nextEntry(position) == NoMore)
534  {
535  // save this for the deletion
536  ItemLink current = position;
537  iterateNext(position, nextBucket);
538  // perform the removal now that we've moved
539  removeChainLink(current, NoLink);
540  }
541  // we're at the head of the chain and we need to remove this
542  // item. This will move the next item into the bucket slot, which
543  // means if we just remove the item, our position information will already
544  // be correct.
545  else
546  {
547  removeChainLink(position, NoLink);
548  }
549  }
550  // we're past the bucket position in the chain. To delete this,
551  // we need to know the previous item. However, if we advance to the
552  // next position first, that information will still be correct after deletion.
553  else
554  {
555  // save the bucket chain anchor and current position for the removal.
556  ItemLink current = position;
557  ItemLink currentBucket = nextBucket - 1;
558  // we need this also
559  ItemLink previous = position;
560  // advance to the next position.
561  iterateNext(position, nextBucket);
562  // now find the previous entry so we can remove the item
563  locatePreviousEntry(previous, currentBucket);
564  // do the removal.
565  removeChainLink(current, previous);
566  }
567 }

571  * Iterate to the next occupied entry of the list in reverse
572  * bucket order.
573  *
574  * @param position The starting position.
575  * @param currentBucket The current bucket we're iterating on.
576  *
577  */
578 void HashContents::iterateNextReverse(ItemLink &position, ItemLink &currentBucket)
579 {
580  // The position will always be valid because we're backing up the chain to the
581  // bucket anchor. If we're at the bucket anchor, we then locate the last position
582  // in the next occupied bucket
583  if (position == currentBucket)
584  {
585  // step to the next bucket and see if we have a next entry
586  currentBucket++;
587  locateNextBucketEnd(position, currentBucket);
588  }
589  // we're traversing
590  else
591  {
592  // we don't maintain back links, so we need to scan. Not
593  // terribly efficient, but the buckets don't tend to get very long
594  // and we don't do this all that often.
595  locatePreviousEntry(position, currentBucket);
596  }

600 /**
601  * Locate the end of the next occupied hash bucket in this
602  * collection.
603  *
604  * @param position The returned position (NoMore indicates we've hit the end)
605  * @param currentBucket
606  * The starting bucket to search. Updated to the new bucket
607  * if something is found.
608  */
609 void HashContents::locateNextBucketEnd(ItemLink &position, ItemLink &currentBucket)
610 {
611  // search each of the buckets in turn looking for an occupied one.
612  while (currentBucket < bucketSize)
613  {
614  // if we have an entry in the first position, this is the one we want
615  if (isInUse(currentBucket))
616  {
617  // set the pointer at the beginning
618  position = currentBucket;
619  // locate the last entry in the bucket
620  while(nextEntry(position) != NoMore)
621  {
622  position = nextEntry(position);
623  }
624  return;
625  }
626  // try the next bucket
627  currentBucket++;
628  }
629 
630  // indicate we've hit the iteration end
631  position = NoMore;
632 }

701 {

710  // clear out the previous value for each chain start

712  previous = NoLink;
713 
714  // ok, run the chain searching for an index match.
715  while (position != NoMore && isInUse(position))
716  {
717  // have a match? return to the caller. All of the position
718  // stuff should be set now
719  if (isItem(position, item))
720  {
721  return true;
722  }
723  // remember the previous position (needed for chain updates)
724  previous = position;
725  // step to the next link in the chain
726  position = nextEntry(position);
727  }
728  }
729 
730  // hit the end of the bucket list, we don't have this item
731  return false;
732 }

741  * @return An array of all matching items.

744 {

747

750  // get a result array
751  ArrayClass *result = new_array(count);
752

754  for (size_t i = 1; i <= count; i++)

765  removeChainLink(position, previous);
766  }
767 
768  // return our result array
769  return result;

1241  // values with identical indexes in the same relative order
1242  // after the merge (very important when merging MethodDirectories,
