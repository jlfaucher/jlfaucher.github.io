65 void signalHandler(int sig)
66 {
67 
68 #ifdef ORXAP_DEBUG
69  switch (sig)
70  {
71  case (SIGINT):
72  printf("\n*** Rexx interrupted.\n");
73  break;
74  case (SIGTERM):
75  printf("\n*** Rexx terminated.\n*** Closing Rexx !\n"); /* exit(0); */
76  break;
77  case (SIGSEGV):
78  printf("\n*** Segmentation fault.\n*** Closing Rexx !\n");
79  break;
80  case (SIGFPE):
81  printf("\n*** Floating point error.\n*** Closing Rexx\n");
82  break;
83  case (SIGBUS):
84  printf("\n*** Bus error.\n*** Closing Rexx\n");
85  break;
86  case (SIGPIPE):
87  printf("\n*** Broken pipe.\n*** Closing Rexx\n");
88  break;
89  default:
90  printf("\n*** Error,closing REXX !\n");
91  break;
92  }
93 #endif
94 
95  // if the signal is a ctrl-C, we perform a halt operation
96  if (sig == SIGINT)
97  {
98  Interpreter::haltAllActivities(GlobalNames::SIGINT_STRING);
99  return;
100  }
101  else if (sig == SIGTERM)
102  {
103  Interpreter::haltAllActivities(GlobalNames::SIGTERM_STRING);
104  return;
105  }
106  else if (sig == SIGHUP)
107  {
108  Interpreter::haltAllActivities(GlobalNames::SIGHUP_STRING);
109  return;
110  }
111  else
112  {
113  exit(0);
114  }
115 }

118 void SystemInterpreter::startInterpreter()
119 {
120 
121  /* Set the cleanup handler for unconditional process termination */
122  struct sigaction new_action;
123  struct sigaction old_action;
124 
125  /* Set up the structure to specify the new action */
126  new_action.sa_handler = signalHandler;
127  old_action.sa_handler = NULL;
128  sigfillset(&new_action.sa_mask);
129 // new_action.sa_flags = SA_RESTART;
130  new_action.sa_flags = 0; // do not use SA_RESTART or ctrl-c will not work as expected!
131 
132 /* Termination signals are set by Object REXX whenever the signals were not set */
133 /* from outside (calling C-routine). The SIGSEGV signal is not set any more, so */
134 /* that we now get a coredump instead of a hang up */
135 
136  sigaction(SIGINT, NULL, &old_action);
137  sigaction(SIGTERM, NULL, &old_action);
138  sigaction(SIGHUP, NULL, &old_action);
139  if (old_action.sa_handler == NULL) /* not set by ext. exit handler*/
140  {
141  sigaction(SIGINT, &new_action, NULL); /* exitClear on SIGINT signal */
142  sigaction(SIGTERM, &new_action, NULL); /* exitClear on SIGTERM signal */
143  sigaction(SIGHUP, &new_action, NULL); /* exitClear on SIGHUP signal */
144  }
145 
146  // Set SIGPIPE to ignore so that ADDRESS WITH pipes or rxsock sockets
147  // may return EPIPE error codes instead of the interpreter being killed
148  signal(SIGPIPE, SIG_IGN);
149 }
