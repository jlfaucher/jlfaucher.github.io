91 class RexxInstructionBaseLoop : public RexxBaseBlockInstruction
92 {
93  public:
94  inline RexxInstructionBaseLoop() {;}
95 
96  void live(size_t)override;
97  void liveGeneral(MarkReason reason)override;
98  void flatten(Envelope *)override;
99 
100  // required by RexxInstruction. For most subclasses, the default
101  // is sufficient.
102  void execute(RexxActivation *, ExpressionStack *)override;
103 
104  // most DO blocks are loops. The simple styles will need to override.
105  EndBlockType getEndStyle()override { return LOOP_BLOCK; }
106  // Most DOs are loops...Simple DO will override again.
107  bool isLoop()override { return true; };
108  RexxVariableBase* getCountVariable()override { return countVariable; }
109 
110  // specific to Do loops. Most subclasses can rely on the default
111  virtual void reExecute(RexxActivation *, ExpressionStack *, DoBlock *);
112 
113  // most loops will want to override these two
114  virtual void setup(RexxActivation *context, ExpressionStack *stack, DoBlock *doblock);
115  virtual bool iterate(RexxActivation *context, ExpressionStack *stack, DoBlock *doblock, bool first);
116 
117  void endLoop(RexxActivation *context);
118  inline void setCountVariable(RexxVariableBase *v) { countVariable = v; }
119 
120  protected:
121  RexxVariableBase *countVariable; // optional variable for a counter
122 };
