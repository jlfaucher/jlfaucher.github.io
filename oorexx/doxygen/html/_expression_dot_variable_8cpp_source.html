87 /**
88  * Perform generalized live marking on an object. This is
89  * used when mark-and-sweep processing is needed for purposes
90  * other than garbage collection.
91  *
92  * @param reason The reason for the marking call.
93  */
94 void RexxDotVariable::liveGeneral(MarkReason reason)
95 {
96  // if this is an alert that the image is getting saved,
97  // we try to resolve any dot variables we can so that
98  // we don't have to add entries to the old-to-new
99  // table during normal operations.
100  if (reason == PREPARINGIMAGE)
101  {
102  RexxObject *t = OREF_NULL; // required for the findClass call
103 
104  // any code in the image is part of the Rexx package, so we resolve
105  // from that.
106  RexxClass *rexxQueue = TheRexxPackage->findClass(variableName, t);
107  // set with anything that is returned, including OREF_NULL
108  setField(cachedValue, t);
109  }
110 
111  memory_mark_general(variableName);
112  memory_mark_general(cachedValue);
113 }

132 /**
133  * Evaluate a dot variable symbol.
134  *
135  * @param context The current evaluation context.
136  * @param stack The current evaluation stack.
137  *
138  * @return The dot symbol value.
139  */
140 RexxObject * RexxDotVariable::evaluate(RexxActivation *context, ExpressionStack *stack )
141 {
142  // if we previously resolved this from a package source, then
143  // we saved the value for quick access.
144  if (cachedValue != OREF_NULL)
145  {
146  // evaluate always pushes on the stack.
147  stack->push(cachedValue);
148  // trace this if tracing intermediates
149  context->traceDotVariable(variableName, cachedValue);
150  return cachedValue;
151  }
152 
153  // if this gets set by resolveDotVariable, the value
154  // can be cached for quick lookup
155  RexxObject *value = OREF_NULL;
156 
157  // try first from the environment
158  RexxObject *result = context->resolveDotVariable(variableName, value);
159  if (result == OREF_NULL)
160  {
161  // might be a special rexx name
162  result = context->rexxVariable(variableName);
163 
164  if (result == OREF_NULL)
165  {
166  // add a period to the name
167  result = variableName->concatToCstring(".");
168  }
169  }
170  else
171  {
172  // the value returned by resolveDotVariable might
173  // be capable to caching, so set this for the next call
174  // (this might still be null if it came from a dynamic
175  // source)
176  setField(cachedValue, value);
177  }
178 
179  // evaluate always pushes on the stack.
180  stack->push(result);
181  // trace this if tracing intermediates
182  context->traceDotVariable(variableName, result);
183  return result;
184 }
