51 /**
52  * The main entry point for rxapi.exe.
53  *
54  * When rxapi is installed as a service, the invoker might be the service
55  * control manager, or the interpreter could be starting rxapi as a service
56  * because rxapi was not running.
57  *
58  * When rxapi is not installed as a service, then the invoker is most likely the
59  * interpreter starting up rxapi. Although, the user could also be invoking
60  * rxapi from the command line.
61  *
62  * The third possibility is that rxapi is being invoked from the command line
63  * with arguments to either install, uninstall, or query rxapi as a service.
64  *
65  *
66  * @param hInstance
67  * @param hPrevInstance
68  * @param lpCmdLine The only arg we are interested in, the command line
69  * arguments.
70  * @param nCmdShow
71  *
72  * @return 0
73  */
74 int APIENTRY WinMain(HINSTANCE hInstance, HINSTANCE hPrevInstance, LPSTR lpCmdLine, int nCmdShow)
75 {
76  try
77  {
78  // create a connection object that will be the server target
79  SysServerNamedPipeConnectionManager *c = new SysServerNamedPipeConnectionManager();
80  // try to create the named pipe used for this server. If this fails, we
81  // likely have a instance of the daemon already running, so just fail quietly.
82  if (!c->bind())
83  {
84  delete c;
85  return ERROR_ACCESS_DENIED;
86  }
87  apiServer.initServer(c); // start up the server
88  apiServer.listenForConnections(); // go into the message loop
89  }
90  catch (ServiceException *e)
91  {
92  // we don't need this, so delete it now
93  delete e;
94  }
95  apiServer.terminateServer(); // shut everything down
96 
97  return 0;
98 }
