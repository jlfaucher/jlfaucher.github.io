151 /**
152  * wait for the semaphore to be posted.
153  *
154  * @param t An optional timeout in seconds, either String or TimeSpan.
155  * Returns immediately if the timeout is zero,
156  * waits indefinitely if it's negative.
157  * Default is -1.
158  *
159  * @return true if the semaphore was posted,
160  * false if there was a timeout.
161  */
162 RexxObject* EventSemaphoreClass::wait(RexxObject *t)
163 {
164  wholenumber_t timeout;
165 
166  if (t == OREF_NULL)
167  {
168  timeout = -1;
169  }
170  else
171  {
172  RexxObject *seconds;
173 
174  // we allow timeout to be a TimeSpan or a String object
175  RexxClass *TimeSpan = TheRexxPackage->findClass(GlobalNames::TIMESPAN);
176  if (t->isInstanceOf(TimeSpan))
177  {
178  ProtectedObject p;
179  seconds = t->sendMessage(GlobalNames::TOTALSECONDS, p);
180  }
181  else
182  {
183  seconds = t;
184  }
185  double secs = floatingPointArgument(seconds, "timeout");
186  // wait() is uint32, so we can wait for 4294967 seconds at most
187  if (secs >= 0 && secs <= 4294967)
188  {
189  // we need milliseconds (the product will always fit a unint32)
190  timeout = (wholenumber_t)(secs * 1000.0);
191  }
192  else
193  {
194  // negative or too large, we just wait forever
195  timeout = -1;
196  }
197  }
198 
199  if (timeout < 0)
200  {
201  UnsafeBlock releaser;
202 
203  semaphore.wait();
204  return TheTrueObject;
205  }
206  // for a zero wait, we just return the current posted status
207  else if (timeout == 0)
208  {
209  return booleanObject(semaphore.posted());
210  }
211  else
212  {
213  UnsafeBlock releaser;
214 
215  return booleanObject(semaphore.wait((uint32_t)timeout));
216  }
217 }
