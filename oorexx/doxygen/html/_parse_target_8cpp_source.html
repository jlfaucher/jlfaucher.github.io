76 /**
77  * Step to the "next" string to parse, resetting all of the
78  * cursor movement values.
79  *
80  * @param context The current execution context.
81  */
82 void RexxTarget::next(RexxActivation *context)
83 {
84  // if we have an argument list and we have an argument value in that
85  // position, grab it. Otherwise, our target string is just ""
86  if (arglist != OREF_NULL)
87  {
88  if (next_argument > argcount)
89  {
90  string = GlobalNames::NULLSTRING;
91  }
92  else
93  {
94  // get the next element
95  string = (RexxString *)arglist[next_argument - 1];
96  // omitted argument? use a null string
97  if (string == OREF_NULL)
98  {
99  string = GlobalNames::NULLSTRING;
100  }
101  }
102  }
103  else
104  {
105  // not PARSE ARG, so beyond the first template is always a null string
106  if (next_argument != 1)
107  {
108  string = GlobalNames::NULLSTRING;
109  }
110  }
111  next_argument++;
112  // make sure we have a string
113  string = string->requestString();
114  // now handle translation options
115  if (translate[parse_upper])
116  {
117  string = string->upper();
118  }
119  else if (translate[parse_lower])
120  {
121  string = string->lower();
122  }
123 
124  // reset the stack to the entry top, and push this value on to protect it.
125  stack->setTop(stackTop);
126  stack->push(string);
127 
128  // if tracing results or intermediates, show the string being parsed.
129  context->traceResult(string);
130 
131  // reset all of the parsing positions
132  start = 0;
133  pattern_end = 0;
134  pattern_start = 0;
135  // get the length for quicker access
136  string_length = string->getLength();
137  subcurrent = 0; // no sub piece to process yet
138 }

159 /**
160  * Move the parse pointer forward by a relative offset
161  * (unsigned numeric trigger)
162  *
163  * @param offset The offset to move.
164  */
165 void RexxTarget::forward(size_t offset)
166 {
167  // the start position is the last postion and the end
168  // position is the start position + the offset we're moving
169  start = pattern_start;
170  end = start + offset;
171 
172  // if we've gone past the end, trunk to the string length
173  if (end >= string_length)
174  {
175  end = string_length;
176  }
177 
178  // if there was no forward movement, the special Rexx rule kicks in
179  if (end <= start)
180  {
181  // we match to the end, and start from the current start next time
182  end = string_length;
183  pattern_start = start;
184  }
185  // normal movement, everthing moves from the end
186  else
187  {
188  pattern_start = end;
189  }
190 
191  // our pattern is zero length because it is not a string match.
192  pattern_end = pattern_start;
193  // the section we parse into words starts with the last match position.
194  subcurrent = start;
195 }

198 /**
199  * Move the parsing position forward by a relative offset
200  * (>numeric trigger), which uses different rules for end <=
201  * start.
202  *
203  * @param offset The offset amount to move.
204  */
205 void RexxTarget::forwardLength(size_t offset)
206 {
207  // the start position is our last start position (which will
208  // be the beginning of the last string match.
209  start = pattern_start;
210  end = start + offset;
211 
212  // perform movement past the string end tests.
213  if (end >= string_length)
214  {
215  end = string_length;
216  }
217 
218  // The match pattern is the end position and is a zero-length
219  // string.
220  pattern_start = end;
221  pattern_end = pattern_start;
222  // The piece we parse into words begins at the last match position.
223  subcurrent = start;
224 }

227 /**
228  * Move to an absolute string postion (unsigned numeric
229  * trigger).
230  *
231  * @param offset The target string position.
232  */
233 void RexxTarget::absolute(size_t offset)
234 {
235  // make the offset origin zero (but be careful of a 0 offset already)
236  if (offset > 0)
237  {
238  offset--;
239  }
240 
241  // our parsing start position is the end of the previous match
242  start = pattern_end;
243  // if we are moving backward, we match to the end of the string
244  // and reset things.
245  if (offset <= start)
246  {
247  end = string_length;
248  pattern_start = offset;
249  }
250  // forward movement. Use the string between the end of the last
251  // match and the current match position
252  else
253  {
254  end = offset;
255  // cap at the end of the string
256  if (end >= string_length)
257  {
258  end = string_length;
259  }
260 
261  pattern_start = end;
262  }
263  // this is a zero length match pattern.
264  pattern_end = pattern_start;
265  subcurrent = start;
266 }

268 /**
269  * Move backward by an offset. This is the -n form, which
270  * employs the backward movement rule to cause a match to the
271  * end of the string.
272  *
273  * @param offset The offset amount to move.
274  */
275 void RexxTarget::backward(size_t offset)
276 {
277  // the current string starts with the last pattern and goes to
278  // the end
279  start = pattern_start;
280  end = string_length;
281 
282  // relative movement is from start of last match. If
283  // this would take us past the beginning, just go to the start.
284  if (offset > pattern_start)
285  {
286  pattern_start = 0;
287  }
288  else
289  {
290  pattern_start -= offset;
291  }
292  // all numeric patterns are zero length
293  pattern_end = pattern_start;
294  subcurrent = start;
295 }

298 /**
299  * Backward movement for a <n pattern. This does not
300  * use the "match to the end of the string" semantics
301  * that the -n pattern used.
302  *
303  * @param offset The offset to move.
304  */
305 void RexxTarget::backwardLength(size_t offset)
306 {
307  // the start position will be the final movement postion.
308  // the end position will be the last match position, which
309  // allows parsing characters immediately before a match position.
310 
311  // move relative to the last match position, but we cannot
312  // go past the beginning.
313  if (offset > pattern_start)
314  {
315  start = 0;
316  }
317  else
318  {
319  start = pattern_start - offset;
320  }
321  // the end is the old last match position
322  end = pattern_start;
323  // the new match position is the current start
324  pattern_start = start;
325  // the pattern end is the old pattern start
326  pattern_end = pattern_start;
327  // and set the location for word parsing.
328  subcurrent = start;
329 }

332 /**
333  * Seach the string from the end of the last match.
334  *
335  * @param needle The search needle.
336  */
337 void RexxTarget::search(RexxString *needle)
338 {
339  // all searches start from the end of the last match position
340  // and so does the section to be parsed up
341  start = pattern_end;
342  // just use pos to search
343  end = string->pos(needle, start);
344  // if not found, we match to the end of the string
345  if (end == 0)
346  {
347  end = string_length;
348  pattern_start = string_length;
349  pattern_end = string_length;
350  }
351  else
352  {
353  // make origin zero
354  end--;
355  // mark the pattern start
356  pattern_start = end;
357  // the pattern end is start + needle length
358  pattern_end = pattern_start + needle->getLength();
359  }
360  // again, prepare for word operations
361  subcurrent = start;
362 }

399 /**
400  * Get the next word from the parsed substring.
401  *
402  * @return The next blank delimited word, or a NULLSTRING if
403  * we've used up the string section.
404  */
405 RexxString *RexxTarget::getWord()
406 {
407  // already moved past the end of the string? This is a NULLSTRING result
408  if (subcurrent >= end)
409  {
410  return GlobalNames::NULLSTRING;
411  }
412  else /* need to scan off a word */
413  {
414  // get some pointers for scanning
415  const char *scan = string->getStringData() + subcurrent;
416  const char *endScan = string->getStringData() + end;
417  // NOTE: All string objects have a terminating NULL, so the
418  // scan for nonblanks is guaranteed to stop before getting into
419  // trouble, which eliminates the need to check against the
420  // length */
421 
422  // we recognize both blanks and tabs as whitespace
423  while (*scan == ' ' || *scan == '\t')
424  {
425  scan++;
426  }
427  // set the new location
428  subcurrent = scan - string->getStringData();
429  // scanned past the end? just return a null string.
430  if (subcurrent >= end)
431  {
432  return GlobalNames::NULLSTRING;
433  }
434  // we have the start of a real word here...
435  else
436  {
437  // now we're looking for a blank for the end of the string.
438  // scan is the start of the word, endScan remains the same,
439  // endPosition will be the end of the word
440  const char *scanner = scan;
441  const char *endPosition = NULL;
442  // scan for a blank
443  while (scanner < endScan)
444  {
445  if (*scanner == ' ' || *scanner == '\t')
446  {
447  endPosition = scanner;
448  break;
449  }
450  scanner++;
451  }
452 
453  size_t length = 0;
454  // if no blank was found, we use the rest of the pattern piece.
455  if (endPosition == NULL)
456  {
457  length = end - subcurrent;
458  // we've used up this string section
459  subcurrent = end;
460  }
461  // consume that piece of the string
462  else
463  {
464  // set the new location parsing location (NOTE: we also consume this
465  // blank character for doing our next scan bit).
466  subcurrent = endPosition - string->getStringData() + 1;
467  length = endPosition - scan;
468  }
469 
470  // is this the entire string? Just return it directly rather than
471  // create a new string object.
472  if (length == string_length)
473  {
474  return string;
475  }
476  else
477  {
478  // extract a subpiece from the string
479  return new_string(scan, length);
480  }
481  }
482  }
483  // should never get here.
484  return GlobalNames::NULLSTRING;
485 }

487 /**
488  * Skip a word in the parse string section. This is
489  * used for the placeholder '.' pattern.
490  */
491 void RexxTarget::skipWord()
492 {
493  // something left to process?
494  if (subcurrent < end)
495  {
496  // get some scanning pointers
497  const char *scan = string->getStringData() + subcurrent;
498  const char *endScan = string->getStringData() + end;
499  // NOTE: All string objects have a terminating NULL, so the
500  // scan for nonblanks is guaranteed to stop before getting into
501  // trouble, which eliminates the need to check against the
502  // length
503  while (*scan == ' ' || *scan == '\t')
504  {
505  scan++;
506  }
507  // set the new location
508  subcurrent = scan - string->getStringData();
509  const char *endPosition = NULL;
510  // something left? Then we need to establish the word end position
511  if (subcurrent < end)
512  {
513  // look for the next blank
514  const char *endPosition = NULL;
515  while (scan < endScan)
516  {
517  if (*scan == ' ' || *scan == '\t')
518  {
519  endPosition = scan;
520  break;
521  }
522  scan++;
523  }
524  // if we did not found anything, this is used up
525  if (endPosition == NULL)
526  {
527  subcurrent = end;
528  }
529  // we know the end position...also consume the terminating white space
530  else
531  {
532  subcurrent = endPosition - string->getStringData() + 1;
533  }
534  }
535  }
536 }

538 /**
539  * Return the remainder of the parsed section of string.
540  *
541  * @return The string remainder.
542  */
543 RexxString *RexxTarget::remainder()
544 {
545  // already used up? this is ""
546  if (subcurrent >= end)
547  {
548  return GlobalNames::NULLSTRING;
549  }
550  // get the rest of the current piece
551 
552  // get the length, and then clear the string section
553  size_t length = end - subcurrent;
554  size_t offset = subcurrent;
555  // consume the rest
556  subcurrent = end;
557  // if this is the entire string, we can just return
558  // the string without making a new string object
559  if (length == string_length)
560  {
561  return string;
562  }
563 
564  // extract a new string piece
565  RexxString *word = string->extract(offset, length);
566  return word;
