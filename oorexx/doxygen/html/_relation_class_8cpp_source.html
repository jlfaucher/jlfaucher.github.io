75 /**
76  * construct a RelationClass with a given size.
77  *
78  * @param capacity The required capacity.
79  */
80 RelationClass::RelationClass(size_t capacity)
81 {
82  // get a suggested bucket size for this capacity
83  // NOTE: all of this needs to be done at the top-level constructor
84  // because of the way C++ constructors work. As each
85  // previous contructor level gets called, the virtual function
86  // pointer gets changed to match the class of the contructor getting
87  // called. We don't have access to our allocateContents() override
88  // until the final constructor is run.
89  size_t bucketSize = calculateBucketSize(capacity);
90  contents = allocateContents(bucketSize, bucketSize *2);
91 }

94 /**
95  * Create a new table instance from Rexx code.
96  *
97  * @param args The new arguments.
98  * @param argCount The count of new arguments.
99  *
100  * @return The constructed instance.
101  */
102 RexxObject *RelationClass::newRexx(RexxObject **args, size_t argCount)
103 {
104  // this class is defined on the object class, but this is actually attached
105  // to a class object instance. Therefore, any use of the this pointer
106  // will be touching the wrong data. Use the classThis pointer for calling
107  // any methods on this object from this method.
108  RexxClass *classThis = (RexxClass *)this;
109 
110  // create the new identity table item (this version does not have a backing contents yet).
111  Protected<RelationClass> temp = new RelationClass(true);
112  // finish setting this up.
113  classThis->completeNewObject(temp, args, argCount);
114 
115  // make sure this has been completely initialized
116  temp->initialize();
117  return temp;
118 }
