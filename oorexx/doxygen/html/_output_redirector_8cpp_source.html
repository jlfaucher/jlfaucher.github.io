88  */
89 void OutputRedirector::write(RexxString *line)
90 {
91  // if we've had a previous writeBuffer, we
92  // could have the tail end of the buffer still
93  // pending. Consider that a complete line, then
94  // write this new line
95  flushBuffer();
96  writeLine(line);
97 }

100 /**
101  * Write a buffer of data potentially consisting of multiple
102  * lines to the collector. We will parse out as many lines as
103  * we can, then will hold the remainder until we receive another
104  * buffer.
105  *
106  * @param data The data buffer to process
107  * @param len The length of the buffer.
108  */
109 void OutputRedirector::writeBuffer(const char *data, size_t len)
110 {
111  // seems silly, but this is easy
112  if (len == 0)
113  {
114  return;
115  }
116 
117  const char *bufferEnd = data + len;
118  // we might have a buffer here. Need to merge this with the data, then we can
119  // chop up the rest of this buffer
120  resolveBuffer(data, bufferEnd);
121  // the process of resolving the buffer could have consumed all of this new
122  // data, so
123  if (data < bufferEnd)
124  {
125  consumeBuffer(data, bufferEnd);
126  }
127 }

130 /**
131  * Flush any remaining buffered data to the accumulator.
132  */
133 void OutputRedirector::flushBuffer()
134 {
135  if (bufferedData != OREF_NULL)
136  {
137  // On Windows, there seems to be a boundary condition that causes
138  // a dangling '\r' on the last read buffer. If we have this at the end
139  // of the buffer, we need to chop that off.
140  if (bufferedData->endsWith('\r'))
141  {
142  // yep, all but the last part of the tail is a complete line,
143  // and we need to step over the first buffer character
144  Protected<RexxString> newLine = new_string(bufferedData->getStringData(), bufferedData->getLength() - 1);
145  writeLine(newLine);
146  }
147  // the dangling buffer is all data
148  else
149  {
150  writeLine(bufferedData);
151  }
152  }
153  bufferedData = OREF_NULL;
154 }

157 /**
158  * Check to see if we have the tail end of a previous buffer
159  * still left. We need to combine this with the front of the new
160  * buffer to create a complete line and add that to the
161  * accumulator.
162  *
163  * @param data The start of the buffer.
164  * @param bufferEnd The end of the buffer.
165  */
166 void OutputRedirector::resolveBuffer(const char *&data, const char *bufferEnd)
167 {
168  // best case is we don't need to do anything.
169  if (bufferedData == OREF_NULL)
170  {
171  return;
172  }
173  // one complication of the scan is the possibility that our
174  // buffer ended with the first half of a linend pair. We need to check
175  // that possibility first
176  if (bufferedData->endsWith('\r'))
177  {
178  // rats, see if this was a true boundary condition
179  if (*data == '\n')
180  {
181  // change the buffer start
182  data++;
183  // yep, all but the last part of the tail is a complete line,
184  // and we need to step over the first buffer character
185  Protected<RexxString> newLine = new_string(bufferedData->getStringData(), bufferedData->getLength() - 1);
186  writeLine(newLine);
187 
188  // we no longer have a tail piece
189  bufferedData = OREF_NULL;
190  return;
191  }
192 
193  // not a split, the '\r' is considered real data
194  }
195 
196  const char *lineEnd = NULL;
197  const char *nextLine = NULL;
198 
199  // scan for a line in the new buffer
200  scanLine(data, bufferEnd, lineEnd, nextLine);
201 
202  // did we find a lineend? if not, we need to merge our old buffer with the new
203  // data and hold it again.
204  if (lineEnd == NULL)
205  {
206  // save this and cast off the previous buffer
207  bufferedData = new_string(bufferedData->getStringData(), bufferedData->getLength(), data, bufferEnd - data);
208  // mark that we've used up all of the new data
209  data = bufferEnd;
210  // and we're finished here
211  return;
212  }
213 
214  // we have a complete line now split in two pieces
215  Protected<RexxString> newLine = new_string(bufferedData->getStringData(), bufferedData->getLength(), data, lineEnd - data);
216  writeLine(newLine);
217 
218  // we're done with this
219  bufferedData = OREF_NULL;
220 
221  // now update the data buffer position
222  if (nextLine == NULL)
223  {
224  // end on a linend, we're done with this new buffer
225  data = bufferEnd;
226  }
227  else
228  {
229  // start scanning at the next position
230  data = nextLine;
231  }
232 }

235 /**
236  * Consume as much of a received buffer as individual
237  * lines as we can, saving the rest.
238  *
239  * @param data The start of the buffer.
240  * @param bufferEnd The end of the buffer.
241  */
242 void OutputRedirector::consumeBuffer(const char *data, const char *bufferEnd)
243 {
244  const char *lineEnd = NULL;
245  const char *nextLine = NULL;
246 
247  for (;;)
248  {
249  // scan for a line
250  scanLine(data, bufferEnd, lineEnd, nextLine);
251  // NULL lineEnd means we did not find a line terminator. We save the remainder
252  // of the buffer and quit
253  if (lineEnd == NULL)
254  {
255  // we just save this as a string object.
256  bufferedData = new_string(data, bufferEnd - data);
257  return;
258  }
259  // convert to a string object and add to the accumulator
260  Protected<RexxString> newLine = new_string(data, lineEnd - data);
261  writeLine(newLine);
262 
263  // the buffer could have ended with a linend
264  // we've magically used up the entire buffer and can leave now
265  if (nextLine == NULL)
266  {
267  return;
268  }
269  // step forward to that position and scan again
270  data = nextLine;
271  }
272 }

275 /**
276  * Useful method for scanning a buffer looking for logical
277  * lines with the buffer. This uses the same line termination
278  * rules as the platform.
279  *
280  * @param lineStart The start location for scanning.
281  * @param bufferLen The length of buffer we have to work with.
282  * @param lineEnd The return pointer to the end of the scanned line. Returns
283  * NULL if no line end is found.
284  * @param nextLine Pointer to the start of the next line (past the linend)
285  * returns NULL if this would be past the end of the buffer.
286  */
287 void OutputRedirector::scanLine(const char *lineStart, const char *bufferEnd, const char *&lineEnd, const char *&nextLine)
288 {
289  lineEnd = NULL;
290  nextLine = NULL;
291 
292  // now scan the buffer looking for a line end.
293  for (; lineStart < bufferEnd; lineStart++)
294  {
295  // special handling for carriage return characters. we check the next character
296  // next character and if it is a newline, then we step over both characters and
297  // consider this a new line. Otherwise, we skip the character
298  // and leave the \r as data within the line.
299  if (*lineStart == '\r')
300  {
301 
302  // this is a little bit of a quandry. We found a \r as the last character
303  // of the buffer. We can't make a decision here, so punt and say we don't have
304  // a line
305  if (lineStart == bufferEnd - 1)
306  {
307  return;
308  }
309 
310  // is the next character a newline?
311  if (*(lineStart + 1) == '\n')
312  {
313  lineEnd = lineStart;
314  nextLine = lineStart + 2;
315  // did this happen at the end of the buffer?
316  // no next Line to note
317  if (nextLine == bufferEnd)
318  {
319  nextLine = NULL;
320  }
321  return;
322  }
323  }
324 
325  // Now check for a newline.
326  else if (*lineStart == '\n')
327  {
328  lineEnd = lineStart;
329  nextLine = lineStart + 1;
330  // did this happen at the end of the buffer?
331  // no next Line to note
332  if (nextLine == bufferEnd)
333  {
334  nextLine = NULL;
335  }
336  return;
337  }
338  }
339 
340  // if we scan all of the buffer without finding a line,
341  // the returns are already set to NULL;
342 }

398 /**
399  * Initialize and validate the stem condition for updates.
400  */
401 void StemOutputTarget::init()
402 {
403  // only one per customer!
404  if (initialized)
405  {
406  return;
407  }
408  initialized = true;
409  // replace is the easy way to handle we just reset everything
410  if (option == OutputOption::REPLACE || option == OutputOption::DEFAULT)
411  {
412  // empty everything out (could already be empty
413  stem->empty();
414  // set the stem.0 value to zero
415  stem->setElement((size_t)0, IntegerZero);
416  // our next element to write is 1
417  index = 1;
418  }
419  // asked to append...this is a little more work.
420  else
421  {
422  // get the stem.0 value
423  RexxObject *count = stem->getElement(size_t(0));
424  // if completely not there, then we'll just treat this like a replace
425  if (count == OREF_NULL)
426  {
427  // empty everything out (could already be empty
428  stem->empty();
429  // set the stem.0 value to zero
430  stem->setElement((size_t)0, IntegerZero);
431  // our next element to write is 1
432  index = 1;
433  }
434  // the stem.0 value must be an integer size. We'll start writing at the next position
435  else
436  {
437  // a valid whole number index
438  if (!count->unsignedNumberValue(index, Numerics::ARGUMENT_DIGITS))
439  {
440  reportException(Error_Invalid_whole_number_stem_array_index, stem->getName(), count);
441  }
442  // we write to the next position
443  index ++;
444  }
445  }
446 }

620 /**
621  * Initialize and validate the stream condition for updates.
622  *
623  * @param option The append/replace option for the redirect.
624  */
625 void StreamOutputTarget::init()
626 {
627  // only one per customer!
628  if (initialized)
629  {
630  return;
631  }
632  initialized = true;
633 
634  // create an instance of the stream class for the given name
635  RexxClass *streamClass = TheRexxPackage->findClass(GlobalNames::STREAM);
636  ProtectedObject result;
637  stream = streamClass->sendMessage(GlobalNames::NEW, name, result);
638 
639  RexxString *openResult = OREF_NULL;
640 
641  // If replace is specified, we open this WRITE REPLACE
642  if (option == OutputOption::REPLACE || option == OutputOption::DEFAULT)
643  {
644  openResult = (RexxString *)stream->sendMessage(GlobalNames::OPEN, GlobalNames::WRITE_REPLACE, result);
645  }
646  // asked to append...just a different open option
647  else
648  {
649  openResult = (RexxString *)stream->sendMessage(GlobalNames::OPEN, GlobalNames::WRITE_APPEND, result);
650  }
651  // the open should return ready
652  if (!openResult->strCompare(GlobalNames::OPENREADY))
653  {
654  reportException(Error_Execution_file_not_writeable, name, openResult);
655  }
656 }
