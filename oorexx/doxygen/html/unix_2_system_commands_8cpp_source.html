80 // a thread that writes INPUT data to the command pipe
81 class InputWriterThread : public SysThread
82 {
83 
84 public:
85  inline InputWriterThread() : SysThread(),
86  pipe(0), inputBuffer(NULL), bufferLength(0), error(0) { }
87  inline ~InputWriterThread() { terminate(); }
88 
89  void start(int _pipe)
90  {
91  pipe = _pipe;
92  SysThread::createThread();
93  }
94 
95  virtual void dispatch()
96  {
97 
98  if (inputBuffer != NULL && bufferLength > 0)
99  {
100  if (write(pipe, inputBuffer, bufferLength) < 0)
101  {
102  // We may receive EPIPE, which we only expect if a spawned
103  // command finishes while we're still busy trying to pipe
104  // our input. The command may simply not need (i. e. want
105  // to read) our input. We dont' consider this an error.
106  if (errno != EPIPE)
107  {
108  error = errno;
109  }
110  }
111  close(pipe);
112  }
113  }
114 
115  int pipe; // the pipe we read the data from
116  const char *inputBuffer; // the buffer of data to write
117  size_t bufferLength; // the length of the read data
118  int error; // and error that resulted.
119 };

122 // a thread that reads ERROR data from the command pipe
123 class ErrorReaderThread : public SysThread
124 {
125 
126 public:
127  inline ErrorReaderThread() : SysThread(),
128  pipe(0), errorBuffer(NULL), bufferLength(0), error(0) { }
129  inline ~ErrorReaderThread()
130  {
131  if (errorBuffer != NULL)
132  {
133  free(errorBuffer);
134  }
135  terminate();
136  }
137 
138  void start(int _pipe)
139  {
140  pipe = _pipe;
141  SysThread::createThread();
142  }
143 
144  virtual void dispatch()
145  {
146  bufferLength = 0;
147  size_t bufferAllocation = PIPE_BUF;
148 
149  // allocate the initial buffer
150  if ((errorBuffer = (char *)malloc(PIPE_BUF)) == NULL)
151  {
152  error = errno;
153  return;
154  }
155  ssize_t length;
156 
157  // Read until we hit EOF or an error. Whenever the buffer gets too
158  // small, we reallocate it with twice the size.
159  while ((length = read(pipe, &errorBuffer[bufferLength], bufferAllocation - bufferLength)) > 0)
160  {
161  bufferLength += length;
162  // do we need to increase our buffer allocation?
163  if (bufferLength >= bufferAllocation)
164  {
165  bufferAllocation += PIPE_BUF; // add another increment
166  char *largerBuffer = (char *)realloc(errorBuffer, bufferAllocation);
167  if (largerBuffer == NULL)
168  {
169  error = errno;
170  return;
171  }
172  errorBuffer = largerBuffer;
173  }
174  }
175  if (length < 0) // error reading pipe
176  {
177  error = errno;
178  return;
179  }
180  close(pipe);
181  }
182 
183  int pipe; // the pipe we read the data from
184  char *errorBuffer; // dynamically allocated once we start reading
185  size_t bufferLength; // the length of the read data
186  int error; // and error that resulted.
187 };

200 /* Handle "export" command in same process */
201 bool sys_process_export(RexxExitContext *context, const char * cmd, RexxObjectPtr &rc, int flag)
202 {
203  char *Env_Var_String = NULL; /* Environment variable string for */
204  size_t size, allocsize; /* size of the string */
205  char **Environment; /* environment pointer */
206  char *np;
207  size_t i,j,k,l,iLength, copyval;
208  char namebufcurr[1281]; /* buf for extracted name */
209  char cmd_name[1281]; /* name of the envvariable setting */
210  char *array, *runarray, *runptr, *endptr, *maxptr;
211  char temparray[1281];
212  const char *st;
213  char *tmpptr;
214  char name[1281]; /* is the name + value + = */
215  char value[1281]; /* is the part behind = */
216  char *del = NULL; /* ptr to old unused memory */
217  char *hit = NULL;
218  bool HitFlag = false;
219  l = 0;
220  j = 0;
221  allocsize = 1281 * 2;
222 
223  memset(temparray, '\0', sizeof(temparray));
224 
225  Environment = getEnvironment(); /* get the environment */
226  if (flag == EXPORT_FLAG)
227  {
228  st = &cmd[6];
229  }
230  else if (flag == UNSET_FLAG)
231  {
232  st = &cmd[5];
233  }
234  else
235  {
236  st = &cmd[3];
237  }
238  while ((*st) && (*st == ' '))
239  {
240  st++;
241  }
242  strcpy(name, st);
243  iLength = strlen(st) + 1;
244 
245 /* if string == EXPORT_FLAG or string == SET and only blanks are delivered */
246 
247  if ( ((flag == EXPORT_FLAG) || (flag == SET_FLAG)) && (iLength == 1) )
248  {
249  return false;
250  }
251 
252  if (!putflag)
253  { /* first change in the environment ? */
254  /* copy all entries to dynamic memory */
255  for (;*Environment != NULL; Environment++)
256  { /*for all entries in the env */
257  size = strlen(*Environment)+1; /* get the size of the string */
258  Env_Var_String = (char *)malloc(size); /* and alloc space for it */
259  memcpy(Env_Var_String,*Environment,size);/* copy the string */
260  putenv(Env_Var_String); /* and chain it in */
261  }
262  }
263  putflag = 1; /* prevent do it again */
264  Environment = getEnvironment(); /* reset the environment pointer */
265 
266 /* do we have a assignment operator? If not return true */
267 /* The operating system treads this like no command, and so do we */
268 
269  if ( !(strchr(name, '=')) && (flag != UNSET_FLAG) ) /*only set and export */
270  {
271 /* we do not have a assignment operator, but maybe a '|' for a */
272 /* controlled output */
273  if ( (strchr(name, '|')) || (strchr(name, '>')) || (strstr(name, ">>")) )
274  {
275  return false;
276  }
277  else
278  {
279  // this worked ok (well, sort of)
280  rc = context->False();
281  return true;
282  }
283  }
284 
285 /* no '=' for unset, so force a shell error message */
286 
287  if ( (strchr(name, '=')) && (flag == UNSET_FLAG) )
288  {
289  return false;
290  }
291 
292  for (i=0; (i<iLength) && (name[i]!='='); i++)
293  {
294  cmd_name[i] = name[i];
295  }
296 
297 /* lets try handling variables in the assignment string */
298 
299  cmd_name[i] = '\0'; /* copy the terminator */
300 
301  i++; /* the place after'=' */
302 
303 /* lets search the value for variable part(s) */
304 
305  strcpy(value, &(name[i])); /* value contains the part behind '=' */
306  array = (char *) malloc(1281);
307  strcpy(array, cmd_name);
308  array[strlen(cmd_name)] = '=';
309  array[i] = '\0'; /* copy the terminator */
310  runarray = array + strlen(array);
311  runptr = value;
312  endptr = runptr + strlen(value); /*this is the end of the input*/
313  maxptr = array + MAX_VALUE -1; /* this is the end of our new string */
314 
315  while ((tmpptr = (strchr(runptr, '$'))) != 0)
316  {
317  Environment = getEnvironment(); /* get the beginning of the environment*/
318  HitFlag = true; /* if not true inputvalue= outputvalue*/
319  copyval = tmpptr - runptr;
320  if (copyval) /* runarray should keep the 'real' environment */
321  {
322  while ((runarray + copyval) > maxptr)
323  {
324  array = (char *) realloc(array, allocsize);
325  runarray = array + strlen(array);
326  maxptr = array + allocsize - 1;
327  allocsize = allocsize * 2;
328  }
329  memcpy(runarray,runptr, copyval);
330  runarray= runarray + copyval; /* a new place to copy */
331  *runarray = '\0';
332  runptr = tmpptr; /* now runptr is at the place of $ */
333  }
334  runptr++;
335  for (j = 0;(*runptr != '/') && (*runptr != ':') && (*runptr != '$') &&
336  (*runptr); j++)
337  {
338  memcpy(&(temparray[j]), runptr,1); /*temparray is the env var to search*/
339  runptr++;
340  }
341 
342  temparray[j] = '\0'; /* lets see what we can do */
343  np = NULL;
344 
345  for (;(*Environment != NULL) && (hit == NULL) ;Environment++)
346  {
347  np = *Environment;
348 
349  for (k=0;(*np!='=')&&(k<255);np++,k++)
350  {
351  memcpy(&(namebufcurr[k]),np,1); /* copy the character */
352  }
353 
354  namebufcurr[k] = '\0'; /* copy the terminator */
355 
356  if (!strcmp(temparray,namebufcurr)) /* have a match ? */
357  {
358  hit = *Environment;
359  /* copy value to new string*/
360  }
361  }
362  if (hit) /* if we have found an entry of the env var in the env list */
363  {
364  np ++; /* don't copy equal */
365  while ((runarray + strlen(np)) > maxptr)
366  {
367  array = (char *) realloc(array, allocsize);
368  runarray = array + strlen(array);
369  maxptr = array + allocsize - 1;
370  allocsize = allocsize * 2;
371  }
372  strcpy(runarray, np);
373  runarray = runarray + strlen(np);
374  *runarray = '\0';
375  hit = NULL;
376  }
377  }
378 
379  if (HitFlag == true)
380  {
381  if (runptr < endptr)
382  {
383  while ((runarray + strlen(runptr)) > maxptr)
384  {
385  array = (char *) realloc(array, allocsize);
386  runarray = array + strlen(array);
387  maxptr = array + allocsize - 1;
388  allocsize = allocsize * 2;
389  }
390  strcpy(runarray, runptr); /* if there is a part after a var */
391  runarray = runarray + strlen(runptr);
392  *runarray = '\0';
393  }
394  }
395  else /* no hit so lets copy the value as it is */
396  {
397  while ((runarray + strlen(value)) > maxptr)
398  {
399  array = (char *) realloc(array, allocsize);
400  runarray = array + strlen(array);
401  maxptr = array + allocsize - 1;
402  allocsize = allocsize * 2;
403  }
404  strcpy(runarray,value);
405  runarray = runarray + strlen(runptr);
406  *runarray = '\0';
407  }
408 
409  Environment = getEnvironment(); /* get the beginning of the environment*/
410 
411  for (;*Environment != NULL;Environment++)
412  {
413  np = *Environment;
414 
415  for (i=0;(*np!='=')&&(i<255);np++,i++)
416  {
417  memcpy(&(namebufcurr[i]),np,1); /* copy the character */
418  }
419 
420  namebufcurr[i] = '\0'; /* copy the terminator */
421 
422  if (!strcmp(cmd_name,namebufcurr))/* have a match ? */
423  {
424  del = *Environment; /* remember it for deletion */
425  }
426  }
427  /* find the entry in the environ */
428  if (flag != UNSET_FLAG)
429  {
430  size = strlen(array)+1;
431  Env_Var_String = (char *)malloc(size);/* get the memory */
432  memcpy(Env_Var_String, array, size);
433  int errCode = putenv(Env_Var_String);
434  if (errCode != 0)
435  {
436  // non-zero is an error condition
437  context->RaiseCondition("ERROR", context->String(cmd), NULL, context->WholeNumberToObject(errno));
438  }
439  else
440  {
441  rc = context->False();
442  }
443  }
444 
445  if (del) /* if there was a old one */
446  {
447  free(del); /* free it */
448  }
449  rc = context->False();
450  return true;
451 }

454 /* Returns a copy of s without quotes. Escaped characters are kept unchanged */
455 char *unquote(const char *s)
456 {
457  if ( s == NULL )
458  {
459  return NULL;
460  }
461  size_t size = strlen(s) + 1;
462  char *unquoted = (char*)malloc(sizeof(char)*size);
463  if ( unquoted == NULL )
464  {
465  return NULL;
466  }
467  char *u = unquoted;
468  char c;
469  bool escape = false;
470  do
471  {
472  c = *s;
473  if ( escape )
474  {
475  *u++ = *s;
476  escape = false;
477  }
478  else if ( c == '\\' )
479  {
480  escape = true;
481  }
482  else if ( c != '"' )
483  {
484  *u++ = *s;
485  }
486  s++;
487  }
488  while ( c != '\0' );
489  return unquoted;
490 }

493 /* Handle "cd XXX" command in same process */
494 bool sys_process_cd(RexxExitContext *context, const char * cmd, RexxObjectPtr rc)
495 {
496  const char * st;
497  const char *home_dir = NULL; /* home directory path */
498  char *dir_buf = NULL; /* full directory path */
499  size_t bufSize;
500  const char *slash; /* ptr to '/' */
501  struct passwd *ppwd;
502 
503  st = &cmd[2];
504  while ((*st) && (*st == ' '))
505  {
506  st++;
507  }
508  if ((!*st) || (strlen(cmd) == 2))
509  {
510  home_dir = getenv("HOME");
511  if (!home_dir)
512  {
513  return false;
514  }
515  dir_buf = (char *)malloc(strlen(home_dir)+1);
516  strcpy(dir_buf, home_dir);
517  } /* if no user name */
518  else if (*(st) == '~' && (*(st+1) == '\0' || *(st+1) == '/'|| *(st+1) == ' ' ))
519  {
520  if (*(st+1) == '/') /* if there is a path */
521  {
522  st +=2; /* jump over '~/' */
523  /* get home directory path */
524  home_dir = getenv("HOME"); /* from the environment */
525  if (!home_dir) /* if no home dir info */
526  {
527  return false;
528  }
529  /* get space for the buf */
530  bufSize = strlen(home_dir) + strlen(st) + 2;
531  dir_buf = (char *)malloc(bufSize);
532  if (!dir_buf)
533  {
534  return false;
535  }
536  /* merge the strings */
537  snprintf(dir_buf, bufSize, "%s/%s", home_dir, st);
538  }
539  else
540  {
541  /* get home directory path */
542  home_dir = getenv("HOME"); /* from the environment */
543  /* get space for the buf */
544  bufSize = strlen(home_dir) + 2;
545  dir_buf = (char *)malloc(bufSize);
546  if (!dir_buf)
547  {
548  return false;
549  }
550  snprintf(dir_buf, bufSize, "%s/", home_dir);
551  }
552  }
553  else if (*(st) == '~') /* cmd is '~username...' */
554  {
555  st++; /* jump over '~' */
556  slash = strchr(st,'/'); /* search for '/' */
557  if (!slash) /* if no '/' */
558  {
559  /* rest of string is username */
560  ppwd = getpwnam(st); /* get info about the user */
561  if (ppwd == NULL || ppwd->pw_dir == NULL)
562  {
563  return false;
564  }
565  /* get space for the buf */
566  bufSize = strlen(ppwd->pw_dir) + 2;
567  dir_buf = (char *)malloc(bufSize);
568  if (!dir_buf)
569  {
570  return false;
571  }
572  /* merge the strings */
573  snprintf(dir_buf, bufSize, "%s/", ppwd->pw_dir);
574  }
575  else /* there is a slash */
576  {
577  char username[256]; // need to copy the user name
578  memcpy(username, st, slash - st);
579  username[slash - st] = '\0';
580 
581  ppwd = getpwnam(username); /* get info about the user */
582  if (ppwd == NULL || ppwd->pw_dir == NULL)
583  {
584  return false;
585  }
586  slash++; /* step over the slash */
587  /* get space for the buf */
588  bufSize = strlen(ppwd->pw_dir) + strlen(slash) + 2;
589  dir_buf = (char *)malloc(bufSize);
590  if (!dir_buf)
591  {
592  return false;
593  }
594  /* merge the strings */
595  snprintf(dir_buf, bufSize, "%s/%s", ppwd->pw_dir, slash);
596  }
597  }
598  else
599  {
600  dir_buf = strdup(st);
601  }
602 
603  char *unquoted = unquote(dir_buf);
604  if (unquoted == NULL)
605  {
606  return false;
607  }
608  int errCode = chdir(unquoted);
609  if (errCode < 0)
610  {
611  errCode = errno;
612  }
613  free(unquoted);
614 
615  free(dir_buf);
616  if (errCode != 0)
617  {
618  // non-zero is an error condition
619  context->RaiseCondition("ERROR", context->String(cmd), NULL, context->WholeNumberToObject(errCode));
620  }
621  else
622  {
623  rc = context->False();
624  }
625  return true;
626 }

629 /**
630  * Breaks up a command string into whitespace-delimited pieces. Whitespace-
631  * delimited double-quoted strings are treated as a single argument.
632  *
633  * @param parm_cmd The command string.
634  * @param argPtr The "argv" array.
635  *
636  * @return True if command parsed successfully, else false.
637  */
638 bool scan_cmd(const char *parm_cmd, char **argPtr)
639 {
640  char *cmd = strdup(parm_cmd); /* Allocate for copy */
641 
642  char *end = cmd + strlen(cmd); /* Find the end of the command*/
643 
644  /* This loop scans our copy of the command, setting pointers in */
645  /* the args[] array to point to each of the arguments, and null- */
646  /* terminating each one of them. */
647 
648  /* LOOP INVARIANT: */
649  /* pos points to the next character of the command to be examined. */
650  /* i indicates the next element of the args[] array to be loaded. */
651  size_t i = 0; /* Start with args[0] */
652  logical_t quoted = false;
653  for (char *pos = cmd; pos < end; pos++)
654  {
655  while (*pos==' ' || *pos=='\t')
656  {
657  pos++; /* Skip to first non-white */
658  }
659 
660  if (*pos == '\0') /* If we're at the end, */
661  {
662  break; /* get out. */
663  }
664 
665  /* If at this point, we've used up all but one of the available */
666  /* elements of our args[] array, let the user know, and we must */
667  /* terminate. */
668  if (i == MAX_COMMAND_ARGS)
669  {
670  return false;
671  }
672 
673  if (*pos == '"') // start of a quoted part?
674  {
675  quoted = true; // we're now within quotes
676  *pos++ = '\0'; // remove and skip the quote character
677  }
678 
679  argPtr[i++] = pos; /* Point to current argument */
680  /* and advance i to next */
681  /* element of args[] */
682  if (quoted)
683  { // look for next quote followed by whitespace or end of command
684  while (!(*pos == '\0' ||
685  (*(pos - 1) == '"' && (*pos == ' ' || *pos == '\t' ))))
686  {
687  pos++;
688  }
689  if (*(pos - 1) == '"')
690  {
691  quoted = false; // we're again outside of any quotes
692  *(pos - 1) = '\0'; // remove the quote character
693  }
694  }
695  else
696  { // look for next whitespace or end of command
697  while (*pos != ' ' && *pos != '\t' && *pos != '\0')
698  {
699  pos++;
700  }
701  }
702  *pos = '\0'; /* Null-terminate this arg */
703 
704  }
705 
706  /* Finally, put a null pointer in args[] to indicate the end. */
707  argPtr[i] = NULL;
708  return true;
709 }

711 /**
712  * Try to handle special commands like "cd" or "export" internally. This will
713  * work if we have a simple commend with no redirection, piping, or such.
714  *
715  * @param context The Exit oontext.
716  * @param cmd The command string.
717  * @param rc The operating system return code.
718  *
719  * @return True if command was handled, false otherwise.
720  */
721 logical_t handleCommandInternally(RexxExitContext *context, char *cmd, RexxObjectPtr rc)
722 {
723  /* check for redirection symbols, ignore them when enclosed in double quotes.
724  escaped quotes are ignored. */
725  bool noDirectInvoc = false;
726  bool inQuotes = false;
727  bool escape = false;
728  size_t i;
729  for (i = 0; i<strlen(cmd); i++)
730  {
731  if (escape)
732  {
733  escape = false;
734  }
735  else if (cmd[i] == '\\')
736  {
737  escape = true;
738  }
739  else if (cmd[i] == '"')
740  {
741  inQuotes = !inQuotes;
742  }
743  else
744  {
745  /* if we're in the unquoted part and the current character is one of */
746  /* the redirection characters or the & for multiple commands then we */
747  /* will no longer try to invoke the command directly */
748  if (!inQuotes && (strchr("<>|&;", cmd[i]) != NULL))

809 /**
810  * A redirecting test command handler.
811  *
812  * @param context The Exit context.
813  * @param address The environment name.
814  * @param command The command name and arguments.
815  * @param ioContext The IO Redirector context.
816  */
817 RexxObjectPtr RexxEntry ioCommandHandler(RexxExitContext *context, RexxStringObject address, RexxStringObject command, RexxIORedirectorContext *ioContext)
818 {
819  pid_t pid;
820  int status;
821  char* argv[MAX_COMMAND_ARGS + 1];
822  // If SYSSHELLPATH could ever grow longer than 128 chars
823  // then this array size will need to be increased.
824  char shell[128 + 1 + 256 + 1]; // SYSSHELLPATH plus "/" plus environment name
825 
826  CSTRING environment = context->CString(address);
827  CSTRING commandString = context->CString(command);
828 
829  if (Utilities::strCaselessCompare("path", environment) == 0)
830  { // For the no-shell "direct" environment we need to break the command
831  // string into a command name and separate arguments each (we have no
832  // shell to do this for us).
833  if (!scan_cmd(commandString, argv))
834  { // too many arguments
835  return ErrorFailure(context, commandString);
836  }
837  // posix_spawnp() will fault if argv[0] is NULL. Fix this.
838  if (argv[0] == NULL)
839  {
840  argv[0] = (char*)"";
841  argv[1] = NULL;
842  }
843  }
844  else
845  { // Set up the full path to the requested shell.
846  strcpy(shell, SYSSHELLPATH);
847  if (shell[strlen(shell) - 1] != '/')
848  { // append slash if we don't have one yet
849  strcat(shell, "/");
850  }
851  if (strlen(environment) == 0 ||
852  Utilities::strCaselessCompare("command", environment) == 0 ||
853  Utilities::strCaselessCompare("system", environment) == 0)
854  { // for environments "", "command" and "system" we use "sh" as shell
855  strcat(shell, "sh");
856  }
857  else
858  { // for all other environment names we use this name in lower case
859  strcat(shell, environment);
860  Utilities::strlower(shell);
861  }
862 
863  argv[0] = (char*)shell;
864  argv[1] = (char*)"-c";
865  argv[2] = (char*)commandString;
866  argv[3] = NULL;
867  }
868 
869  if (ioContext->IsRedirectionRequested())
870  {
871  InputWriterThread inputThread; // separate thread if we need to write INPUT
872  ErrorReaderThread errorThread; // separate thread if we need to read ERROR
873  bool need_to_write_input = false;
874  bool need_to_read_output = false;
875  bool need_to_read_error = false;
876  int input[2], output[2], error[2];
877 
878  posix_spawn_file_actions_t action;
879  posix_spawn_file_actions_init(&action);
880 
881  // Create stdin, stdout, and stderr pipes as requested. pipe() returns
882  // two file descriptors: [0] is the pipe read end, [1] is the pipe
883  // write end. The child process (i. e. the command to be run) will
884  // inherit both the read and write end of all created pipes. Of each
885  // end *we* will only ever use one end, and the child the other end.
886  // Therefore we and the child will close any unused ends. In addition,
887  // we also prepare any file actions that are to be be done by the child.
888  // These actions are collected with posix_spawn_file_actions()
889 
890  // is stdin redirection requested?
891  if (ioContext->IsInputRedirected())
892  {
893  if (pipe(input) != 0) // create stdin pipe
894  {
895  return ErrorRedirection(context, errno);
896  }
897  posix_spawn_file_actions_adddup2(&action, input[0], 0); // stdin reads from pipe
898  posix_spawn_file_actions_addclose(&action, input[1]); // close unused write end in child
899  need_to_write_input = true;
900  }
901 
902  // is stdout redirection requested?
903  if (ioContext->IsOutputRedirected())
904  {
905  if (pipe(output) != 0) // create stdout pipe
906  {
907  return ErrorRedirection(context, errno);
908  }
909  // do we want interleaved stdout and stderr redirection?
910  // this is a special case .. we just redirect stderr to stdout upfront
911  if (ioContext->AreOutputAndErrorSameTarget())
912  {
913  posix_spawn_file_actions_adddup2(&action, output[1], 2); //stderr writes to pipe
914  }
915  posix_spawn_file_actions_adddup2(&action, output[1], 1); //stdout writes to pipe
916  posix_spawn_file_actions_addclose(&action, output[0]); // close unused read end in child
917  need_to_read_output = true;
918  }
919 
920  // now stderr redirection
921  // if both stdout and stderr are to be redirected to the same object, then
922  // everything was already set up in the previous stdout redirection step
923  if (ioContext->IsErrorRedirected() && !ioContext->AreOutputAndErrorSameTarget())
924  {
925  if (pipe(error) != 0) // create stderr pipe
926  {
927  return ErrorRedirection(context, errno);
928  }
929  posix_spawn_file_actions_adddup2(&action, error[1], 2); // stderr writes to pipe
930  posix_spawn_file_actions_addclose(&action, error[0]); // close unused read end in child
931  need_to_read_error = true;
932  }
933 
934  // ok, everything is set up accordingly, let's fork the redirected command
935  if (posix_spawnp(&pid, argv[0], &action, NULL, argv, getEnvironment()) != 0)
936  {
937  return ErrorFailure(context, commandString);
938  }
939 
940  // Close all unneeded read- and write ends
941  if (need_to_write_input)
942  {
943  close(input[0]); // we close our unused stdout pipe write end
944  }
945  if (need_to_read_output)
946  {
947  close(output[1]); // we close our unused stdout pipe write end
948  }
949  if (need_to_read_error)
950  {
951  close(error[1]); // we close our unused stderr pipe write end
952  }
953 
954  if (need_to_write_input)
955  {
956  ioContext->ReadInputBuffer(&inputThread.inputBuffer, &inputThread.bufferLength);
957  // we start a separate thread to write INPUT data to the input pipe
958  inputThread.start(input[1]);
959  }
960  if (need_to_read_error)
961  {
962  // we start a separate thread to read ERROR data from the error pipe
963  errorThread.start(error[0]);
964  }
965 
966  // do we need to read OUTPUT from the stdout pipe?
967  // we run this in the main thread instead of starting another new thread
968  // if we have stdout and sterr interleaved, we'll read both of them here
969  if (need_to_read_output)
970  {
971  // According to POSIX.1 PIPE_BUF is at least 512 bytes; on Linux
972  // it's 4096 bytes.
973  char outputBuffer[PIPE_BUF];
974  ssize_t outputLength;
975 
976  while ((outputLength = read(output[0], &outputBuffer, PIPE_BUF)) > 0)
977  {
978  ioContext->WriteOutputBuffer(outputBuffer, outputLength);
979  }
980  if (outputLength != 0) // 0 is EOF, success
981  {
982  return ErrorRedirection(context, errno);
983  }
984  close(output[0]);
985  }
986 
987  // did we start an ERROR thrad?
988  if (need_to_read_error)
989  {
990  // wait for the ERROR thread to finish
991  errorThread.waitForTermination();
992  if (errorThread.bufferLength > 0)
993  { // return what the ERROR thread read from its pipe
994  ioContext->WriteErrorBuffer(errorThread.errorBuffer, errorThread.bufferLength);
995  }
996  // the ERROR thread may have encountered an error .. raise it now
997  if (errorThread.error != 0)
998  {
999  return ErrorRedirection(context, errorThread.error);
1000  }
1001  }
1002 
1003  // if we started an INPUT thread, wait until it finishes
1004  if (need_to_write_input)
1005  {
1006  inputThread.waitForTermination();
1007  // the INPUT thread may have encountered an error .. raise it now
1008  if (inputThread.error != 0)
1009  {
1010  return ErrorRedirection(context, inputThread.error);
1011  }
1012  }
1013 
1014  posix_spawn_file_actions_destroy(&action);
1015  }
1016  else // no redirection requested
1017  {
1018  // try to handle special commands like 'cd' or 'export' internally
1019  RexxObjectPtr rc = NULLOBJECT;
1020 
1021  if (handleCommandInternally(context, (char*)commandString, rc))
1022  { // the command was handled in this thread
1023  return rc;
1024  }
1025  else
1026  { // to run the command we spawn another thread
1027  if (posix_spawnp(&pid, argv[0], NULL, NULL, argv, getEnvironment()) != 0)
1028  {
1029  return ErrorFailure(context, commandString);
1030  }
1031  }
1032  }
1033 
1034  // wait for our child, the forked command
1035  waitpid(pid, &status, 0);
1036  int rc;
1037  if (WIFEXITED(status)) // child process ended normal
1038  {
1039  rc = WEXITSTATUS(status); // set return code
1040  }
1041  else
1042  { // child process died
1043  rc = -(WTERMSIG(status));
1044  if (rc == 1) // process was stopped
1045  {
1046  rc = -1;
1047  }
1048  }
1049 
1050  // unknown command code?
1051  // a FAILURE will be raised for any command returning 127
1052  if (rc == UNKNOWN_COMMAND)
1053  {
1054  return ErrorFailure(context, commandString);
1055  }
1056  else if (rc != 0)
1057  {
1058  // for any non-zero return code we raise an ERROR condition
1059  context->RaiseCondition("ERROR", context->String(commandString),
1060  NULL, context->WholeNumberToObject(rc));
1061  return NULLOBJECT;
1062  }
1063  return context->False(); // zero return code
1064 }
