49 /**
50  * Default virtual method for handling a run() methods on
51  * an activity dispatcher.
52  */
53 void TranslateDispatcher::run()
54 {
55  Protected<RoutineClass> program;
56 
57  // default name is a null string if one not given
58  Protected<RexxString> name = GlobalNames::NULLSTRING;
59 
60  // have an actual name?
61  if (programName != NULL)
62  {
63  // get string version of the name
64  name = new_string(programName);
65  }
66 
67  // if not an instore request, then this must be a resolvable name
68  if (instore == NULL)
69  {
70  Protected<RexxString> fullname = activity->resolveProgramName(name, OREF_NULL, OREF_NULL, RESOLVE_DEFAULT);
71  if (fullname == OREF_NULL)
72  {
73  reportException(Error_Program_unreadable_notfound, name);
74  }
75 
76  // go translate the image */
77  program = LanguageParser::createProgram(fullname);
78  }
79  // this is an instore program, so it needs to be translated as is
80  else
81  {
82  // this is handled by the language parser
83  program = LanguageParser::processInstore(instore, name);
84  // not valid, so give the unreadable error
85  if (program == OREF_NULL)
86  {
87  reportException(Error_Program_unreadable_name, name);
88  }
89  }
90 
91  // is this a request to save to a file?
92  if (outputName != NULL)
93  {
94  // go save this program using the indicated format
95  program->save(outputName, encode);
96  }
97 }

116 /**
117  * Default virtual method for handling a run() methods on
118  * an activity dispatcher.
119  */
120 void TranslateInstoreDispatcher::run()
121 {
122  Protected<RexxString> name = GlobalNames::NULLSTRING; // name of the invoked program
123  // have a name given, then use that instead
124  if (programName != NULL)
125  {
126  name = new_string(programName);
127  }
128 
129  RXSTRING instore[2];
130 
131  MAKERXSTRING(instore[0], const_cast<char *>(source->strptr), source->strlength);
132  MAKERXSTRING(instore[1], NULL, 0);
133 
134  // do the translation
135  Protected<RoutineClass> program = LanguageParser::processInstore(instore, name);
136  if (program == OREF_NULL)
137  {
138  reportException(Error_Program_unreadable_name, name);
139  }
140 
141  // copy the image information back
142  image->strptr = instore[1].strptr;
143  image->strlength = instore[1].strlength;
144 }
