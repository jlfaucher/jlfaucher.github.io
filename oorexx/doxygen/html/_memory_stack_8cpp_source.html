48 /**
49  * Create a new stack for the memory object from temporary
50  * memory.
51  *
52  * @param size The base size of the object.
53  * @param stksize The number of items we wish to have in the stack.
54  * @param temporary A dummy argument just to get this new method invoked...
55  * causes the stack to be allocated from temporary memory.
56  *
57  * @return A newly allocated stack object.
58  */
59 void *LiveStack::operator new(size_t size, size_t stksize)
60 {
61  // This is a special allocation. We use this if we need to expand the livestack
62  // during a GC operation, which of course is when we are not able to allocate from
63  // the Rexx heap.
64  return memoryObject.temporaryObject(size + ((stksize-1) * sizeof(RexxObject *)));
65 }

93  * Reallocate the stack to one that is larger by a given multiplier.
94  *
95  * @param delta The size to expand by
96  *
97  * @return A newly allocated stack with the entries from this
98  * stack copied over to it.
99  */
100 LiveStack *LiveStack::reallocate(size_t delta)
101 {
102  // create a new stack that is larger by the given multiplier
103  LiveStack *newStack = new (size + delta) LiveStack (size + delta);

110 /**
111  * Ensure that the live stack is going to be large enough to fit
112  * all of the objects that might be marked from a large
113  * collection. This allows us to proactively expand the stack
114  * before a GC event is taking place, and potentially raise a
115  * real Rexx EOM condition before we're at a critical point.
116  *
117  * @param needed The number of live stack slots this object
118  * might require. It is assumed that the stack
119  * needs to be expanded at this point.
120  *
121  * @return A newly allocated stack with the entries from this
122  * stack copied over to it.
123  */
124 LiveStack *LiveStack::ensureSpace(size_t needed)
125 {
126  // NB: we've already determined expansion is needed, so
127  // this should be non-zero. We only expand by LiveStackSize increments
128  size_t shortFall = Memory::roundUp(needed - size, Memory::LiveStackSize);
129 
130  // now expand by that increment
131  return reallocate(shortFall);
132 }

191 /**
192  * Remove an element from the stack and NULL out the stack slot.
193  *
194  * @param element The element to remove.
195  * @param search indicates whether we should search the stack for the item.
196  */
197 void PushThroughStack::remove(RexxInternalObject *element, bool search)
198 {
199  // first check the top item (which is easy)...we just leave the empty slot
200  // if we found it.
201  if (stack[current] == element)
202  {
203  stack[current] = OREF_NULL;
204  }
205  else
206  {
207  // not top element, search it if requested
208  if (search)
209  {
210  for (size_t i= 0; i < size; i++)
211  {
212  if (stack[i] == element)
213  {
214  stack[i] = OREF_NULL;
215  break;
216  }
217  }
218  }
219  }
220 }
