66 /**
67  * Test is a string is an abbreviation of a target string.
68  *
69  * @param info The test string. We're testing if info is an
70  * abbreviation of the receiving string.
71  * @param _length The minumum length for being a valid abbreviation.
72  *
73  * @return .true if this qualfies as an abbreviation, .false if not.
74  */
75 RexxObject *RexxString::abbrev(RexxString *info, RexxInteger *_length)
76 {
77  // make sure this is a good string argument
78  info = stringArgument(info, ARG_ONE);
79  size_t len2 = info->getLength();
80 
81  // get the optional minimum length...the default is the length of the
82  // test string.
83  size_t checkLen = optionalLengthArgument(_length, len2, ARG_TWO);
84  size_t len1 = getLength();
85 
86  // if info is a null string and the minimum length is zero, then
87  // this passes.
88  if (checkLen == 0 && len2 == 0)
89  {
90  return TheTrueObject;
91  }
92 
93  // we can reject some possibilities based on length checks
94  // alone:
95  // 1) the receiving string is a null string and the info is non-null
96  // 2) the info length is less than the minimum
97  // 3) the info length is longer than the receiving length
98  if (len1 == 0 || (len2 < checkLen) || (len1 < len2))
99  {
100  return TheFalseObject;
101  }
102 
103  // we do a comparison for the info length
104  return booleanObject(memcmp(getStringData(), info->getStringData(), len2) == 0);
105 }

150 /**
151  * Compare two strings, returning the point where they differ.
152  *
153  * @param string2 the other comparison string.
154  * @param pad The optional pad character.
155  *
156  * @return The point of mismatch, or 0 if the two strings
157  * compare equal.
158  */
159 RexxInteger *RexxString::compare(RexxString *other, RexxString *pad)
160 {
161  size_t length1 = getLength();
162  other = stringArgument(other, ARG_ONE);
163  size_t length2 = other->getLength();
164  char padChar = optionalPadArgument(pad, ' ', ARG_TWO);
165 
166  const char *string1;
167  const char *string2;
168  size_t leadLength;
169  size_t padLength;
170 
171  // padding decisions and compare lengths are based on the shorter of
172  // the two strings.
173  if (length1 > length2)
174  {
175  string1 = getStringData();
176  string2 = other->getStringData();
177  leadLength = length2;
178  padLength = length1 - length2;
179  }
180  else
181  {
182  string2 = getStringData();
183  string1 = other->getStringData();
184  leadLength = length1;
185  padLength = length2 - length1;
186  }
187 
188  // scan the leading length for a mismatch
189  for (size_t i = 0; i < leadLength; i++)
190  {
191  // if the strings do not compare equal, return this index position
192  // as the result.
193  if (string1[i] != string2[i])
194  {
195  return new_integer(i + 1);
196  }
197  }
198 
199  // step to the trailing portion, if any
200  string1 += leadLength;
201  for (size_t i = 0; i < padLength; i++)
202  {
203  // if this does not match the pad character, return the position
204  if (string1[i] != padChar)
205  {
206  return new_integer(leadLength + i + 1);
207  }
208  }
209 
210  // no mismatches found, return zero as a result
211  return IntegerZero;
212 }

281 /**
282  * The copies method.
283  *
284  * @param _copies The number of copies of the string to make.
285  *
286  * @return The string with the copies.
287  */
288 RexxString *RexxString::copies(RexxInteger *_copies)
289 {
290  // the copy count is required, and must
291  size_t count = nonNegativeArgument(_copies, ARG_ONE);
292  size_t len = getLength();
293 
294  // if no copies have been requested or we're making copies of
295  // a null string, the result is a null string
296  if (count == 0 || len == 0)
297  {
298  return GlobalNames::NULLSTRING;
299  }
300 
301  // if just a single copy requested, we can just return this string
302  if (count == 1)
303  {
304  return this;
305  }
306 
307  // get a result string large enough for the requested number of copies
308  RexxString *retval = raw_string(len * count);
309  StringBuilder builder(retval);
310 
311  // copy in count times
312  while (count--)
313  {
314  builder.append(getStringData(), len);
315  }
316  return retval;
317 }

442 /**
443  * Change occurrences of values in a string into another value.
444  *
445  * @param needle The search needle we're changing.
446  * @param newNeedle The replacement string
447  * @param countArg The optional count of occurrences to change.
448  *
449  * @return A new string with the replacements made.
450  */
451 RexxString *RexxString::changeStr(RexxString *needle, RexxString *newNeedle, RexxInteger *countArg)
452 {
453  // both needles must be string arguments
454  needle = stringArgument(needle, ARG_ONE);
455  newNeedle = stringArgument(newNeedle, ARG_TWO);
456 
457  // we'll only change up to a specified count. If not there, we do everything.
458  size_t count = optionalNonNegative(countArg, Numerics::MAX_WHOLENUMBER, ARG_THREE);
459  // if no change is requested, return the original string.
460  if (count == 0)
461  {
462  return this;
463  }
464 
465  // find the number of matches (up to count) in the string
466  count = StringUtil::countStr(getStringData(), getLength(), needle, count);
467 
468  // if we don't have anything we can change, return the original string.
469  if (count == 0)
470  {
471  return this;
472  }
473 
474  size_t needleLength = needle->getLength();
475  size_t newLength = newNeedle->getLength();
476  // get the proper size result string based on the number of changes we need to make.
477 
478  RexxString *result = raw_string(getLength() - (count * needleLength) + (count * newLength));
479  StringBuilder builder(result);
480 
481  const char *source = getStringData();
482 
483  const char *newPtr = newNeedle->getStringData();
484  size_t start = 0;
485  // now make changes until we either hit our target count
486  for (size_t i = 0; i < count; i++)
487  {
488  // find the next occurrence. Since we already know how many
489  // occurrences there are, we should never not get a hit
490  size_t matchPos = pos(needle, start);
491  // get the prefix length to copy
492  size_t copyLength = (matchPos - 1) - start;
493  builder.append(source + start, copyLength);
494  // not changing to a null string? copy over an occurrence of the new value.
495  builder.append(newPtr, newLength);
496 
497  // step to the next search start position
498  start = matchPos + needleLength - 1;
499  }
500 
501  // have something left at the end, copy on to the end of the string
502  builder.append(source + start, getLength() - start);
503  return result;
504 }

670 /**
671  * Translate characters in a string into another character.
672  *
673  * @param tableo The output translate table.
674  * @param tablei The input translate table,
675  * @param pad The pad character.
676  * @param _start The starting position for the translate.
677  * @param _range The length to apply the translation to.
678  *
679  * @return The string with the translations applied.
680  */
681 RexxString *RexxString::translate(RexxString *tableo, RexxString *tablei, RexxString *pad,
682  RexxInteger *_start, RexxInteger *_range)
683 {
684  // if there are no input or output table specified, and no pad, this is just an uppercase.
685  if (tableo == OREF_NULL && tablei == OREF_NULL && pad == OREF_NULL)
686  {
687  return upperRexx(_start, _range);
688  }
689 
690  // get our tables...a null string is the default, which means we use the
691  // pad character for everything.
692  tableo = optionalStringArgument(tableo, GlobalNames::NULLSTRING, ARG_ONE);
693  size_t outTableLength = tableo->getLength();
694
