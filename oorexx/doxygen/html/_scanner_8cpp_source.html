166 /**
167  * Find the next special character and validate
168  * against a target character.
169  *
170  * @param target The type of special we expect to find.
171  * @param location A location object to record the object location against.
172  *
173  * @return true if we found the special, false otherwise.
174  */
175 bool LanguageParser::nextSpecial(unsigned int target, SourceLocation &location )
176 {
177  // find the start of the next token.
178  unsigned int inch;
179 
180  // locate the next token. Blanks are not significant here
181  CharacterClass tokenClass = locateToken(inch, false);
182  /* have something else on this line? */
183  if (tokenClass != CLAUSE_EOF && tokenClass != CLAUSE_EOL)
184  {
185  // is the next character a match for our target?
186  if (getChar() == target)
187  {
188  // step the position and record where we are as the end position
189  stepPosition();
190  endLocation(location);
191  return true;
192  }
193  }
194  return false; // didn't find the one we're looking for
195 }

198 /**
199  * Scan over a Rexx comment, including checking for
200  * comment nesting.
201  */
202 void LanguageParser::scanComment()
203 {
204  // comments can nest, so we need to keep track of our nesting level while
205  // parsing.
206  int level = 1;
207  // step over the starting comment delimiter
208  stepPosition(2);
209 
210  // we'll need to remember the starting lineNumber if we have an error.
211  size_t startLine = lineNumber;
212 
213  // loop until our nesting level goes to zero. We can also terminate
214  // via an error if we hit the end of the file without finding the
215  // closing delimiter.
216  while (level > 0)
217  { /* while still in a comment nest */
218 
219  // hit the end of a line?, just step to the next line and keep going
220  if (!moreChars())
221  {
222  nextLine();
223 
224  // but out of lines? We've got a missing comment terminator
225  if (!moreLines())
226  {
227  // record the ending position in our current clause
228  clause->setEnd(lineNumber, lineOffset);
229  // update the error information
230  clauseLocation = clause->getLocation();
231  syntaxError(Error_Unmatched_quote_comment, new_integer(startLine));
232  }
233  // keep scanning for the terminator
234  continue;
235  }
236 
237  // get the current char and step the position.
238  unsigned int inch = nextChar();
239  // at the end delimiter
240  if (inch == '*' && getChar() == '/')
241  {
242  // this is our closer, step over the position and reduce the level.
243  stepPosition();
244  level--;
245  }
246  // start of a new nested character?
247  else if (inch == '/' && getChar() == '*')
248  {
249  // we have a new nesting level to process
250  stepPosition();
251  level++;
252  }
253  }
254 }

257 /**
258  * Locate the next "real" token in the parsing stream,
259  * skipping extra blanks and comments.
260  *
261  * @param character If this is a "normal" character, this is the character value.
262  * @param blanksSignificant
263  * Indicates if blank characters should be considered as
264  * significant in the current context.
265  *
266  * @return A type of the next token character. If the type is
267  * NORMAL_CHAR, the character is also returned in the
268  * character reference argument.
269  */
270 CharacterClass LanguageParser::locateToken(unsigned int &character, bool blanksSignificant)
271 {
272  // default to having an invalid character
273  character = INVALID_CHARACTER;
274 
275  // no more lines? indicate we've hit the end of the file.
276  if (!moreLines())
277  {
278  return CLAUSE_EOF;
279  }
280  // or, we could be at the end of the line...still a clause terminator,
281  // but for a different reason.
282  else if (!moreChars())
283  {
284  return CLAUSE_EOL;
285  }
286 
287  // ok, we will scan as long as we have lines left.
288  while (moreChars())
289  {
290  // next character from the line.
291  unsigned char inch = getChar();
292  // a recognized whitespace character?
293  if (inch==' ' || inch=='\t')
294  {
295  // if blanks are significant, then return the indicator.
296  if (blanksSignificant)
297  {
298  return SIGNIFICANT_BLANK;
299  }
300  else
301  {
302  // an ignorable blank...just step over it and try for another
303  stepPosition();
304  }
305  }
306  // possible continuation character? We accept either ',' or '-' as a
307  // continuation, but we also need to recognize that "--" is a line comment.
308  else if (inch == ',' || inch == '-')
309  {
310  // line comment? Just truncate the line and process as if
311  // we ran out of characters and hit the end of the line.
312  if (inch == '-' && followingChar() == '-')
313  {
314  truncateLine();
315  return CLAUSE_EOL;
316  }
317 
318  // assume for now that this is a real character...we need to
319  // check for an end-of-line, ignoring any following blanks or comments.
320  character = inch;
321 
322  // remember the current positions
323  size_t startOffset = lineOffset;
324  size_t startLine = lineNumber;
325  // step over the current position to continue scanning.
326  stepPosition();
327 
328  // skip blanks and tokens. If we find anything else, we backup and treat
329  // the character we found as a real character.
330  while (true)
331  {
332  // hit the end of the clause while scanning for a continuation?
333  if (!moreChars())
334  {
335  // we've hit the end of the line without finding anything
336  // other than comments and white space. If there are
337  // more lines in the file, we step to the next line.
338  // The continuation is functionally equivalent to a
339  // blank, so if blanks are significant, we can stop now,
340  // otherwise we need to continue scanning for a token character.
341  if (moreLines())
342  {
343  nextLine();
344  if (blanksSignificant)
345  {
346  return SIGNIFICANT_BLANK;
347  }
348  }
349  break;
350  }
351 
352  // grab the next character and check for comments or white space
353  unsigned int inch2 = getChar();
354 
355  // blanks are common, so check them first.
356  if (inch2 == ' ' || inch2 == '\t')
357  {
358  stepPosition();
359  continue;
360  }
361  // start of a traditional comment type?
362  // we just step over the comment and continue. Note, this might
363  // actually step one or more lines...that's fine, the continuation still
364  // holds.
365  if (inch2 == '/' && followingChar() == '*')
366  {
367  scanComment();
368  continue;
369  }
370 
371  // line comment after a continuation char? Truncate the line at this
372  // position and loop around again to the end-of-line test so we
373  // will handle the continuation.
374  if (inch2 == '-' && followingChar() == '-')
375  {
376  truncateLine();
377  continue;
378  }
379 
380  // found something other than ignorable stuff, so the ',' or '-' is
381  // a real character. Return this now, after stepping back to the
382  // saved position
383  position(startLine, startOffset);
384  character = inch;
385  return NORMAL_CHAR;
386  }
387  }
388  // could be a comment start. Swallow that and continue. Comments
389  // are always ignored. Note that his could leave us on a new line.
390  else if (inch == '/' && followingChar() == '*')
391  {
392  scanComment();
393  }
394  else
395  {
396  // got a normal character, so we're done.
397  character = inch;
398  return NORMAL_CHAR;
399  break;
400  }
401  }
402 
403  // hit the end of the line without finding anything. Indicate end of clause
404  return CLAUSE_EOL;
405 }

408 /**
409  * Convert and check a hex constant, packing it down into a
410  * string object.
411  *
412  * @param start the start of the string data in the current
413  * source line.
414  *
415  * @param length The length of the raw string data.
416  *
417  * @return A Rexx string packed into 8-bit character form.
418  */
419 RexxString* LanguageParser::packHexLiteral(size_t start, size_t length)
420 {
421  // ""X is legal...it is just a null string
422  // the rest is not nearly as easy :-)
423  if (length == 0)
424  {
425  return GlobalNames::NULLSTRING;
426  }
427 
428  // this is our counter for group packing
429  int groupCount = 0;
430  // our count of nibble characters we find...we can calculate the result length from this
431  int nibbleCount = 0;
432  // a pointer for scanning the data
433  // scanning right-to-left makes it easier to identify correct whitespace positioning
434  const char *inPointer = current + start + length - 1;
435 
436  // update the current clause location in case there are any errors
437  clauseLocation = clause->getLocation();
438 
439  // first scan is to check REXX rules for validity of grouping
440 
441  // scan the entire input string from right to left
442  for (size_t i = length; i > 0; i--)
443  {
444  // do we have a white space character?
445  if (*inPointer == ' ' || *inPointer == '\t')
446  {
447  // now check to see if this is in a valid position. We do not allow
448  // blanks at the start or the end of the string, and blanks may
449  // only appear at even hex digit (byte) boundaries.
450  if (i == 1 || i == length) // no blank at string start or end
451  {
452  syntaxError(Error_Invalid_hex_hexblank, new_integer(i));
453 
454  }
455  // not evenly divisible by two...bad blank placement
456  else if ((groupCount & 1) != 0)
457  {
458  syntaxError(Error_Invalid_hex_invhex_group);
459  }
460 
461  // we start a new group now
462  groupCount = 0;
463  }
464 
465  // non-blank character...for now, just count how many we have.
466  else
467  {
468  // keep track of how large this group is and how many
469  // total nibbles we have.
470  groupCount++;
471  nibbleCount++;
472  }
473 
474  inPointer--; // step the input position
475  }
476 
477  // second scan is to create the string value determined by the
478  // hex constant.
479 
480  // reset the scan pointer
481  inPointer = current + start;
482 
483  // this tells us how to process the first nibble
484  size_t nibbleStart = (nibbleCount & 1);
485 
486  // get the final value size, rounding up if there are an odd
487  // number of nibbles. The first grouping of the string can have
488  // an odd numbers, allowing things like 'a'x rather than '0a'x.
489  size_t characterCount = nibbleCount / 2 + nibbleStart;
490 
491  // allocate a string we can pack the nibbles into
492  RexxString *value = raw_string(characterCount);
493 
494  // this is for poking characters into the result
495  size_t outPosition = 0;
496 
497  // scan the string again, packing the result number of
498  // characters.
499  for (size_t i = 0; i < characterCount; i++)
500  {
501  unsigned char byte = 0;
502 
503  // get the nibble character and pack it.
504  unsigned char nibble = (unsigned char)*inPointer;
505  // scan over white space, if we're there.
506  while (nibble == ' ' || nibble == '\t')
507  {
508  nibble = (unsigned char)*(++inPointer);
509  }
510 
511  // if we had an odd number of nibbles, the first time
512  // we execute this loop, we'll only process 1 nibble. After
513  // the first time, we'll always do two at a time. Since
514  // we validated that the groups have an even number of non-blank
515  // characters after the first, this will handle things fine.
516  // Note also, since we do not allow leading blanks, we'll be
517  // positioned correctly at the start.
518  for (size_t k = nibbleStart; k < 2; k++)
519  {
520  // this should be a real character now
521  nibble = (unsigned char)*inPointer++;
522  // validate the digit and convert to a base binary value.
523  // regular digit
524  if (nibble >= '0' && nibble <= '9')
525  {
526  nibble -= '0';
527  }
528  // lowercase hex
529  else if (nibble >= 'a' && nibble <= 'f')
530  {
531  nibble -= 'a';
532  nibble += 10;
533  }
534  // uppercase hex
535  else if (nibble >= 'A' && nibble <= 'F')
536  {
537  nibble -= 'A';
538  nibble += 10;
539  }
540  else
541  {
542  // invalid character
543  clauseLocation = clause->getLocation();
544  char errorOutput = (char)nibble;
545  syntaxError(Error_Invalid_hex_invhex, new_string(&errorOutput, 1));
546  }
547  // shift over the last nibble and add in the new one
548  byte <<= 4;
549  byte += nibble;
550  }
551  // we always process two nibbles at a time from here.
552  nibbleStart = 0;
553  // store in the output string
554  value->putChar(outPosition++, byte);
555  }
556 
557 // return this...the caller will handle making this a common string.
558  return value;
559 }

613  }

617  }

620  else

633

635  inPointer = current + start;
636 
637  // this tells us how to process the first byte
638  size_t byteSize = (bitCount & 7);
639 
640  // get the final value size, rounding up if there are extra bits

642  // an odd number of bits.

645

647  // first byte size. After the first group, we always process
648  // in groups of 8.
649  if (byteSize == 0)
650  {

652  }
653 
654  // allocate a string we can pack the nibbles into
655  RexxString *value = raw_string(characterCount);
656 
657  // this is for poking characters into the result
658  size_t outPosition = 0;
659 
660  // scan the string again, packing the result number of
661  // characters.
662  for (size_t i = 0; i < characterCount; i++)
663  {
664  unsigned char byte = 0;
665 
666  // if we had an odd number of bits, the first time
667  // we execute this loop, we'll process fewer than 8 bits. After
668  // the first time, we'll always do eight at a time. Since
669  // we validated that the groups have an even number of non-blank
670  // characters after the first, this will handle things fine.
671  // Note also, since we do not allow leading blanks, we'll be
672  // positioned correctly at the start.
673  for (size_t k = 0; k < byteSize; k++)
674  {
675  // get the bit character and pack it.
676  unsigned char bit = (unsigned char)*inPointer++;
677  // We can have white space between nibbles, so we need to do this here.
678  while (bit == ' ' || bit == '\t')
679  {
680  bit = (unsigned char)*inPointer++;
681  }
682  // shift our accumulator
683  byte <<= 1;
684  // if this is a one bit, add this in
685  if (bit == '1')
686  {
687  byte++;
688  }
689  // other option is a 0, else give an error
690  else if (bit != '0')
691  {
692  // invalid character
693  clauseLocation = clause->getLocation();
694  char errorOutput = (char)bit;
695  syntaxError(Error_Invalid_hex_invbin, new_string(&errorOutput, 1));
696  }
697  }
698  // we always process 8 bits after the first byte
699  byteSize = 8;
700  // store in the output string
701  value->putChar(outPosition++, byte);
702  }
703 
704  // return this...the caller will handle making this a common string.
705  return value;
706 }

1201 /**
1202  * Scan a symbol from the source and return a token
1203  * identifying the type of symbol (constant, dot, variable, etc.).
1204  *
1205  * @return A token object describing the symbol.
1206  */
1207 RexxToken *LanguageParser::scanSymbol()
1208 {
1209 
1210  // we're in a clean scan state now
1211  SymbolScanState state = EXP_START;
1212  size_t eoffset = 0; // position of exponential sign for backing up.
1213  size_t start = lineOffset; // remember token start position
1214  int dotCount = 0; // no periods yet
1215 
1216  // set the start position for the token
1217  SourceLocation location;
1218  startLocation(location);
1219 
1220  unsigned int inch = getChar(); // ok, get the current character to start this off
1221  // ok, loop through the token until we've consumed it all.
1222  for (;;)
1223  {
1224  // keep a count of periods...we use this to determine stem/compound and numeric values
1225  if (inch == '.')
1226  {
1227  dotCount++;
1228  }
1229 
1230  // finite state machine to establish numeric constant (with possible
1231  // included sign in exponential form)
1232 
1233  switch (state)
1234  {
1235  // this is our beginning state...we know nothing about this symbol yet.
1236 
1237  case EXP_START:
1238  {
1239  // have a digit at the start? Potential number, so
1240  // we're looking for digits here.
1241  if (inch >= '0' && inch <= '9')
1242  {
1243  state = EXP_DIGIT;
1244  }
1245  // if this is a dot, then we've got a starting decimal
1246  // point. This could be a number or an environment symbol
1247  else if (inch == '.')
1248  {
1249  state = EXP_SPOINT;
1250  }
1251  // a non-numeric character. A number is not possible.
1252  else
1253  {
1254  state = EXP_EXCLUDED;
1255  }
1256  break;
1257  }
1258 
1259  // we're scanning digits, still potentially a number.
1260  case EXP_DIGIT:
1261  {
1262  // is this a period? Since we're scanning digits, this
1263  // is must be the first period and is a decimal point.
1264  // switch to scanning the part after the decimal.
1265  if (inch=='.')
1266  {
1267  state = EXP_POINT;
1268  }
1269  // So far, the form is "digitsE"...this can still be a number,
1270  // but know we're looking for an exponent.
1271  else if (inch=='E' || inch == 'e')
1272  {
1273  state = EXP_E;
1274  }
1275  // other non-digit? We're no longer scanning a number.
1276  else if (inch < '0' || inch > '9')
1277  {
1278  state = EXP_EXCLUDED;
1279  }
1280  // if we encounter a digit, the state is unchanged
1281  break;
1282  }
1283 
1284  // we're scanning from a leading decimal point. How we
1285  // go from here depends on the next character.
1286  case EXP_SPOINT:
1287  {
1288  // not a digit immediately after the period, we're
1289  // scanning a normal symbol from here.
1290  if (inch < '0' || inch > '9')
1291  {
1292  state = EXP_EXCLUDED; /* not a number */
1293  }
1294  // second character is a digit, so we're scanning the
1295  // part after the decimal.
1296  else
1297  {
1298  state = EXP_POINT;
1299  }
1300  break;
1301  }
1302 
1303  // scanning after a decimal point. From here, we could hit
1304  // the 'E' for exponential notation.
1305  case EXP_POINT:
1306  {
1307  // potential exponential, switch scan to the exponent part.
1308  if (inch == 'E' || inch == 'e')
1309  {
1310  state = EXP_E;
1311  }
1312  // non-digit other than an 'E'?, no longer a valid numeric.
1313  else if (inch < '0' || inch > '9')
1314  {
1315  state = EXP_EXCLUDED;
1316  }
1317  // if we find a digit, the state is unchanged.
1318  break;
1319  }
1320 
1321  // we have a valid number up to an 'E'...now we can have digits or
1322  // a sign for the exponent. The digit will be handled here, but
1323  // the +/- is either a symbol terminator or part of the symbol.
1324  // we check that at the end-of-symbol processing
1325  case EXP_E:
1326  {
1327  // switching to process the exponent digits
1328  if (inch >= '0' && inch <= '9')
1329  {
1330  state = EXP_EDIGIT;
1331  }
1332  // we handle the sign situation below.
1333  break;
1334  }
1335 
1336  // we're scanning a potential numeric value, and we've just
1337  // had the sign, so we're looking for digits after that. If there
1338  // are no digits, then the sign actually terminated the symbol, so
1339  // we need to back up.
1340  case EXP_ESIGN:
1341  {
1342  // found a digit here? switching into exponent scan mode.
1343  if (inch >= '0' && inch <= '9')
1344  {
1345  state = EXP_EDIGIT;
1346  }
1347  else
1348  {
1349  // non-digit cannot be a number.
1350  state = EXP_EXCLUDED;
1351  }
1352  break;
1353  }
1354 
1355  // scanning for exponent digits. No longer numeric if we find a non-digit.
1356  case EXP_EDIGIT:
1357  {
1358  if (inch < '0' || inch > '9')
1359  {
1360  state = EXP_EXCLUDED;
1361  }
1362  break; /* go get the next character */
1363  }
1364 
1365  // once EXP_EXCLUDED is reached the state doesn't change. We're
1366  // just consuming symbol characters from here.
1367  case EXP_EXCLUDED:
1368  {
1369  break; /* go get the next character */
1370  }
1371  }
1372 
1373  // handled all of the states, now handle the termination checks.
1374  stepPosition();
1375 
1376  // did we step past an exponential sign but found an invalid exponent?
1377  if (eoffset != 0 && state == EXP_EXCLUDED)
1378  {
1379  // we need to back up the scan pointer to the sign position and
1380  // stop...this is the end of the symbol.
1381  lineOffset = eoffset;
1382  break; /* and we're finished with this */
1383  }
1384 
1385  // have we reached the end of the line? Also done.
1386  if (!moreChars())
1387  {
1388  break;
1389  }
1390 
1391  // get the next character and validate as a symbol character.
1392  inch = getChar();
1393  // if this was a good symbol character, run around the loop again and
1394  // see how this impacts the state machine
1395  if (translateChar(inch) != 0)
1396  {
1397  continue;
1398  }
1399 
1400  // we have a non-symbol character abutting the symbol characters. If
1401  // we just scanned the 'E' (or 'e') in a potential exponent number,
1402  // a '+' or '-' is potentially part of the symbol value.
1403  if (state == EXP_E && (inch == '+' || inch == '-'))
1404  {
1405  // the sign might be at the end of the line. If there
1406  // are no characters after that, no point in switching states.
1407  if (!haveNextChar())
1408  {
1409  // this is not a number and we've found the end position
1410  state = EXP_EXCLUDED;
1411  break;
1412  }
1413 
1414  // this only works if there are only digits after this point.
1415  // we need to remember this position in case we have to back up.
1416  eoffset = lineOffset;
1417  // step past the sign and switch the scanning state to look for
1418  // the exponent digits after a sign.
1419  stepPosition();
1420  state = EXP_ESIGN;
1421 
1422  // everything is all set up so we can back up. Now get the next
1423  // character and see how things go from here.
1424  inch = getChar();
1425  // if this was a good symbol character, run around the loop again and
1426  // see how this impacts the state machine
1427  if (translateChar(inch) != 0)
1428  {
1429  continue;
1430  }
1431 
1432  // this is not a number...mark it so and also back up to before
1433  // the sign position.
1434  state = EXP_EXCLUDED;
1435  // we need to back up the scan pointer to the sign position and
1436  // stop...this is the end of the symbol.
1437  lineOffset = eoffset;
1438  break;
1439  }
1440  else
1441  {
1442  // We've reached a non-symbol character. State remains in whatever
1443  // the last state was.
1444  break;
1445  }
1446  }
1447 
1448  // ok, we've located the end position of the symbol token. The final state
1449  // will tell us what sort of symbol we have. Generally, EXP_EXCLUDED indicates
1450  // a non-numeric value, all other states will be valid numbers.
1451 
1452  // lineOffset is now one character past the end of the symbol.
1453  size_t length = lineOffset - start;
1454  // we're going to need to copy this into a string object.
1455  RexxString *value = raw_string(length);
