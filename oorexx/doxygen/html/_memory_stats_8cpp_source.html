64 /**
65  * Accumulate memory statistics for a scanned object
66  *
67  * @param memStats The memory stats we're recording into.
68  * @param obj The object we're recording.
69  */
70 void SegmentStats::recordObject(MemoryStats *memStats, RexxInternalObject *obj)
71 {
72  // tally the size of the object
73  size_t bytes = obj->getObjectSize();
74  totalBytes += bytes;
75  // if the object is alive, log the type information.
76  if (obj->isObjectLive(memoryObject.markWord))
77  {
78  memStats->logObject(obj);
79  liveBytes += bytes;
80  liveObjects++;
81  }
82  // this is a dead object
83  else
84  {
85  deadObjects++;
86  deadBytes += bytes;
87  }
88 }
