67  * Determine the location of the running program. This returns the full path
68  * of the currently running executable.
69  *
70  * @return A character string of the path (does not need to be freed by the caller)
71  */
72 const char *SysProcess::getExecutableFullPath()
73 {
74  if (executableFullPath != NULL)
75  {
76  return executableFullPath;
77  }
78 
79  char modulePath[MAX_PATH];
80 
81  // NULL means the current executable
82  GetModuleFileName(NULL, modulePath, sizeof(modulePath));
83 
84  // save a copy of this
85  executableFullPath = strdup(modulePath);
86  return executableFullPath;
87 }

90 /**
91  * Determine the location of the rexxapi.dll library. This returns the
92  * directory portion of the library's path with a trailing backslash.
93  *
94  * @return A character string of the location (does not need to be freed by the caller)
95  */
96 const char *SysProcess::getLibraryLocation()
97 {
98  if (libraryLocation != NULL)
99  {
100  return libraryLocation;
101  }
102 
103  // look up rexxapi.dll
104  HMODULE module = GetModuleHandle("rexxapi");
105 
106  char modulePath[MAX_PATH];
107  GetModuleFileName(module, modulePath, sizeof(modulePath));
108 
109  size_t pathLength = strlen(modulePath);
110 
111  // scan backwards to find the last directory delimiter
112  for (; pathLength > 0; pathLength--)
113  {
114  // is this the directory delimiter?
115  if (modulePath[pathLength - 1] == '\\')
116  {
117  // terminate the string after the first encountered backslash and quit
118  modulePath[pathLength] = '\0';
119  break;
120  }
121  }
122 
123  // belt-and-braces, make sure we found a directory
124  if (pathLength == 0)
125  {
126  return NULL;
127  }
128 
129  // save a copy of this
130  libraryLocation = strdup(modulePath);
131  return libraryLocation;
132 }
