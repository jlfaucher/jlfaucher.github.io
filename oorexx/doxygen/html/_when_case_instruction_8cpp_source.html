58 RexxInstructionCaseWhen::RexxInstructionCaseWhen(size_t count, QueueClass *list, RexxToken *thenToken)
59 {
60  expressionCount = count;
61 
62  // now copy the expressions from the sub term stack
63  // NOTE: The expressionss are in last-to-first order on the stack.
64  initializeObjectArray(count, expressions, RexxInternalObject, list);
65 
66  //get the location from the THEN token and use its location to set

68  // own, but using the start of the THEN gives a fuller picture of things.

134 /**
135  * Execute a WHEN instruction attached to a SELECT CASE.
136  *
137  * @param context The current execution context.
138  * @param stack The current evaluation stack.
139  */
140 void RexxInstructionCaseWhen::execute(RexxActivation *context, ExpressionStack *stack)
141 {
142  context->traceInstruction(this);
143 
144  // This should be us. It really isn't possible to jump into a middle of a select
145  // and get to a WHEN without raising an error.
146  DoBlock *doBlock = context->topBlockInstruction();
147  // get the case expression
148  RexxObject *caseValue = doBlock->getCase();
149 
150  for (size_t i = 0; i < expressionCount; i++)
151  {
152  // and the compare target (which needs tracing, but only as an intermediate
153  RexxObject *compareValue = expressions[i]->evaluate(context, stack);
154  context->traceResult(compareValue);
155  // now perform the compare using the "==" operator method.
156  // NOTE that the case value is the left hand side.
157  RexxObject *result = caseValue->callOperatorMethod(OPERATOR_STRICT_EQUAL, compareValue);
158  context->traceResult(result);
159 
160  // Remove the compare object from the stack
161  stack->toss();
162 
163  // evaluate and decide if we execute this WHEN...we stop with the first true result
164  if (result->truthValue(Error_Logical_value_when_case))
165  {
166  // do a pause and return
167  context->pauseInstruction();
168  return;
169  }
170  }
171 
172  // we execute the ELSE branch if everything evaluates to false
173  context->setNext(else_location->nextInstruction);
174  context->pauseInstruction();
175 }
