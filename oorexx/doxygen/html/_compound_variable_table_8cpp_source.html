64 /**
65  * Copy all of the entries from another compound table into this
66  * table.
67  *
68  * @param other The source tail collection.

70 void CompoundVariableTable::copyFrom(CompoundVariableTable &other)
71 {
72  // do a traversal of the other table an insert an identical entry into our
73  // table.
74  CompoundTableElement *entry = other.first();
75  while (entry != OREF_NULL)
76  {
77  // create an entry in this table with the same tail name
78  CompoundTableElement *newEntry = findEntry(entry->getName(), true);
79  // copy the value into the entry, and continue the iteration
80  newEntry->setValue(entry->variableValue);
81  entry = other.next(entry);
82  }
83 }

114 /**
115  * Locate a compound variable item and optionally create
116  * a new table entry if not found.
117  *
118  * @param tail The constructed tail for comparisons.
119  * @param create indicates whether we create a new entry if this is not found.
120  *
121  * @return The entry for the given name, or NULL if not found and
122  * not asked to create the entry.
123  */
124 CompoundTableElement *CompoundVariableTable::findEntry(CompoundVariableTail &tail, bool create)
125 {
126  CompoundTableElement *anchor = root; // get our anchor position and keep a previous
127  CompoundTableElement *previous = anchor; // pointer for backing up and insertions.
128 
129  int rc = 0; // used for the traversal, but also used for insertion afterward
130 
131  // do a standard binary tree search.
132  while (anchor)
133  {
134  // do the name comparison with the tail
135  // our comparison is done first on length, second on value.
136  // the rc will be 1, 0, or -1 based on which name is the "larger".
137  rc = tail.compare(anchor->getName());
138  // compare greater, take the right branch
139  if (rc > 0)
140  {
141  previous = anchor;
142  anchor = anchor->right;
143  continue;
144  }
145  // if the target is shorter, go to the left
146  else if (rc < 0)
147  {
148  previous = anchor;
149  anchor = anchor->left;
150  continue;
151  }
152  // the names matched, found the one we're looking for.
153  else
154  {
155  return anchor;
156  }
157  }
158 
159  // if not a create operation, we return a failure result
160  if (!create)
161  {
162  return OREF_NULL;
163  }
164 
165 
166  // need to insert a new element into the tree
167  anchor = new_compoundElement(tail.makeString());
168  // no previous means the tree was empty, so this is our new
169  // anchor value
170  if (!previous)
171  {
172  // the anchor does not have a parent for backing up.
173  anchor->setParent(OREF_NULL);
174  setRoot(anchor);
175  }
176  else
177  {
178  // a real insertion. hook us up to the parent
179  anchor->setParent(previous);
180  // we use the last decision point to determine whether
181  // we're setting the left child or the right child. We got
182  // here because that child link was null.
183  previous->setChild(rc, anchor);
184  // balance the tree from the inserted node, if necessary
185  balance(anchor);
186  }
187  return anchor;
188 }

191 /**
192  * Balance the compound variable tree.
193  *
194  * @param node The location of the just inserted node. We balance
195  * from here.
196  */
197 void CompoundVariableTable::balance(CompoundTableElement *node)
198 {
199  // if this is the root node, nothing to do here.
200  if (node == root)
201  {
202  return;
203  }
204 
205  // we want the inserted node's parent. Since this is not the
206  // root node, this is non-null.
207  CompoundTableElement *parentNode = node->parent;
208  // the initial depth is 1
209  unsigned short depth = 1;
210 
211  while (parentNode != OREF_NULL)
212  {
213  if (parentNode->isRightChild(node))
214  {
215  // update the parent nodes depth
216  parentNode->rightDepth = depth;
217 
218  unsigned short workingDepth = parentNode->leftDepth + 1;
219  // if the updated depth is greater than one more than the left
220  // depth, we need to move stuff around
221  if (depth > workingDepth)
222  {
223  // we need to adjust the right branch
224  moveNode(parentNode, false);
225  // adjust our depth and go around again.
226  depth = parentNode->rightDepth;
227  }
228  else
229  {
230  // the left is shorter, we're still playing catchup. No
231  // balancing required.
232  if (workingDepth < depth)
233  {
234  return ;
235  }
236  }
237  }
238  // coming up from the left
239  else
240  {
241  // same process as the other branch
242  parentNode->leftDepth = depth;
243  unsigned short workingDepth = parentNode->rightDepth + 1;
244  if (depth > workingDepth)
245  {
246  // adjusting the left depth this time
247  moveNode(parentNode, true);
248  depth = parentNode->leftDepth;
249  }
250  else
251  {
252  // no adjustment needed
253  if (workingDepth < depth)
254  {
255  return ;
256  }
257  }
258  }
259  depth++; // increment the depth and go up a level to check again.
260  node = parentNode;
261  parentNode = parentNode->parent;
262  }
263 }

266 /**
267  * Move a node of the tree.
268  *
269  * @param anchor The anchor position for the movement (which might
270  * get updated).
271  * @param toright A flag indicating movement to the right or left.
272  */
273 void CompoundVariableTable::moveNode(CompoundTableElement *&anchor, bool toright)
274 {
275  // save where we are
276  CompoundTableElement *temp = anchor;
277  CompoundTableElement *work;
278 
279  // moving to the right?
280  if (toright)
281  {
282  // move the right banch to the left side.
283  work = temp->left;
284  CompoundTableElement *work1 = temp->left = work->right;
285  // adjust the depth values
286  temp->leftDepth = work->rightDepth;
287  // was the right moved. Need to adjust it's parent
288  if (work1 != OREF_NULL)
289  {
290  work1->setParent(temp); /* set its parent correctly */
291  }
292  // the old left node is now the parent of the starting point
293  // and is one level deeper
294  work->setRight(temp);
295  work->rightDepth++;
296  }
297  else
298  {
299  // moving left branch to the rigth
300  work = temp->right;
301  CompoundTableElement *work1 = temp->right = work->left;
302  temp->rightDepth = work->leftDepth;
303  // if we moved a real node, set its parent
304  if (work1 != OREF_NULL)
305  {
306  work1->setParent(temp);
307  }
308  // old right node is now the parent of our starting point
309  work->setLeft(temp);
310  work->leftDepth++;
311  }
312  // now we need to adjust the parents a bit
313  // the new parent of the moved node is the anchor node's
314  // parent (work is now the parent of the anchor node)
315  work->setParent(temp->parent);
316  CompoundTableElement *work2 = temp->parent;
317  temp->setParent(work);
318  // the original parent might have been NULL, which means we just
319  // moved the root element
320  if (work2 == OREF_NULL)
321  {
322  setRoot(work);
323  }
324  // if the anchor was originally on the left, set the new left node
325  else if (work2->isLeftChild(temp))
326  {
327  work2->setLeft(work);
328  }
329  else
330  {
331  work2->setRight(work);
332  }
333  // this is the new anchor position
334  anchor = work;
335 }

358 /**
359  * Find a the "first" leaf node starting from a given
360  * node. The first node is the one that first in iteration
361  * order, which will be the deepest node. Since we balance,
362  * this will either be the left-mode node or the right child
363  * of the left-most node.
364  *
365  * @param node The starting node.
366  *
367  * @return The final end node.
368  */
369 CompoundTableElement *CompoundVariableTable::findLeaf(CompoundTableElement *node)
370 {
371  for (;;)
372  {
373  // go as far left as we can.
374 
375  // It's just a jump to the left...
376  while (node->left != OREF_NULL)
377  {
378  node = node->left;
379  }
380  // if we have no right node, we're done
381  if (node->right == OREF_NULL)
382  {
383  return node;
384  }
385 
386  // And then a step to the right.
387  // With your hand on your hips.
388  // You bring your knees in tight.
389  // But it's the pelvic thrust.
390  // They really drive you insane.
391  // Let's do the Time Warp again.
392  // Let's do the Time Warp again.
393  node = node->right;
394  }
395 }

398 /**
399  * Step to the "next" node while iterating.
400  *
401  * @param node The node we're moving from.
402  *
403  * @return The next node in iteration order.
404  */
405 CompoundTableElement *CompoundVariableTable::next(CompoundTableElement *node)
406 {
407  // get the parent node of our starting point. We're generally
408  // coming up from a leaf node while doing this, so we'll have already
409  // visited the nodes below us.
410  CompoundTableElement *parentNode = node->parent;
411  // not at the root?
412  if (parentNode != OREF_NULL)
413  {
414  // if coming up from the right side, it is time to visit the
415  // parent node.
416  if (parentNode->isRightChild(node))
417  {
418  return parentNode;
419  }
420  // we're coming up from the left side. If the parent node has
421  // a right child, then drill down from that point to find another
422  // leaf node.
423  if (parentNode->right != OREF_NULL)
424  {
425  return findLeaf(parentNode->right);
426  }
427  // coming up from the left and no right child. Time to visit this node.
428  return parentNode;
429  }
430  // we've reached the top, nothing more to do.
431  return OREF_NULL;
432 }

467 /**
468  * Search for a compound entry. This version is optimized for
469  * "find-but-don't create" usage.
470  *
471  * @param tail The tail name we're searching for.
472  *
473  * @return The located position, if any.
474  */
475 CompoundTableElement *CompoundVariableTable::findEntry(CompoundVariableTail &tail)
476 {
477  CompoundTableElement *anchor = root;
478 
479  while (anchor != NULL)
480  {
481  // do the name comparison
482  int rc = tail.compare(anchor->getName());
483  // right side larter
484  if (rc > 0)
485  {
486  anchor = anchor->right;
487  }
488  // left side smaller
489  else if (rc < 0)
490  {
491  anchor = anchor->left;
492  }
493  // equal, honey, I'm home....
494  else
495  {
496  return anchor;
497  }
498  }
499  // not found
500  return OREF_NULL;
501 }
