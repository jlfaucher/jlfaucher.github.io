65 /**
66  * Execute a NUMERIC instruction.
67  *
68  * @param context The current execution context.
69  * @param stack The current evaluation stack.
70  */
71 void RexxInstructionNumeric::execute(RexxActivation *context, ExpressionStack *stack )
72 {
73  // trace if necessary
74  context->traceInstruction(this);
75 
76  // now process the specific functions encoded in the flags
77 
78  // NUMERIC DIGITS
79  if (numericFlags[numeric_digits])
80  {
81  // no expression? Just set digits back to default
82  if (expression == OREF_NULL)
83  {
84  wholenumber_t defaultDigits = context->getPackage()->getDigits();
85  // digits cannot be less than or equal to fuzz
86  if (defaultDigits <= context->fuzz())
87  {
88  reportException(Error_Expression_result_digits, defaultDigits, context->fuzz());
89  }
90  // set the value
91  context->setDigits(defaultDigits);
92  }
93  // expression version
94  else
95  {
96  // need to evaluate
97  RexxObject *result = expression->evaluate(context, stack);
98  context->traceKeywordResult(GlobalNames::DIGITS, result);
99 
100  size_t setting;
101 
102  // this must be an a positive numeric value
103  if (!result->requestUnsignedNumber(setting, number_digits()) || setting < 1)
104  {
105  reportException(Error_Invalid_whole_number_digits, result);
106  }
107  // digits cannot be less than or equal to fuzz
108  if ((wholenumber_t)setting <= context->fuzz())
109  {
110  reportException(Error_Expression_result_digits, setting, context->fuzz());
111  }
112  // set the value
113  context->setDigits(setting);
114  }
115  }
116  // NUMERIC FUZZ
117  else if (numericFlags[numeric_fuzz])
118  {
119  // no expression resets to default
120  if (expression == OREF_NULL)
121  {
122  wholenumber_t defaultFuzz = context->getPackage()->getFuzz();
123  // cannot be greater than or equal to digits
124  if (defaultFuzz >= context->digits())
125  {
126  reportException(Error_Expression_result_digits, context->digits(), defaultFuzz);
127  }
128  // set the value
129  context->setFuzz(defaultFuzz);
130  }
131  else
132  {
133  // get the expression value and convert to a numeric
134  RexxObject *result = expression->evaluate(context, stack);
135  context->traceKeywordResult(GlobalNames::FUZZ, result);
136 
137  size_t setting;
138  /* bad value? */
139  if (!result->requestUnsignedNumber(setting, number_digits()))
140  {
141  reportException(Error_Invalid_whole_number_fuzz, result);
142  }
143 
144  // cannot be greater than or equal to digits
145  if ((wholenumber_t)setting >= context->digits())
146  {
147  reportException(Error_Expression_result_digits, context->digits(), setting);
148  }
149  // change the setting
150  context->setFuzz(setting);
151  }
152  }
153  // NUMERIC FORM
154  else if (numericFlags[numeric_form])
155  {
156  // NON VALUE form?
157  if (expression == OREF_NULL)
158  {
159  // if default form, set that
160  if (numericFlags[numeric_form_default])
161  {
162  context->setForm(context->getPackage()->getForm());
163  }
164  else
165  {
166  // set it to what was specified.
167  context->setForm(numericFlags[numeric_engineering] ? Numerics::FORM_ENGINEERING : Numerics::FORM_SCIENTIFIC);
168  }
169  }
170  else
171  {
172  // evaluate the expression and get as a string value.
173  RexxObject *result = expression->evaluate(context, stack);
174  context->traceKeywordResult(GlobalNames::FORM, result);
175  RexxString *stringResult = result->requestString();
176 
177  // Scientific form?
178  if (stringResult->strCompare(GlobalNames::SCIENTIFIC))
179  {
180  context->setForm(Numerics::FORM_SCIENTIFIC);
181  }
182  // Engineering form?
183  else if (stringResult->strCompare(GlobalNames::ENGINEERING))
184  {
185  context->setForm(Numerics::FORM_ENGINEERING);
186  }
187  else
188  {
189  reportException(Error_Invalid_subkeyword_form, result);
190  }
191  }
192  }
193  context->pauseInstruction(); // do debug pause if necessary
194 }
