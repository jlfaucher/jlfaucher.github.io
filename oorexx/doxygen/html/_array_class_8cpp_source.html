81 /**
82  * Create a new array from Rexx code.
83  *
84  * @param arguments The pointer to the arguments (size items)
85  * @param argCount The number of arguments.
86  *
87  * @return An allocated Array item of the target class.
88  */
89 RexxObject* ArrayClass::newRexx(RexxObject **arguments, size_t argCount)
90 {
91  // this method is defined as an instance method, but this is actually attached
92  // to a class object instance. Therefore, any use of the this pointer
93  // will be touching the wrong data. Use the classThis pointer for calling
94  // any methods on this object from this method.
95  RexxClass *classThis = (RexxClass *)this;
96 
97  // creating an array of the default size?
98  if (argCount == 0)
99  {
100  Protected<ArrayClass> temp = new(0, DefaultArraySize) ArrayClass;
101  // finish setting this up.
102  classThis->completeNewObject(temp);
103  return temp;
104  }
105 
106  // Special case for 1-dimensional. This could be a single integer size or
107  // an array of sizes to create a multi-dimension array.
108  if (argCount == 1)
109  {
110  RexxObject *currentDim = arguments[0];
111  // specified as an array of dimensions?
112  // this gets special handling
113  if (currentDim != OREF_NULL && isArray(currentDim))
114  {
115  return createMultidimensional((ArrayClass *)currentDim, classThis);
116  }
117 
118  // Make sure it's an integer
119  size_t totalSize = validateSize(currentDim, ARG_ONE);
120 
121  // allocate a new item
122  Protected<ArrayClass> temp = new(totalSize) ArrayClass();
123  // if we're creating an explicit 0-sized array, create
124  // a dimension array so dimension cannot be changed later.
125  if (totalSize == 0)
126  {
127  temp->dimensions = new(1) NumberArray(1);
128  }
129 
130  // finish the class initialization and init calls.
131  classThis->completeNewObject(temp);
132  return temp;
133  }
134  else
135  {
136  // more than one argument, so all arguments must be valid size values.
137  return createMultidimensional(arguments, argCount, classThis);
138  }
139 }

191 /**
192  * Helper routine for creating a multidimensional array.
193  *
194  * @param dims Pointer to an array of pointers to dimension objects.
195  * @param count the number of dimensions to create
196  *
197  * @return The created array
198  */
199 ArrayClass* ArrayClass::createMultidimensional(RexxObject **dims, size_t count, RexxClass *classThis)
200 {
201  // Working with a multi-dimension array, so get a dimension array
202  Protected<NumberArray> dim_array = new(count) NumberArray(count);
203 
204  // we need to calculate total size needed for this array
205  // since we multiply each additional size in, start with a zeroth size of 1
206  size_t totalSize = 1;
207  for (size_t i = 0; i < count; i++)
208  {
209  // each dimension must be there and must be a non-negative integer value
210  // ...a dimension of 0 really does not make sense, right, but does work.
211  // this creates a multi-dimensional array of zero size which will get resized
212  // on the first assignment. Doesn't really make sense, but it's perfectly legal.
213  RexxObject *currentDim = dims[i];
214  size_t currentSize = nonNegativeArgument(currentDim, i + 1);
215  // going to do an overflow? By dividing, we can detect a
216  // wrap situation.
217  if (currentSize != 0 && (MaxFixedArraySize + 1) / currentSize < totalSize)
218  {
219  reportException(Error_Incorrect_method_array_too_big, MaxFixedArraySize);
220  }
221  // keep running total size and put integer object into current position
222  totalSize *= currentSize;
223 
224  // add this to our dimensions array, using a new integer object so
225  // we know what is there.
226  dim_array->put(currentSize, i + 1);
227  }
228 
229  // a final sanity check for out of bounds
230  if (totalSize > MaxFixedArraySize)
231  {
232  reportException(Error_Incorrect_method_array_too_big, MaxFixedArraySize);
233  }
234 
235  // create a new array item
236 
237  Protected<ArrayClass> temp = new(totalSize) ArrayClass;
238 
239  // set the dimension array
240  temp->dimensions = dim_array;
241 
242  // finish the class initialization and init calls.
243  classThis->completeNewObject(temp);
244  return temp;
245 }

268 /**
269  * Common routine for use by Array and subclass operator methods.
270  *
271  * @param size The base object size.
272  * @param items The number of items we wish to hold.
273  * @param maxSize The maximum size to allocate.
274  * @param type The Rexx class type code for this object.
275  *
276  * @return A newly allocated object.
277  */
278 ArrayClass* ArrayClass::allocateNewObject(size_t size, size_t items, size_t maxSize, size_t type)
279 {
280  size_t bytes = size;
281  // we never create below a minimum size
282  maxSize = std::max(maxSize, MinimumArraySize);
283  // and use at least the size as the max
284  maxSize = std::max(maxSize, items);
285  // add in the max size value. Note that we subtract one since
286  // the first item is contained in the base object allocation.
287  bytes += sizeof(RexxInternalObject *) * (maxSize - 1);
288  // now allocate the new object with that size. We also give a hint to
289  // the language process about how many objects we can potentially mark during
290  // a garbage collection.
291  ArrayClass *newArray = (ArrayClass *)new_object(bytes, type, maxSize);
292 
293  // now fill in the various control bits. Ideally, this
294  // really should be done in the constructor, but that gets really too
295  // complicated.
296  newArray->arraySize = items;
297  newArray->maximumSize = maxSize;
298  newArray->lastItem = 0;
299  newArray->itemCount = 0;
300  // no expansion yet, use ourself as the expansion
301  newArray->expansionArray = newArray;
302  // and return the new array.
303  return newArray;
304 }

341 /**
342  * Copy the array (and the expansion array, if necessary)
343  *
344  * @return A copy of the array object.
345  */
346 RexxInternalObject* ArrayClass::copy()
347 {
348  // make a copy of of the main array
349  ArrayClass *newArray = (ArrayClass *)RexxObject::copy();
350  // have we expanded in the past? Then we need to
351  // copy the expansion array too. NOTE: It would be tempting
352  // to just copy the expansion array and return that, but
353  // the possibilty that the array might have object variables
354  // (e.g., because of an OBJECTNAME= call) means we really can't do
355  // that directly. It is easier to just copy both pieces when needed.
356  if (hasExpanded())
357  {
358  newArray->expansionArray = (ArrayClass *)expansionArray->copy();
359  }
360  else
361  {
362  // make sure the new array is updated to point to itself
363  newArray->expansionArray = newArray;
364  }
365  return newArray;
366 }

735 /**
736  * Insert an element into the given index location.
737  *
738  * @param value The value to insert. This can be omitted,
739  * which will insert an empty slot at the
740  * indicated position.
741  * @param index The target index. This can be .nil, which will insert

743  * or a single-dimensional index of the position where the
744  * value will be inserted after.
745  *
746  * @return The index of the inserted valued.
747  */

749 {
750  checkMultiDimensional("INSERT");
751 
752  size_t position; // array position
753 
754  if (index == TheNilObject)
755  {
756  position = 1; // the insertion point of the item is 1
757  }
758  else if (index == OREF_NULL)
759  {
760  position = lastItem + 1; // inserting after the last item
761  }
762  else

842  * starting at the given index.
843  *
844  * @param index The index of the first item to shift.
845  *
846  * @param elements The number of elements to shift.
847  */
848 void ArrayClass::openGap(size_t index, size_t elements)
849 {
850  // is this larger than our current last element? If so, we have nothing to move
851  // but do need to expand the array size to accommodate the additional members
852  if (index > lastItem)
853  {
854  ensureSpace(index + elements - 1);
855  }
856  else

859  ensureSpace(lastItem + elements);

862  char *_end = (char *)slotAddress(lastItem + 1);
863

865  char *_start = (char *)slotAddress(index);
866  char *_target = (char *)slotAddress(index + elements);

868  memmove(_target, _start, _end - _start);

871  // explicit assignment rather than put to avoid old-to-new
872  // tracking issues
873  for (size_t i = index; i <= index + elements - 1; i++)
874  {
875  zeroItem(i);
876  }

878  if (lastItem != 0)
879  {
880  lastItem += elements;
881  }
882  }

886 /**
887  * Close a gap in the array item.
888  *
889  * @param index The gap to close.
890  * @param elements The size of the gap to close.
891  */
892 void ArrayClass::closeGap(size_t index, size_t elements)
893 {
894  // if we're beyond the current last item, nothing to do here
895  if (index > lastItem)
896  {
897  // if the index is within the size bounds of the
898  // array, we need to adjust the size
899  if (index <= size())
900  {
901  shrink(elements);
902  }
903  return;
904  }
905 
906  // cap the number of elements we're shifting.
907  elements = std::min(elements, lastItem - index + 1);
908 
909  // explicitly null out the slots of the gap we're closing to
910  // ensure that any oldspace tracking issues are resolved.
911  // use the clear method to ensure the item count is getting updated
912  // appropriately.
913  for (size_t i = index; i < index + elements; i++)
914  {
915  clearArrayItem(i);
916  }
917 
918  // we could have cleared out the last item when
919  // we cleared the gap, thus removing the need to
920  // shift anything
921  if (lastItem < index)
922  {
923  // if the index is within the size bounds of the
924  // array, we need to adjust the size
925  if (index <= size())
926  {
927  shrink(elements);
928  }
929  return;
930  }
931 
932  // get the address of first element and the first item to move.
933  char *_target = (char *)slotAddress(index);
934  char *_start = (char *)slotAddress(index + elements);
935  // and the end location of the real data
936  char *_end = (char *)slotAddress(lastItem + 1);
937  // shift the array over
938  memmove(_target, _start, _end - _start);
939  // because we track the item count, we need to ensure that
940  // the positions that we just copied out of are cleared out, otherwise
941  // the item count will not get incremented when these slots get reused.
942  _start = (char *)slotAddress(lastItem + 1 - elements);
943  memset(_start, 0, _end - _start);
944  // adjust the last element position (NOTE: because we needed to shift,
945  // we know that lastItem is non-zero)
946  lastItem -= elements;
947  // reduce the size
948  shrink(elements);
949 }

1348  *
1349  * @param index The array of objects representing the index.
1350  * @param indexCount The count of index items.
1351  * @param argPosition The argument position (used for error
1352  * reporting)

1356  * item in the array.

1359  * on the bounds checking flags.

1361 bool ArrayClass::validateMultiDimensionIndex(RexxObject **index, size_t indexCount,
1362  size_t argPosition, size_t boundsError, size_t &position)
1363 {
1364  size_t numSubscripts = indexCount;
1365  size_t numDimensions = dimensions->size();
1366 
1367  // do we have the right number of subscripts for this array?
1368  // this is worth processing.
1369  if (numSubscripts == numDimensions)
1370  {
1371  // a multiplier for translating into an absolute index position
1372  size_t multiplier = 1;
1373  size_t offset = 0;
1374 
1375  // now loop through all of the index values
1376  for (size_t i = 1; i <= numSubscripts; i++)
1377  {
1378  // each subscript is required.
1379  RexxObject *value = index[i - 1];
1380  // this must be a positive whole number and is required.
1381  position = positionArgument(value, argPosition + i);
1382

1384  size_t dimension = dimensions->get(i);
1385  // is this position larger than the current dimension? Check how
1386  // the out of bounds situation should be handled.
1387  if (position > dimension)

1390  // upper bounds on all dimensions here.
1391  if (boundsError & ExtendUpper)
1392  {
1393  // go extend this
1394  extendMulti(index, indexCount, argPosition);
1395  // Ok, now try validating again now that the array has expanded.

1397  // need to extend a second time.
1398  return validateMultiDimensionIndex(index, indexCount, argPosition, boundsError, position);

1400  // probably a get request, so just say this is out of bounds
1401  else
1402  {
1403  return false;
1404  }
1405  }
1406 
1407  // calculate the next offset position
1408  offset += multiplier * (position - 1);
1409  // and get the total dimension base size.
1410  multiplier *= dimension;
1411  }
1412  // get the accumulated position value...we now have a good value within the bounds
1413  position = offset + 1;
1414  return true;
1415  }
1416  // we have a mismatch between the index and the number of dimensions...issue the appropriate error
1417  else if (numSubscripts < numDimensions)
1418  {

1420  }
1421  // must be too many subscripts
1422  else
1423  {
1424  reportException(Error_Incorrect_method_maxsub, numDimensions);
1425  }
1426  return false;
1427 }
