92 /**
93  * Now resolve conflicts between the input and output
94  * streams and check for duplicate output targets.
95  */
96 void CommandIOContext::resolveConflicts()
97 {
98  // not yet determined that we have dual output
99  bool dualOutput = false;
100 
101  // if we have both error and output specified, we need to check that these
102  // are the same and collapse them to a single target
103  if (error != OREF_NULL && output != OREF_NULL && output->isSameTarget(error))
104  {
105  error = output;
106  // if we detect a conflict between the input and output, then
107  // we need to adjust both.
108  dualOutput = true;
109  }
110 
111  // now check if we have an input conflict
112  if (input != OREF_NULL)
113  {
114  // now see if these are the same target
115  if (output != OREF_NULL && output->needsBuffering(input))
116  {
117  // make this buffered until the command returns
118  output = new BufferingOutputTarget(output);
119  // if we've already established that output and error
120  // are the same, then adjust error to point to the buffered
121  // input
122  if (dualOutput)
123  {
124  error = output;
125  }
126  }
127  // now we still could have a conflict beteen the input ane error, so check that too
128  else if (error != OREF_NULL && error->needsBuffering(input))
129  {
130  // make this buffered until the command returns
131  output = new BufferingOutputTarget(output);
132  }
133  }
134 }
