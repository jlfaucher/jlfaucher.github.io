69 BOOL __stdcall WinConsoleCtrlHandler(DWORD dwCtrlType)
70 /******************************************************************************/
71 /* Arguments: Report record, registration record, context record, */
72 /* dispatcher context */
73 /* */
74 /* DESCRIPTION : For Control Break conditions issue a halt to activation */
75 /* Control-C or control-Break is pressed. */
76 /* */
77 /* Returned: Action code */
78 /******************************************************************************/
79 {
80  // check to condition for all threads of this process */
81 
82  if ((dwCtrlType == CTRL_CLOSE_EVENT) || (dwCtrlType == CTRL_SHUTDOWN_EVENT))
83  {
84  return false; /* send to system */
85  }
86 
87  /* if RXCTRLBREAK=NO then ignore SIGBREAK exception */
88  if (dwCtrlType == CTRL_BREAK_EVENT || dwCtrlType == CTRL_LOGOFF_EVENT)
89  {
90  char envp[65];
91  if (GetEnvironmentVariable("RXCTRLBREAK", envp, sizeof(envp)) > 0 && strcmp("NO", envp) == 0)
92  {
93  return true; /* ignore signal */
94  }
95  }
96 
97  if (dwCtrlType == CTRL_LOGOFF_EVENT)
98  {
99  return false; /* send to system */
100  }
101 
102  // we need to do something about this one, let the system interpreter handle
103  return SystemInterpreter::processSignal(dwCtrlType);
104 }

106 /**
107  * Handle system-specific once-per-process startup tasks.
108  *
109  * @param mod The module handle.
110  */
111 void SystemInterpreter::processStartup(HINSTANCE mod)
112 {
113  moduleHandle = mod;
114  // now do the platform independent startup
115  Interpreter::processStartup();
116  // Allow Rexx console output to use virtual terminal (VT) sequences.
117  // This is based on the examples Microsoft gives in
118  // https://docs.microsoft.com/en-us/windows/console/console-virtual-terminal-sequences
119  // The Microsoft example shows
120  // a) SetConsoleMode STD_OUTPUT_HANDLE or'ing in ENABLE_VIRTUAL_TERMINAL_PROCESSING and DISABLE_NEWLINE_AUTO_RETURN
121  // b) SetConsoleMode STD_INPUT_HANDLE or'ing in ENABLE_VIRTUAL_TERMINAL_INPUT
122  // In addition to a) we also run SetConsoleMode for STD_ERROR_HANDLE
123  // but we don't do b) because this breaks PULL line editing (cursor
124  // keys emit ANSI characters instead of moving around).
125 
126  // VT support was added around 10.0.10586, so we might or might not
127  // have some of these defines. If missing, make them nop's.
128 #ifndef ENABLE_VIRTUAL_TERMINAL_PROCESSING
129 #define ENABLE_VIRTUAL_TERMINAL_PROCESSING 0
130 #endif
131 #ifndef DISABLE_NEWLINE_AUTO_RETURN
132 #define DISABLE_NEWLINE_AUTO_RETURN 0
133 #endif
134 
135  DWORD mode;
136  HANDLE h;
137 
138  h = GetStdHandle(STD_OUTPUT_HANDLE);
139  if (h != NULL && h != INVALID_HANDLE_VALUE && GetConsoleMode(h, &mode) != 0)
140  {
141  SetConsoleMode(h, mode | ENABLE_VIRTUAL_TERMINAL_PROCESSING | DISABLE_NEWLINE_AUTO_RETURN);
142  }
143  h = GetStdHandle(STD_ERROR_HANDLE);
144  if (h != NULL && h != INVALID_HANDLE_VALUE && GetConsoleMode(h, &mode) != 0)
145  {
146  SetConsoleMode(h, mode | ENABLE_VIRTUAL_TERMINAL_PROCESSING | DISABLE_NEWLINE_AUTO_RETURN);
147  }
148 
149  // Because of using the stand-alone runtime library or when using different compilers,
150  // the std-streams of the calling program and the REXX.DLL might be located at different
151  // addresses and therefore _file might be -1. If so, std-streams are reassigned to the
152  // file standard handles returned by the system
153  if ((_fileno(stdin) < 0) && (GetFileType(GetStdHandle(STD_INPUT_HANDLE)) != FILE_TYPE_UNKNOWN))
154  {
155  *stdin = *_fdopen(_open_osfhandle((intptr_t)GetStdHandle(STD_INPUT_HANDLE), _O_RDONLY), "r");
156  }
157  if ((_fileno(stdout) < 0) && (GetFileType(GetStdHandle(STD_OUTPUT_HANDLE)) != FILE_TYPE_UNKNOWN))
158  {
159  *stdout = *_fdopen(_open_osfhandle((intptr_t)GetStdHandle(STD_OUTPUT_HANDLE), _O_APPEND), "a");
160  }
161  if ((_fileno(stderr) < 0) && (GetFileType(GetStdHandle(STD_ERROR_HANDLE)) != FILE_TYPE_UNKNOWN))
162  {
163  *stderr = *_fdopen(_open_osfhandle((intptr_t)GetStdHandle(STD_ERROR_HANDLE), _O_APPEND), "a");
164  }
165  // enable trapping for CTRL_C exceptions
166  SetConsoleCtrlHandler(&WinConsoleCtrlHandler, true);
167 
168  // we never want to see a critical-error-handler message box.
169  // according to MS docs: "Best practice is that all applications call
170  // the process-wide SetErrorMode function with a parameter of
171  // SEM_FAILCRITICALERRORS at startup
172  SetErrorMode(SEM_FAILCRITICALERRORS);
173 }

243 /**
244  * Process a signal based on current command handler settings.
245  *
246  * @param dwCtrlType The type of exception.
247  *
248  * @return true if we handled the signal, false if it should be
249  * passed up the handler chain.
250  */
251 bool SystemInterpreter::processSignal(DWORD dwCtrlType)
252 {
253  /* Ignore Ctrl+C if console is running in console */
254  if (exceptionConsole)
255  {
256  GenerateConsoleCtrlEvent(CTRL_BREAK_EVENT, exceptionHostProcessId);
257  return true; /* ignore signal */
258  }
259 
260  if (exceptionHostProcess)
261  {
262  GenerateConsoleCtrlEvent(CTRL_C_EVENT, exceptionHostProcessId);
263  TerminateProcess(exceptionHostProcess, -1);
264  }
265  // if this is a ctrl_C, try to halt all of the activities. If we hit this
266  // in a situation where we still have one pending, then we'll allow the system
267  // to kill the process.
268  if (dwCtrlType == CTRL_C_EVENT)
269  {
270  return Interpreter::haltAllActivities(OREF_NULL);
271  }
272  return true; /* ignore signal */
273 }

276 /**
277  * Retrieve the value of an envinment variable into a smart buffer.
278  *
279  * @param variable The name of the environment variable.
280  * @param buffer The buffer used for the return.
281  *
282  * @return true if the variable exists, false otherwise.
283  */
284 bool SystemInterpreter::getEnvironmentVariable(const char *variable, FileNameBuffer &buffer)
285 {
286  // do the request first with no buffer...this will return the size of the variable, which
287  // will allow us to ensure the buffer is large enough
288  DWORD dwSize = GetEnvironmentVariable(variable, NULL, 0);
289  if (dwSize > 0)
290  {
291  // now request the variable again
292  buffer.ensureCapacity(dwSize);
293  GetEnvironmentVariable(variable, (char *)buffer, dwSize);
294  return true;
295  }
296  // make sure this is a null string
297  buffer = "";
298  return false;
299 }
