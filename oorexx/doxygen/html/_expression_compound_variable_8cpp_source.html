59 /**
60  * Allocate storage for a compound variable object.
61  *
62  * @param size The base object size
63  * @param tailCount The number of tail elements in the variable.
64  *
65  * @return storage for the object.
66  */
67 void * RexxCompoundVariable::operator new(size_t size, size_t tailCount)
68 {
69  // belt-and-braces...it would be a strange compound variable indeed with
70  // no tail pieces.
71  if (tailCount == 0)
72  {
73  // this object is normal sized, minus the dummy tail element
74  return new_object(size - sizeof(RexxObject *), T_CompoundVariableTerm);
75  }
76  else
77  {
78  // adjust the size based on the number of tail elements.
79  return new_object(size + ((tailCount - 1) * sizeof(RexxObject *)), T_CompoundVariableTerm);
80  }
81 }

295 /**
296  * Expose a compound variable as part of a PROCEDURE
297  * instruction.
298  *
299  * @param context The current execution context.
300  * @param parent The parent (calling) context.
301  * @param stack The current evaluation stack.
302  */
303 void RexxCompoundVariable::procedureExpose(RexxActivation *context, RexxActivation *parent)
304 {
305  // first get (and possible create) the compound variable in the parent context.
306  CompoundTableElement *variable = parent->exposeLocalCompoundVariable(stemName, stemIndex, &tails[0], tailCount);
307  // get the stem index from the current level. This may end up
308  // creating the stem that holds the exposed value.
309  StemClass *stem_table = context->getLocalStem(stemName, stemIndex);
310  // have the stem expose this
311  stem_table->expose(variable);
312  // trace resolved compound name
313  context->traceCompoundName(stemName, &tails[0], tailCount, variable->getName());
314 }

317 /**
318  * Expose an object compound variable in a method.
319  *
320  * @param context The current execution context.
321  * @param stack The current evaluation stack.
322  * @param object_dictionary
323  */
324 void RexxCompoundVariable::expose(RexxActivation *context, VariableDictionary *object_dictionary)
325 {
326  // get the stem in the source dictionary
327  StemClass *source_stem = object_dictionary->getStem(stemName);
328  // new tail for the compound variable
329  CompoundVariableTail resolved_tail(context, &tails[0], tailCount);
330  // first get (and possible create) the compound variable in the
331  // object context.
332  CompoundTableElement *variable = source_stem->exposeCompoundVariable(resolved_tail);
333  // get the stem index from the current level. This may end up
334  // creating the stem that holds the exposed value.
335  StemClass *stem_table = context->getLocalStem(stemName, stemIndex);
336  // have the stem expose this
337  stem_table->expose(variable);
338  // trace resolved compound name
339  context->traceCompoundName(stemName, &tails[0], tailCount, variable->getName());
340 }
