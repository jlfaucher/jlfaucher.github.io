123 /**
124  * Execute a dynamic ::CONSTANT directive as if it is a real
125  * instruction.
126  *
127  * @param context The current execution context.
128  * @param stack The current evaluation stack.
129  */
130 void ConstantDirective::execute(RexxActivation *context, ExpressionStack *stack )
131 {
132  context->traceInstruction(this);
133  // get the expression value
134  RexxObject *result = expression->evaluate(context, stack);
135  // trace the result
136  context->traceResult(result);
137  // complete the method for the constant with the resolved value
138  constantCode->setValue(result);
139  // do debug pause
140  context->pauseInstruction();
141 }
