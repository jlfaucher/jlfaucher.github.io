53 /**
54  * Build a tail value from a set of tail elements without doing
55  * variable resolution.
56  *
57  * @param tails An array of tail elements.
58  * @param count The count of tail elements.
59  */
60 void CompoundVariableTail::buildUnresolvedTail(RexxInternalObject **tails, size_t count)
61 {
62  bool first = true;
63 
64  for (size_t i = 0; i < count; i++)
65  {
66  // only add a dot after the first piece
67  if (!first)
68  {
69  addDot();
70  }
71 
72  first = false;
73  RexxInternalObject *part = tails[i];
74  // this could be ommitted in a [] list
75  if (part != OREF_NULL)
76  {
77  // if this is a variable, just copy the name. Otherwise, copy the value
78  if (isOfClass(VariableTerm, part))
79  {
80  ((RexxSimpleVariable *)part)->getName()->copyIntoTail(this);
81  }
82  else
83  {
84  part->stringValue()->copyIntoTail(this);
85  }
86  }
87  }
88  length = current - tail; // set the final, updated length
89 }

92 /**
93  * Construct a tail piece from its elements, using a
94  * variable dictionary as the context for resolving
95  * variables.
96  *
97  * @param dictionary The source variable dictionary.
98  * @param tails
99  * @param tailCount
100  */
101 void CompoundVariableTail::buildTail(VariableDictionary *dictionary, RexxInternalObject **tails, size_t tailCount)
102 {
103  // a single element is easiest, and common enough that we optimize for it.
104  if (tailCount == 1)
105  {
106  // get the tail value
107  RexxInternalObject *_tail = tails[0]->getValue(dictionary);
108  // if it is an integer type, we might be able to address the string representation directly.
109  if (isInteger(_tail))
110  {
111  RexxString *rep = ((RexxInteger *)_tail)->getStringrep();
112  if (rep != OREF_NULL)
113  {
114  useStringValue(rep);
115  return;
116  }
117  }
118  // if this is directly a string, we can use this directly
119  if (isString(_tail))
120  {
121  useStringValue((RexxString *)_tail);
122  return;
123  }
124  // some other type of object, or an integer without a string
125  // rep. We need to have it do the copy operation.
126  _tail->copyIntoTail(this);
127  length = current - tail; // set the final, updated length
128  }
129  else
130  {
131  // copy the first element, then the remainder of the elements
132  // with a dot in between.
133  tails[0]->getValue(dictionary)->copyIntoTail(this);
134  for (size_t i = 1; i < tailCount; i++)
135  {
136  addDot();
137  tails[i]->getValue(dictionary)->copyIntoTail(this);
138  }
139 
140  length = current - tail; // set the final, updated length
141  }
142 }

197 /**
198  * Resolve the "stem.[a,b,c]=" pieces to a fully qualified stem
199  * name.
200  *
201  * @param tails The array of tails.
202  * @param count The count of tails.
203  */
204 void CompoundVariableTail::buildTail(RexxInternalObject **tails, size_t count)
205 {
206  bool first = true;
207  // copy in each piece, with dots added after the first.
208 
209  for (size_t i = 0; i < count; i++)
210  {
211  if (!first)
212  {
213  addDot();
214  }
215  first = false;
216 
217  RexxInternalObject *part = (RexxInternalObject *)tails[i];
218  // omitted pices are null strings
219  if (part == OREF_NULL)
220  {
221  part = GlobalNames::NULLSTRING;
222  }
223  part->copyIntoTail(this);
224  }
225  length = current - tail;
226 }

256 /**
257  * Build a tail from a single string and a number index
258  * value. This is used by the stem sort, and the if the tail
259  * piece does not end with a dot, one is provided.
260  *
261  * @param _tail the tail prefix.
262  * @param index the index value.
263  */
264 void CompoundVariableTail::buildTail(RexxString *_tail, size_t index)
265 {
266  // the tail prefix can be optional,
267  if (_tail != OREF_NULL)
268  {
269  _tail->copyIntoTail(this);
270  length = length + _tail->getLength();
271  // if we don't have a dot on the end of this tail piece, add one
272  if (!_tail->endsWith('.'))
273  {
274  addDot();
275  }
276  length = current - tail;
277  }
278 
279  size_t numberLength = Numerics::formatWholeNumber(index, current);
280  current += numberLength;
281  length += numberLength;
282  remainder -= length;
283 }

300 /**
301  * Ensure we have sufficient space for adding another
302  * tail element to our buffer.
303  *
304  * @param needed The size of the required element.
305  */
306 void CompoundVariableTail::expandCapacity(size_t needed)
307 {
308  // update the accumulated length
309  length = current - tail;
310  // if we have previously allocated a buffer, expand the size (this must be
311  // a BIG tail!)
312  if (!temp.isNull())
313  {
314  temp->expand(needed + ALLOCATION_PAD);
315  tail = temp->getData();
316  current = tail + length;
317  remainder += needed + ALLOCATION_PAD;
318  }
319  // allocate a completly new buffer, copying our
320  // existing tail data into it.
321  else
322  {
323 
324  size_t newLength = length + needed + ALLOCATION_PAD;
325  temp = (BufferClass *)new_buffer(newLength);
326  tail = temp->getData();
327  current = tail + length;
328  memcpy(tail, buffer, length);
329  remainder = newLength - length;
330  }
331 }

334 /**
335  * Create a fully resolved compound name from this tail.
336  * This appends the stem name to the constructed tail.
337  *
338  * @param stem The stem name.
339  *
340  * @return The fully resolved compound variable name as a string object.
341  */
342 RexxString *CompoundVariableTail::createCompoundName(RexxString *stem)
343 {
344  size_t len1 = stem->getLength();
345  // create a new string of the required length (note, this assumes the
346  // first period of the compound name is in the stem name.
347  RexxString *result = (RexxString *)raw_string(len1 + length);
348  // copy in the stem name and the tail piece.
349  char *data = result->getWritableData();
350  if (len1 != 0)
351  {
352  memcpy(data, stem->getStringData(), len1);
353  data += len1;
354  }
355  if (length != 0)
356  {
357  memcpy(data, tail, length);
358  }
359  return result;
360 }
