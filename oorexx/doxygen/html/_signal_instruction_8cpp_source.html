108  *
109  * @param labels The directory of label objects for this code section.
110  */
111 void RexxInstructionSignal::resolve(StringTable *labels)
112 {
113  // The section might not have any labels, but if it does, get our
114  // name from the directory. Note, we don't raise an error now, we
115  // wait until the instruction is actually executed.
116  if (labels != OREF_NULL)
117  {
118  // this links the signal instruction directly to the label
119  // instruction that is the target.
120  targetInstruction = (RexxInstruction *)labels->get(targetName);
121  }
122 }

125 /**
126  * Execute this instruction
127  *
128  * @param context The current program context.
129  * @param stack The expression stack
130  */
131 void RexxInstructionSignal::execute(RexxActivation *context, ExpressionStack *stack)
132 {
133  // trace the instruction if necessary
134  context->traceInstruction(this);
135 
136  // normal signal instruction
137 
138  // the target should have been resolved during the resolve phase.
139  // if we don't have anything, raise the error now
140  if (targetInstruction == OREF_NULL)
141  {
142  reportException(Error_Label_not_found_name, targetName);
143  }
144  // tell the activation we're changing course. Debug pauses
145  // are handled after the signal
146  context->signalTo(targetInstruction);
147 }

203 /**
204  * Execute this instruction
205  *
206  * @param context The current program context.
207  * @param stack The expression stack
208  */
209 void RexxInstructionDynamicSignal::execute(RexxActivation *context, ExpressionStack *stack)
210 {
211  // trace the instruction if necessary
212  context->traceInstruction(this);
213 
214  // evaluate the expression in the current context.
215  RexxObject *result = dynamicName->evaluate(context, stack);
216  // expression results require tracing
217  context->traceKeywordResult(GlobalNames::VALUE, result);
218  // force to a string value
219  RexxString *stringResult = result->requestString();
220  stack->push(stringResult);
221  // the context handles locating the target label
222  context->signalValue(stringResult);
223 }

292 /**
293  * Execute a SIGNAL ON/OFF instruction. This either activates
294  * or deactivates the signal trap. Calling of a target only
295  * happens when a trap is activated.
296  *
297  * @param context The current program context.
298  * @param stack The current context evaluation stack.
299  */
300 void RexxInstructionSignalOn::execute(RexxActivation *context, ExpressionStack *stack)
301 {
302  // do trace stuff.
303  context->traceInstruction(this);
304 
305  // if we do not have a target name set, this is a SIGNAL OFF instruction. Just
306  // disable the trap.
307  if (targetName != OREF_NULL)
308  {
309  // wax on...
310  context->trapOn(conditionName, this, true);
311  }
312  else
313  {
314  // wax off...
315  context->trapOff(conditionName, true);
316  }
317 }

320 /**
321  * Resolve a call target at the end of block processing.
322  *
323  * @param labels The table of label instructions in the current context.
324  */
325 void RexxInstructionSignalOn::resolve(StringTable *labels)
326 {
327  // if there is a labels table, see if we can find a label object from the context.
328  // SIGNALS only go to labels, but we don't report an error until the trap is triggered.
329  if (labels != OREF_NULL)
330  {
331  // see if there is a matching label. If we get something,
332  // we're finished.
333  targetInstruction = (RexxInstruction *)labels->get((RexxString *)targetName);
334  }
335 }

338 /**
339  * Process a trapped condition.
340  *
341  * @param context The current program context.
342  * @param conditionObj
343  * The condition object for the trap
344  */
345 void RexxInstructionSignalOn::trap(RexxActivation *context, DirectoryClass *conditionObj)
346 {
347  // trapping a condition turns off the tracp
348  context->trapOff(conditionName, true);
349  // this should have been resolved already...raise the error now if
350  // we don't have a resolved label.
351  if (targetInstruction == OREF_NULL)
352  {
353  reportException(Error_Label_not_found_name, targetName);
354  }
355 
356  // set the new condition object
357  context->setConditionObj(conditionObj);
358  // and jump to the condition target
359  context->signalTo(targetInstruction);
360 }
