66 /**
67  * Save all globally available string objects in the image.
68  *
69  * @return An array of all global string objects.
70  */
71 ArrayClass *MemoryObject::saveStrings()
72 {
73  // pass one, count how many string objects we have
74  #undef GLOBAL_NAME
75  #define GLOBAL_NAME(name, value) stringCount++;
76 
77  size_t stringCount = 0;
78  #include "GlobalNames.h"
79 
80  // get an array to contain all of the string values
81  ArrayClass *stringArray = new_array(stringCount);
82 
83  // redefine GLOBAL_NAME to save each string in the array
84  #undef GLOBAL_NAME
85  #define GLOBAL_NAME(name, value) stringArray->put(GlobalNames::name, stringCount); stringCount++;
86 
87  // the index gets incremented as we go
88  stringCount = 1;
89  #include "GlobalNames.h"
90 
91  return stringArray; // and return the saved string array
92 }
