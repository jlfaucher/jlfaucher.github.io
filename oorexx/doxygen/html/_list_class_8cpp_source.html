111 /**
112  * Initialize the list contents, either directly from the
113  * low level constructor or from the INIT method.
114  *
115  * @param capacity The requested capacity.
116  */
117 void ListClass::initialize(size_t capacity)
118 {
119  // only do this if we have no contents already
120  if (contents == OREF_NULL)
121  {
122  // we don't want to create a zero sized contents, so cap the
123  // minimum size at the default
124  capacity = std::max(capacity, DefaultListSize);
125  // and allocate the initial contents item.
126  contents = new (capacity) ListContents(capacity);
127  }
128 }

186 /**
187  * Convert a provided index value into the
188  * target item in the contents.
189  *
190  * @param index The provided index.
191  * @param position The argument position
192  *
193  * @return The validated location of the targetted entry.
194  */
195 ListContents::ItemLink ListClass::validateIndex(RexxObject *index, size_t position)
196 {
197  // this is required
198  requiredArgument(index, position);
199 
200  size_t item_index;
201  // converted using the ARGUMENT_DIGITS value
202  if (!index->unsignedNumberValue(item_index, Numerics::ARGUMENT_DIGITS))
203  {
204  reportException(Error_Incorrect_method_index, index);
205  }
206  // if not valid, return the no link marker
207  if (!contents->isIndexValid(item_index))
208  {
209  return ListContents::NoLink;
210  }
211  return item_index;
212 }

215 /**
216  * Convert a provided index value into the
217  * target item in the contents. For this version, .nil is a
218  * valid index and is indicated by returning NoLink. Otherwise,
219  * this must point to a real value
220  *
221  * @param index The provided index.
222  * @param position The argument position
223  *
224  * @return The validated location of the targetted entry.
225  */
226 ListContents::ItemLink ListClass::validateInsertionIndex(RexxObject *index, size_t position)
227 {
228  // this is a special index for insertion operations
229  if (index == TheNilObject)
230  {
231  return ListContents::AtBeginning;
232  }

234  {
235  return ListContents::AtEnd;
236  }
237  // map this to a valid index position
238  return requiredIndex(index, position);
239 }

264 /**
265  * Expand the contents of a collection. We try for double the
266  * size.
267  */
268 void ListClass::expandContents()
269 {
270  // double the bucket size for small Lists
271  // for Lists above the limit, just add half of the actual bucket size
272  size_t size = contents->capacity();
273  size += size <= ExpansionDoubleLimit ? size : size / 2;
274  expandContents(size);
275 }

278 /**
279  * Expand the contents of a collection to a given bucket
280  * capacity.
281  */
282 void ListClass::expandContents(size_t capacity )
283 {
284  // allocate a new table with the requested capacity, then merge the
285  // contents back into the the new one before replacing.
286  Protected<ListContents> newContents = new (capacity) ListContents(capacity);
287  contents->mergeInto(newContents);
288 
289  setField(contents, (ListContents *)newContents);
290 }

293 /**
294  * Ensure that the collection has sufficient space for a
295  * mass incoming addition.
296  *
297  * @param delta The number of entries we wish to add.
298  */
299 void ListClass::ensureCapacity(size_t delta)
300 {
301  // not enough space? time to expand. We'll go to
302  // double the bucket size for small Lists
303  // for Lists above the limit, just add half of the actual bucket size
304  if (!contents->hasCapacity(delta))
305  {
306  size_t size = contents->capacity();
307  size += size <= ExpansionDoubleLimit ? size : size / 2;
308  expandContents(size);
309  }
310 }

416 /**
417  * Low-level section method for a list.
418  *
419  * @param index The starting index (non-validated)
420  * @param count The count of items to include.
421  *
422  * @return A new list item containing the requested items.
423  */
424 ListClass *ListClass::section(size_t index, size_t count)
425 {
426  // create a new list to add things in. It would be nice to allocate
427  // this at the required size, but we don't really know how many
428  // items we'll put in the list, so just use a default value.
429  Protected<ListClass> result = new ListClass;
430 
431  // grab as many items as we can.
432  while (index != ListContents::NoMore && count-- > 0)
433  {
434  // append to the list
435  result->append(contents->get(index));
436  index = contents->nextIndex(index);
437  }
438  return result;
439 }
