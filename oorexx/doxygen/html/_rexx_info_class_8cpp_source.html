174 /**
175  * Initialize the static values of a RexxInfo object.
176  */
177 void RexxInfo::initialize()
178 {
179  char buffer[100]; // buffer for building the string
180  char work[20]; // working buffer for the date
181 
182  strcpy(work, __DATE__); // copy the build date
183  char *month = strtok(work, " "); // now we need to tokenize the different bits
184  char *day = strtok(NULL, " ");
185  char *year = strtok(NULL, " ");
186 
187  // make sure we don't have a leading zero on the day
188  if (*day == '0')
189  {
190  day++;
191  }
192  // now format into a usable date and make into a string object.
193  snprintf(buffer, sizeof(buffer), "%s %s %s", day, month, year);
194  interpreterDate = new_string(buffer);
195 
196  // now build the version string
197  snprintf(buffer, sizeof(buffer), "%d.%d.%d", ORX_VER, ORX_REL, ORX_MOD);
198  interpreterVersion = new_string(buffer);
199 
200  languageLevel = new_string(Interpreter::languageLevel);
201 
202  // now handle the platform-specific values
203  endOfLine = new_string(SysFileSystem::getLineEnd());
204  directorySeparator = new_string(SysFileSystem::getSeparator());
205  pathSeparator = new_string(SysFileSystem::getPathSeparator());
206  platformName = new_string(SystemInterpreter::getPlatformName());
207  interpreterName = Interpreter::getVersionString();
208 }
