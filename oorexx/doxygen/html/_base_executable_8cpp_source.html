84 /**
85  * Set the source object into a routine or method executable.
86  * This is generally used to attach a source context to a
87  * native method or function defined on a source directive. Since
88  * native functions can be referenced in multiple packages, but are
89  * managed in the package manager context, this may end up
90  * returning a copy of the executable.
91  *
92  * @param s The new source.
93  *
94  * @return Either the same executable object, or a new copy with the
95  * context set.
96  */
97 BaseExecutable *BaseExecutable::setPackageObject(PackageClass *s)
98 {
99  // set this into a source object context. If we get a
100  // new object returned, we need to make a copy of the base
101  // executable object also
102  BaseCode *setCode = code->setPackageObject(s);
103  // we're cool if these are equal
104  if (setCode == code)
105  {
106  return this;
107  }
108  // make a copy of this executable, and set the new code into it.
109  BaseExecutable *newBase = (BaseExecutable *)this->copy();
110  newBase->code = setCode;
111  return newBase;
112 }

159 /**
160  * Common handling static method for processing method or
161  * routine source. This sorts out the string vs. array
162  * argument and returns everything as an array if valid.
163  *
164  * @param source The input object.
165  * @param position The position (for error reporting)
166  *
167  * @return An array of the source lines.
168  */
169 ArrayClass *BaseExecutable::processExecutableSource(RexxObject *source, const char *position)
170 {
171  Protected<ArrayClass> sourceArray;
172 
173  // if this is a string object, then convert to a a single element array.
174  if (isString(source))
175  {
176  sourceArray = new_array((RexxString *)source);
177  }
178  else
179  {
180  // request this as an array. If not convertable, then we'll use it as a string
181  sourceArray = source->requestArray();
182  // couldn't convert?
183  if (sourceArray == (ArrayClass *)TheNilObject)
184  {
185  // get the string representation
186  RexxString *sourceString = source->makeString();
187  // still can't convert? This is an error
188  if (sourceString == (RexxString *)TheNilObject)
189  {
190  return OREF_NULL;
191  reportException(Error_Incorrect_method_no_method, position);
192  }
193  // wrap an array around the value
194  sourceArray = new_array(sourceString);
195  }
196  else
197  {
198  // validate the array contents are correct.
199  stringArrayArgument(sourceArray, position);
200  }
201  }
202  return sourceArray;
203 }

206 /**
207  * Process the new arguments for either a Routine
208  * or Method class. This decodes all of the arguments
209  * and processes them into acceptable forms. This includes
210  * figuring out the different source types and the
211  * different source contexts.
212  *
213  * @param init_args The original argument pointer passed to the new method.
214  * This will be advanced over any of the arguments we
215  * consume to leave the remaining arguments to be passed
216  * to an init method.
217  * @param argCount The count of arguments. This will be decremented for
218  * any arguments we use.
219  * @param name The name option of the call.
220  * @param sourceArray
221  * The array that will be used to create this object.
222  * @param sourceContext
223  * The optional source context this should inherit from.
224  */
225 void BaseExecutable::processNewExecutableArgs(RexxObject **&init_args, size_t &argCount, RexxString *&name,
226  Protected<ArrayClass> &sourceArray, PackageClass *&sourceContext)
227 {
228  RexxObject *pgmname; // method name
229  RexxObject *source; // Array or string object
230 
231  // do the initial parse of the new arguments.
232  RexxClass::processNewArgs(init_args, argCount, init_args, argCount, 2, pgmname, (RexxObject **)&source);
233  // get the method name as a string
234  name = stringArgument(pgmname, "name");
235  // make sure there is something for the second arg.
236  requiredArgument(source, "source");
237 
238  // figure out the source section.
239  sourceArray = processExecutableSource(source, "source");
240 
241  // if not a valid source, give an error
242  if (sourceArray == OREF_NULL)
243  {
244  reportException(Error_Incorrect_method_no_method, "source");
245  }
246 
247  // now process an optional sourcecontext argument
248  sourceContext = OREF_NULL;
249  // retrieve extra parameter if exists
250  // if not explicitly specified, then the program scope is used
251  if (argCount == 0)
252  {
253  // using the calling source context, so get the package from the top activation if
254  // there is one.
255  // see if we have an active context and use the current source as the basis for the lookup
256  RexxActivation *currentContext = ActivityManager::currentActivity->getCurrentRexxFrame();
257  if (currentContext != OREF_NULL)
258  {
259  sourceContext = currentContext->getPackage();
260  }
261  }
262  else
263  {
264  RexxObject *option;
265  // parse off an additional argument
266  RexxClass::processNewArgs(init_args, argCount, init_args, argCount, 1, option, NULL);
267  // if there are more than 3 options passed, it is possible this one was omitted
268  // we're don
269  if (option == OREF_NULL)
270  {
271  return;
272  }
273 
274  if (isOfClass(Method, option) || isOfClass(Routine, option))
275  {
276  sourceContext = ((BaseExecutable *)option)->getPackage();
277  }
278  else if (isOfClass(Package, option))
279  {
280  sourceContext = (PackageClass *)option;
281  }
282  else
283  {
284  // this must be a string (or convertable) and have a specific value
285  option = option->requestString();
286  if (option == TheNilObject)
287  {
288  reportException(Error_Incorrect_method_argType, IntegerThree, "Method, Routine, Package, or String object");
289  }
290  // default given? set option to NULL (see code below)
291  if (!((RexxString *)option)->strCaselessCompare("PROGRAMSCOPE"))
292  {
293  reportException(Error_Incorrect_call_list, "NEW", IntegerThree, "\"PROGRAMSCOPE\", Method, Routine, or Package object", option);
294  }
295 
296  // using the calling source context, so get the package from the top activation if
297  // there is one.
298  // see if we have an active context and use the current source as the basis for the lookup
299  RexxActivation *currentContext = ActivityManager::currentActivity->getCurrentRexxFrame();
300  if (currentContext != OREF_NULL)
301  {
302  sourceContext = currentContext->getPackage();
303  }
304  }
305  }
306 }

310  * Process the newFile() arguments for either a Routine
311  * or Method class. This decodes all of the arguments
312  * and processes them into acceptable forms. This includes
313  * figuring out the different source contexts.
314  *
315  * @param filename The filename option of the call.
316  * @param sourceContext
317  * The optional source context this should inherit from.
318  */
319 void BaseExecutable::processNewFileExecutableArgs(RexxString *&filename,
320  PackageClass *&sourceContext)
321 {
322  // get the filename as a string
323  filename = stringArgument(filename, "name");
324 
325  // now process an optional sourcecontext argument
326  if (sourceContext == OREF_NULL)
327  {
328  // using the calling source context, so get the package from the top activation if
329  // there is one.
330  // see if we have an active context and use the current source as the basis for the lookup
331  RexxActivation *currentContext = ActivityManager::currentActivity->getCurrentRexxFrame();
332  if (currentContext != OREF_NULL)
333  {
334  sourceContext = currentContext->getPackage();
335  }
336  }
337 
338  if (isOfClass(Package, sourceContext))
339  {
340  return;
341  }
342 
343  if (isOfClass(Method, sourceContext) || isOfClass(Routine, sourceContext))
344  {
345  sourceContext = ((BaseExecutable *)sourceContext)->getPackage();
346  }
347  else
348  {
349  // this must be a string (or convertable) and have a specific value
350  RexxString *source = sourceContext->requestString();
351  if (source == TheNilObject)
352  {
353  reportException(Error_Incorrect_method_argType, IntegerTwo, "Method, Routine, Package, or String object");
354  }
355  // default given?
356  if (!(source->strCaselessCompare("PROGRAMSCOPE")))
357  {
358  reportException(Error_Incorrect_call_list, "NEWFILE", IntegerTwo, "\"PROGRAMSCOPE\", Method, Routine, or Package object", source);
359  }
360 
361  // using the calling source context, so get the package from the top activation if
362  // there is one.
363  // see if we have an active context and use the current source as the basis for the lookup
364  RexxActivation *currentContext = ActivityManager::currentActivity->getCurrentRexxFrame();
365  if (currentContext != OREF_NULL)

367  sourceContext = currentContext->getPackage();
