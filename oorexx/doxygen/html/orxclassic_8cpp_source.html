191 RexxRoutine1(int, // Return type
192  TestFVariablePool, // Function name
193  RexxArrayObject, arr) // Array of shvblocks
194 {
195  RexxReturnCode retc;
196  size_t members = context->ArrayItems(arr);
197  size_t ctr;
198  PSHVBLOCK blocks = NULL;
199  PSHVBLOCK nextblock = NULL;
200  PSHVBLOCK currentblock = NULL;
201  unsigned int tempint;
202  RexxStringObject str;
203  size_t len;
204 
205  // for a RXSHV_FETCH, RXSHV_SYFET or RXSHV_PRIV operation we may either let
206  // RexxVariablePool allocate memory for a returned value, or malloc it here
207  // for RexxVariablePool to use it
208  // we need to remember who did which memory allocation as the one will have
209  // to be freed with RexxFreeMemory and the other with free
210  char* rexxAlloc = (char*)malloc((members + 1) * sizeof(char));
211 
212  // set up the shvblocks from the array
213  for (ctr = 1; ctr <= members; ctr++) {
214  nextblock = (PSHVBLOCK)malloc(sizeof(SHVBLOCK));
215  if (currentblock != NULL) {
216  currentblock->shvnext = nextblock;
217  }
218  currentblock = nextblock;
219  if (blocks == NULL) {
220  blocks = currentblock;
221  }
222  currentblock->shvnext = NULL;
223  RexxObjectPtr entry = context->ArrayAt(arr, ctr);
224  RexxObjectPtr val = context->SendMessage0(entry, "shvcode");
225  context->ObjectToUnsignedInt32(val, &tempint);
226  currentblock->shvcode = (unsigned char)tempint;
227  val = context->SendMessage0(entry, "shvret");
228  context->ObjectToUnsignedInt32(val, &tempint);
229  currentblock->shvret = (unsigned char)tempint;
230  val = context->SendMessage0(entry, "shvnamelen");
231  context->ObjectToUnsignedInt32(val, &tempint);
232  currentblock->shvnamelen = (size_t)tempint;
233  val = context->SendMessage0(entry, "shvvaluelen");
234  context->ObjectToUnsignedInt32(val, &tempint);
235  currentblock->shvvaluelen = (size_t)tempint;
236 
237  // all request codes require a name
238  val = context->SendMessage0(entry, "shvname");
239  str = context->ObjectToString(val);
240  currentblock->shvname.strptr = context->StringData(str);
241  currentblock->shvname.strlength = context->StringLength(str);
242 
243  switch (currentblock->shvcode)
244  {
245  case RXSHV_SET:
246  case RXSHV_SYSET:
247  // a Set request also requires a value
248  val = context->SendMessage0(entry, "shvvalue");
249  str = context->ObjectToString(val);
250  currentblock->shvvalue.strptr = (char *)context->StringData(str);
251  currentblock->shvvalue.strlength = context->StringLength(str);
252  break;
253 
254  case RXSHV_FETCH:
255  case RXSHV_SYFET:
256  case RXSHV_PRIV:
257  len = currentblock->shvvaluelen;
258  currentblock->shvvalue.strlength = len;
259  // allocate a value buffer or let RexxVariablePool allocate it
260  currentblock->shvvalue.strptr = len == 0 ? NULL: (char *)malloc(len);
261  rexxAlloc[ctr] = len == 0; // remember wo did it
262  break;
263 
264  case RXSHV_DROPV:
265  case RXSHV_SYDRO:
266  default:
267  // allow invalid request codes to be handled by RexxVariablePool
268  break;
269  }
270  }
271 
272  // call the variable pool interface
273  retc = RexxVariablePool(blocks);
274 
275  // set the array to the shvblocks
276  currentblock = blocks;
277  for (ctr = 1; ctr <= members; ctr++)
278  {
279  RexxObjectPtr entry = context->ArrayAt(arr, ctr);
280  context->SendMessage1(entry, "shvret=", context->UnsignedInt32ToObject((uint32_t)currentblock->shvret));
281  switch (currentblock->shvcode)
282  {
283  case RXSHV_FETCH:
284  case RXSHV_SYFET:
285  case RXSHV_PRIV:
286  context->SendMessage1(entry, "shvvalue=", context->NewString(currentblock->shvvalue.strptr, currentblock->shvvalue.strlength));
287  context->SendMessage1(entry, "shvvaluelen=", context->UnsignedInt32ToObject((uint32_t)currentblock->shvvaluelen));
288  // memory allocated by RexxVariablePool must be freed with RexxFreeMemory
289  if (currentblock->shvvalue.strptr != NULL)
290  {
291  if (rexxAlloc[ctr] == 1)
292  RexxFreeMemory(currentblock->shvvalue.strptr);
293  else
294  free(currentblock->shvvalue.strptr);
295  }
296  break;
297 
298  default:
299  break;
300  }
301  nextblock = currentblock->shvnext;
302  free(currentblock);
303  currentblock = nextblock;
304  }
305  free(rexxAlloc);
306 
307  return retc;
308 }

310 RexxRoutine1(size_t, // Return type
311  TestFNVariablePool, // Function name
312  RexxArrayObject, arr) // Array of shvblocks
313 
314 {
315  RexxReturnCode retc = 0;
316  size_t members = context->ArrayItems(arr);
317  size_t ctr = 0;
318  SHVBLOCK block;
319 
320  while (retc != RXSHV_LVAR && ctr < members) {
321  block.shvnext = NULL;
322  block.shvname.strptr = NULL;
323  block.shvname.strlength = 0;
324  block.shvvalue.strptr = NULL;
325  block.shvvalue.strlength = 0;
326  block.shvnamelen = 0;
327  block.shvvaluelen = 0;
328  block.shvcode = RXSHV_NEXTV;
329  block.shvret = 0;
330  retc = RexxVariablePool(&block);
331 
332  ctr++;
333  RexxObjectPtr entry = context->ArrayAt(arr, ctr);
334  context->SendMessage1(entry, "shvret=", context->UnsignedInt32ToObject((uint32_t)block.shvret));
335  context->SendMessage1(entry, "shvname=", context->NewString(block.shvname.strptr, block.shvname.strlength));
336  context->SendMessage1(entry, "shvvalue=", context->NewString(block.shvvalue.strptr, block.shvvalue.strlength));
337 
338  if (block.shvname.strptr != NULL) {
339  // this memory must be freed this way since it was allocated with RexxAllocateMemeory
340  RexxFreeMemory((void *)block.shvname.strptr);
341  }
342  if (block.shvvalue.strptr != NULL) {
343  // this memory must be freed this way since it was allocated with RexxAllocateMemeory
344  RexxFreeMemory(block.shvvalue.strptr);
345  }
346  }
347 
348  return ctr;
349 }
