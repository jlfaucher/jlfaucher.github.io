171 /**
172  * Evaluate an operator expression term
173  *
174  * @param context The current execution context.
175  * @param stack The expression stack from the context.
176  *
177  * @return The operation result object.
178  */
179 RexxObject *RexxBinaryOperator::evaluate(RexxActivation *context, ExpressionStack *stack )
180 {
181  // evaluate both expression terms
182  RexxObject *left = left_term->evaluate(context, stack);
183  RexxObject *right = right_term->evaluate(context, stack);
184  // and finally evaluate the operation itself. The left term is
185  // the target of the message and determines how this is invoked.
186  RexxObject *result = left->callOperatorMethod(oper, right);
187  // replace top two stack elements with our result
188  stack->operatorResult(result);
189  // trace the result if necessary
190  context->traceOperator(operatorName(), result);
191  return result;
192 }

195 /**
196  * Evaluate a Unary operation.
197  *
198  * @param context The current execution context.
199  * @param stack The context evaluation stack.
200  *
201  * @return The operation result.
202  */
203 RexxObject *RexxUnaryOperator::evaluate(RexxActivation *context, ExpressionStack *stack )
204 {
205  // we only have a single term to evaluate
206  RexxObject *term = left_term->evaluate(context, stack);
207  // and forward to the operator type
208  RexxObject *result = term->callOperatorMethod(oper, OREF_NULL);
209  // we only replace one term on the stack.
210  stack->prefixResult(result);
211  context->tracePrefix(operatorName(), result);
212  return result;
213 }
