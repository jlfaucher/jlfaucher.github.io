402 /**
403  * Resolve a token to a potential subkeyword.
404  *
405  * @param token The token to resolve.
406  * @param Table The table to search.
407  * @param Table_Size The size of the table.
408  *
409  * @return The numeric identifier for the constant. Returns 0 if not
410  * found in the target table.
411  */
412 int RexxToken::resolveKeyword(RexxString *token, KeywordEntry *table, size_t tableSize)
413 {
414  const char *name = token->getStringData();
415  size_t length = token->getLength();
416 
417  // search this table using a binary search
418 
419  // NOTE: We need to use signed numbers for this, otherwise
420  // the numbers can wrap
421  int lower = 0; // set initial lower bound
422  int upper = (int)tableSize - 1; // set the upper bound
423  char firstChar = *name; // get the first character for fast compares
424 
425  // loop until the range converges
426  while (lower <= upper)
427  {
428  // find a new middle location
429  int middle = lower + ((upper - lower) / 2);
430 
431  // only compare on the name if the first character matches
432  if (*table[middle].name == firstChar)
433  {
434  int rc = memcmp(name, table[middle].name, std::min(length, table[middle].length));
435  // if this compared equal, then compare the lengths...if not equal, the longer is
436  // the greater of the two.
437  if (rc == 0)
438  {
439  // same length, they are equal
440  if (length == table[middle].length)
441  {
442  // return the keyword code.
443  return table[middle].keywordCode;
444  }
445  // not equal, figure out how to divide the range
446  else if (length > table[middle].length)
447  {
448  // new lower bound
449  lower = middle + 1;
450  }
451  else
452  {
453  // going the other direction
454  upper = middle - 1;
455  }
456  }
457  // did not compare equal, so figure out which way to move
458  else if (rc > 0)
459  {
460  lower = middle + 1;
461  }
462  else
463  {
464  upper = middle - 1;
465  }
466  }
467  // still on the first char compares
468  else if (*table[middle].name < firstChar)
469  {
470  lower = middle + 1;
471  }
472  else
473  {
474  upper = middle - 1;
475  }
476  }
477  // zero is not found
478  return 0;
479 }

485 /**
486  * Resolve a token value to a subkeyword identifier.
487  *
488  * @return The subkeyword identifier.
489  */
490 InstructionSubKeyword RexxToken::subKeyword()
491 {
492  // not a symbol? not a keyword
493  if (!isSymbol())
494  {
495  return SUBKEY_NONE;
496  }
497  return static_cast<InstructionSubKeyword>(resolveKeyword(stringValue, subKeywords, tabSize(subKeywords)));
498 }

500 /**
501  * Resolve a token to a keyword value.
502  *
503  * @return The keyword identifier.
504  */
505 InstructionKeyword RexxToken::keyword()
506 {
507  // not a symbol? not a keyword
508  if (!isSymbol())
509  {
510  return KEYWORD_NONE;
511  }
512  return static_cast<InstructionKeyword>(resolveKeyword(stringValue, keywordInstructions, tabSize(keywordInstructions)));
513 }

517  *

520 BuiltinCode RexxToken::builtin()
521 {
522  // This can be a symbol or a literal
523  if (!isSymbolOrLiteral())
524  {

526  }
527  return static_cast<BuiltinCode>(resolveKeyword(stringValue, builtinFunctions, tabSize(builtinFunctions)));
