54 class OutputRedirector : public RexxInternalObject
55 {
56  public:
57 
58  // Note, we never directly create instances of this
59  // class so we don't need new, live, etc. These must

63 
64  virtual void init () { ; }

66  virtual void cleanup() { flushBuffer(); }

69 
70  virtual bool needsBuffering(InputRedirector *d);

72 
73  void write(RexxString *l);

76 
77  protected:
78  void flushBuffer();

80  void consumeBuffer(const char *data, const char *bufferEnd);
81 
82  OutputOption::Enum option; // REPLACE/APPEND option

84  RexxString *bufferedData; // if we're doing writes using buffered data, this is the tail of the last write.
