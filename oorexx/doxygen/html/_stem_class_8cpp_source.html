333 /**
334  * Remove an item from the collection. This is essentially
335  * equivalent to a drop operation on the stem variable.
336  *
337  * @param tails The set of tail indexes.
338  * @param argCount The number of indexes.
339  *
340  * @return The removed object. If nothing was removed, this returns
341  * .nil.
342  */
343 RexxInternalObject *StemClass::remove(RexxObject **tailElements, size_t argCount)
344 {
345  // if asked to remove the default value, reset this back to the name
346  if (argCount == 0)
347  {
348  // replace with the name and return the old value.
349  RexxInternalObject *oldValue = value;
350  setField(value, getName());
351  return oldValue;
352  }
353 
354  // compose the tail element
355  CompoundVariableTail resolved_tail((RexxInternalObject **)tailElements, argCount);
356  CompoundTableElement *compound = findCompoundVariable(resolved_tail);
357  // if there's a variable there, and it has a real value, then
358  // we have something to remove
359  if (compound != OREF_NULL && compound->getVariableValue() != OREF_NULL)
360  {
361  // get the value, which is the return value, and drop the variable.
362  RexxInternalObject *oldValue = compound->getVariableValue();
363  compound->drop();
364  return oldValue;
365  }
366  return TheNilObject; // nothing dropped.
367 }

453 /**
454  * Resolve the "stem.[a,b,c]=" to the equivalent stem.a.b.c= form,
455  * with all of the indices taken as constants
456  *
457  * @param tailElements
458  * The array of arguments. the first argument is the
459  * assigned value. All additional arguments are the
460  * tail pieces.
461  * @param argCount The count of arguments.
462  *
463  * @return Returns nothing.
464  */
465 RexxObject *StemClass::bracketEqual(RexxObject **tailElements, size_t argCount)
466 {
467 
468  // no value to set? This is an error
469  if (argCount == 0)
470  {
471  reportException(Error_Incorrect_method_noarg, IntegerOne);
472  }
473  RexxObject *newValue = requiredArgument(tailElements[0], ARG_ONE);
474 
475  // if the argument count is 1, we're just setting the default value
476  if (argCount == 1)
477  {
478  // stem value as default? don't allow this as it leads to recursion loops
479  if (isStem(newValue))
480  {
481  reportException(Error_Execution_nostem);
482  }
483 
484  setField(value, newValue);
485  // this clears out all of our elements and gets marked as having an
486  // explicit value.
487  tails.clear();
488  dropped = false;
489  return OREF_NULL;
490  }
491 
492  // create a searchable tail from the array elements
493  // and set the variable value
494  CompoundVariableTail resolved_tail((RexxInternalObject **)(tailElements + 1), argCount - 1);
495  CompoundTableElement *variable = getCompoundVariable(resolved_tail);
496  variable->set(newValue);
497  return OREF_NULL;
498 }

778 /**
779  * Retrieve a compound variable, returning the default value if the
780  * variable does not exist. This includes NOVALUE handling.
781  *
782  * @param context The current execution context.
783  * @param stemVariableName
784  * The name of the stem variable (needed to compose default value)
785  * @param resolved_tail
786  * The resolved tail.
787  *
788  * @return The variable value.
789  */
790 RexxObject *StemClass::evaluateCompoundVariableValue(RexxActivation *context, RexxString *stemVariableName, CompoundVariableTail &resolved_tail)
791 {
792  CompoundTableElement *variable = findCompoundVariable(resolved_tail);
793  // if we do not have a variable, then we need to sort out what the
794  // variable value nees to be

1397 /**
1398  * The merge sort routine. This will partition the data in to
1399  * two sections, mergesort each partition, then merge the two
1400  * partitions together.
1401  *
1402  * @param sd The sort information.
1403  * @param comparator The comparator object used for the compares.
1404  * @param strings The input set of strings.
1405  * @param working The working array (same size as the sorted array).
1406  * @param left The left bound of the partition.
1407  * @param right The right bounds of the parition.
1408  */
1409 void StemClass::mergeSort(SortData *sd, int (*comparator)(SortData *, RexxString *, RexxString *), RexxString **strings, RexxString **working, size_t left, size_t right)
1410 {
1411  size_t len = right - left + 1;
1412  // use insertion sort for small arrays
1413  if (len <= 7) {
1414  for (size_t i = left + 1; i <= right; i++) {
1415  RexxString *current = strings[i];
1416  RexxString *prev = strings[i - 1];
1417  if (comparator(sd, current, prev) < 0) {
1418  size_t j = i;
1419  do {
1420  strings[j--] = prev;
1421  } while (j > left && comparator(sd, current, prev = strings[j - 1]) < 0);
1422  strings[j] = current;
1423  }
1424  }
1425  return;
1426  }
1427 
1428  size_t mid = (right + left) / 2;
1429  mergeSort(sd, comparator, strings, working, left, mid);
1430  mergeSort(sd, comparator, strings, working, mid + 1, right);
1431  merge(sd, comparator, strings, working, left, mid + 1, right);
1432 }

1435 /**
1436  * Perform a merge of two sort partitions.
1437  *
1438  * @param sd The sort descriptor.
1439  * @param comparator The comparator used to compare elements.
1440  * @param strings The input strings.
1441  * @param working A working array used for the merge operations.
1442  * @param left The left bound for the merge.
1443  * @param mid The midpoint for the merge.
1444  * @param right The right bound of merge (inclusive).
1445  */
1446 void StemClass::merge(SortData *sd, int (*comparator)(SortData *, RexxString *, RexxString *), RexxString **strings, RexxString **working, size_t left, size_t mid, size_t right)
1447 {
1448  size_t leftEnd = mid - 1;
1449  // merging
1450 
1451  // if arrays are already sorted - no merge
1452  if (comparator(sd, strings[leftEnd], strings[mid]) <= 0) {
1453  return;
1454  }
1455 
1456  size_t leftCursor = left;
1457  size_t rightCursor = mid;
1458  size_t workingPosition = left;
1459 
1460  // use merging with exponential search
1461  do
1462  {
1463  RexxString *fromVal = strings[leftCursor];
1464  RexxString *rightVal = strings[rightCursor];
1465  if (comparator(sd, fromVal, rightVal) <= 0)
1466  {
1467  size_t leftInsertion = find(sd, comparator, strings, rightVal, -1, leftCursor + 1, leftEnd);
1468  size_t toCopy = leftInsertion - leftCursor + 1;
1469  arraycopy(strings, leftCursor, working, workingPosition, toCopy);
1470  workingPosition += toCopy;
1471  working[workingPosition++] = rightVal;
1472  // now we've added this
1473  rightCursor++;
1474  // step over the section we just copied...which might be
1475  // all of the remaining section
1476  leftCursor = leftInsertion + 1;
1477  }
1478  else
1479  {
1480  size_t rightInsertion = find(sd, comparator, strings, fromVal, 0, rightCursor + 1, right);
1481  size_t toCopy = rightInsertion - rightCursor + 1;
1482  arraycopy(strings, rightCursor, working, workingPosition, toCopy);
1483  workingPosition += toCopy;
1484  // insert the right-hand value
1485  working[workingPosition++] = fromVal;
1486  leftCursor++;
1487  rightCursor = rightInsertion + 1;
1488  }
1489  } while (right >= rightCursor && mid > leftCursor);
1490 
1491  // copy rest of array. If we've not used up the left hand side,
1492  // we copy that. Otherwise, there are items on the right side
1493  if (leftCursor < mid)
1494  {
1495  arraycopy(strings, leftCursor, working, workingPosition, mid - leftCursor);
1496  }
1497  else
1498  {
1499  arraycopy(strings, rightCursor, working, workingPosition, right - rightCursor + 1);
1500  }
1501 
1502  arraycopy(working, left, strings, left, right - left + 1);
1503 }

1544  size_t delta = 1;
1545  while (checkPoint <= right)
1546  {
1547  // if this is too big, then we're moving to the right
1548  if (comparator(sd, val, strings[checkPoint]) > limit)
1549  {
1550  // the left bound is at least this
1551  left = checkPoint + 1;
1552  }
1553  else
1554  {
1555  // we've found a right limit. We can stop scanning here
1556  right = checkPoint - 1;
1557  break;
1558  }
1559  // step the delta amount
1560  checkPoint += delta;
1561  // and double the movement amount
1562  delta = delta * 2;
1563  }
1564  // we should have now limited the bounds for the insertion point
1565  // now start in the middle and shrink the range with each comparison
1566  while (left <= right)
1567  {
1568  // start in the middle of the current range
1569  checkPoint = (left + right) / 2;
1570  if (comparator(sd, val, strings[checkPoint]) > limit)
1571  {
1572  // pull in the left end of the range
1573  left = checkPoint + 1;
1574  }
1575  else
1576  {
1577  // chop the right range
1578  right = checkPoint - 1;
1579  }
