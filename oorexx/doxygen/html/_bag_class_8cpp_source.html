75  * construct a BagClass with a given size.
76  *
77  * @param capacity The required capacity.
78  */
79 BagClass::BagClass(size_t capacity)
80 {
81  // get a suggested bucket size for this capacity
82  // NOTE: all of this needs to be done at the top-level constructor
83  // because of the way C++ constructors work. As each
84  // previous contructor level gets called, the virtual function
85  // pointer gets changed to match the class of the contructor getting
86  // called. We don't have access to our allocateContents() override
87  // until the final constructor is run.
88  size_t bucketSize = calculateBucketSize(capacity);
89  contents = allocateContents(bucketSize, bucketSize *2);
90 }

136 /**
137  * Create a list bag and populate with the argument items.
138  *
139  * @param args Pointer to the arguments.
140  * @param argCount The number of arguments.
141  *
142  * @return The populated list object.
143  */
144 BagClass *BagClass::ofRexx(RexxObject **args, size_t argCount)
145 {
146  // create a list item of the appopriate type.
147  Protected<BagClass> newBag = (BagClass *)newRexx(NULL, 0);
148 
149  // add all of the arguments
150  for (size_t i = 0; i < argCount; i++)
151  {
152  RexxObject *item = args[i];
153  // omitted arguments not allowed here.
154  if (item == OREF_NULL)
155  {
156  reportException(Error_Incorrect_method_noarg, i + 1);
157  }
158  newBag->put(item);
159  }
160  return newBag;
161 }
