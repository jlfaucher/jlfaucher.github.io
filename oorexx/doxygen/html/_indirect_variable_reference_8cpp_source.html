118 /**
119  * Parse a variable's string contents into an array of
120  * "good" variable retrievers.
121  *
122  * @param context The current execution context.
123  *
124  * @return An array of variable retrievers.
125  */
126 ArrayClass *RexxVariableReference::list(RexxActivation *context)
127 {
128  // evaluate the variable.
129  // this also traces and does the NOVALUE checks.
130  RexxObject *value = variableObject->evaluate(context, context->getStack());
131  // force to string form
132  Protected<RexxString> nameString = value->requestString();
133 
134  // get this as a list of words
135  Protected<ArrayClass> list = ((RexxString *)nameString)->subWords(OREF_NULL, OREF_NULL);
136 
137  size_t count = list->size();
138 
139  // now process all of the names, replacing them with retrievers
140  for (size_t i = 1; i <= count; i++)
141  {
142  // get the next name
143  RexxString *variable_name = (RexxString *)list->get(i);
144 
145  // get the first character
146  unsigned int character = variable_name->getChar(0);
147  // report obvious non-variable names
148  // start with a period
149  if (character == '.')
150  {
151  reportException(Error_Invalid_variable_period, variable_name);
152  }
153  // or a digit
154  else if (character >= '0' && character <= '9')
155  {
156  reportException(Error_Invalid_variable_number, variable_name);
157  }
158  // now convert into a variable reference
159  RexxVariableBase *retriever = VariableDictionary::getVariableRetriever(variable_name);
160  // if it did not convert (invalid characters, for example), report this
161  if (retriever == OREF_NULL)
162  {
163  reportException(Error_Symbol_expected_indirect, variable_name);
164  }
165  // replace the name with the retriever
166  list->put(retriever, i);
167  }
168  return list;
169 }
