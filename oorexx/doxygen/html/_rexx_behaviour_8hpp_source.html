126  inline RexxBehaviour *getSavedPrimitiveBehaviour()
127  {
128  uintptr_t behaviourID = (uintptr_t)getClassType();
129  // if this is an internal class, normalize this so we can
130  // restore this to the correct value if we add additional internal classes.
131  if (isInternalClass())
132  {
133  behaviourID -= T_Last_Exported_Class;
134  behaviourID |= INTERNALCLASS;
135  }
136  return (RexxBehaviour *)behaviourID;
137  }
138 
139  static inline RexxBehaviour *restoreSavedPrimitiveBehaviour(RexxBehaviour *b)
140  {
141  uintptr_t behaviourID = (uintptr_t)b;
142  // if this is an internal class, we need to convert back to
143  // the relative internal class id
144  if ((behaviourID & INTERNALCLASS) != 0)
145  {
146  behaviourID &= ~INTERNALCLASS; // turn off the internal marker
147  behaviourID += T_Last_Exported_Class; // turn back into an internal class
148  }
149  return &primitiveBehaviours[behaviourID]; // translate back into proper behaviour
150  }
