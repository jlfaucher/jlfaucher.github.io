209 RexxObject *MessageClass::notify(RexxObject *notificationTarget)

212 
213  classArgument(notificationTarget, TheRexxPackage->findClass(GlobalNames::MessageNotification, t), "notification target");

216  if (interestedParties == OREF_NULL)

219  }
220  // we always add this because we can reuse the message object.
221  interestedParties->append(notificationTarget);

234 /**
235  * Wait until message completion, either because the message has
236  * completed execution or because it terminated with an error.
237  *
238  * @return Always return nothing.
239  */
240 RexxObject *MessageClass::wait()
241 {
242  // we've not completed execution yet, so wait
243  if (!isComplete())
244  {
245  // we can wait, even if the message has not been triggered
246  // yet. The message object keeps the list of waiting activities
247  // and will wake them up when it finally is triggered and runs.
248 
249  // make sure we're not about to create a deadlock situation.
250  if (startActivity != OREF_NULL)
251  {
252  startActivity->checkDeadLock(ActivityManager::currentActivity);
253  }
254 
255  // we might be the first one to wait, so create the activities
256  // list if we are
257  if (waitingActivities == OREF_NULL)
258  {
259  setField(waitingActivities, new_array());
260  }
261  // add our activity to the list
262  waitingActivities->append(ActivityManager::currentActivity);
263  // and wait for the wake up call. We are handling this like
264  // it is a guard wait, which has its own semaphore rules.
265  ActivityManager::currentActivity->waitReserve(this);
266  }
267 
268  // always return no result value
269  return OREF_NULL;
270 }

273 /**
274  * Return the message result object. This will wait if
275  * the message has not completed.
276  *
277  * @return The result object (if any)
278  */
279 RexxObject *MessageClass::result()
280 {
281  // go wait, if necessary
282  wait();
283 
284  // did running this message cause an error? If so, we raise the same error
285  // condition here.
286  if (raiseError())
287  {
288  ActivityManager::currentActivity->reraiseException(condition);
289  }
290  // since this is requested via a method that will give an error if used
291  // in an expression, return .nil if there is no return value.
292  return resultOrNil(resultObject);
293 }

357 /**
358  * Verify we're not reusing a message object incorrectly.
359  */
360 void MessageClass::checkReuse()
361 {
362  // Once start has been called, this can no longer be reused.
363  if (isActivated())
364  {
365  reportException(Error_Execution_message_reuse);
366  }
367 
368  // we need the current activity to handle this
369  Activity *myActivity = ActivityManager::currentActivity;
370  // if we are waiting for things to start on another activity, this
371  // is also a reuse.
372  if (startPending() && myActivity != startActivity )
373  {
374  reportException(Error_Execution_message_reuse);
375  }
376 }

416 /**
417  * Dispatch the message object via a send()
418  *
419  * @return The result object.
420  */
421 RexxObject *MessageClass::dispatch()
422 {
423  // we need the current activity to handle this
424  Activity *myActivity = ActivityManager::currentActivity;
425 
426  // ok, now tell the stack frame we're running under that
427  // we want to be notified of any errors here.
428  myActivity->getTopStackFrame()->setObjNotify(this);
429 
430  // mark what activity we're running this under for
431  // deadlock detection
432  setField(startActivity, myActivity);
433 
434  // get a protected object for the result
435  ProtectedObject result(myActivity);
436  // now issue the message.
437  if (startscope != OREF_NULL)
438  {
439  receiver->messageSend(message, (RexxObject **)args->data(), args->size(), startscope, result);
440  }
441  else
442  {
443  receiver->messageSend(message, (RexxObject **)args->data(), args->size(), result);
444  }
445  resultObject = result;
446  setResultReturned();
447  // notify any waiters and return the result object
448  sendNotification();
449  return resultObject;
450 }

514 /**
515  * Execute this message asynchronously by spawning a
516  * new thread and dispatching this message on the new thread.
517  *
518  * @param _receiver The optional receiver object.
519  *
520  * @return returns nothing as a instruction message send.
521  */
522 RexxObject *MessageClass::start()
523 {
524  // make sure we're not trying to reuse a message object after a start operation
525  checkReuse();
526 
527  // clear all of the error/result flags before sending
528  clearCompletion();
529 
530  // ok, mark this as pending dispatch so that it can't be
531  // started a second time.
532  setStartPending();
533 
534  // validate that the scope override is valid
535  receiver->validateScopeOverride(startscope);
536 
537  // spawn a new activity off of the old activity
538  Activity *oldActivity = ActivityManager::currentActivity;
539  Activity *newActivity = oldActivity->spawnReply();
540 
541  // save current frame info in a StringTable as it has caused the creation of a new activity
542  Activity::setCallerStackFrameAsStringTable(oldActivity, newActivity, true); // create stackframe from parent/caller
543 
544  // mark which activity we're running on, then dispatch the message
545  // on the new activity (which is sitting waiting for work to perform)
546  setField(startActivity, newActivity);
547  newActivity->run(this);
548  // we have no return value.
549  return OREF_NULL;
550 }

642 /**
643  * Notify all interested parties after this message completed.
644  */
645 void MessageClass::sendNotification()
646 {
647  // we're no longer interested in any errors that occur.
648  ActivityManager::currentActivity->getTopStackFrame()->setObjNotify(OREF_NULL);
649  // if we have waiting activities, iterate over them and tell their activities to wake up
650  if (waitingActivities != OREF_NULL)
651  {
652  size_t count = waitingActivities->items();
653  for (size_t i = 1; i <= count; i++)
654  {
655  // get each activity and give them a poke.
656  Activity *waitingActivity = (Activity *)waitingActivities->get(i);
657  // we process this like it is a guard variable reserve, so
658  // we wake it up as if it was a guard wait.
659  waitingActivity->guardPost();
660  }
661  // clear the list so that we don't anchor those activities needlessly
662  waitingActivities = OREF_NULL;
663  }
664 
665  // now see if we have any interested parties to notify.
666  if (interestedParties != OREF_NULL)
667  {
668  size_t count = interestedParties->lastIndex();
669  for (size_t i = 1; i <= count; i++)
670  {
671  // get each message and give them a poke.
672  RexxObject *waitingMessage = (RexxObject *)interestedParties->get(i);
673  ProtectedObject result;
674  waitingMessage->sendMessage(GlobalNames::MessageComplete, this, result);
675  }
676  }
677 
678  // indicate we've notified everybody
679  setAllNotified();
680 }

828 {
829  // this class is defined on the object class, but this is actually attached
830  // to a class object instance. Therefore, any use of the this pointer
831  // will be touching the wrong data. Use the classThis pointer for calling
832  // any methods on this object from this method.
833  RexxClass *classThis = (RexxClass *)this;
834 
835  size_t num_args = argCount;
836 
837  // the first two arguments are required
838  if (num_args < 2 )
839  {
840  reportException(Error_Incorrect_method_minarg, IntegerTwo);
841  }
842 
843  // The receiver and the message name are required
844  RexxObject *_target = msgArgs[0];
845  requiredArgument(_target, "message target");
846  RexxObject *_message = msgArgs[1];
847  requiredArgument(_message, "message name");
848  ProtectedObject msgName;
849  ProtectedObject _startScope;
850  // decode the message argument into name and scope
851  RexxObject::decodeMessageName(_target, _message, msgName, _startScope);
852 
853  Protected<ArrayClass> argPtr;
854 
855  // are there arguments to be sent with the message?
856  if (num_args > 2 )
857  {
858  // get 3rd arg only concerned w/ 1st
859  // this is now required
860  char option = optionArgument(msgArgs[2], "AI", "argument style");
861  // arguments passed as an array
862  if (option == 'A')
863  {
864  // this must be here, and there can only be one additional argument
865  if (num_args < 4)
866  {
867  reportException(Error_Incorrect_method_minarg, IntegerFour);
868  }
869  if (num_args > 4)
870  {
871  reportException(Error_Incorrect_method_maxarg, IntegerFour);
872  }
873 
874  // get this directly and this is required to be an array
875  argPtr = arrayArgument(msgArgs[3], "message arguments");
876  }
877  // specified as individual arguments, so we just build an array
878  // directly from that. There need not be anything following the
879  // option
880  else if (option == 'I' )
881  {
882  argPtr = new_array(argCount - 3, msgArgs + 3);
883  }
884  }
885 
886  // if no arguments provided, default to a null array
887  if (argPtr == OREF_NULL)
888  {
889  argPtr = new_array();
890  }
891 
892  // we're all ready to create this now
893  MessageClass *newMessage = new MessageClass(_target, msgName, _startScope, argPtr);
894  ProtectedObject p(newMessage);
895 
896  // handle Rexx class completion
897  classThis->completeNewObject(newMessage);
898
