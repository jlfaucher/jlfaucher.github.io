114 /**
115  * Evaluate a logical expresion list.
116  *
117  * @param context The execution context.
118  * @param stack The evaluation stack.
119  *
120  * @return The result of the operation, either .true or .false.
121  */
122 RexxObject *RexxExpressionList::evaluate(RexxActivation *context, ExpressionStack *stack)
123 {
124  // loop through the expression list evaulating and then testing for the
125  // logical value
126  size_t count = expressionCount;
127 
128  // save the top of the stack for popping values off later.
129  size_t stacktop = stack->location();
130 
131  // create a result array with a matching size
132  Protected<ArrayClass> result = new_array(expressionCount);
133 
134  // there are no optional values in the list, so evaluate unconditionally.
135  for (size_t i = 0; i < count; i++)
136  {
137  // evaluate and trace
138  RexxInternalObject *expr = expressions[i];
139  // if this is a real expression (omitted expressions are permitted)
140  if (expr != OREF_NULL)
141  {
142  RexxObject *value = expr->evaluate(context, stack);
143  // trace this as an argument value
144  context->traceArgument(value);
145 
146  // add this to the created array
147  result->put(value, i + 1);
148  }
149  }
150 
151  // remove the arguments from the stack and push our result
152  stack->setTop(stacktop);
153  stack->push(result);
154 
155  // TODO: Need to reassess how this final result is traced.
156  // trace the array result and return it
157  context->traceResult(result);
158  return result;
159 }
