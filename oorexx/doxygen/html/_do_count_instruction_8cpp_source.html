244 /**
245  * Perform a test on whether this loop iteration should
246  * be executed or not.
247  *
248  * @param context The current execution context.
249  * @param stack The current evaluation stack.
250  * @param doblock The doblock for this instruction instance.
251  * @param first True if this is the first iteration, false for
252  * reExecute iterations.
253  *
254  * @return true if we should execute the loop block, false if
255  * we should terminate the loop.
256  */
257 bool RexxInstructionDoCountUntil::iterate(RexxActivation *context, ExpressionStack *stack, DoBlock *doblock, bool first)
258 {
259  // we don't check the UNTIL condition on the first iteration
260  if (first)
261  {
262  return doblock->checkFor();
263  }
264 
265  return !whileLoop.checkUntil(context, stack) && doblock->checkFor();
266 }
