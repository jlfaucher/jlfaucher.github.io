87 /**
88  * Static method for creating a new MethodClass instance.
89  *
90  * @param name The name given to the method and package.
91  * @param source The code source as an array of strings.
92  * @param sourceContext
93  * A parent source context that this method will inherit
94  * a package environment from.
95  *
96  * @return An executable method object.
97  */
98 MethodClass *LanguageParser::createMethod(RexxString *name, ArrayClass *source, PackageClass *sourceContext)
99 {
100  // check whether the data is representing an encoded compiled Rexx program (rexxc with the '/e' switch),
101  // i.e. the second entry consists of the string "/**/@REXX@" only;
102  if (source->items()>1 && source->get(2)->stringValue()->strCompare(ENCODED_NEEDLE))
103  {
104  Protected<RexxString> strSource=source->toString(GlobalNames::LINES, GlobalNames::LINEFEED); // use single LF for concatenation
105  Protected<BufferClass> program_buffer = new_buffer(strSource->getStringData(), strSource->getLength());
106 
107  // try to restore a compiled and encoded program first
108  Protected<MethodClass> method = MethodClass::restore(name, program_buffer);
109  if (method != (MethodClass *)OREF_NULL)
110  {
111  return method;
112  }
113  }
114 
115  // create the appropriate array source, then the parser, then generate the
116  // code.
117  Protected<ProgramSource> programSource = new ArrayProgramSource(source);
118  Protected<LanguageParser> parser = new LanguageParser(name, programSource);
119  return parser->generateMethod(sourceContext);
120 }

578 /**
579  * Generate a method object from a source collection.
580  *
581  * @param sourceContext
582  * An optional source context used to add additional visibility to
583  * dynamically generated methods.
584  *
585  * @return A method object represented by the leading code block
586  * of the source. Ideally, this code should not contain directives,
587  * but since this was allowed in the past, we need to continue
588  * to allow this.
589  */
590 MethodClass *LanguageParser::generateMethod(PackageClass *sourceContext)
591 {
592  // initialize, and compile all of the source.
593  compileSource();
594  // get the main section of the source package and make a method
595  // object from it. This is the package "main" executable.
596  package->mainExecutable = new MethodClass(name, mainSection);
597  // since we've explicitly made this a method, there is no
598  // longer an init code section to this package.
599  package->initCode = OREF_NULL;
600  // if we have a source context, then we need to inherit this
601  // context before doing the install so that anything from the parent
602  // context is visible during the install processing.
603  package->inheritPackageContext(sourceContext);
604 
605  // force the package to resolve classes/libraries now.
606  installPackage();
607  // return the main executable.
608  return (MethodClass *)package->mainExecutable;
609 }

996 /**
997  * Extract a clause from the source and return as a clause
998  * object. The clause object contains a list of all of the
999  * tokens contained within the clause and is used by
1000  * the parser to determine the type of instruction and
1001  * create the instruction parse tree.
1002  *
1003  * @return true if we managed to get a clause, false otherwise.
1004  */
1005 bool LanguageParser::nextClause()
1006 {
1007  SourceLocation location; // location of the clause
1008  SourceLocation tokenLocation; // location of each token
1009  RexxToken *token = OREF_NULL;
1010 
1011  // if reclamed is true, we have previously given up on processing a clause, so
1012  // the tokens are already available. This is not typically the case.
1013  if (!flags.test(reclaimed))
1014  {
1015  // we keep one clause object, just reset the tokens in the clause.
1016  clause->newClause();
1017  // scan over all null clauses. These could be blank lines, comments,
1018  // semi-colons, etc. Those are not interesting.
1019  for (;;)
1020  {
1021  // record the start position
1022  clause->setStart(lineNumber, lineOffset);
1023  // get the next source token. White space tokens are ignored in this context,
1024  // as we're looking for the beginning of a real clause.
1025  token = sourceNextToken(OREF_NULL);
1026  // OREF_NULL indicates we've hit the end of the source. Mark us as
1027  // finished and return
1028  if (token->isEndOfFile())
1029  {
1030  flags.set(noClause);
1031  return false;
1032  }
1033  // we have a token, but is it an end of clause marker (explicit or implicit
1034  // semicolon). If not, we've got a real clause to process.
1035  if (!token->isEndOfClause())
1036  {
1037  break;
1038  }
1039  // reset the clause and start again.
1040  clause->newClause();
1041  }
1042  // get the clause start position from the first token, and
1043  // make a copy for the start
1044  tokenLocation = token->getLocation();
1045  location = tokenLocation;
1046  // record this in the clause for potential error reporting.
1047  clause->setLocation(location);
1048 
1049  // now consume tokens until we find an end of clause marker.
1050  for (;;)
1051  {
1052  // get the next token. Blanks can now be significant inside
1053  // a clause.
1054  token = sourceNextToken(token);
1055  // and save the position again.
1056  tokenLocation = token->getLocation();
1057  // and quit once we've hit an end-of-clause (note, we'll never
1058  // see a null return here because there will be an end of clause
1059  // implied by the end of line. We only see the null case when
1060  // we're scanning for a new clause.
1061  if (token->isEndOfClause())
1062  {
1063  break;
1064  }
1065  }
1066 
1067  // the location of the last scanned token is the clause end position.
1068  location.setEnd(tokenLocation);
1069  // and set this in the class to give it the full bounds.
1070  clause->setLocation(location);
1071  }
1072 
1073  // no reclaimed clause now, regardless of how we got here.
1074  flags.reset(reclaimed);
1075  // always set the error information for the clause being processed.
1076  clauseLocation = clause->getLocation();
1077  // we have a clause
1078  return true;
1079 }

1166 /**
1167  * Translate a block of REXX code (delimited by possible
1168  * directive instructions
1169  *
1170  * @return A RexxCode object for this block.
1171  */
1172 RexxCode *LanguageParser::translateBlock()
1173 {
1174  // reset for a fresh translation
1175  initializeForTranslation();
1176 
1177  // add a dummy instruction at the front. All other instructions get chained off of this.
1178  RexxInstruction *instruction = new RexxInstruction(OREF_NULL, KEYWORD_FIRST);
1179  // this is the bottom of the control stack, and also the
1180  // first clause of the code stream.
1181  firstInstruction = instruction;
1182  lastInstruction = instruction;
1183 
1184  pushDo(instruction);
1185 
1186  // get a location for this block of code from the first and last instructions
1187  SourceLocation blockLocation;
1188 
1189  // save the block start position
1190  blockLocation.setStart(lineNumber == 0 ? 1 : lineNumber, lineOffset);
1191 
1192  // time to start actual parsing. Continue until we reach the end
1193  nextClause();
1194 
1195  for (;;)
1196  {
1197  // start with no instruction
1198  instruction = OREF_NULL;
1199  // At this point, we want to consume any label clauses, since they are
1200  // not real instructions.
1201  while (!noClauseAvailable())
1202  {
1203  // resolve this clause into an instruction
1204  instruction = nextInstruction();
1205  // if nothing is returned, this must be a directive, which terminates
1206  // parsing of this block.
1207  if (instruction == OREF_NULL)
1208  {
1209  break;
1210  }
1211  // not a label, break out of the loop
1212  if (!instruction->isType(KEYWORD_LABEL))
1213  {
1214  break;
1215  }
1216 
1217  // labels are not allowed within DO/LOOP/IF/SELECT groups
1218  InstructionKeyword type = topDo()->getType();
1219  // We are
1220  // - inside a DO, if top isControl, i. e. KEYWORD_LOOP_* or _DO_*
1221  // - inside a SELECT if top is KEYWORD_SELECT / SELECT_CASE / WHENTHEN / ENDWHEN / OTHERWISE
1222  // - inside an IF if top is IFTHEN or KEYWORD_ELSE
1223  // If ENDTHEN is at the top this is somewhat special. In this case
1224  // this label here may be preceding an ELSE (so we are inside the IF)
1225  // or something else (meaning we're just outside the IF block).
1226  // We handle this in the KEYWORD_ELSE path below.
1227  bool withinIf = type == KEYWORD_IFTHEN || type == KEYWORD_ELSE;
1228  bool withinSelect = type == KEYWORD_SELECT || type == KEYWORD_SELECT_CASE ||
1229  type == KEYWORD_WHENTHEN || type == KEYWORD_ENDWHEN || type == KEYWORD_OTHERWISE;
1230  if (topDo()->isControl() || withinIf || withinSelect)
1231  {
1232  // get our label name
1233  resetPosition(1); // firstToken() won't do ..
1234  RexxToken *token = nextReal();
1235  syntaxError(
1236  withinIf ? Error_Unexpected_label_if :
1237  (withinSelect ? Error_Unexpected_label_select :
1238  Error_Unexpected_label_do), token);
1239  }
1240  // append the label and try again
1241  addClause(instruction);
1242  nextClause();
1243  // need to zero this out in case we break the loop for an end of file
1244  instruction = OREF_NULL;
1245  }
1246  // ok, have we hit the end of the file or the end of the block?
1247  if (noClauseAvailable() || instruction == OREF_NULL)
1248  {
1249  // see what we have at the top of the control stack, we probably
1250  // have some cleanup to do.
1251  InstructionKeyword controltype = topDoType();
1252  // handle any then or when terminators
1253  while (controltype == KEYWORD_ENDTHEN || controltype == KEYWORD_ENDWHEN)
1254  {
1255  // pop these from the stack and flush and pending control instructions
1256  popDo();
1257  flushControl(OREF_NULL);
1258  controltype = topDoType();
1259  }
1260  // In theory, at this point we should be at the first instruction.
1261  // if not, we have an unclosed block instruction, which is an error
1262  if (!topDoIsType(KEYWORD_FIRST))
1263  {
1264  blockSyntaxError(topDo());
1265  }
1266  // remove the top instruction from the stack, and we're done
1267  popDo();
1268  break;
1269  }
1270 
1271  // now check if we need to adjust the control stack for this new instruction.
1272  InstructionKeyword type = instruction->getType();
1273  // if this is not an ELSE, we might have a pending THEN to finish.
1274  if (type != KEYWORD_ELSE)
1275  {
1276  // get the type at the top of the stack.
1277  InstructionKeyword controltype = topDoType();
1278  // we might need to pop off multiple pending thens while end of an IF or WHEN
1279  while (controltype == KEYWORD_ENDTHEN || controltype == KEYWORD_ENDWHEN)
1280  {
1281  popDo();
1282  flushControl(OREF_NULL);
1283  controltype = topDoType();
1284  }
1285  }
1286 
1287  // now check the actual disposition of this instruction. If it is a control type
1288  // add it immediately to the stream. Control types are IF, SELECT, or any of the
1289  // DO/LOOP types.
1290  if (instruction->isControl())
1291  {
1292  addClause(instruction);
1293  }
1294  // if this is an ELSE, we don't add a new level, but rather flush
1295  // any pending control levels.
1296  else if (type != KEYWORD_ELSE)
1297  {
1298  flushControl(instruction);
1299  }
1300  // validate allowed instructions in a SELECT
1301  if (topDoIsType(KEYWORD_SELECT, KEYWORD_SELECT_CASE) &&
1302  (type != KEYWORD_WHEN && type != KEYWORD_WHEN_CASE && type != KEYWORD_OTHERWISE && type != KEYWORD_END))
1303  {
1304  syntaxError(Error_When_expected_whenotherwise, topDo());
1305  }
1306 
1307  // now handle the different instructions to ensure they
1308  // are valid in their particular contexts.
1309  switch (type)
1310  {
1311  // a WHEN clause. The top of the Do stack must be a select.
1312  case KEYWORD_WHEN:
1313  case KEYWORD_WHEN_CASE:
1314  {
1315  // the top of the queue must be a SELECT instruction, but
1316  // we have TWO varieties of this. The second requires the
1317  // WHEN to be converted to a different type.
1318  RexxInstruction *second = topDo();
1319  if (second->isType(KEYWORD_SELECT) || second->isType(KEYWORD_SELECT_CASE))
1320  {
1321  // let the select know that another WHEN was added
1322  ((RexxInstructionSelect *)second)->addWhen((RexxInstructionIf *)instruction);
1323  }
1324  }
1325 
1326  // processing of an IF instruction, and also a WHEN (from above)
1327  case KEYWORD_IF:
1328  {
1329  // we need to finish the IF instruction.
1330  // get the next token
1331  RexxToken *token = nextReal();
1332  RexxInstruction *second;
1333  // Did the line end with no THEN? It must be on the next
1334  // line,
1335  if (token->isEndOfClause())
1336  {
1337  // get the next full clause, which should start with
1338  // the THEN keyword.
1339  // did we hit the end of file?, this is an error
1340  if (!nextClause())
1341  {
1342  // select appropriate error message: either for the IF here,
1343  // or for the WHEN / WHEN_CASE we fell through from above
1344  syntaxError(type == KEYWORD_IF ? Error_Then_expected_if : Error_Then_expected_when, instruction);
1345  }
1346 
1347  // now check the next token and ensure it is a THEN keyword.
1348  token = nextReal();
1349  // Not a THEN keyword? This is an error
1350  if (token->keyword() != KEYWORD_THEN)
1351  {
1352  // select appropriate error message: either for the IF here,
1353  // or for the WHEN / WHEN_CASE we fell through from above
1354  syntaxError(type == KEYWORD_IF ? Error_Then_expected_if : Error_Then_expected_when, instruction);
1355  }
1356  // create a new then clause attached to the IF
1357  second = thenNew(token, (RexxInstructionIf *)instruction);
1358  // now get the next token.. and ensure is something after the THEN
1359  token = nextReal();
1360 
1361  // if this is a clause end (e.g, end of line), parse off a clause
1362  // to make sure there is something there
1363  if (token->isEndOfClause())
1364  {
1365  if (!nextClause())
1366  {
1367  syntaxError(Error_Incomplete_do_then, instruction);
1368  }
1369  }
1370  else
1371  {
1372  // step back a token and trim the THEN off of the clause.
1373  previousToken();
1374  trimClause();
1375  }
1376  }
1377  // we have THEN on the same line as the IF
1378  else
1379  {
1380  // attach a THEN to the IF
1381  second = thenNew(token, (RexxInstructionIf *)instruction);
1382  // there might not be anything after the THEN, so we have to check
1383 
1384  token = nextReal();
1385  // if the THEN was the last thing, ensure we have something on the
1386  // next line
1387  if (token->isEndOfClause())
1388  {
1389  // we must have a clause, else there is an error
1390  if (!nextClause())
1391  {
1392  syntaxError(Error_Incomplete_do_then, instruction);
1393  }
1394  }
1395  else
1396  {
1397  // ok, back up one token and trim the clause to remove the THEN
1398  previousToken();
1399  trimClause();
1400  }
1401  }
1402 
1403  // add this to the instruction list and also add this to the
1404  // DO stack in case there is an ELSE to process.
1405  addClause(second);
1406  pushDo(second);
1407  // back to the top to get a new instruction
1408  continue;
1409  }
1410 
1411  // we have an ELSE instruction. Need to verify this is in a correct context.
1412  case KEYWORD_ELSE:
1413  {
1414  // ok, the top instruction is the key. It must be the
1415  // tail end of a THEN instruction that we can attach to.
1416  RexxInstruction *second = topDo();
1417  if (!second->isType(KEYWORD_ENDTHEN))
1418  {
1419  syntaxError(Error_Unexpected_then_else);
1420  }
1421 
1422  // no label is allowed immediately before the ELSE keyword
1423  if (lastInstruction->isType(KEYWORD_LABEL))
1424  {
1425  // unfortunately the label name isn't readily available
1426  // probably a rare error .. just use "ELSE" as message insert
1427  syntaxError(Error_Unexpected_label_if, new_string("ELSE"));
1428  }
1429 
1430  // ok, add the ELSE to the instruction list, pop the
1431  // THEN off of the control stack, push the ELSE on to the stack,
1432  // and hook them together. The ELSE will need to be completed
1433  // once the next instruction has been parsed off.
1434  addClause(instruction);
1435  second = popDo();
1436  // this is the ELSE
1437  pushDo(instruction);
1438  // join the THEN and ELSE together
1439  ((RexxInstructionElse *)instruction)->setParent((RexxInstructionEndIf *)second);
1440  ((RexxInstructionEndIf *)second)->setEndInstruction((RexxInstructionEndIf *)instruction);
1441 
1442  // check for a dangling ELSE now.
1443  RexxToken *token = nextReal();
1444 
1445  // if the next token is the end of the line (or potentially, a semicolon)
1446  // we're not at the end.
1447  if (token->isEndOfClause())
1448  {
1449  // dangling ELSE if we can't get another clause.
1450  if (!nextClause())
1451  {
1452  syntaxError(Error_Incomplete_do_else, instruction);
1453  }
1454  }
1455  // ELSE followed by an instruction on the same line. We need to
1456  // push the token we grabbed back on to the clause and trim off the
1457  // ELSE part we've already process.
1458  else
1459  {
1460  previousToken();
1461  trimClause();
1462  }
1463  // continue parsing instructions.
1464  continue;
1465  }
1466 
1467 
1468  // found the start of an OTHERWISE group. We need to verify that
1469  // we're really working on a SELECT.
1470  case KEYWORD_OTHERWISE:
1471  {
1472  // we must have a SELECT at the top of the control stack
1473  RexxInstruction *second = topDo();
1474  if (!second->isType(KEYWORD_SELECT) && !second->isType(KEYWORD_SELECT_CASE))
1475  {
1476  syntaxError(Error_Unexpected_when_otherwise);
1477  }
1478 
1479  // hook the otherwise up to the SELECT and push the otherwise
1480  // on to the top of the stack until we find an END.
1481  ((RexxInstructionSelect *)second)->setOtherwise((RexxInstructionOtherwise *)instruction);
1482  pushDo(instruction);
1483 
1484  // we could have the OTHERWISE on the same line as its following instruction, so
1485  // we need to trim the instruction
1486  RexxToken *token = nextReal();
1487  if (!token->isEndOfClause())
1488  {
1489  previousToken();
1490  trimClause();
1491  // we have a clause, go back around and process this
1492  continue;
1493  }
1494  // on the next line, fall through to the bottom of the loop
1495  // to get the next clause.
1496  break;
1497  }
1498 
1499  // An END instruction. This could be the closure for a DO, LOOP, or SELECT.
1500  // its matchup should be on the top of the control stack.
1501  case KEYWORD_END:
1502  {
1503  // ok, pop the top instruction. If this is the
1504  // correct type, we're finished with this. If not the
1505  // correct type, we have an error.
1506  RexxInstruction *second = popDo();
1507  type = second->getType();
1508  // that the top of the stack is an acceptable block-type instruction.
1509  // This includes all of the DO/LOOP variants, both Select types, and Otherwise
1510  if (!second->isBlock())
1511  {
1512  // we have a couple of specific errors based on what sort of instruction is
1513  // on the top.
1514  if (type == KEYWORD_ELSE)
1515  {
1516  syntaxError(Error_Unexpected_end_else);
1517  }
1518  else if (type == KEYWORD_IFTHEN || type == KEYWORD_WHENTHEN)
1519  {
1520  syntaxError(Error_Unexpected_end_then);
1521  }
1522  else
1523  {
1524  syntaxError(Error_Unexpected_end_nodo);
1525  }
1526  }
1527 
1528  // ok, we have something good, now there is some additional
1529  // processing required for specific block instructions
1530 
1531  // if the top of the stack is an OTHERWISE, then the
1532  // END is really closing the SELECT. Replace the Otherwise
1533  // with the Select.
1534  if (type == KEYWORD_OTHERWISE)
1535  {
1536  second = popDo();
1537  }
1538 
1539  // Now do the appropriate closure action based on the instruction type.
1540  if (type == KEYWORD_SELECT || type == KEYWORD_SELECT_CASE)
1541  {
1542  ((RexxInstructionSelect *)second)->matchEnd((RexxInstructionEnd *)instruction, this);
1543  }
1544  else /* must be a DO block */
1545  {
1546  ((RexxBlockInstruction *)second)->matchEnd((RexxInstructionEnd *)instruction, this);
1547  }
1548 
1549  // We've just completed a large block instruction. It is possible that
1550  // this was part of something like "if a then do;..end", in which case, the
1551  // conditional instruction is still in a pending state. Flush out the control
1552  // stack to complete now that the block has been closed.
1553  flushControl(OREF_NULL);
1554  break;
1555  }
1556 
1557  // start of new DO group (also picks up LOOP instruction)
1558  // this gets pushed on to the top of the control stack until
1559  // we find an END instruction.
1560  case KEYWORD_DO:
1561  case KEYWORD_LOOP:
1562  case KEYWORD_SIMPLE_BLOCK:
1563  case KEYWORD_LOOP_FOREVER:
1564  case KEYWORD_LOOP_OVER:
1565  case KEYWORD_LOOP_OVER_UNTIL:
1566  case KEYWORD_LOOP_OVER_WHILE:
1567  case KEYWORD_LOOP_OVER_FOR:
1568  case KEYWORD_LOOP_OVER_FOR_UNTIL:
1569  case KEYWORD_LOOP_OVER_FOR_WHILE:
1570  case KEYWORD_LOOP_WITH:
1571  case KEYWORD_LOOP_WITH_UNTIL:
1572  case KEYWORD_LOOP_WITH_WHILE:
1573  case KEYWORD_LOOP_WITH_FOR:
1574  case KEYWORD_LOOP_WITH_FOR_UNTIL:
1575  case KEYWORD_LOOP_WITH_FOR_WHILE:
1576  case KEYWORD_LOOP_CONTROLLED:
1577  case KEYWORD_LOOP_CONTROLLED_UNTIL:
1578  case KEYWORD_LOOP_CONTROLLED_WHILE:
1579  case KEYWORD_LOOP_COUNT:
1580  case KEYWORD_LOOP_COUNT_UNTIL:
1581  case KEYWORD_LOOP_COUNT_WHILE:
1582  case KEYWORD_LOOP_WHILE:
1583  case KEYWORD_LOOP_UNTIL:
1584  {
1585  pushDo(instruction);
1586  break;
1587  }
1588 
1589  // new select group. Again, we push this on the start of the stack
1590  // while it awaits its associated WHEN, OTHERWISE, and END bits.
1591  case KEYWORD_SELECT:
1592  case KEYWORD_SELECT_CASE:
1593  {
1594  pushDo(instruction);
1595  break;
1596  }
1597 
1598  // all other types of instructions don't require additional processing.
1599  default:
1600  break;
1601  }
1602  // grab another clause and go around.
1603  nextClause();
1604  }
1605 
1606  // we might have function calls or call instructions that had deferred resolution,
1607  // take care of those now
1608  resolveCalls();
1609 
1610  // the first instruction is just a dummy we use to anchor
1611  // everything while parsing. We can unchain that now.
1612  firstInstruction = firstInstruction->nextInstruction;
1613  // if this code block does not contain labels (pretty common if
1614  // using an oo style), get rid of those too
1615  if (labels->isEmpty())
1616  {
1617  labels = OREF_NULL;
1618  }
1619 
1620  // a code block need not have any instructions, and it may have leading or
1621  // trailing blanks. The source block started at the very beginning, now we need
1622  // find out the location of whatever terminated the block, which will be either
1623  // a directive clause that has already been parsed or actual end of the source itself.
1624  // if this block was terminated by discovering another directive, then
1625  // we set the end of this block to just before
1626  if (clause != OREF_NULL)
1627  {
1628  // get the next clause. If this clause starts at the beginning of
1629  // a line (common), then we need to use the end of the previous line as the
1630  // end location.
1631  SourceLocation nextLocation = clause->getLocation();
1632  if (nextLocation.getOffset() == 0)
1633  {
1634  size_t previousLine = nextLocation.getLineNumber() - 1;
1635  // this might actually have stepped things back a tick if there truly
1636  // was nothing there. In that case, set the source location to zero to indicate
1637  // there was nothing there
1638  if (blockLocation.getLineNumber() > previousLine)
1639  {
1640  blockLocation.setLineNumber(0);
1641  }
1642  else
1643  {
1644  // we need this to be the end of the previous line
1645  const char *data = NULL;
1646  size_t lineLength = 0;
1647 
1648  source->getLine(previousLine, data, lineLength);
1649  blockLocation.setEnd(previousLine, lineLength);
1650  }
1651  }
1652  // we can just set the end to the beginning of the pending clause
1653  else
1654  {
1655  blockLocation.setEnd(nextLocation.getLineNumber(), nextLocation.getOffset());
1656  }
1657  }
1658  // no active clause, so we set this to the very end of the file
1659  else
1660  {
1661  // we need this to be the end of the previous line
1662  const char *data = NULL;
1663  size_t lineLength = 0;
1664 
1665  source->getLine(source->getLineCount(), data, lineLength);
1666  blockLocation.setEnd(source->getLineCount(), lineLength);
1667  }
1668 
1669 
1670  // now create a code object that is attached to the package.
1671  // this will have all of the information needed to execute this code.
1672  RexxCode *code = new RexxCode(package, blockLocation, firstInstruction, labels, maxStack, variableIndex);
1673 
1674  // we don't automatically create the labels when we translate the block because
1675  // they might have been provided by an interpret. So always clear them out at the
1676  // end of a block.
1677  labels = OREF_NULL;
1678  // and return the code object.
1679  return code;
1680 }

1789 /**
1790  * Resolve dependencies between ::CLASS directives,
1791  * rearranging the order of the directives to preserve
1792  * relative ordering wherever possible. Classes with no
1793  * dependencies in this source file will be done first,
1794  * followed by those with dependencies in the appropriate
1795  * order
1796  */
1797 void LanguageParser::resolveDependencies()
1798 {
1799  // if we have classes, here, we need to sort out the install order
1800  // and configure the source package with this.
1801  if (!classes->isEmpty())
1802  {
1803  // get the count of classes we need to process.
1804  size_t classCount = classes->items();
1805 
1806  // run through the class list having each directive set up its
1807  // dependencies
1808  for (size_t i = 1; i <= classCount; i++)
1809  {
1810  ClassDirective *currentClass = (ClassDirective *)classes->get(i);
1811  currentClass->addDependencies(classDependencies);
1812  }
1813 
1814  // get an array for handling the ordering
1815  ArrayClass *classOrder = new_array(classCount);
1816  ProtectedObject p2(classOrder);
1817 
1818  // now we repeatedly scan the pending directory looking for a class
1819  // with no in-program dependencies - it's an error if there isn't one
1820  // as we build the classes we have to remove them (their names) from
1821  // pending list and from the remaining dependencies
1822  while (!classes->isEmpty())
1823  {
1824  // this is the next one we process
1825  ClassDirective *nextInstall = OREF_NULL;
1826 
1827  // the count will update each time through.
1828  classCount = classes->items();
1829  for (size_t i = 1; i <= classCount; i++)
1830  {
1831  // get the next directive
1832  ClassDirective *currentClass = (ClassDirective *)classes->get(i);
1833  // if this class doesn't have any additional dependencies, pick it next.
1834  if (currentClass->dependenciesResolved())
1835  {
1836  nextInstall = currentClass;
1837  // add this to the class ordering
1838  classOrder->append(nextInstall);
1839  // remove this from the processing list
1840  classes->deleteItem(i);
1841  break;
1842  }
1843  }
1844 
1845  // if nothing was located during this pass, we must have circular dependencies
1846  // this is an error.
1847  if (nextInstall == OREF_NULL)
1848  {
1849  // directive line where we can give as the source of the error
1850  ClassDirective *errorClass = (ClassDirective *)classes->get(1);
1851  clauseLocation = errorClass->getLocation();
1852  syntaxError(Error_Execution_cyclic, name);
1853  }
1854 
1855  // ok, now go remove these from the dependencies
1856  RexxString *className = nextInstall->getName();
1857 
1858  // now go through the pending list telling each of the remaining classes that
1859  // they can remove this dependency from their list
1860 
1861  // the count will update each time through.
1862  classCount = classes->items();
1863  for (size_t i = 1; i <= classCount; i++)
1864  {
1865  ClassDirective *currentClass = (ClassDirective *)classes->get(i);
1866  currentClass->removeDependency(className);
1867  }
1868  }
1869 
1870  // now add this to the package
1871  package->classes = classOrder;
1872  // this requires an install step
1873  package->setNeedsInstallation();
1874  }
1875 
1876  // ok, now we need to fill in any additional bits needed by the package
1877  if (!requires->isEmpty())
1878  {
1879  package->requires = requires;
1880  // this requires an install step
1881  package->setNeedsInstallation();
1882  }
1883  if (!libraries->isEmpty())
1884  {
1885  package->libraries = libraries;
1886  // this requires an install step
1887  package->setNeedsInstallation();
1888  }
1889  if (!routines->isEmpty())
1890  {
1891  package->routines = routines;
1892  }
1893  if (!publicRoutines->isEmpty())
1894  {
1895  package->publicRoutines = publicRoutines;
1896  }
1897  if (!unattachedMethods->isEmpty())
1898  {
1899  package->unattachedMethods = unattachedMethods;
1900  }
1901  if (!resources->isEmpty())
1902  {
1903  package->resources = resources;
1904  }
1905 }

1908 /**
1909  * Flush an pending instructions from the control stack
1910  * for a new added instruction.
1911  *
1912  * @param instruction
1913  * The newly added instruction.
1914  */
1915 void LanguageParser::flushControl(RexxInstruction *instruction)
1916 {
1917  // loop through the control stack
1918  for (;;)
1919  {
1920  // get the type of the instruction at the top of the control
1921  // stack.
1922  InstructionKeyword type = topDoType(); /* get the instruction type */
1923  // is this a pending ELSE clause? */
1924  if (type == KEYWORD_ELSE)
1925  {
1926  // pop the instruction off of the stack
1927  RexxInstruction *second = popDo();
1928  // and create a new end marker for that instruction.
1929  second = endIfNew((RexxInstructionIf *)second);
1930  // have an instruction to add?
1931  if (instruction != OREF_NULL)
1932  {
1933  // add to the current location and don't process any additional
1934  // instructions.
1935  addClause(instruction);
1936  instruction = OREF_NULL;
1937  }
1938  // now add the else terminator behind this.
1939  addClause(second);
1940  // we can go around again on this one.
1941  }
1942  // nested IF-THEN situation?
1943  else if (type == KEYWORD_IFTHEN || type == KEYWORD_WHENTHEN)
1944  {
1945  // get the top item
1946  RexxInstruction *second = popDo();
1947  // have an instruction to add?
1948  if (instruction != OREF_NULL)
1949  {
1950  // insert this here and null it out.
1951  addClause(instruction);
1952  instruction = OREF_NULL;
1953  }
1954  // we need a new end marker
1955  second = endIfNew((RexxInstructionIf *)second);
1956  // we add this clause behind the new one, and also add this
1957  // to the control stack as a pending instruction
1958  addClause(second);
1959  pushDo(second);
1960 
1961  // we're done with this
1962  break;
1963  }
1964  // some other type of construct. We just add the instruction to the
1965  // execution stream
1966  else
1967  {
1968  if (instruction != OREF_NULL)
1969  {
1970  addClause(instruction);
1971  }
1972  // all done flushing
1973  break;
1974  }
1975  }
1976 }

3503  // operator. Unfortunately, there's a problem hidden within
3504  // that approach. Parsing off the first term results in an
3505  // symbols that are located getting handled as variable references.

3507  // slot and creating a variable retriever for all keyword instruction names

3509  // ::method or ::routine section) in the program. So, we will attempt to
3510  // eliminate this particular situation up front just be examining the

3514  // The caller knows how to deal with a missing term.

3519  previousToken();

4293  total++;

4296  // a comma, we have more expressions to parse.
4297  terminatorToken = nextToken();
