135 /**
136  * Process an interation of a DO OVER loop.
137  *
138  * @param context The current execution context.
139  * @param stack The current evaluation stack.
140  *
141  * @return true if we should perform this iteration of the loop,
142  * false otherwise.
143  */
144 bool DoBlock::checkOver(RexxActivation *context, ExpressionStack *stack)
145 {
146  // the array was stored in the too field
147  ArrayClass *overArray = (ArrayClass *)to;
148  // are we past the end of the array?
149  if (overArray->lastIndex() < overIndex)
150  {
151  return false; // time to get out of here.
152  }
153 
154  // get the next element from the array. This should be a
155  // non-sparse array, but we need to double check anyway.
156  RexxObject *result = (RexxObject *)overArray->get(overIndex);
157  // use .nil for any empty slots
158  if (result == OREF_NULL)
159  {
160  result = TheNilObject; /* use .nil instead */
161  }
162 
163  // assign the control variable and trace this result
164  control->assign(context, result);
165  overIndex++;
166  return true;
167 }

170 /**
171  * Perform control variable checks on a DO/LOOP iteration.
172  *
173  * @param context The current execution context.
174  * @param stack The current evaluation stack.
175  * @param increment a flag indicating whether to increment the value or
176  * not. This will be true on the first loop iteration and
177  * false on all further iterations.
178  *
179  * @return True if the loop should continue, false if we've hit
180  * a termination condition.
181  */
182 bool DoBlock::checkControl(RexxActivation *context, ExpressionStack *stack, bool increment)
183 {
184  Protected<RexxObject> result = (RexxObject *)OREF_NULL;
185 
186  // if this is time to increment the value, perform the plus operation
187  // to add in the BY increment.
188  if (increment)
189  {
190  // get the control variable value and trace
191  result = control->evaluate(context, stack);
192  context->traceResult(result);
193  // increment using the plus operator
194  result = result->callOperatorMethod(OPERATOR_PLUS, by);
195  context->traceResult(result); // trace if necessary
196 
197  // the control variable gets set immediately, and and the assignment will also get traced
198  // increment result
199  control->assign(context, result);
200  }
201  else
202  {
203  // get the control variable value without tracing. We've
204  // already traced the initial assignment as part of the setup. This
205  // prevents getting an extra add looking item traced.
206  result = control->getValue(context);
207  }
208 
209 
210  // if we have a termination condition, do the compare now by calling the operator method.
211  if (to != OREF_NULL)
212  {
213  if (result->callOperatorMethod(compare, to) == TheTrueObject)
214  {
215  return false; // time to stop if this is true
216  }
217  }
218 
219  // do we have a forCount? perform that test now
220  if (forCount != SIZE_MAX)
221  {
222  return checkFor();
223  }
224 
225  return true; // still looping
226 }
