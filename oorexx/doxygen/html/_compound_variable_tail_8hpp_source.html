129  // apend data to the constructed tail
130  inline void append(const char *newData, size_t stringLen)
131  {
132  ensureCapacity(stringLen); // make sure have have space
133  memcpy(current, newData, stringLen); // copy this into the buffer
134  current += stringLen; // step the pointer
135  remainder -= stringLen; // adjust the lengths
136  }
