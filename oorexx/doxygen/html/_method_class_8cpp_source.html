379 /**
380  * Set the entire set of method attributes with one call. Used
381  * during source compilation.
382  *
383  * @param _access The access setting.
384  * @param _protected The protected setting.
385  * @param _guarded The guarded setting.
386  */
387 void MethodClass::setAttributes(AccessFlag _access, ProtectedFlag _protected, GuardFlag _guarded)
388 {
389  switch (_access)
390  {
391  case PRIVATE_SCOPE:
392  setPrivate();
393  break;
394 
395  case PACKAGE_SCOPE:
396  setPackageScope();
397  break;
398 
399  // covers PUBLIC and DEFAULT
400  default:
401  break;
402  }
403 
404  if (_protected == PROTECTED_METHOD)
405  {
406  setProtected();
407  }
408 
409  // both GUARDED and DEFAULT are guarded, so check for the reverse.
410  if (_guarded == UNGUARDED_METHOD)
411  {
412  setUnguarded();
413  }
414 }

441 /**
442  * Static method used for constructing new method objects in
443  * various contexts (such as the define method on the Class class).
444  *
445  * @param pgmname The name of the method we're creating.
446  * @param source The method source (either a string or an array).
447  * @param scope The scope that the new method object will be given.
448  * @param position The position used for reporting errors. This is the position
449  * of the source argument for the calling method context.
450  *
451  * @return The constructed method object.
452  */
453 MethodClass *MethodClass::newMethodObject(RexxString *pgmname, RexxObject *source, RexxClass *scope, const char *position)
454 {
455  // this is used in contexts where an existing method object is allowed...perform this
456  // check here and just return the original object if it is already a method.
457  if (isMethod(source))
458  {
459  return ((MethodClass *)source)->newScope(scope);
460  }
461 
462  // validate, and potentially transform, the method source object.
463  ArrayClass *newSourceArray = processExecutableSource(source, position);
464 
465  // if not a valid source, give an error
466  if (newSourceArray == OREF_NULL)
467  {
468  reportException(Error_Incorrect_method_no_method_type, position);
469  }
470 
471  // this method is called when methods are added to class, object, directory, etc.
472  // we want to inherit from the current execution source context if we can.
473 
474  PackageClass *sourceContext = OREF_NULL;
475 
476  // see if we have an active context and use the current source as the basis for the lookup
477  RexxActivation *currentContext = ActivityManager::currentActivity->getCurrentRexxFrame();
478  if (currentContext != OREF_NULL)
479  {
480  sourceContext = currentContext->getPackage();
481  }
482 
483  // create a method and give it the target scope
484  MethodClass *method = LanguageParser::createMethod(pgmname, newSourceArray, sourceContext);
485  method->setScope(scope);
486  return method;
487 }

490 /**
491  * Create a new method from REXX code contained in a string or an
492  * array
493  *
494  * @param init_args The array of arguments to the new method.
495  * @param argCount The argument count.
496  *
497  * @return A new method object.
498  */
499 MethodClass *MethodClass::newRexx(RexxObject **init_args, size_t argCount)
500 {
501  // this method is defined as an instance method, but this is actually attached
502  // to a class object instance. Therefore, any use of the this pointer
503  // will be touching the wrong data. Use the classThis pointer for calling
504  // any methods on this object from this method.
505  RexxClass *classThis = (RexxClass *)this;
506 
507  RexxString *programName;
508  Protected<ArrayClass> sourceArray;
509  PackageClass *sourceContext;
510 
511  // parse all of the options
512  processNewExecutableArgs(init_args, argCount, programName, sourceArray, sourceContext);
513 
514  // go create a method from whatever we were given for source.
515  ProtectedObject newMethod = LanguageParser::createMethod(programName, sourceArray, sourceContext);
516 
517  // finish up the object creation. Set the correct instance behavior (this could
518  // be a subclass), check for uninit methods, and finally, send an init message using any
519  // left over arguments.
520  classThis->completeNewObject(newMethod, init_args, argCount);
521  return newMethod;
522 }

551 /**
552  * Create a method from an external library source.
553  *
554  * @param methodName The method name.
555  * @param libraryDescriptor
556  *
557  * @return The resolved method object, or OREF_NULL if unable to
558  * load the routine.
559  */
560 MethodClass *MethodClass::loadExternalMethod(RexxString *methodName, RexxString *libraryDescriptor)
561 {
562  Protected<RexxString> name = stringArgument(methodName, "name");
563  Protected<RexxString> descriptor = stringArgument(libraryDescriptor, "descriptor");
564  // convert external into words
565  Protected<ArrayClass> _words = StringUtil::words(descriptor->getStringData(), descriptor->getLength());
566  // "LIBRARY libbar [foo]"
567  if (_words->size() > 0 && ((RexxString *)(_words->get(1)))->strCaselessCompare("LIBRARY"))
568  {
569  RexxString *library = OREF_NULL;
570  // the default entry point name is the internal name
571  RexxString *entry = name;
572 
573  // full library with entry name version?
574  if (_words->size() == 3)
575  {
576  library = (RexxString *)_words->get(2);
577  entry = (RexxString *)_words->get(3);
578  }
579  else if (_words->size() == 2)
580  {
581  library = (RexxString *)_words->get(2);
582  }
583  else // wrong number of tokens
584  {
585  reportException(Error_Translation_bad_external, descriptor);
586  }
587  // get the native code for this library
588  NativeCode *nmethod = PackageManager::loadMethod(library, entry);
589  // raise an exception if this entry point is not found.
590  if (nmethod == OREF_NULL)
591  {
592  return (MethodClass *)TheNilObject;
593  }
594  // turn into a real method object
595  return new MethodClass(name, nmethod);
596  }
597  else
598  {
599  // unknown external type
600  reportException(Error_Translation_bad_external, descriptor);
601  }
602  return OREF_NULL;
603 }
