50 /**
51  * Initialize the interpreter instance.
52  *
53  * @param i Our interpreter instance container.
54  * @param options The options used to initialize us. We can add additional
55  * platform-specific options if we wish.
56  */
57 void SysInterpreterInstance::initialize(InterpreterInstance *i, RexxOption *options)
58 {
59  instance = i;
60 
61  externalTraceEnabled = false; // off by default
62  /* scan current environment, */
63  const char *rxTraceBuf = getenv("RXTRACE");
64  if (rxTraceBuf != NULL)
65  {
66  if (!Utilities::strCaselessCompare(rxTraceBuf, "ON")) /* request to turn on? */
67  {
68  externalTraceEnabled = true; // turn on tracing of top-level activations for this instance
69  }
70  }
71 
72  // add our default search extension as both upper and lower case
73  addSearchExtension(".REX");
74  addSearchExtension(".rex");
75 }

92 {
93  // if the extension is not already in the extension list, add it
94  RexxString *ext = new_string(name);
95  if (!instance->searchExtensions->hasItem(ext))
96  {
97  instance->searchExtensions->append(ext);

116 /**
117  * Build a search path used for this resolution step.
118  *
119  * @param parentDir The location of the program calling us (can be null).
120  * @param extensionPath
121  * The system extension path (can be null).
122  */
123 SysSearchPath::SysSearchPath(const char *parentDir, const char *extensionPath)
124 {
125  const char *sysPath = getenv("PATH");
126  const char *rexxPath = getenv("REXX_PATH");
127 
128 #ifdef ORX_REXXPATH
129  if (rexxPath == NULL)
130  {
131  rexxPath = ORX_REXXPATH;
132  }
133 #endif
134 
135  addPath(parentDir);
136  // add on the current directory after the parent dir
137  addPath(".");
138  // next comes the extension path defined on the instance
139  addPath(extensionPath);
140  // followed by the REXX_PATH
141  addPath(rexxPath);
142  // and finally the PATH
143  addPath(sysPath);
144 }
