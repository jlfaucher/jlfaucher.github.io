82 /**
83  * Allocate a MutableBuffer object from Rexx code.
84  *
85  * @param args The pointer to the arrays.
86  * @param argc The count of arguments.
87  *
88  * @return A new mutable buffer object.
89  */
90 MutableBuffer *MutableBuffer::newRexx(RexxObject **args, size_t argc)
91 {
92  // this class is defined on the object class, but this is actually attached
93  // to a class object instance. Therefore, any use of the this pointer
94  // will be touching the wrong data. Use the classThis pointer for calling
95  // any methods on this object from this method.
96  RexxClass *classThis = (RexxClass *)this;
97 
98  // default string value
99  RexxString *string = GlobalNames::NULLSTRING;
100  size_t bufferLength = DEFAULT_BUFFER_LENGTH;
101  size_t defaultSize = 0;
102  // if we have at least one argument, then the first
103  // argument is an initial string value
104  if (argc >= 1)
105  {
106  if (args[0] != NULL)
107  {
108  string = stringArgument(args[0], ARG_ONE);
109  }
110  }
111 
112  // have a minimum buffer size specified
113  if (argc >= 2)
114  {
115  bufferLength = optionalLengthArgument(args[1], DEFAULT_BUFFER_LENGTH, ARG_TWO);
116  }
117 
118  // remember this as the default
119  defaultSize = bufferLength;
120 
121  // input string longer than demanded minimum size? expand accordingly
122  if (string->getLength() > bufferLength)
123  {
124  bufferLength = string->getLength();
125  }
126  // allocate the new object
127  Protected<MutableBuffer> newBuffer = new MutableBuffer(bufferLength, defaultSize);
128 
129  newBuffer->dataLength = string->getLength();
130  // copy the content
131  newBuffer->copyData(0, string->getStringData(), string->getLength());
132 
133  // handle Rexx class completion
134  classThis->completeNewObject(newBuffer, args, argc > 2 ? argc - 2 : 0);
135  return newBuffer;
136 }

251 /**
252  * Set the length of the data in the buffer. The limit is
253  * the current capacity of the buffer. If the length is
254  * extended beyond the current length, the extra characters
255  * of the buffer will be filled with nulls.
256  *
257  * @param newLength The new datalength. This is capped to the capacity of
258  * the buffer.
259  *
260  * @return The actual length the data has been set to. If the
261  * target length is greater than the capacity, the capacity
262  * value is returned.
263  */
264 size_t MutableBuffer::setDataLength(size_t newLength)
265 {
266  // cap the data length at the capacity
267  size_t capacity = getCapacity();
268  if (newLength > capacity)
269  {
270  newLength = capacity;
271  }
272 
273  size_t oldLength = getLength();
274  // set the new buffer length
275  dataLength = newLength;
276  // do we need to pad?
277  if (newLength > oldLength)
278  {
279  setData(oldLength, '\0', newLength - oldLength);
280  }
281 
282  return newLength;
283 }

410 /**
411  * insert string at given position
412  *
413  * @param str The string to insert.
414  * @param pos The starting position.
415  * @param len The insertion length.
416  * @param pad Any pad character.
417  *
418  * @return The same mutable buffer object.
419  */
420 MutableBuffer *MutableBuffer::insert(RexxObject *str, RexxObject *pos, RexxObject *len, RexxObject *pad)
421 {
422  // force this into string form
423  RexxString * string = stringArgument(str, ARG_ONE);
424 
425  // we're using optional length because 0 is valid for insert.
426  size_t begin = optionalNonNegative(pos, 0, ARG_TWO);
427  size_t insertLength = optionalLengthArgument(len, string->getLength(), ARG_THREE);
428 
429  char padChar = optionalPadArgument(pad, ' ', ARG_FOUR);
430 
431  size_t copyLength = std::min(insertLength, string->getLength());
432  size_t padLength = insertLength - copyLength;
433 
434 
435  // if inserting within the current bounds, we only need to add the length
436  // if inserting beyond the end, we need to make sure we add space for the gap too
437  if (begin < dataLength)
438  {
439  // if inserting a zero length string, this is simple!
440  if (insertLength == 0)
441  {
442  return this;
443  }
444  ensureCapacity(insertLength);
445  }
446  else
447  {
448  ensureCapacity(insertLength + (begin - dataLength));
449  }
450 
451 
452  // create space in the buffer
453  if (begin < dataLength)
454  {
455  openGap(begin, insertLength, dataLength - begin);
456  }
457  // else if we need to pad before the insertion
458  else if (begin > dataLength)
459  {
460  setData(dataLength, padChar, begin - dataLength);
461  }
462  // copy in the string contents
463  copyData(begin, string->getStringData(), copyLength);
464  // do we need data padding?
465  if (padLength > 0)
466  {
467  setData(begin + string->getLength(), padChar, padLength);
468  }
469  // inserting after the end? the resulting length is measured from the insertion point
470  if (begin > dataLength)
471  {
472  dataLength = begin + insertLength;
473  }
474  else
475  {
476  // just add in the inserted length
477  dataLength += insertLength;
478  }
479  return this;
480 }

483 /**
484  * Overlay a string value on the buffer data.
485  *
486  * @param str The string to overlay.
487  * @param pos The position to start the overlay.
488  * @param len The length of the overlay.
489  * @param pad An optional pad character.
490  *
491  * @return The same mutable buffer.
492  */
493 MutableBuffer *MutableBuffer::overlay(RexxObject *str, RexxObject *pos, RexxObject *len, RexxObject *pad)
494 {
495  RexxString *string = stringArgument(str, ARG_ONE);
496  size_t begin = optionalPositionArgument(pos, 1, ARG_TWO) - 1;
497  size_t replaceLength = optionalLengthArgument(len, string->getLength(), ARG_THREE);
498 
499  char padChar = optionalPadArgument(pad, ' ', ARG_FOUR);
500 
501  // make sure we have room for this
502  ensureCapacity(begin + replaceLength);
503 
504  // is our start position beyond the current data end?
505  if (begin > dataLength)
506  {
507  // add padding to the gap
508  setData(dataLength, padChar, begin - dataLength);
509  }
510 
511  // now overlay the string data
512  copyData(begin, string->getStringData(), std::min(replaceLength, string->getLength()));
513  // do we need additional padding?
514  if (replaceLength > string->getLength())
515  {
516  // pad the section after the overlay
517  setData(begin + string->getLength(), padChar, replaceLength - string->getLength());
518  }
519 
520  // did this add to the size?
521  if (begin + replaceLength > dataLength)
522  {
523  //adjust upward
524  dataLength = begin + replaceLength;
525  }
526  return this;
527 }

552 /**
553  * Replace a target substring within a string with
554  * a new string value. This is similar overlay, but
555  * replacing might cause the characters following the
556  * replacement position to be shifted to the left or
557  * right.
558  *
559  * @param str The replacement string.
560  * @param pos The target position (required).
561  * @param len The target length (optional). If not specified, the
562  * length of the replacement string is used, and this
563  * is essentially an overlay operation.
564  * @param pad A padding character if padding is required. The default
565  * pad is a ' '. Padding only occurs if the replacement
566  * position is beyond the current data length.
567  *
568  * @return The target mutablebuffer object.
569  */
570 MutableBuffer *MutableBuffer::replaceAt(RexxObject *str, RexxObject *pos, RexxObject *len, RexxObject *pad)
571 {
572  RexxString *string = stringArgument(str, "new");
573  size_t begin = positionArgument(pos, "position") - 1;
574  size_t newLength = string->getLength();
575  size_t replaceLength = optionalLengthArgument(len, newLength, "length");
576 
577  char padChar = optionalPadArgument(pad, ' ', "pad");
578  size_t finalLength;
579 
580  // if replaceLength extends beyond the end of the string
581  // then we cut it.
582  if (begin > dataLength)
583  {
584  replaceLength = 0;
585  }
586  else if (begin + replaceLength > dataLength)
587  {
588  replaceLength = dataLength - begin;
589  }
590 
591  // We need to add the delta between the excised string and the inserted
592  // replacement string.
593  //
594  // If this extends beyond the end of the string, then we require space for
595  // the position + the replacement string length. Else we find the required
596  // size (may be smaller than before)
597  if (begin > dataLength)
598  {
599  finalLength = begin - replaceLength + newLength;
600  }
601  else
602  {
603  finalLength = dataLength - replaceLength + newLength;
604  }
605 
606  // make sure we have room for this
607  ensureCapacity(finalLength);
608 
609  // is our start position beyond the current data end?
610  // NB: Even though we've adjusted the buffer size, the dataLength is still
611  // the original entry length.
612  if (begin > dataLength)
613  {
614  // add padding to the gap
615  setData(dataLength, padChar, begin - dataLength);
616  // now overlay the string data
617  copyData(begin, string->getStringData(), newLength);
618  }
619  else
620  {
621  // if the strings are of different lengths, we need to adjust the size
622  // of the gap we're copying into. Only adjust if there is a real gap
623  if (replaceLength != newLength && begin + replaceLength < dataLength)
624  {
625  // snip out the original string
626  adjustGap(begin, replaceLength, newLength);
627  }
628  // now overlay the string data
629  copyData(begin, string->getStringData(), newLength);
630  }
631 
632  // and finally adjust the length
633  dataLength = finalLength;
634  // our return value is always the target mutable buffer
635  return this;
636 }

701  dataLength = std::min(dataLength, newsize);
702  newBuffer->copyData(0, data->getData(), dataLength);
703  // replace the old buffer
704  setField(data, newBuffer);
705  // and update the size....
706  bufferLength = newsize;
707  }
708  return this;
709 }

971 MutableBuffer *MutableBuffer::changeStr(RexxString *needle, RexxString *newNeedle, RexxInteger *countArg)
972 {

974  newNeedle = stringArgument(newNeedle, ARG_TWO);
975 
976  // we'll only change up to a specified count. If not there, we do everything.
977  size_t count = optionalNonNegative(countArg, Numerics::MAX_WHOLENUMBER, ARG_THREE);
978 
979  // if no change is requested, return the original string.
980  if (count == 0)
981  {
982  return this;
983  }
984 
985  size_t needleLength = needle->getLength();
986  // if needle is the nullstring, return the original string.
987  if (needleLength == 0)
988  {
989  return this;
990  }

992  const char *newData = newNeedle->getStringData();
993 
994  // an inplace update has complications, depending on whether the new string is shorter,
995  // the same length, or longer
996 
997  size_t matches;
998 
999  // simplest case...same length strings. We can just overlay the existing occurrences
1000  if (needleLength == newLength)
1001  {
1002  const char *source = getStringData();
1003  size_t sourceLength = getLength();
1004  size_t _start = 0; /* set a zero starting point */
1005  // search for at most 'count' occurences
1006  for (matches = 1; matches <= count; matches++)
1007  {
1008  // search for the next occurrence
1009  size_t matchPos = StringUtil::pos(source, sourceLength, needle, _start, sourceLength);
1010  if (matchPos == 0) // no more match? we're done
1011  {
1012  break;
1013  }
1014  copyData(matchPos - 1, newData, newLength);
1015  // step to the next search position
1016  _start = matchPos + newLength - 1;
1017  }
1018  // no need to update the result length as it will be the same
1019  }
1020 
1021  // this will be a shorter thing, so we can do things in place as if we were using two buffers
1022  else if (needleLength > newLength)
1023  {

1026  size_t _start = 0;
1027  // get our string bounds
1028  const char *source = getStringData();
1029  size_t sourceLength = getLength();
1030  // search for at most 'count' occurences
1031  for (matches = 0; matches < count; matches++)
1032  {
1033  // search for the next occurrence
1034  size_t matchPos = StringUtil::pos(source, sourceLength, needle, _start, sourceLength);
1035  if (matchPos == 0) // no more match? we're done
1036  {
1037  break;
1038  }
1039

1042  // if this skipped over characters, we need to copy those
1043  if (copyLength != 0)
1044  {
1045  copyData(copyOffset, source + _start, copyLength);
1046  copyOffset += copyLength;
1047  }
1048  // replacing with a non-null string, copy the replacement string in
1049  if (newLength != 0)
1050  {
1051  copyData(copyOffset, newData, newLength);
1052  copyOffset += newLength;
1053  }
1054  _start = matchPos + needleLength - 1; /* step to the next position */
1055  }
1056  // we likely have some remainder that needs copying
1057  if (_start < sourceLength)
1058  {
1059  copyData(copyOffset, source + _start, sourceLength - _start);
1060  }
1061 
1062  // update the result length
1063  dataLength = sourceLength - matches * (needleLength - newLength);
1064  }

1080  size_t resultLength = getLength() + matches * (newLength - needleLength);
1081  ensureCapacity(resultLength);
1082

1084

1087  size_t _start = 0;

1089  const char *source = getStringData() + growth;
1090  size_t sourceLength = getLength();
1091  // this shifts everything to the end of the buffer. From there,
1092  // we pull pieces back into place.
1093  openGap(0, growth, sourceLength);
1094  // this is our scan offset
1095  for (size_t i = 0; i < matches; i++)
1096  {
1097  // look for each instance and replace
1098  size_t matchPos = StringUtil::pos(source, sourceLength, needle, _start, sourceLength);
1099  size_t copyLength = (matchPos - 1) - _start; /* get the next length to copy */
1100  // if this skipped over characters, we need to copy those
1101  if (copyLength != 0)

1116  {
