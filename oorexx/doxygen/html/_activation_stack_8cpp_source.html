145 /**
146  * Expand the capacity of the stack to add at least entries
147  * additional values on the stack.
148  *
149  * @param entries
150  */
151 void ActivationStack::expandCapacity(size_t entries)
152 {
153  ActivationFrameBuffer *next;
154  entries = std::max(entries, (size_t)DefaultFrameBufferSize);
155  // do we have an unused one we're holding ready that has enough room?
156  if (unused != OREF_NULL && unused->hasCapacity(entries))
157  {
158  // just activate this one for use
159  next = unused;
160  unused = OREF_NULL;
161  }
162  else
163  {
164  // create a new frame buffer
165  next = new_activationFrameBuffer(entries);
166  }
167  // chain the existing buffer off of the new one
168  next->push(current);
169  // set this up as the new current stack
170  current = next;
171 }
