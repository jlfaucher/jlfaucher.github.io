60 int main(int argc, char *argv[], char *envp[])
61 {
62  int args = argc - 2; // number of RXSUBCOM [ACTION] args
63 
64  if (args < 0) // no RXSUBCOM argument specified
65  {
66  // The RXSUBCOM parameters are incorrect
67  parmerr(Error_RXSUBC_general);
68  }
69 
70  CASE("REGISTER")
71  {
72  if (args != 3) // REGISTER envname dllname procname
73  {
74  // The RXSUBCOM REGISTER parameters are incorrect
75  parmerr(Error_RXSUBC_register);
76  }
77  return RexxRegisterSubcomDll(argv[2], argv[3], argv[4], NULL, RXSUBCOM_DROPPABLE);
78  }
79 
80  CASE("QUERY")
81  {
82  if (args < 1 || args > 2) // QUERY envname [dllname]
83  {
84  // The RXSUBCOM QUERY parameters are incorrect
85  parmerr(Error_RXSUBC_query);
86  }
87  unsigned short flags;
88  return RexxQuerySubcom(argv[2], args == 1 ? NULL : argv[3], &flags, NULL);
89  }
90 
91  CASE("DROP")
92  {
93  if (args < 1 || args > 2) // DROP envname [dllname]
94  {
95  // The RXSUBCOM DROP parameters are incorrect
96  parmerr(Error_RXSUBC_drop);
97  }
98  return RexxDeregisterSubcom(argv[2], args == 1 ? NULL : argv[3]);
99  }
100 
101  CASE("LOAD")
102  {
103  if (args < 1 || args > 2) // LOAD envname [dllname]
104  {
105  // The RXSUBCOM LOAD parameters are incorrect
106  parmerr(Error_RXSUBC_load);
107  }
108  return RexxLoadSubcom(argv[2], args == 1 ? NULL : argv[3]);
109  }
110 
111  // The RXSUBCOM parameters are incorrect
112  parmerr(Error_RXSUBC_general);
113  return 0; // this will never be reached
114 }
