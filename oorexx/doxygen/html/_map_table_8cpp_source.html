109 /**
110  * Place an item into a hash collection using a key.
111  *
112  * @param value The inserted value.
113  * @param index The insertion key.
114  *
115  * @return The retrieved object. Returns OREF_NULL if the object
116  * was not found.
117  */
118 void MapTable::put(size_t value, RexxInternalObject *index)
119 {
120  // try to insert in the existing hash tables...if this
121  // fails, we're full and need to reallocate.
122  if (!contents->put(value, index))
123  {
124  // reallocate and try again
125  reallocateContents();
126  contents->put(value, index);
127  }
128 }

150 /**
151  * Reallocate the hash bucket to a larger version after
152  * a failed put() operation.
153  */
154 void MapTable::reallocateContents()
155 {
156  // create a new bucket and merge the old bucket into it, then replace the contents
157  // with the new ones.
158  MapBucket *newContents = new (contents->totalSize * 2) MapBucket(contents->totalSize * 2);
159  contents->merge(newContents);
160  contents = newContents;
161 }
