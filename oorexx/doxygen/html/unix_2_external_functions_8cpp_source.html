93 /******************************************************************************/
94 /* Name: SysExternalFunction */
95 /* */
96 /* Notes: Handles searching for and executing an external function. The */
97 /* search order is: */
98 /* 1) Macro-space pre-order functions */
99 /* 2) Registered external functions */
100 /* 3) REXX programs with same extension (if applicable) */
101 /* 4) REXX programs with default extension */
102 /* 5) Macro-space post-order functions */
103 /******************************************************************************/
104 bool SystemInterpreter::invokeExternalFunction(
105  RexxActivation *activation, /* Current Activation */
106  Activity *activity, /* activity in use */
107  RexxString *target, /* Name of external function */
108  RexxObject **arguments, /* Argument array */
109  size_t argcount, /* count of arguments */
110  RexxString *calltype, /* Type of call */
111  ProtectedObject &result)
112 {
113  if (activation->callMacroSpaceFunction(target, arguments, argcount, calltype, RXMACRO_SEARCH_BEFORE, result))
114  {
115  return true;
116  }
117  /* no luck try for a registered func */
118  if (PackageManager::callNativeRoutine(activity, target, arguments, argcount, result))
119  {
120  return true;
121  }
122  /* have activation do the call */
123  if (activation->callExternalRexx(target, arguments, argcount, calltype, result))
124  {
125  return true;
126  }
127  /* function. If still not found, */
128  /* then raise an error */
129  if (activation->callMacroSpaceFunction(target, arguments, argcount, calltype, RXMACRO_SEARCH_AFTER, result))
130  {
131  return true;
132  }
133 
134  return false;
135 }

185 /* Subroutine Name: BuildEnvlist */
186 /* */
187 /* Descriptive Name: Build saved environment block */

189 /* Function: Builds a block containing all of the */

195 RexxObject* SystemInterpreter::buildEnvlist()
196 {
197  BufferClass *newBuffer; /* Buffer object to hold env */
198  char **Environment; /* environment pointer */
199  size_t size = 0; /* size of the new buffer */
200  char *New; /* starting address of buffer */
201  Environment = getEnvironment(); /* get the ptr to the environ */
202 
203  for (; *Environment != NULL; Environment++)
204  {
205  size += strlen(*Environment); /* calculate the size for all */
206  size++; /* environment variables+'\0' */
207  } /* now get current dir */
208  if (!size)
209  {
210  return OREF_NULL; /* no envrionment ! */
211  }
212 
213  FileNameBuffer curr_dir;
214 
215  // start with a copy of the current working directory
216  SysFileSystem::getCurrentDirectory(curr_dir);
217  size_t dirLen = curr_dir.length();
218 
219  size += dirLen; /* add the space for curr dir */
220  size++; /* and its terminating '\0' */
221  size += sizeof(size_t); /* this is for the size itself*/
222  /* Now we have the size for */
223  /* allocating the new buffer */
224  newBuffer = new_buffer(size); /* let's do it */
225  /* Get starting address of buf*/
226  New = newBuffer->getData();
227  ((ENVENTRY *)New)->size = size; /* first write the size */
228  New += sizeof(size_t); /* update the pointer */
229  strcpy(New, (const char *)curr_dir); // add the current dir to the buffer
230  New += dirLen + 1; /* update the pointer */
231  Environment = getEnvironment(); /* reset to begin of environ */
232  /* Loop through environment */
233  /* and copy all entries to the*/
234  /* buffer, each terminating */
235  /* with '\0' */
236  for (; *Environment != NULL; Environment++)
237  {
238  // copy the entry and step over it
239  size_t len = strlen(*Environment);
240  strcpy(New, *Environment);
241  New += len + 1;
242  }
243  return newBuffer; /* return the pointer */
244 }

258 void SystemInterpreter::restoreEnvironment(
259  void *CurrentEnv) /* saved environment */
260 {
261  char *current; /* ptr to saved environment */
262  size_t size; /* size of the saved space */
263  size_t length; /* string length */
264  char *begin; /* begin of saved space */
265  char **Environment; /* environment pointer */
266 
267  char *del = NULL; /* ptr to old unused memory */
268  char *Env_Var_String; /* enviornment entry */
269  char namebufsave[256], namebufcurr[256];
270  char *np;
271  int i;
272 
273  Environment = getEnvironment(); /* get the current environment*/
274 
275  begin = current = (char *)CurrentEnv; /* get the saved space */
276  size = ((ENVENTRY *)current)->size; /* first read out the size */
277  current += sizeof(size_t); /* update the pointer */
278  if (chdir(current) == -1) /* restore the curr dir */
279  {
280  char msg[1024];
281  snprintf(msg, sizeof(msg), "Error restoring current directory: %s", current);
282  reportException(Error_System_service_service, msg);
283  }
284  current += strlen(current); /* update the pointer */
285  current++; /* jump over '\0' */
286  if (!putflag)
287  { /* first change in the */
288  /* environment ? */
289  /* copy all entries to dynamic memory */
290  /*for all entries in the env */
291  for (; *Environment != NULL; Environment++)
292  {
293  length = strlen(*Environment) + 1; /* get the size of the string */
294  /* and alloc space for it */
295  Env_Var_String = (char *)malloc(length);
296  memcpy(Env_Var_String, *Environment, length); /* copy the string */
297  putenv(Env_Var_String); /* and make it part of env */
298  }
299  putflag = 1; /* prevent do it again */
300  }
301  /* Loop through the saved env */
302  /* entries and restore them */
303  for (; (size_t)(current - begin) < size; current += (strlen(current) + 1))
304  {
305  Environment = getEnvironment(); /* get the environment */
306  del = NULL;
307  np = current;
308  /* extract the the name */
309  /* from the saved enviroment */
310  for (i = 0; (*np != '=') && (i < 255); np++, i++)
311  {
312  memcpy(&(namebufsave[i]), np, 1); /* copy the character */
313  }
314  memcpy(&(namebufsave[i]), "\0", 1); /* copy the terminator */
315  /* find the entry in the env */
316  for (; *Environment != NULL; Environment++)
317  {
318  np = *Environment;
319  /* extract the the name */
320  /* from the current env */
321  for (i = 0; (*np != '=') && (i < 255); np++, i++)
322  {
323  memcpy(&(namebufcurr[i]), np, 1); /* copy the character */
324  }
325  memcpy(&(namebufcurr[i]), "\0", 1); /* copy the terminator */
326 
327  if (!strcmp(namebufsave, namebufcurr))
328  { /* have a match ? */
329  del = *Environment; /* remember it for deletion */
330  break; /* found, so get out of here */
331  }
332  }
333  if (putenv(current) == -1)
334  {
335  reportException(Error_System_service_service, "Error restoring environment variable");
336  }
337  if (del) /* if there was an old entry */
338  {
339  free(del); /* free it */
340  }
341  }
