52 /**
53  * A descriptor for a single source line.
54  */
55 class LineDescriptor
56 {
57  public:
58  LineDescriptor() : position(0), length(0) { }
59 
60  // clear the descriptor
61  inline void clear()
62  {
63  position = 0;
64  length = 0;
65  }
66 
67  // map the descriptor to a line position/length
68  inline void getLine(const char *dataPointer, const char *&linePointer, size_t &lineLength)
69  {
70  linePointer = dataPointer + position;
71  lineLength = length;
72  }
73 
74  size_t position; // position with the buffer
75  size_t length; // length of the line
76 };

79 /**
80  * Base class for data holding program source. This also acts
81  * as a source provider for "sourceless" code units.
82  */
83 class ProgramSource: public RexxInternalObject
84 {
85 public:
86  void *operator new(size_t);
87  inline void operator delete(void *) { ; }
88 
89  ProgramSource() : firstLine(1), lineCount(0) { };
90  inline ProgramSource(RESTORETYPE restoreType) { ; };
91 
92  // each subclass will need to implement this.
93  // default setup is to do nothing.
94  virtual void setup() { }
95  // a default implementation that returns nothing.
96  virtual void getLine(size_t lineNumber, const char *&data, size_t &length)
97  {
98  data = NULL;
99  length = 0;
100  }
101 
102  // provides with real source need to override this
103  virtual bool isTraceable() { return false; }
104  // provides the starting line location. Can be non-zero
105  // if this is an interpret
106  virtual size_t getFirstLine() { return firstLine; }
107  size_t getLineCount() { return lineCount; }
108 
109  RexxString *getStringLine(size_t lineNumber);
110  RexxString *getStringLine(size_t position, size_t startOffset, size_t endOffset = SIZE_MAX);
111  RexxString *extract(SourceLocation &location);
112  ArrayClass *extractSourceLines(SourceLocation &location);
113 
114 protected:
115 
116  size_t firstLine; // the first line of the source for parsing
117  size_t lineCount; // count of lines in the source file.
118 };
