360 {
361  // make sure we have a valid name and delegate to the source object

363  ProtectedObject p(name);

545 PackageClass *PackageClass::newRexx(
546  RexxObject **init_args, /* subclass init arguments */
547  size_t argCount, /* number of arguments passed */
548  size_t named_argCount)
549 /******************************************************************************/
550 /* Function: Create a new packag from REXX code contained in a file or an */
551 /* array */
552 /******************************************************************************/
553 {
554  RexxObject *pgmname; /* method name */
555  RexxObject *_source; /* Array or string object */
556  size_t initCount = 0; /* count of arguments we pass along */
557  RexxActivity *activity = ActivityManager::currentActivity;
558  InterpreterInstance *instance = activity->getInstance();
559 
560  /* break up the arguments */
561 
562  RexxClass::processNewArgs(init_args, argCount, &init_args, &initCount, 2, (RexxObject **)&pgmname, (RexxObject **)&_source);
563 
564  PackageClass *package = OREF_NULL;
565 
566  /* get the package name as a string */
567  RexxString *nameString = stringArgument(pgmname, OREF_positional, "name");
568  ProtectedObject p_nameString(nameString);
569  if (_source == OREF_NULL)
570  {
571  // if no directly provided source, resolve the name in the global context and have the instance
572  // load the file.
573  RexxString *resolvedName = instance->resolveProgramName(nameString, OREF_NULL, OREF_NULL);
574  ProtectedObject n(resolvedName);
575  package = instance->loadRequires(activity, nameString, resolvedName);
576  }
577  else
578  {
579  // add this to the instance context
580  RexxArray *sourceArray = arrayArgument(_source, OREF_positional, OREF_SOURCE);
581  ProtectedObject p(sourceArray);
582  package = instance->loadRequires(activity, nameString, sourceArray);
583  }
584 
585  ProtectedObject p_package(package);
586 
587  /* Give new object its behaviour */
588  package->setBehaviour(((RexxClass *)this)->getInstanceBehaviour());
589  if (((RexxClass *)this)->hasUninitDefined())
590  {
591  package->hasUninit(); /* Make sure everyone is notified. */
592  }
593  /* now send an INIT message */
594  package->sendMessage(OREF_INIT, init_args, initCount, named_argCount);
595  return package; /* return the new method */
596 }
